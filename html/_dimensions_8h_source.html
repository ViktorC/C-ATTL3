32 template<typename IndexType, std::size_t Rank>
33 class Dimensions : public DimExpression<Dimensions<IndexType,Rank>,IndexType,Rank> {
34  template<typename OtherIndexType, std::size_t OtherRank>
35  friend class Dimensions;
36 public:
37  inline Dimensions() :
38  values(Rank, 1) { }
39  inline Dimensions(const std::initializer_list<IndexType>& values) :
40  Dimensions() {
41  assert(values.size() <= Rank);
42  std::copy(values.begin(), values.end(), this->values.begin());
43  }
44  inline Dimensions(const std::array<IndexType,Rank>& array) :
45  Dimensions() {
46  assert(array.size() <= Rank);
47  std::copy(array.begin(), array.end(), values.begin());
48  }
49  template<typename OtherDerived>
50  inline Dimensions(const DimExpression<OtherDerived,IndexType,Rank>& dims) :
51  Dimensions() {
52  for (std::size_t i = 0; i < Rank; ++i)
53  values[i] = dims(i);
54  }
55  template<std::size_t Ranks = 1>
56  inline Dimensions<IndexType,Rank + Ranks> promote() const {
57  Dimensions<IndexType,Rank + Ranks> promoted;
58  std::copy(values.begin(), values.end(), promoted.values.begin() + Ranks);
59  return promoted;
60  }
61  template<std::size_t Ranks = 1>
62  inline Dimensions<IndexType,Rank - Ranks> demote() const {
63  static_assert(Rank > Ranks, "rank must be greater than the number of ranks to demote by");
64  Dimensions<IndexType,Rank - Ranks> demoted;
65  std::copy(values.begin() + Ranks, values.end(), demoted.values.begin());
66  return demoted;
67  }
68  inline IndexType operator()(std::size_t i) const {
69  if (i < 0 || i >= Rank)
70  throw std::invalid_argument("illegal index value: " + i);
71  return values[i];
72  }
73  inline IndexType& operator()(std::size_t i) {
74  if (i < 0 || i >= Rank)
75  throw std::invalid_argument("illegal index value: " + i);
76  return values[i];
77  }
78  inline operator std::array<IndexType,Rank>() const {
79  std::array<IndexType,Rank> array;
80  std::copy(values.begin(), values.end(), array.begin());
81  return array;
82  }
83 private:
84  std::vector<IndexType> values;
85 };

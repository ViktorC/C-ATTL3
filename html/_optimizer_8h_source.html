39 template<typename Scalar, std::size_t Rank, bool Sequential>
40 class Optimizer {
41  static_assert(std::is_floating_point<Scalar>::value, "non floating-point scalar type");
42  static_assert(Rank > 0 && Rank < 4, "illegal optimizer rank");
43 protected:
44  typedef NeuralNetwork<Scalar,Rank,Sequential> Net;
45  typedef DataProvider<Scalar,Rank,Sequential> Provider;
46  typedef Tensor<Scalar,Rank + Sequential + 1> Data;
47 public:
48  inline Optimizer(LossSharedPtr<Scalar,Rank,Sequential> loss) :
49  loss(loss) {
50  assert(loss != nullptr);
51  }
52  virtual ~Optimizer() = default;
53  inline bool verify_gradients(Net& net, Provider& provider, Scalar step_size = 1e-5, Scalar abs_epsilon = Utils<Scalar>::EPSILON2,
54  Scalar rel_epsilon = Utils<Scalar>::EPSILON3) const {
55  assert(net.get_input_dims() == provider.get_obs_dims());
56  assert(net.get_output_dims() == provider.get_obj_dims());
57  assert(step_size > 0);
58  assert(abs_epsilon >= 0 && rel_epsilon > 0);
59  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(provider.instances());
60  provider.reset();
61  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
62  * back-propagate is to be divided by the number of observations in the batch. */
63  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) /
64  (Scalar) provider.instances());
65  bool failure = false;
66  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
67  for (unsigned i = 0; i < layers.size(); ++i) {
68  Layer<Scalar,Rank>& layer = *(layers[i]);
69  if (layer.is_parametric()) {
70  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) <<
71  std::string(28, '-') << std::endl;
72  Matrix<Scalar>& params = layer.get_params();
73  const Matrix<Scalar>& param_grads = layer.get_param_grads();
74  for (int j = 0; j < params.rows(); ++j) {
75  for (int k = 0; k < params.cols(); ++k) {
76  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
77  Scalar ana_grad = param_grads(j,k);
78  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
79  Scalar param = params(j,k);
80  params(j,k) = param + step_size;
81  /* Compute the numerical gradients in training mode to ensure that the means
82  * and standard deviations used for batch normalization are the same as those
83  * used during the analytic gradient computation. */
84  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true),
85  data_pair.second).mean();
86  params(j,k) = param - step_size;
87  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true),
88  data_pair.second).mean();
89  params(j,k) = param;
90  Scalar num_grad = (loss_inc - loss_dec) / (2 * step_size);
91  std::cout << "\t\tNumerical gradient = " << num_grad;
92  if (!Utils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
93  std::cout << " *****FAIL*****";
94  failure = true;
95  }
96  std::cout << std::endl;
97  }
98  }
99  }
100  }
101  // Empty the network caches.
102  net.empty_caches();
103  return !failure;
104  }
105  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0) {
106  assert(net.get_input_dims() == training_prov.get_obs_dims());
107  assert(net.get_output_dims() == training_prov.get_obj_dims());
108  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
109  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
110  assert(epochs > 0);
111  // Fit the optimizer parameters to the network.
112  fit(net);
113  Scalar prev_test_loss = Utils<Scalar>::MAX;
114  unsigned cons_loss_inc = 0;
115  // Start the optimization iterations.
116  for (unsigned i = 0; i <= epochs; ++i) {
117  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
118  // Train.
119  if (i != 0) {
120  training_prov.reset();
121  std::cout << "\ttraining loss: " << std::to_string(train(net, training_prov, i)) << std::endl;
122  }
123  // Validate.
124  test_prov.reset();
125  Scalar test_loss = test(net, test_prov, i);
126  std::cout << "\ttest loss: " << std::to_string(test_loss);
127  if (test_loss >= prev_test_loss) {
128  cons_loss_inc++;
129  std::cout << " *****INCREASED LOSS*****";
130  if (early_stop > 0 && cons_loss_inc >= early_stop)
131  return prev_test_loss;
132  } else
133  cons_loss_inc = 0;
134  std::cout << std::endl << std::endl;
135  prev_test_loss = test_loss;
136  }
137  return prev_test_loss;
138  }
139 protected:
140  virtual void fit(Net& net) = 0;
141  virtual Scalar train(Net& net, Provider& training_prov, unsigned epoch) = 0;
142  virtual Scalar test(Net& net, Provider& test_prov, unsigned epoch) = 0;
143  inline static std::vector<Layer<Scalar,Rank>*> get_layers(Net& net) {
144  return net.get_layers();
145  }
146  inline static Data propagate(Net& net, Data in) {
147  return net.propagate(std::move(in), true);
148  }
149  inline static void backpropagate(Net& net, Data out_grads) {
150  net.backpropagate(std::move(out_grads));
151  }
152  inline static bool is_parametric(Layer<Scalar,Rank>& layer) {
153  return layer.is_parametric();
154  }
155  inline static Matrix<Scalar>& get_params(Layer<Scalar,Rank>& layer) {
156  return layer.get_params();
157  }
158  inline static const Matrix<Scalar>& get_param_grads(Layer<Scalar,Rank>& layer) {
159  return layer.get_param_grads();
160  }
161  inline static const void enforce_constraints(Layer<Scalar,Rank>& layer) {
162  layer.enforce_constraints();
163  }
164  const LossSharedPtr<Scalar,Rank,Sequential> loss;

170 template<typename Scalar, std::size_t Rank, bool Sequential>
171 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
172  typedef Optimizer<Scalar,Rank,Sequential> Base;
173 public:
174  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size) :
175  Base::Optimizer(loss),
176  reg(reg),
177  batch_size(batch_size) {
178  assert(reg != nullptr);
179  assert(batch_size > 0);
180  }
181  virtual ~SGDOptimizer() = default;
182 protected:
183  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch) {
184  // TODO Handle overflow.
185  Scalar training_loss = 0;
186  Scalar instances = (Scalar) training_prov.instances();
187  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
188  // Perform an entire training epoch.
189  while (training_prov.has_more()) {
190  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
191  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
192  training_loss += Base::loss->function(out, data_pair.second).sum();
193  /* Again, the loss on a batch is the mean of the losses on the observations in the batch and not their
194  * sum (see the last line of the function), thus the gradients of the loss function w.r.t the output of
195  * the network have to be divided by the number of instances in the batch. */
196  Base::backpropagate(net, Base::loss->d_function(out, data_pair.second) / instances);
197  for (unsigned k = 0; k < layers.size(); ++k) {
198  Layer<Scalar,Rank>& layer = *(layers[k]);
199  if (Base::is_parametric(layer)) {
200  update_params(layer, k, epoch - 1);
201  Base::enforce_constraints(layer);
202  }
203  }
204  }
205  return training_loss / instances;
206  }
207  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch) {
208  // TODO Handle overflow.
209  Scalar obj_loss = 0;
210  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
211  // Perform an entire test epoch.
212  while (test_prov.has_more()) {
213  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
214  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), data_pair.second).sum();
215  }
216  Scalar mean_obj_loss = obj_loss / test_prov.instances();
217  Scalar reg_loss = 0;
218  for (unsigned j = 0; j < layers.size(); ++j)
219  reg_loss += reg->function(Base::get_params(*(layers[j])));
220  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
221  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
222  return mean_obj_loss + reg_loss;
223  }
224  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
225  const RegPenSharedPtr<Scalar> reg;
226  const unsigned batch_size;
227 };

266  param_grads_vec = std::vector<Matrix<Scalar>>(layers.size());
267  for (unsigned i = 0; i < param_grads_vec.size(); ++i) {
268  Layer<Scalar,Rank>& layer = *(layers[i]);
269  Matrix<Scalar>& param_grads = param_grads_vec[i];
270  param_grads = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).rows(),
271  Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).cols());
272  param_grads.setZero(param_grads.rows(), param_grads.cols());
273  }
274  }
275  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
276  Scalar learning_rate = calculate_learning_rate(epoch);
277  Matrix<Scalar>& param_grads = param_grads_vec[i];
278  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
279  param_grads = momentum * param_grads - learning_rate * (Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
280  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params));
281  params += param_grads;
282  }
283  Scalar calculate_learning_rate(unsigned epoch) {
284  return init_learning_rate / (1.0 + annealing_rate * epoch);
285  }
286  const Scalar init_learning_rate;
287  const Scalar annealing_rate;
288  const Scalar momentum;
289  std::vector<Matrix<Scalar>> param_grads_vec;
290 };

381 template<typename Scalar, std::size_t Rank, bool Sequential>
382 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
383 public:
384  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
385  Scalar decay = 5e-2, Scalar epsilon = Utils<Scalar>::EPSILON2) :
386  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, reg, batch_size),
387  decay(decay),
388  epsilon(epsilon) {
389  assert(decay >= 0 && decay <= 1);
390  assert(epsilon > 0);
391  }
392 protected:
393  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
394  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
395  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
396  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
397  Layer<Scalar,Rank>& layer = *(layers[i]);
398  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
399  pgus.param_grad = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).rows(),
400  Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).cols());
401  pgus.param_grad.setZero(pgus.param_grad.rows(), pgus.param_grad.cols());
402  pgus.param_update = Matrix<Scalar>(pgus.param_grad.rows(), pgus.param_grad.cols());
403  pgus.param_update.setZero(pgus.param_update.rows(), pgus.param_update.cols());
404  }
405  }
406  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
407  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
408  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
409  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
410  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
411  pgus.param_grad = (1 - decay) * pgus.param_grad + decay * param_grads.cwiseProduct(param_grads);
412  Matrix<Scalar> weight_updates = -param_grads.array() * (pgus.param_update.array() + epsilon).sqrt() /
413  (pgus.param_grad.array() + epsilon).sqrt();
414  params += weight_updates;
415  pgus.param_update = (1 - decay) * pgus.param_update + decay * weight_updates.cwiseProduct(weight_updates);
416  }
417  const Scalar decay;
418  const Scalar epsilon;
419  struct ParamGradAndUpdateSqrs {
420  Matrix<Scalar> param_grad;
421  Matrix<Scalar> param_update;
422  };
423  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
424 };

426 template<typename Scalar, std::size_t Rank, bool Sequential>
427 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
428 public:
429  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
430  Scalar learning_rate = 1e-3, Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3,
431  Scalar epsilon = Utils<Scalar>::EPSILON2) :
432  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, reg, batch_size),
433  learning_rate(learning_rate),
434  l1_decay(l1_decay),
435  l2_decay(l2_decay),
436  epsilon(epsilon) {
437  assert(learning_rate > 0);
438  assert(l1_decay >= 0 && l1_decay <= 1);
439  assert(l2_decay >= 0 && l2_decay <= 1);
440  assert(epsilon > 0);
441  }
442  virtual ~AdamOptimizer() = default;
443 protected:
444  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
445  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
446  pgn_vec = std::vector<ParamGradNorms>(layers.size());
447  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
448  Layer<Scalar,Rank>& layer = *(layers[i]);
449  ParamGradNorms& vel = pgn_vec[i];
450  vel.param_grad_l1 = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).rows(),
451  Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).cols());
452  vel.param_grad_l1.setZero(vel.param_grad_l1.rows(), vel.param_grad_l1.cols());
453  vel.param_grad_l2 = Matrix<Scalar>(vel.param_grad_l1.rows(), vel.param_grad_l1.cols());
454  vel.param_grad_l2.setZero(vel.param_grad_l2.rows(), vel.param_grad_l2.cols());
455  }
456  }
457  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
458  ParamGradNorms& grad_norms = pgn_vec[i];
459  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - l1_decay, epoch + 1) + epsilon);
460  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - l2_decay, epoch + 1) + epsilon);
461  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
462  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
463  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
464  grad_norms.param_grad_l1 = (1 - l1_decay) * grad_norms.param_grad_l1 + l1_decay * param_grads;
465  grad_norms.param_grad_l2 = (1 - l2_decay) * grad_norms.param_grad_l2 +
466  l2_decay * param_grads.cwiseProduct(param_grads);
467  params -= (learning_rate * (grad_norms.param_grad_l1 * l1_corr).array() /
468  ((grad_norms.param_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
469  }
470  const Scalar learning_rate;
471  const Scalar l1_decay;
472  const Scalar l2_decay;
473  const Scalar epsilon;
474  struct ParamGradNorms {
475  Matrix<Scalar> param_grad_l1;
476  Matrix<Scalar> param_grad_l2;
477  };
478  std::vector<ParamGradNorms> pgn_vec;
479 };

515  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 1) + Base::epsilon);
516  Scalar l1_next_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 2) + Base::epsilon);
517  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - Base::l2_decay, epoch + 1) + Base::epsilon);
518  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
519  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
520  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
521  grad_norms.param_grad_l1 = (1 - Base::l1_decay) * grad_norms.param_grad_l1 + Base::l1_decay * param_grads;
522  grad_norms.param_grad_l2 = (1 - Base::l2_decay) * grad_norms.param_grad_l2 + Base::l2_decay *
523  param_grads.cwiseProduct(param_grads);
524  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * param_grads +
525  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.param_grad_l1).array() /
526  ((grad_norms.param_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();
527  }

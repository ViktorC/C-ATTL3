74  inline static bool almost_equal(Scalar n1, Scalar n2, Scalar abs_epsilon = EPSILON1,
75  Scalar rel_epsilon = EPSILON1) {
76  Scalar diff = std::abs(n1 - n2);
77  if (diff <= abs_epsilon)
78  return true;
79  Scalar max = std::max(std::abs(n1), std::abs(n2));
80  return diff <= max * rel_epsilon;
81  }

159  template<std::size_t Rank>
160  inline static Tensor<Scalar,Rank - 1> join_first_two_ranks(Tensor<Scalar,Rank> tensor) {
161  static_assert(Rank > 1, "illegal tensor rank");
162  Dimensions<int,Rank> dims = get_dims<Rank>(tensor);
163  int lowest_dim = dims(0);
164  Dimensions<int,Rank - 1> joined_dims = dims.template demote<>();
165  joined_dims(0) *= lowest_dim;

179  inline static void shuffle_mat_rows(Matrix<Scalar>& mat) {
180  Eigen::PermutationMatrix<Eigen::Dynamic,Eigen::Dynamic> perm(mat.rows());
181  perm.setIdentity();
182  // Shuffle the indices of the identity matrix.
183  std::random_shuffle(perm.indices().data(), perm.indices().data() + perm.indices().size());
184  mat = perm * mat;
185  }
186  template<std::size_t Rank>
187  inline static void shuffle_tensor_rows(Tensor<Scalar,Rank>& tensor) {
188  static_assert(Rank > 1, "illegal tensor rank");
189  Dimensions<int,Rank - 1> dims = get_dims<Rank>(tensor).template demote<>();
190  Matrix<Scalar> mat = map_tensor_to_mat<Rank>(std::move(tensor));
191  shuffle_mat_rows(mat);
192  tensor = map_mat_to_tensor<Rank>(std::move(mat), dims);
193  }

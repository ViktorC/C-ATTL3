127 class FCLayer : public KernelLayer<Scalar,Rank> {
128  typedef KernelLayer<Scalar,Rank> Base;
129  typedef Tensor<Scalar,Rank + 1> Data;
130 public:
131  inline FCLayer(const Dimensions<int,Rank>& input_dims, std::size_t output_size, WeightInitSharedPtr<Scalar> weight_init,
132  Scalar max_norm_constraint = 0) :
133  Base::KernelLayer(input_dims, Dimensions<int,Rank>({ (int) output_size }), weight_init,
134  input_dims.get_volume() + 1, output_size, max_norm_constraint) { }
135  inline Layer<Scalar,Rank>* clone() const {
136  return new FCLayer(*this);
137  }
138 protected:
139  inline void empty_cache() {
140  biased_in = Matrix<Scalar>(0, 0);
141  }
142  inline Data pass_forward(Data in, bool training) {
143  assert(Utils<Scalar>::template get_dims<Rank + 1>(in).template demote<>() == Base::input_dims);
144  assert(in.dimension(0) > 0);
145  unsigned input_size = Base::input_dims.get_volume();
146  // Add a 1-column to the input for the bias trick.
147  biased_in = Matrix<Scalar>(in.dimension(0), input_size + 1);
148  biased_in.leftCols(input_size) = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(in));
149  biased_in.col(input_size).setOnes();
150  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>((biased_in * Base::weights).eval(), Base::output_dims);
151  }
152  inline Data pass_back(Data out_grads) {
153  assert(Utils<Scalar>::template get_dims<Rank + 1>(out_grads).template demote<>() == Base::output_dims);
154  assert(out_grads.dimension(0) > 0 && biased_in.rows() == out_grads.dimension(0));
155  Matrix<Scalar> out_grads_mat = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(out_grads));
156  // Compute the gradients of the outputs with respect to the weights.
157  Base::weight_grads = biased_in.transpose() * out_grads_mat;
158  if (Base::is_input_layer())
159  return Utils<Scalar>::template get_null_tensor<Rank + 1>();
160  /* Remove the bias row from the weight matrix, transpose it, and compute gradients w.r.t. the
161  * previous layer's output. */
162  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>((out_grads_mat *
163  Base::weights.topRows(Base::input_dims.get_volume()).transpose()).eval(), Base::input_dims);
164  }
165 private:
166  // Staged computation caches
167  Matrix<Scalar> biased_in;
168 };

171 template<typename Scalar>
172 class ConvLayer : public KernelLayer<Scalar,3> {
173  typedef KernelLayer<Scalar,3> Base;
174  typedef Tensor<Scalar,4> Data;
175  typedef std::array<int,4> RankwiseArray;
176 public:
177  inline ConvLayer(const Dimensions<int,3>& input_dims, std::size_t filters, WeightInitSharedPtr<Scalar> weight_init,
178  std::size_t receptor_size = 3, std::size_t padding = 1, std::size_t stride = 1, std::size_t dilation = 0,
179  Scalar max_norm_constraint = 0) :
180  /* For every filter, there is a column in the weight matrix with the same number of
181  * elements as the size of the receptive field (F * F * D) + 1 for the bias row. */
182  Base::KernelLayer(input_dims, Dimensions<int,3>({ calculate_output_dim(input_dims(0), receptor_size, padding, dilation, stride),
183  calculate_output_dim(input_dims(1), receptor_size, padding, dilation, stride), (int) filters }), weight_init,
184  receptor_size * receptor_size * input_dims(2) + 1, filters, max_norm_constraint),
185  filters(filters),
186  receptor_size(receptor_size),
187  padding(padding),
188  stride(stride),
189  dilation(dilation),
190  padded_height(input_dims(0) + 2 * padding),
191  padded_width(input_dims(1) + 2 * padding),
192  dil_receptor_size(receptor_size + (receptor_size - 1) * dilation) {
193  assert(filters > 0);
194  assert(receptor_size > 0);
195  assert(stride > 0);
196  assert(input_dims(1) + 2 * padding >= receptor_size + (receptor_size - 1) * dilation &&
197  input_dims(2) + 2 * padding >= receptor_size + (receptor_size - 1) * dilation);
198  }
199  inline Layer<Scalar,3>* clone() const {
200  return new ConvLayer(*this);
201  }
202 protected:
203  inline void empty_cache() {
204  biased_in_vec = std::vector<Matrix<Scalar>>(0);
205  }
206  inline Data pass_forward(Data in, bool training) {
207  assert(Utils<Scalar>::template get_dims<4>(in).template demote<>() == Base::input_dims);
208  assert(in.dimension(0) > 0);
209  // Spatial padding.
210  std::array<std::pair<int,int>,4> paddings;
211  paddings[0] = std::make_pair(0, 0);
212  paddings[1] = std::make_pair(padding, padding);
213  paddings[2] = std::make_pair(padding, padding);
214  paddings[3] = std::make_pair(0, 0);
215  Data padded_in = in.pad(paddings);
216  // Free the memory occupied by the now-expendable input tensor.
217  in = Utils<Scalar>::template get_null_tensor<4>();
218  int rows = padded_in.dimension(0);
219  int patches = Base::output_dims(0) * Base::output_dims(1);
220  int receptor_vol = receptor_size * receptor_size * Base::input_dims(2);
221  int height_rem = padded_height - dil_receptor_size;
222  int width_rem = padded_width - dil_receptor_size;
223  // Prepare the base offsets and extents for slicing and dilation.
224  RankwiseArray row_offsets({ 0, 0, 0, 0 });
225  RankwiseArray row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
226  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, padded_in.dimension(3) });
227  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
228  RankwiseArray dil_strides({ 1, (int) dilation + 1, (int) dilation + 1, 1 });
229  Data out(padded_in.dimension(0), Base::output_dims(0), Base::output_dims(1), Base::output_dims(2));
230  biased_in_vec = std::vector<Matrix<Scalar>>(padded_in.dimension(0));
231  /* 'Tensor-row' by 'tensor-row', stretch the receptor locations into row vectors, form a matrix out of
232  * them, and multiply it by the weight matrix. */
233  for (int i = 0; i < rows; ++i) {
234  row_offsets[0] = i;
235  patch_offsets[0] = i;
236  int patch_ind = 0;
237  {
238  Matrix<Scalar> in_mat_i(patches, receptor_vol);
239  for (int j = 0; j <= height_rem; j += stride) {
240  patch_offsets[1] = j;
241  for (int k = 0; k <= width_rem; k += stride) {
242  patch_offsets[2] = k;
243  Data patch;
244  // If the patch is dilated, skip the 'internal padding' when stretching it into a row.
245  if (dilation > 0)
246  patch = padded_in.slice(patch_offsets, patch_extents).stride(dil_strides);
247  else
248  patch = padded_in.slice(patch_offsets, patch_extents);
249  in_mat_i.row(patch_ind++) = Utils<Scalar>::template map_tensor_to_mat<4>(std::move(patch));
250  }
251  }
252  assert(patch_ind == patches);
253  // Set the appended column's elements to 1.
254  biased_in_vec[i] = Matrix<Scalar>(patches, receptor_vol + 1);
255  Matrix<Scalar>& biased_in = biased_in_vec[i];
256  biased_in.col(receptor_vol).setOnes();
257  biased_in.leftCols(receptor_vol) = in_mat_i;
258  }
259  /* Flatten the matrix product into a row vector, reshape it into a 'single-row' sub-tensor, and
260  * assign it to the output tensor's corresponding 'row'. */
261  Matrix<Scalar> out_i = biased_in_vec[i] * Base::weights;
262  out.slice(row_offsets, row_extents) = Utils<Scalar>::template map_mat_to_tensor<4>(MatrixMap<Scalar>(out_i.data(),
263  1, out_i.rows() * out_i.cols()).matrix(), Base::output_dims);
264  }
265  return out;
266  }
267  inline Data pass_back(Data out_grads) {
268  assert(Utils<Scalar>::template get_dims<4>(out_grads).template demote<>() == Base::output_dims);
269  assert(out_grads.dimension(0) > 0 && biased_in_vec.size() == (unsigned) out_grads.dimension(0));
270  int rows = out_grads.dimension(0);
271  int depth = Base::input_dims(2);
272  int height_rem = padded_height - dil_receptor_size;
273  int width_rem = padded_width - dil_receptor_size;
274  Dimensions<int,3> comp_patch_dims({ (int) receptor_size, (int) receptor_size, Base::input_dims(2) });
275  RankwiseArray out_grads_row_offsets({ 0, 0, 0, 0 });
276  RankwiseArray out_grads_row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
277  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, depth });
278  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
279  RankwiseArray strides({ 1, (int) dilation + 1, (int) dilation + 1, 1 });
280  Data prev_out_grads(rows, padded_height, padded_width, depth);
281  prev_out_grads.setZero();
282  Base::weight_grads.setZero(Base::weight_grads.rows(), Base::weight_grads.cols());
283  for (int i = 0; i < rows; ++i) {
284  out_grads_row_offsets[0] = i;
285  Matrix<Scalar> prev_out_grads_mat_i;
286  {
287  Data slice_i = out_grads.slice(out_grads_row_offsets, out_grads_row_extents);
288  MatrixMap<Scalar> out_grads_mat_map_i = MatrixMap<Scalar>(slice_i.data(), Base::output_dims(0) *
289  Base::output_dims(1), filters);
290  // Accumulate the gradients of the outputs w.r.t. the weights for each observation a.k.a 'tensor-row'.
291  Base::weight_grads += biased_in_vec[i].transpose() * out_grads_mat_map_i;
292  if (Base::is_input_layer())
293  continue;
294  /* Remove the bias row from the weight matrix, transpose it, and compute the gradients w.r.t. the
295  * previous layer's output. */
296  prev_out_grads_mat_i = out_grads_mat_map_i * Base::weights.topRows(Base::weights.rows() - 1).transpose();
297  }
298  /* Given the gradients w.r.t. the stretched out receptor patches, perform a 'backwards' convolution
299  * to get the gradients w.r.t. the individual input nodes. */
300  int patch_ind = 0;
301  patch_offsets[0] = i;
302  for (int j = 0; j <= height_rem; j += stride) {
303  patch_offsets[1] = j;
304  for (int k = 0; k <= width_rem; k += stride) {
305  patch_offsets[2] = k;
306  // Accumulate the gradients where the receptor-patch-tensors overlap.
307  if (dilation > 0) {
308  prev_out_grads.slice(patch_offsets, patch_extents).stride(strides) +=
309  Utils<Scalar>::template map_mat_to_tensor<4>(prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
310  } else {
311  prev_out_grads.slice(patch_offsets, patch_extents) += Utils<Scalar>::template map_mat_to_tensor<4>(
312  prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
313  }
314  }
315  }
316  assert(patch_ind == prev_out_grads_mat_i.rows());
317  }
318  // Cut off the padding.
319  RankwiseArray no_padding_offsets({ 0, (int) padding, (int) padding, 0 });
320  RankwiseArray no_padding_extents({ rows, Base::input_dims(0), Base::input_dims(1), Base::input_dims(2) });
321  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
322  }
323  static int calculate_output_dim(int input_dim, int receptor_size, int padding, int dilation, int stride) {
324  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
325  }
326 private:
327  const std::size_t filters;
328  const std::size_t receptor_size;
329  const std::size_t padding;
330  const std::size_t stride;
331  const std::size_t dilation;
332  const int padded_height;
333  const int padded_width;
334  const int dil_receptor_size;
335  // Staged computation caches
336  std::vector<Matrix<Scalar>> biased_in_vec;
337 };

494 template<typename Scalar, std::size_t Rank>
495 class SoftmaxActivationLayer : public ActivationLayer<Scalar,Rank> {
496 public:
497  inline SoftmaxActivationLayer(const Dimensions<int,Rank>& dims, Scalar epsilon = Utils<Scalar>::EPSILON2) :
498  ActivationLayer<Scalar,Rank>::ActivationLayer(dims),
499  epsilon(epsilon) { }
500  inline Layer<Scalar,Rank>* clone() const {
501  return new SoftmaxActivationLayer(*this);
502  }
503 protected:
504  inline Matrix<Scalar> activate(const Matrix<Scalar>& in) {
505  /* First subtract the value of the greatest coefficient from each element row-wise
506  * to avoid an overflow due to raising e to great powers. */
507  Matrix<Scalar> out = (in.array().colwise() - in.array().rowwise().maxCoeff()).exp();
508  return out.array().colwise() / (out.array().rowwise().sum() + epsilon);
509  }
510  inline Matrix<Scalar> d_activate(const Matrix<Scalar>& in, const Matrix<Scalar>& out,
511  const Matrix<Scalar>& out_grads) {

636  typedef Tensor<Scalar,4> Data;
637  typedef std::array<int,4> RankwiseArray;
638 public:

640  input_dims(input_dims),

643  receptor_size(receptor_size),
644  stride(stride),
645  receptor_area(receptor_size * receptor_size),
646  height_rem(input_dims(0) - receptor_size),
647  width_rem(input_dims(1) - receptor_size),
648  input_layer(false),
649  params(0, 0),
650  param_grads(0, 0) {
651  assert(input_dims(0) >= (int) receptor_size && input_dims(1) >= (int) receptor_size);
652  assert(receptor_size > 0);
653  assert(stride > 0);
654  }
655  inline const Dimensions<int,3>& get_input_dims() const {
656  return input_dims;
657  }
658  inline const Dimensions<int,3>& get_output_dims() const {
659  return output_dims;
660  }
661 protected:
662  virtual void init_cache() = 0;
663  virtual Scalar reduce(const RowVector<Scalar>& patch, unsigned patch_ind) = 0;
664  virtual RowVector<Scalar> d_reduce(Scalar grad, unsigned patch_ind) = 0;

667  }
668  inline void set_input_layer(bool input_layer) {
669  this->input_layer = input_layer;
670  }
671  inline void init() { }
672  inline Matrix<Scalar>& get_params() {
673  return params;
674  }
675  inline Matrix<Scalar>& get_param_grads() {
676  return param_grads;
677  }
678  inline void enforce_constraints() { };
679  inline Data pass_forward(Data in, bool training) {
680  assert(Utils<Scalar>::template get_dims<4>(in).template demote<>() == input_dims);
681  assert(in.dimension(0) > 0);
682  rows = in.dimension(0);
683  int depth = input_dims(2);
684  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
685  RankwiseArray patch_extents({ 1, (int) receptor_size, (int) receptor_size, 1 });
686  Data out(rows, output_dims(0), output_dims(1), output_dims(2));
687  init_cache();
688  int patch_ind = 0;
689  for (int i = 0; i < rows; ++i) {
690  patch_offsets[0] = i;
691  int out_j = 0;
692  for (int j = 0; j <= height_rem; j += stride, ++out_j) {
693  patch_offsets[1] = j;
694  int out_k = 0;
695  for (int k = 0; k <= width_rem; k += stride, ++out_k) {
696  patch_offsets[2] = k;
697  for (int l = 0; l < depth; ++l) {
698  patch_offsets[3] = l;
699  Data patch = in.slice(patch_offsets, patch_extents);
700  out(i,out_j,out_k,l) = reduce(Utils<Scalar>::template map_tensor_to_mat<4>(std::move(patch)), patch_ind++);
701  }
702  }
703  }
704  }
705  return out;
706  }
707  inline Data pass_back(Data out_grads) {
708  assert(Utils<Scalar>::template get_dims<4>(out_grads).template demote<>() == output_dims);
709  assert(out_grads.dimension(0) > 0 && rows == out_grads.dimension(0));
710  if (input_layer)
711  return Utils<Scalar>::template get_null_tensor<4>();
712  int depth = input_dims(2);
713  Dimensions<int,3> patch_dims({ (int) receptor_size, (int) receptor_size, 1 });
714  RankwiseArray patch_offsets({ 0, 0, 0, 0});
715  RankwiseArray patch_extents({ 1, patch_dims(0), patch_dims(1), patch_dims(2) });
716  Data prev_out_grads(rows, input_dims(0), input_dims(1), depth);
717  prev_out_grads.setZero();
718  int patch_ind = 0;
719  for (int i = 0; i < rows; ++i) {
720  patch_offsets[0] = i;
721  int out_grads_j = 0;
722  for (int j = 0; j <= height_rem; j += stride, ++out_grads_j) {
723  patch_offsets[1] = j;
724  int out_grads_k = 0;
725  for (int k = 0; k <= width_rem; k += stride, ++out_grads_k) {
726  patch_offsets[2] = k;
727  for (int l = 0; l < depth; ++l) {
728  patch_offsets[3] = l;
729  prev_out_grads.slice(patch_offsets, patch_extents) +=
730  Utils<Scalar>::template map_mat_to_tensor<4>(d_reduce(out_grads(i,out_grads_j,out_grads_k,l),
731  patch_ind++), patch_dims);
732  }
733  }
734  }
735  }
736  return prev_out_grads;
737  }
738  static int calculate_output_dim(int input_dim, int receptor_size, int stride) {
739  return (input_dim - receptor_size) / stride + 1;
740  }
741  const Dimensions<int,3> input_dims;
742  const Dimensions<int,3> output_dims;
743  const std::size_t receptor_size;
744  const std::size_t stride;
745  const int receptor_area;
746  const int height_rem;
747  const int width_rem;
748  bool input_layer;
749  // No actual parameters.
750  Matrix<Scalar> params;
751  Matrix<Scalar> param_grads;
752  // Keep track of the input rows.
753  int rows;
754 };

834 template<typename Scalar, std::size_t Rank>
835 class BatchNormLayerBase : public Layer<Scalar,Rank> {
836 public:
837  virtual ~BatchNormLayerBase() = default;
838  inline const Dimensions<int,Rank>& get_input_dims() const {
839  return dims;
840  }
841  inline const Dimensions<int,Rank>& get_output_dims() const {
842  return dims;
843  }
844 protected:
845  typedef Tensor<Scalar,Rank + 1> Data;
846  inline BatchNormLayerBase(const Dimensions<int,Rank>& dims, int depth, Scalar norm_avg_decay, Scalar epsilon) :
847  dims(dims),
848  depth(depth),
849  norm_avg_decay(norm_avg_decay),
850  epsilon(epsilon),
851  input_layer(false),
852  avg_means(depth, dims.get_volume() / depth),
853  avg_inv_sds(depth, dims.get_volume() / depth),
854  avgs_init(false),
855  params(2 * depth, dims.get_volume() / depth),
856  param_grads(2 * depth, dims.get_volume() / depth),
857  cache_vec(depth) {
858  assert(norm_avg_decay >= 0 && norm_avg_decay <= 1 &&
859  "norm avg decay must not be less than 0 or greater than 1");
860  assert(epsilon > 0 && "epsilon must be greater than 0");
861  }
862  inline bool is_input_layer() const {
863  return input_layer;
864  }
865  inline void set_input_layer(bool input_layer) {
866  this->input_layer = input_layer;
867  }
868  inline void init() {
869  for (int i = 0; i < params.rows(); i += 2) {
870  // Gamma
871  params.row(i).setOnes();
872  // Beta
873  params.row(i + 1).setZero();
874  }
875  param_grads.setZero(params.rows(), params.cols());
876  avg_means.setZero(avg_means.rows(), avg_means.cols());
877  avg_inv_sds.setZero(avg_means.rows(), avg_inv_sds.cols());
878  avgs_init = false;
879  }
880  inline void empty_cache() {
881  for (unsigned i = 0; i < cache_vec.size(); ++i) {
882  Cache& cache = cache_vec[i];
883  cache.inv_in_sd = RowVector<Scalar>(0);
884  cache.std_in = Matrix<Scalar>(0, 0);
885  }
886  }
887  inline Matrix<Scalar>& get_params() {
888  return params;
889  }
890  inline Matrix<Scalar>& get_param_grads() {
891  return param_grads;
892  }
893  inline void enforce_constraints() { }
894  inline Data _pass_forward(Data in, const Dimensions<int,Rank>& output_dims, bool training, int i) {
895  Matrix<Scalar> in_ch_i = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(in));
896  if (training) {
897  Cache& cache = cache_vec[i];
898  RowVector<Scalar> means = in_ch_i.colwise().mean();
899  Matrix<Scalar> norm_in = in_ch_i.rowwise() - means;
900  cache.inv_in_sd = (norm_in.array().square().colwise().mean() + epsilon).sqrt().inverse();
901  cache.std_in = norm_in * cache.inv_in_sd.asDiagonal();
902  in_ch_i = cache.std_in;
903  // Maintain a moving average of means and variances for testing.
904  if (avgs_init) {
905  avg_means.row(i) = (1.0 - norm_avg_decay) * avg_means.row(i) + norm_avg_decay * means;
906  avg_inv_sds.row(i) = (1.0 - norm_avg_decay) * avg_inv_sds.row(i) + norm_avg_decay *
907  cache.inv_in_sd;
908  } else {
909  avg_means.row(i) = means;
910  avg_inv_sds.row(i) = cache.inv_in_sd;
911  avgs_init = true;
912  }
913  } else // For testing, use the moving averages.
914  in_ch_i = (in_ch_i.rowwise() - avg_means.row(i)) * avg_inv_sds.row(i).asDiagonal();
915  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(((in_ch_i * params.row(2 * i).asDiagonal())
916  .rowwise() + params.row(2 * i + 1)).eval(), output_dims);
917  }
918  inline Data _pass_back(Data out_grads, const Dimensions<int,Rank>& output_dims, int i) {
919  int rows = out_grads.dimension(0);
920  Cache& cache = cache_vec[i];
921  Matrix<Scalar> std_in_grads_i;
922  /* Back-propagate the gradient through the batch normalization 'function' and also calculate the
923  * gradients on the betas and gammas. */
924  { // Manage memory by scope restriction.
925  Matrix<Scalar> out_grads_ch_map_i = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(out_grads));
926  param_grads.row(2 * i) = out_grads_ch_map_i.cwiseProduct(cache.std_in).colwise().sum();
927  param_grads.row(2 * i + 1) = out_grads_ch_map_i.colwise().sum();
928  if (input_layer)
929  return Utils<Scalar>::template get_null_tensor<Rank + 1>();
930  std_in_grads_i = out_grads_ch_map_i * params.row(2 * i).asDiagonal();
931  }
932  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(((((rows * std_in_grads_i).rowwise() -
933  std_in_grads_i.colwise().sum()) - cache.std_in *
934  (cache.std_in.cwiseProduct(std_in_grads_i).colwise().sum().asDiagonal())) *
935  ((1.0 / rows) * cache.inv_in_sd).asDiagonal()).eval(), output_dims);
936  }
937  const Dimensions<int,Rank> dims;
938  const int depth;
939  const Scalar norm_avg_decay;
940  const Scalar epsilon;
941  bool input_layer;
942  // Dynamic batch normalization parameters.
943  Matrix<Scalar> avg_means;
944  Matrix<Scalar> avg_inv_sds;
945  bool avgs_init;
946  // Betas and gammas
947  Matrix<Scalar> params;
948  Matrix<Scalar> param_grads;
949  // Staged computation cache_vec
950  struct Cache {
951  RowVector<Scalar> inv_in_sd;
952  Matrix<Scalar> std_in;
953  };
954  std::vector<Cache> cache_vec;
955 };

993  assert(in.dimension(0) > 0);

997  int rows = in.dimension(0);
998  typename Base::Data out(rows, Base::dims(0), Base::dims(1), Base::dims(2));
999  Dimensions<int,3> slice_dims({ Base::dims(0), Base::dims(1), 1 });
1000  std::array<int,4> offsets({ 0, 0, 0, 0 });
1001  std::array<int,4> extents({ rows, slice_dims(0), slice_dims(1), slice_dims(2) });
1002  for (int i = 0; i < Base::depth; ++i) {
1003  offsets[3] = i;
1004  typename Base::Data in_slice_i = in.slice(offsets, extents);
1005  out.slice(offsets, extents) = Base::_pass_forward(std::move(in_slice_i), slice_dims, training, i);
1006  }
1007  return out;
1008  }
1009  }
1010  inline typename Base::Data pass_back(typename Base::Data out_grads) {
1011  assert(Utils<Scalar>::template get_dims<4>(out_grads).template demote<>() == Base::dims);
1012  assert(out_grads.dimension(0) > 0 && Base::cache_vec[0].std_in.rows() == out_grads.dimension(0));
1013  if (Base::depth == 1)
1014  return Base::_pass_back(std::move(out_grads), Base::dims, 0);
1015  else {
1016  int rows = out_grads.dimension(0);
1017  typename Base::Data prev_out_grads;
1018  if (!Base::is_input_layer())

1100  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(out_grads))

1102  }

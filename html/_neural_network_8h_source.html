263  assert(input_dims == lane.get_input_dims());
264  const typename Base::Dims& lane_output_dims = lane.get_output_dims();
265  if (merge_type != SUM) {
266  if (merge_type == CONCAT_HI_RANK) {
267  for (std::size_t i = 0; i < concat_rank; ++i)
268  assert(output_dims(i) == lane_output_dims(i));
269  } else {
270  for (std::size_t i = Rank - 1; i > concat_rank; --i)
271  assert(output_dims(i) == lane_output_dims(i));
272  }
273  output_dims(concat_rank) += lane_output_dims(concat_rank);
274  } else
275  assert(output_dims == lane_output_dims);
276  }

343  inline typename Base::Data propagate(typename Base::Data input, bool training) {
344  Utils<Scalar>::template check_dim_validity<Base::DATA_RANKS>(input);
345  assert(input_dims == Utils<Scalar>::template get_dims<Base::DATA_RANKS>(input).template demote<>());
346  int rows = input.dimension(0);
347  RankwiseArray offsets;
348  RankwiseArray extents = output_dims.template promote<>();
349  offsets.fill(0);
350  extents[0] = rows;
351  typename Base::Data out;
352  if (merge_type != SUM)
353  out = typename Base::Data(extents);
354  unsigned lane_num = lanes.size();
355  unsigned helper_thread_num = lane_num - 1;
356  pthread_t threads[helper_thread_num];
357  pthread_attr_t attr;
358  if (helper_thread_num > 0) {
359  pthread_attr_init(&attr);
360  pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
361  }
362  PropArgs args_arr[lane_num];
363  for (int i = helper_thread_num; i >= 0; --i) {
364  PropArgs args;
365  args.obj = this;
366  args.lane_id = i;
367  args.training = training;
368  args.in = &input;
369  args_arr[i] = args;
370  // Leave the first lane to the main thread.
371  if (i == 0)
372  propagate(&args_arr[i]);
373  else
374  assert(!pthread_create(&threads[i - 1], &attr, propagate, &args_arr[i]));
375  }
376  for (unsigned i = 0; i < lane_num; ++i) {
377  if (i != 0)
378  assert(!pthread_join(threads[i - 1], nullptr));
379  if (merge_type != SUM) {
380  int concat_rank_dim = lanes[i]->get_output_dims()(concat_rank);
381  extents[concat_batch_rank] = concat_rank_dim;
382  out.slice(offsets, extents) = args_arr[i].out;
383  offsets[concat_batch_rank] += concat_rank_dim;
384  } else {
385  if (i == 0)
386  out = std::move(args_arr[i].out);
387  else
388  out += args_arr[i].out;
389  }
390  }
391  if (helper_thread_num > 0)
392  pthread_attr_destroy(&attr);
393  return out;
394  }

710  bool foremost = true) :
711  modules(modules),
712  concat_type(concat_type),
713  concat_rank(concat_type == LOWEST_RANK ? 0 : Rank - 1),

725  } else {
726  for (std::size_t i = 0; i < concat_rank; ++i)
727  assert(input_dims(i) == output_dims(i));
728  }
729  for (unsigned i = 1; i < this->modules.size(); ++i) {
730  Module& module = this->modules[i];
731  const typename Base::Dims& module_input_dims = module.get_input_dims();

765  layers.push_back(module_layers[j]);

771  Dimensions<int,Base::DATA_RANKS> dims = Utils<Scalar>::template get_dims<Base::DATA_RANKS>(input);
772  assert(input_dims == dims.template demote<>());

774  RankwiseArray extents = dims;
775  offsets.fill(0);
776  for (unsigned i = 0; i < modules.size(); ++i) {
777  Module& module = modules[i];
778  int layer_input_concat_rank_dim = module.get_input_dims()(concat_rank);
779  int layer_output_concat_rank_dim = module.get_output_dims()(concat_rank);
780  RankwiseArray out_i_sizes = extents;
781  out_i_sizes[concat_batch_rank] = layer_input_concat_rank_dim + layer_output_concat_rank_dim;
782  typename Base::Data out_i(out_i_sizes);
783  offsets[concat_batch_rank] = 0;
784  extents[concat_batch_rank] = layer_input_concat_rank_dim;
785  out_i.slice(offsets, extents) = input;
786  offsets[concat_batch_rank] = layer_input_concat_rank_dim;
787  extents[concat_batch_rank] = layer_output_concat_rank_dim;
788  out_i.slice(offsets, extents) = module.propagate(std::move(input), training);
789  input = typename Base::Data(std::move(out_i));
790  }
791  return input;
792  }
793  inline typename Base::Data backpropagate(typename Base::Data out_grads) {
794  Utils<Scalar>::template check_dim_validity<Base::DATA_RANKS>(out_grads);
795  assert(output_dims == Utils<Scalar>::template get_dims<Base::DATA_RANKS>(out_grads).template demote<>());
796  RankwiseArray offsets;
797  RankwiseArray extents = input_dims.template promote<>();
798  offsets.fill(0);
799  extents[0] = out_grads.dimension(0);
800  for (int i = modules.size() - 1; i >= 0; --i) {
801  Module& module = modules[i];
802  int layer_input_concat_rank_dim = module.get_input_dims()(concat_rank);
803  int layer_output_concat_rank_dim = module.get_output_dims()(concat_rank);
804  offsets[concat_batch_rank] = layer_input_concat_rank_dim;
805  extents[concat_batch_rank] = layer_output_concat_rank_dim;
806  typename Base::Data out_grads_i = out_grads.slice(offsets, extents);
807  offsets[concat_batch_rank] = 0;
808  extents[concat_batch_rank] = layer_input_concat_rank_dim;
809  out_grads = typename Base::Data(out_grads.slice(offsets, extents) +
810  module.backpropagate(std::move(out_grads_i)));
811  }
812  return out_grads;
813  }

816  const std::size_t concat_rank;

818  bool foremost;

820  typename Base::Dims output_dims;

1272  // Calculate the output sequence length and delay based on the input sequence length.
1273  std::pair<std::size_t,std::size_t> output_seq_info = output_seq_size_func((std::size_t) input_seq_length);
1274  int output_seq_length = (int) output_seq_info.first;
1275  int output_seq_delay = (int) output_seq_info.second;
1276  assert(output_seq_length > 0);
1277  if (reversed) {
1278  RankwiseBoolArray reverse;
1279  reverse.fill(false);
1280  reverse[1] = true;
1281  input = input.reverse(reverse);
1282  }
1283  int output_end = output_seq_length + output_seq_delay;
1284  int time_steps = std::max(input_seq_length, output_end);
1285  // If in training mode, unroll the updated network.
1286  if (training) {
1287  cells = std::vector<Cell>(time_steps - 1);
1288  if (time_steps > 1) {
1289  // Empty the caches of the main cell to reduce the amount of data to copy.
1290  empty_caches();
1291  cells = std::vector<Cell>(time_steps - 1);
1292  // Unroll the network by creating n -1 copies of the main cell;
1293  for (int j = 1; j < time_steps; ++j) {
1294  Cell& cell = cells[j - 1];
1295  cell.state_kernel = KernelPtr<Scalar,Rank>(
1296  (KernelLayer<Scalar,Rank>*) main_cell.state_kernel->clone());
1297  cell.state_act = ActivationPtr<Scalar,Rank>(
1298  (ActivationLayer<Scalar,Rank>*) main_cell.state_act->clone());
1299  // Only copy the kernels and activations that will actually be used.
1300  if (j < input_seq_length)
1301  cell.input_kernel = KernelPtr<Scalar,Rank>(
1302  (KernelLayer<Scalar,Rank>*) main_cell.input_kernel->clone());
1303  if (j >= output_seq_delay && j < output_end) {
1304  cell.output_kernel = KernelPtr<Scalar,Rank>(
1305  (KernelLayer<Scalar,Rank>*) main_cell.output_kernel->clone());
1306  cell.output_act = ActivationPtr<Scalar,Rank>(
1307  (ActivationLayer<Scalar,Rank>*) main_cell.output_act->clone());
1308  }
1309  }
1310  } else
1311  cells = std::vector<Cell>(1);
1312  }
1313  // If the network is stateful and we are in training mode, retain the state.
1314  if (!training || !stateful || batch_size == -1) {
1315  Dimensions<int,Rank + 1> dims = main_cell.input_kernel->get_output_dims().template promote<>();
1316  dims(0) = samples;
1317  state = Tensor<Scalar,Rank + 1>(dims);
1318  state.setZero();
1319  } else if (samples != batch_size) {
1320  std::array<int,Rank + 1> offsets;
1321  std::array<int,Rank + 1> extents = main_cell.input_kernel->get_output_dims().template promote<>();
1322  offsets.fill(0);
1323  extents[0] = samples;
1324  TimeStepData new_state;
1325  if (samples > batch_size) {
1326  new_state = Tensor<Scalar,Rank + 1>(extents);
1327  new_state.setZero();
1328  extents[0] = batch_size;
1329  new_state.slice(offsets, extents) = state;
1330  } else
1331  new_state = state.slice(offsets, extents);
1332  state = std::move(new_state);
1333  }
1334  RankwiseIntArray input_offsets;
1335  RankwiseIntArray input_extents = data_dims;
1336  RankwiseIntArray output_offsets;
1337  RankwiseIntArray output_extents = output_dims.template promote<2>();
1338  input_offsets.fill(0);
1339  output_offsets.fill(0);
1340  input_extents[1] = 1;
1341  output_extents[0] = samples;
1342  typename Root::Data out;
1343  // If the output is a single time step prediction, there is no need to create an output tensor.
1344  if (output_seq_length > 1) {
1345  output_extents[1] = output_seq_length;
1346  out = typename Root::Data(output_extents);
1347  }
1348  output_extents[1] = 1;
1349  Dimensions<int,Rank + 1> input_time_step_dims = input_dims.template promote<>();
1350  input_time_step_dims(0) = samples;
1351  EvalThreadPoolPtr thread_pool = Utils<Scalar>::get_eval_thread_pool();
1352  EvalThreadPoolDevice thread_pool_device(thread_pool.get(), Utils<Scalar>::num_of_eval_threads());
1353  for (int i = 0; i < time_steps; ++i) {
1354  // In inference mode, do not unroll the network.
1355  Cell& cell = !training || i == 0 ? main_cell : cells[i - 1];
1356  // Always apply the state kernel.
1357  state = Root::pass_forward(*cell.state_kernel, std::move(state), training);
1358  // If in inference mode, empty the caches after passing the data through each layer.
1359  if (!training)
1360  Root::empty_cache(*cell.state_kernel);
1361  // If there is an input for the time step...
1362  if (i < input_seq_length) {
1363  typename Root::Data in_i_seq;
1364  if (input_seq_length == 1)
1365  in_i_seq = std::move(input);
1366  else {
1367  in_i_seq = input.slice(input_offsets, input_extents);
1368  input_offsets[1] += 1;
1369  }
1370  if (mul_int) {
1371  if (training) {
1372  /* If multiplicative integration is enabled, cache the factors of the multiplication so that
1373  * the function can be differentiated in the backward pass. */
1374  cell.state_kernel_cache = state;
1375  cell.input_kernel_cache = Root::pass_forward(*cell.input_kernel,
1376  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1377  input_time_step_dims), training);
1378  state.device(thread_pool_device) = state * cell.input_kernel_cache;
1379  } else
1380  state.device(thread_pool_device) = state * Root::pass_forward(*cell.input_kernel,
1381  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1382  input_time_step_dims), training);
1383  } else
1384  state.device(thread_pool_device) += Root::pass_forward(*cell.input_kernel,
1385  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1386  input_time_step_dims), training);
1387  if (!training)
1388  Root::empty_cache(*cell.input_kernel);
1389  }
1390  state = Root::pass_forward(*cell.state_act, std::move(state), training);
1391  if (!training)
1392  Root::empty_cache(*cell.state_act);
1393  // If there is an output for the time step...
1394  if (i >= output_seq_delay && i < output_end) {
1395  TimeStepData out_i = Root::pass_forward(*cell.output_kernel, state, training);
1396  if (!training)
1397  Root::empty_cache(*cell.output_kernel);
1398  // If the output is a single time step prediction, just return it.
1399  if (output_seq_length == 1)
1400  out = Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1401  Root::pass_forward(*cell.output_act, std::move(out_i), training),
1402  output_extents);
1403  else {
1404  out.slice(output_offsets, output_extents) = Utils<Scalar>::template map_tensor_to_tensor<
1405  Rank + 1,Root::DATA_RANKS>(Root::pass_forward(*cell.output_act, std::move(out_i),
1406  training), output_extents);
1407  output_offsets[1] += 1;
1408  }
1409  if (!training)
1410  Root::empty_cache(*cell.output_act);
1411  }
1412  }
1413  batch_size = samples;
1414  this->input_seq_length = input_seq_length;
1415  this->output_seq_length = output_seq_length;
1416  this->output_seq_delay = output_seq_delay;
1417  return out;

1419  inline typename Root::Data backpropagate(typename Root::Data out_grads) {
1420  Utils<Scalar>::template check_dim_validity<Root::DATA_RANKS>(out_grads);
1421  Dimensions<int,Root::DATA_RANKS> data_dims = Utils<Scalar>::template get_dims<Root::DATA_RANKS>(out_grads);
1422  assert(output_dims == data_dims.template demote<2>() && batch_size == data_dims(0) &&
1423  output_seq_length == data_dims(1));
1424  RankwiseIntArray output_offsets;
1425  RankwiseIntArray output_extents = data_dims;
1426  RankwiseIntArray input_offsets;
1427  RankwiseIntArray input_extents = input_dims.template promote<2>();
1428  output_offsets.fill(0);
1429  output_offsets[1] = output_seq_length - 1;
1430  input_offsets.fill(0);
1431  input_offsets[1] = input_seq_length - 1;
1432  output_extents[1] = 1;
1433  input_extents[0] = batch_size;
1434  typename Root::Data prev_out_grads;
1435  if (input_seq_length > 1) {
1436  input_extents[1] = input_seq_length;
1437  prev_out_grads = typename Root::Data(input_extents);
1438  }
1439  input_extents[1] = 1;
1440  TimeStepData state_grads(state.dimensions());
1441  state_grads.setZero();
1442  Dimensions<int,Rank + 1> out_time_step_dims = output_dims.template promote<>();
1443  out_time_step_dims(0) = batch_size;
1444  int output_end = output_seq_length + output_seq_delay;
1445  int time_steps = std::max(input_seq_length, output_end);
1446  EvalThreadPoolPtr thread_pool = Utils<Scalar>::get_eval_thread_pool();
1447  EvalThreadPoolDevice thread_pool_device(thread_pool.get(), Utils<Scalar>::num_of_eval_threads());
1448  for (int i = time_steps - 1; i >= 0; --i) {
1449  Cell& cell = i == 0 ? main_cell : cells[i - 1];
1450  // If there was an output at the time step...
1451  if (i >= output_seq_delay && i < output_end) {
1452  typename Root::Data out_grads_seq_i;
1453  if (output_seq_length == 1)
1454  out_grads_seq_i = std::move(out_grads);
1455  else {
1456  out_grads_seq_i = out_grads.slice(output_offsets, output_extents);
1457  output_offsets[1] -= 1;
1458  }
1459  TimeStepData out_grads_i = Root::pass_back(*cell.output_act,
1460  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(
1461  std::move(out_grads_seq_i), out_time_step_dims));
1462  Root::empty_cache(*cell.output_act);
1463  state_grads.device(thread_pool_device) = Root::pass_back(*cell.output_kernel, std::move(out_grads_i)) +
1464  state_grads;
1465  Root::empty_cache(*cell.output_kernel);
1466  }
1467  // Always back-propagate the state gradient.
1468  state_grads = Root::pass_back(*cell.state_act, std::move(state_grads));
1469  Root::empty_cache(*cell.state_act);
1470  // If there was an input at the time step...
1471  if (i < input_seq_length) {
1472  // If it is the foremost layer, the gradients do not need to be propagated further back.
1473  if (foremost) {
1474  if (mul_int) { // Multiplicative integration.
1475  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads);
1476  cell.state_kernel_cache = Utils<Scalar>::template get_null_tensor<Rank + 1>();
1477  } else // Additive integration.
1478  Root::pass_back(*cell.input_kernel, state_grads);
1479  } else if (input_seq_length == 1) {
1480  if (mul_int) {
1481  prev_out_grads = Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1482  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads), input_extents);
1483  cell.state_kernel_cache = Utils<Scalar>::template get_null_tensor<Rank + 1>();
1484  } else
1485  prev_out_grads = Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1486  Root::pass_back(*cell.input_kernel, state_grads), input_extents);
1487  } else {
1488  if (mul_int) {
1489  prev_out_grads.slice(input_offsets, input_extents) =
1490  Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1491  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads), input_extents);
1492  cell.state_kernel_cache = Utils<Scalar>::template get_null_tensor<Rank + 1>();
1493  } else
1494  prev_out_grads.slice(input_offsets, input_extents) =
1495  Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1496  Root::pass_back(*cell.input_kernel, state_grads), input_extents);
1497  input_offsets[1] -= 1;
1498  }
1499  Root::empty_cache(*cell.input_kernel);
1500  }
1501  // Compute the gradients w.r.t. the state kernel.
1502  if (mul_int) {
1503  state_grads = Root::pass_back(*cell.state_kernel, cell.input_kernel_cache * state_grads);
1504  cell.input_kernel_cache = Utils<Scalar>::template get_null_tensor<Rank + 1>();
1505  } else
1506  state_grads = Root::pass_back(*cell.state_kernel, std::move(state_grads));
1507  Root::empty_cache(*cell.state_kernel);
1508  }
1509  // Roll the network up and accumulate the gradients.
1510  // TODO Single evaluation.
1511  Matrix<Scalar>& u_param_grads = Root::get_param_grads(*main_cell.input_kernel);
1512  Matrix<Scalar>& w_param_grads = Root::get_param_grads(*main_cell.state_kernel);
1513  Matrix<Scalar>& v_param_grads = Root::get_param_grads(*main_cell.output_kernel);
1514  Matrix<Scalar>& state_act_param_grads = Root::get_param_grads(*main_cell.state_act);
1515  Matrix<Scalar>& output_act_param_grads = Root::get_param_grads(*main_cell.output_act);
1516  for (int i = 1; i < time_steps; ++i) {
1517  Cell& cell = cells[i - 1];
1518  w_param_grads += Root::get_param_grads(*cell.state_kernel);
1519  state_act_param_grads += Root::get_param_grads(*cell.state_act);
1520  if (i < input_seq_length)

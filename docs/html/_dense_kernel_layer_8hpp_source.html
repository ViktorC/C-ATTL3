92  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
93  assert((Dimensions<std::size_t,Root::DATA_RANK>(in.dimensions()).template demote<>()) == Base::input_dims);
94  assert(in.dimension(0) > 0);
95  std::size_t rows = in.dimension(0);
96  out_conversion_dims[0] = rows;
97  prev_out_conversion_dims[0] = rows;
98  in_mat_cache = MatrixMap<Scalar>(in.data(), in.dimension(0), Base::input_dims.get_volume());
99  Matrix<Scalar> out_mat = (in_mat_cache * Base::weights->get_values()).rowwise() +
100  Base::bias->get_values().row(0);
101  return TensorMap<Scalar,Root::DATA_RANK>(out_mat.data(), out_conversion_dims);
102  }
103  inline typename Root::Data pass_back(typename Root::Data out_grad) {
104  assert((Dimensions<std::size_t,Root::DATA_RANK>(out_grad.dimensions()).template demote<>()) == Base::output_dims);
105  assert(out_grad.dimension(0) > 0 && out_conversion_dims[0] == out_grad.dimension(0));
106  // Compute the gradient of the outputs with respect to the weights and the bias.
107  MatrixMap<Scalar> out_grad_mat(out_grad.data(), out_grad.dimension(0), Base::output_dims.get_volume());
108  Base::weights->accumulate_grad(in_mat_cache.transpose() * out_grad_mat);
109  Base::bias->accumulate_grad(out_grad_mat.colwise().sum());
110  if (Base::is_input_layer())
111  return typename Root::Data();
112  // Compute the gradient of the previous layer's output.
113  Matrix<Scalar> prev_out_grad_mat = out_grad_mat * Base::weights->get_values().transpose();
114  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grad_mat.data(), prev_out_conversion_dims);
115  }

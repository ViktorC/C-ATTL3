35  inline ResidualNeuralNetwork(std::vector<Module>&& modules, bool foremost = true) :
36  modules(std::move(modules)),
37  foremost(foremost) {
38  assert(this->modules.size() > 0 && "modules must contain at least 1 element");
39  Base& first_module = *this->modules[0];
40  input_dims = first_module.get_input_dims();
41  output_dims = this->modules[this->modules.size() - 1]->get_output_dims();
42  first_module.set_foremost(foremost);
43  typename Base::Dims prev_dims = input_dims;
44  for (std::size_t i = 0; i < this->modules.size(); ++i) {
45  Base& module = *this->modules[i];
46  if (i != 0)
47  module.set_foremost(false);
48  assert(module.get_input_dims() == module.get_output_dims() &&
49  "residual module input-output dimension discrepancy");
50  assert(prev_dims == module.get_input_dims() && "incompatible module dimensions");
51  prev_dims = module.get_output_dims();
52  }
53  }

112  inline typename Base::Data propagate(typename Base::Data input, bool training) {
113  assert(input_dims == (Dimensions<std::size_t,Base::DATA_RANK>(input.dimensions()).template demote<>()));
114  for (std::size_t i = 0; i < modules.size(); ++i)
115  input += modules[i]->propagate(input, training);
116  return input;

118  inline typename Base::Data backpropagate(typename Base::Data out_grad) {
119  assert(output_dims == (Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()));
120  for (int i = modules.size() - 1; i >= 0; --i) {
121  if (foremost && i == 0)
122  return modules[i]->backpropagate(std::move(out_grad));
123  else
124  out_grad += modules[i]->backpropagate(out_grad);
125  }
126  return out_grad;
127  }

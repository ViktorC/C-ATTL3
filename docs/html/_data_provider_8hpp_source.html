112  typedef DataProvider<Scalar,Rank,Sequential> Base;
113 public:
114  inline PartitionDataProvider(Base& orig_provider, std::size_t offset, std::size_t length) :
115  orig_provider(orig_provider),

117  length(length) {

124  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {

128  return instances_read < length && orig_provider.has_more();

131  std::size_t instances_to_read = std::min(batch_size, length - instances_read);

134  }
135  inline void reset() {
136  orig_provider.reset();
137  orig_provider.skip(offset);
138  instances_read = 0;
139  }
140 protected:
141  inline void skip(std::size_t instances) {
142  orig_provider.skip(instances);
143  }
144 private:
145  Base& orig_provider;
146  const std::size_t offset;
147  const std::size_t length;
148  std::size_t instances_read;

156 template<typename Scalar, std::size_t Rank, bool Sequential, bool Shuffle = true>
157 class MemoryDataProvider : public DataProvider<Scalar,Rank,Sequential> {
158  typedef DataProvider<Scalar,Rank,Sequential> Base;
159 public:
167  inline MemoryDataProvider(typename Base::DataPtr obs, typename Base::DataPtr obj) :
168  obs(std::move(obs)),
169  obj(std::move(obj)),
170  offsets() {
171  assert(this->obs != nullptr);
172  assert(this->obj != nullptr);
173  assert(this->obs->dimension(0) == this->obj->dimension(0) &&
174  "mismatched data and obj tensor row numbers");
175  Dimensions<std::size_t,Base::DATA_RANK> obs_dims = this->obs->dimensions();
176  Dimensions<std::size_t,Base::DATA_RANK> obj_dims = this->obj->dimensions();
177  this->obs_dims = obs_dims.template demote<Sequential + 1>();
178  this->obj_dims = obj_dims.template demote<Sequential + 1>();
179  instances = (std::size_t) this->obs->dimension(0);
180  offsets.fill(0);
181  data_extents = obs_dims;
182  obj_extents = obj_dims;
183  if (Shuffle)
184  shuffle_tensor_rows();
185  }
186  inline const Dimensions<std::size_t,Rank>& get_obs_dims() const {
187  return obs_dims;
188  }
189  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {
190  return obj_dims;
191  }
192  inline bool has_more() {
193  return offsets[0] < (int) instances;
194  }
195  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
196  if (!has_more())
197  throw std::out_of_range("no more data left to fetch");
198  std::size_t max_batch_size = std::min(batch_size, instances - offsets[0]);
199  data_extents[0] = max_batch_size;
200  obj_extents[0] = max_batch_size;
201  typename Base::Data data_batch = obs->slice(offsets, data_extents);
202  typename Base::Data obj_batch = obj->slice(offsets, obj_extents);
203  offsets[0] = std::min(instances, offsets[0] + max_batch_size);
204  return std::make_pair(std::move(data_batch), std::move(obj_batch));
205  }
206  inline void reset() {
207  offsets[0] = 0;
208  }
209 protected:
210  inline void skip(std::size_t instances) {
211  offsets[0] = std::min((int) this->instances, (int) (offsets[0] + instances));
212  }
213  inline void shuffle_tensor_rows() {
214  std::size_t rows = obs->dimension(0);
215  MatrixMap<Scalar> obs_mat(obs->data(), rows, obs->size() / rows);
216  MatrixMap<Scalar> obj_mat(obj->data(), rows, obj->size() / rows);
217  // Create an identity matrix.
218  internal::PermMatrix perm(rows);
219  perm.setIdentity();
220  // Shuffle its indices.
221  std::random_shuffle(perm.indices().data(), perm.indices().data() + perm.indices().size());
222  // And apply the same row permutation transformation to both the observations and the objectives.
223  obs_mat = perm * obs_mat;
224  obj_mat = perm * obj_mat;
225  }
226 private:
227  typename Base::DataPtr obs;
228  typename Base::DataPtr obj;
229  Dimensions<std::size_t,Rank> obs_dims;
230  Dimensions<std::size_t,Rank> obj_dims;
231  std::size_t instances;
232  std::array<std::size_t,Base::DATA_RANK> offsets;
233  std::array<std::size_t,Base::DATA_RANK> data_extents;
234  std::array<std::size_t,Base::DATA_RANK> obj_extents;
235 };

243 template<typename Scalar, std::size_t Rank, bool Sequential, bool Binary = false>
244 class JointFileDataProvider : public DataProvider<Scalar,Rank,Sequential> {
245  typedef DataProvider<Scalar,Rank,Sequential> Base;
246 public:
247  virtual ~JointFileDataProvider() = default;
248  inline bool has_more() {
249  for (; current_stream < data_streams.size(); ++current_stream) {
250  std::ifstream& data_stream = data_streams[current_stream];
251  if (data_stream && data_stream.peek() != EOF)
252  return true;
253  }
254  return false;
255  }
256  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
257  if (!has_more())
258  throw std::out_of_range("no more data left to fetch");
259  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(data_streams[current_stream],
260  batch_size);
261  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
262  /* If the data contains fewer batches than expected, the end of the file has been reached and the
263  * rest of the data should be read from the next file. */
264  while (data_pair.first.dimension(0) < batch_size && has_more()) {
265  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(data_streams[current_stream],
266  batch_size - data_pair.first.dimension(0));
267  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
268  // It has to be evaluated into a temporary due to the dimension incompatibility.
269  typename Base::Data obs_concat =
270  data_pair.first.concatenate(std::move(add_data_pair.first),0);
271  data_pair.first = std::move(obs_concat);
272  typename Base::Data obj_concat =
273  data_pair.second.concatenate(std::move(add_data_pair.second), 0);
274  data_pair.second = std::move(obj_concat);
275  }
276  return data_pair;
277  }
278  inline void reset() {
279  for (std::size_t i = 0; i < data_streams.size(); ++i) {
280  std::ifstream& data_stream = data_streams[i];
281  data_stream.clear();
282  _set_to_beg(data_stream);
283  }
284  current_stream = 0;
285  }
286 protected:
287  inline JointFileDataProvider(std::vector<std::string> dataset_paths) :
288  data_streams(dataset_paths.size()),
289  current_stream(0) {
290  assert(!dataset_paths.empty());
291  for (std::size_t i = 0; i < dataset_paths.size(); ++i) {
292  data_streams[i] = std::ifstream(dataset_paths[i], Binary ? std::ios::binary : std::ios::in);
293  std::ifstream& data_stream = data_streams[i];
294  assert(data_stream.is_open());
295  }
296  }
297  inline JointFileDataProvider(std::string dataset_path) :
298  JointFileDataProvider({ dataset_path }) { }
304  virtual inline void _set_to_beg(std::ifstream& data_stream) {
305  data_stream.seekg(0, std::ios::beg);
306  }
316  virtual DataPair<Scalar,Rank,Sequential> _get_data(std::ifstream& data_stream,
317  std::size_t batch_size) = 0;
327  virtual std::size_t _skip(std::ifstream& data_stream, std::size_t instances) = 0;
328  inline void skip(std::size_t instances) {
329  if (!has_more())
330  return;
331  std::size_t skipped = _skip(data_streams[current_stream], instances);
332  while (skipped < instances && has_more())
333  skipped += _skip(data_streams[current_stream], instances - skipped);
334  }
335 private:
336  std::vector<std::ifstream> data_streams;
337  std::size_t current_stream;
338 };

347 class SplitFileDataProvider : public DataProvider<Scalar,Rank,Sequential> {
348  typedef DataProvider<Scalar,Rank,Sequential> Base;
349 public:
350  virtual ~SplitFileDataProvider() = default;
351  inline bool has_more() {
352  for (; current_stream_pair < data_stream_pairs.size(); ++current_stream_pair) {
353  std::pair<std::ifstream,std::ifstream>& stream_pair = data_stream_pairs[current_stream_pair];
354  if (stream_pair.first && (stream_pair.first.peek() != EOF) &&
355  stream_pair.second && (stream_pair.second.peek() != EOF))
356  return true;
357  }
358  return false;
359  }
360  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
361  if (!has_more())
362  throw std::out_of_range("no more data left to fetch");
363  std::pair<std::ifstream,std::ifstream>& first_stream_pair = data_stream_pairs[current_stream_pair];
364  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(first_stream_pair.first,
365  first_stream_pair.second, batch_size);
366  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
367  while (data_pair.first.dimension(0) < batch_size && has_more()) {
368  std::pair<std::ifstream,std::ifstream>& stream_pair = data_stream_pairs[current_stream_pair];
369  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(stream_pair.first,
370  stream_pair.second, batch_size - data_pair.first.dimension(0));
371  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
372  typename Base::Data obs_concat =
373  data_pair.first.concatenate(std::move(add_data_pair.first),0);
374  data_pair.first = std::move(obs_concat);
375  typename Base::Data obj_concat =
376  data_pair.second.concatenate(std::move(add_data_pair.second), 0);
377  data_pair.second = std::move(obj_concat);
378  }
379  return data_pair;
380  }
381  inline void reset() {
382  for (std::size_t i = 0; i < data_stream_pairs.size(); ++i) {
383  std::pair<std::ifstream,std::ifstream>& stream_pair = data_stream_pairs[i];
384  stream_pair.first.clear();
385  stream_pair.second.clear();
386  _set_to_beg(stream_pair.first, stream_pair.second);
387  }
388  current_stream_pair = 0;
389  }
390 protected:
391  inline SplitFileDataProvider(std::vector<std::pair<std::string,std::string>> dataset_path_pairs) :
392  data_stream_pairs(dataset_path_pairs.size()),
393  current_stream_pair(0) {
394  assert(!dataset_path_pairs.empty());
395  for (std::size_t i = 0; i < dataset_path_pairs.size(); ++i) {
396  std::pair<std::string,std::string>& path_pair = dataset_path_pairs[i];
397  std::ifstream obs_stream(path_pair.first, ObsBinary ? std::ios::binary : std::ios::in);
398  assert(obs_stream.is_open());
399  std::ifstream obj_stream(path_pair.second, ObjBinary ? std::ios::binary : std::ios::in);
400  assert(obj_stream.is_open());
401  data_stream_pairs[i] = std::make_pair(std::move(obs_stream), std::move(obj_stream));
402  }
403  }
404  inline SplitFileDataProvider(std::pair<std::string,std::string> dataset_path_pair) :
405  SplitFileDataProvider(std::vector<std::pair<std::string,std::string>>({ dataset_path_pair })) { }
413  virtual inline void _set_to_beg(std::ifstream& obs_input_stream, std::ifstream& obj_input_stream) {
414  obs_input_stream.seekg(0, std::ios::beg);
415  obj_input_stream.seekg(0, std::ios::beg);
416  }
428  virtual DataPair<Scalar,Rank,Sequential> _get_data(std::ifstream& obs_input_stream,
429  std::ifstream& obj_input_stream, std::size_t batch_size) = 0;
440  virtual std::size_t _skip(std::ifstream& obs_input_stream, std::ifstream& obj_input_stream,
441  std::size_t instances) = 0;
442  inline void skip(std::size_t instances) {
443  if (!has_more())
444  return;
445  std::pair<std::ifstream,std::ifstream>& first_stream_pair = data_stream_pairs[current_stream_pair];
446  std::size_t skipped = _skip(first_stream_pair.first, first_stream_pair.second, instances);
447  while (skipped < instances && has_more()) {
448  std::pair<std::ifstream,std::ifstream>& stream_pair = data_stream_pairs[current_stream_pair];
449  skipped += _skip(stream_pair.first, stream_pair.second, instances - skipped);
450  }
451  }
452 private:
453  std::vector<std::pair<std::ifstream,std::ifstream>> data_stream_pairs;
454  std::size_t current_stream_pair;

460 template<typename Scalar>
461 class MNISTDataProvider : public SplitFileDataProvider<Scalar,3,false,true,true> {
462  typedef DataProvider<Scalar,3,false> Root;
463  typedef SplitFileDataProvider<Scalar,3,false,true,true> Base;
464  static constexpr std::size_t OBS_OFFSET = 16;
465  static constexpr std::size_t LABEL_OFFSET = 8;
466  static constexpr std::size_t OBS_INSTANCE_LENGTH = 784;
467  static constexpr std::size_t LABEL_INSTANCE_LENGTH = 1;
468 public:
473  MNISTDataProvider(std::string obs_path, std::string labels_path) :
474  Base::SplitFileDataProvider(std::make_pair(obs_path, labels_path)),
475  obs({ 28u, 28u, 1u }),
476  obj({ 10u, 1u, 1u }),
477  offsets({ 0u, 0u, 0u, 0u }),
478  obs_extents({ 0u, 28u, 28u, 1u }),
479  obj_extents({ 0u, 10u, 1u, 1u }) {
480  Base::reset();
481  }
482  inline const Dimensions<std::size_t,3>& get_obs_dims() const {
483  return obs;
484  }
485  inline const Dimensions<std::size_t,3>& get_obj_dims() const {
486  return obj;
487  }
488 protected:
489  inline void _set_to_beg(std::ifstream& obs_input_stream, std::ifstream& obj_input_stream) {
490  Base::_set_to_beg(obs_input_stream, obj_input_stream);
491  obs_input_stream.ignore(OBS_OFFSET);
492  obj_input_stream.ignore(LABEL_OFFSET);
493  }
494  inline DataPair<Scalar,3,false> _get_data(std::ifstream& obs_input_stream,
495  std::ifstream& obj_input_stream, std::size_t batch_size) {
496  Tensor<Scalar,4> obs(batch_size, 28u, 28u, 1u);
497  Tensor<Scalar,4> obj(batch_size, 10u, 1u, 1u);
498  obj.setZero();
499  std::size_t i;
500  for (i = 0; i < batch_size && obs_input_stream.read(obs_buffer, OBS_INSTANCE_LENGTH); ++i) {
501  // Read and set the label.
502  char label;
503  obj_input_stream.read(&label, LABEL_INSTANCE_LENGTH);
504  obj(i,static_cast<std::size_t>(label),0u,0u) = (Scalar) 1;
505  // Set the image.
506  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(obs_buffer);
507  std::size_t buffer_ind = 0;
508  for (std::size_t height = 0; height < 28; ++height) {
509  for (std::size_t width = 0; width < 28; ++width)
510  obs(i,height,width,0u) = (Scalar) u_buffer[buffer_ind++];
511  }
512  assert(buffer_ind == OBS_INSTANCE_LENGTH);
513  }
514  if (i == batch_size)
515  return std::make_pair(obs, obj);
516  obs_extents[0] = i;
517  obj_extents[0] = i;
518  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
519  }
520  inline std::size_t _skip(std::ifstream& obs_input_stream, std::ifstream& obj_input_stream,
521  std::size_t instances) {
522  // Skip observations.
523  std::streampos curr_obs_pos = obs_input_stream.tellg();
524  obs_input_stream.seekg(0, std::ios::end);
525  std::size_t obs_skip_extent = obs_input_stream.tellg() - curr_obs_pos;
526  obs_input_stream.seekg(curr_obs_pos);
527  obs_input_stream.ignore(instances * OBS_INSTANCE_LENGTH);
528  std::size_t skipped_obs = std::min(instances, obs_skip_extent / OBS_INSTANCE_LENGTH);
529  // Skip labels.
530  std::streampos curr_label_pos = obj_input_stream.tellg();
531  obj_input_stream.seekg(0, std::ios::end);
532  std::size_t label_skip_extent = obj_input_stream.tellg() - curr_label_pos;
533  obj_input_stream.seekg(curr_label_pos);
534  obj_input_stream.ignore(instances * LABEL_INSTANCE_LENGTH);
535  std::size_t skipped_labels = std::min(instances, label_skip_extent / LABEL_INSTANCE_LENGTH);
536  assert(skipped_obs == skipped_labels);
537  return skipped_obs;
538  }
539 private:
540  const Dimensions<std::size_t,3> obs;
541  const Dimensions<std::size_t,3> obj;
542  char obs_buffer[OBS_INSTANCE_LENGTH];
543  std::array<std::size_t,4> offsets;
544  std::array<std::size_t,4> obs_extents;
545  std::array<std::size_t,4> obj_extents;
546 };

562  static constexpr std::size_t NUM_LABELS = CIFARType == CIFAR_10 ? 10 : 100;

580  CIFARDataProvider(std::vector<std::string>({ file_path })) { }

582  return obs;
583  }
584  inline const Dimensions<std::size_t,3>& get_obj_dims() const {
585  return obj;
586  }
587 protected:
588  inline DataPair<Scalar,3,false> _get_data(std::ifstream& data_stream,
589  std::size_t batch_size) {
590  Tensor<Scalar,4> obs(batch_size, 32u, 32u, 3u);
591  Tensor<Scalar,4> obj(batch_size, NUM_LABELS, 1u, 1u);
592  obj.setZero();
593  std::size_t i;
594  for (i = 0; i < batch_size && data_stream.read(buffer, INSTANCE_LENGTH); ++i) {
595  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(buffer);
596  std::size_t buffer_ind = 0;
597  // Set the label.
598  if (CIFARType == CIFAR_100)
599  buffer_ind++;
600  obj(i,u_buffer[buffer_ind++],0u,0u) = (Scalar) 1;
601  // Set the image.
602  for (std::size_t channel = 0; channel < 3; ++channel) {
603  for (std::size_t height = 0; height < 32; ++height) {
604  for (std::size_t width = 0; width < 32; ++width)
605  obs(i,height,width,channel) = (Scalar) u_buffer[buffer_ind++];
606  }
607  }
608  assert(buffer_ind == INSTANCE_LENGTH);
609  }
610  if (i == batch_size)
611  return std::make_pair(obs, obj);
612  obs_extents[0] = i;
613  obj_extents[0] = i;
614  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
615  }

617  std::streampos curr_pos = data_stream.tellg();
618  data_stream.seekg(0, std::ios::end);
619  std::size_t skip_extent = data_stream.tellg() - curr_pos;
620  data_stream.seekg(curr_pos);
621  data_stream.ignore(instances * INSTANCE_LENGTH);
622  return std::min(instances, skip_extent / INSTANCE_LENGTH);
623  }
624 private:
625  const Dimensions<std::size_t,3> obs;
626  const Dimensions<std::size_t,3> obj;
627  char buffer[INSTANCE_LENGTH];
628  std::array<std::size_t,4> offsets;
629  std::array<std::size_t,4> obs_extents;
630  std::array<std::size_t,4> obj_extents;
631 };

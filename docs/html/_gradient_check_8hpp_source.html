24 template<typename Scalar, std::size_t Rank, bool Sequential>
25 class GradientCheck {
26  static_assert(std::is_floating_point<Scalar>::value, "non floating-point scalar type");
27  static_assert(Rank > 0 && Rank < 4, "illegal optimizer rank");
28  typedef DataProvider<Scalar,Rank,Sequential> Provider;
29  typedef NeuralNetwork<Scalar,Rank,Sequential> Net;
30  typedef Loss<Scalar,Rank,Sequential> Loss;
31 public:
48  inline static bool verify_gradients(Provider& provider, Net& net, const Loss& loss, bool verbose = true,
49  Scalar step_size = NumericUtils<Scalar>::EPSILON2 / 2,
50  Scalar abs_epsilon = NumericUtils<Scalar>::EPSILON2,
51  Scalar rel_epsilon = NumericUtils<Scalar>::EPSILON2) {
52  assert(net.get_input_dims() == provider.get_obs_dims());
53  assert(net.get_output_dims() == provider.get_obj_dims());
54  assert(step_size > 0);
55  assert(abs_epsilon >= 0 && rel_epsilon > 0);
56  bool failure = false;
57  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
58  std::size_t instances = data_pair.first.dimension(0);
59  provider.reset();
60  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
61  * back-propagate is to be divided by the number of observations in the batch. */
62  net.backpropagate(loss.d_function(net.propagate(data_pair.first, true),
63  data_pair.second) / (Scalar) instances);
64  std::vector<Parameters<Scalar>*> params_vec = net.get_unique_optimizable_params();
65  for (std::size_t i = 0; i < params_vec.size(); ++i) {
66  Parameters<Scalar>& params = *(params_vec[i]);
67  if (verbose) {
68  std::cout << "Parameter Set " << std::setw(3) << std::to_string(i) <<
69  std::string(28, '-') << std::endl;
70  }
71  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
72  * parameters' gradient. */
73  params.regularize();
74  Matrix<Scalar> params_values = params.get_values();
75  const Matrix<Scalar>& params_grad = params.get_grad();
76  for (int j = 0; j < params_values.rows(); ++j) {
77  for (int k = 0; k < params_values.cols(); ++k) {
78  if (verbose)
79  std::cout << "\tparam[" << i << "," << j << "," << k << "]:" << std::endl;
80  Scalar ana_grad = params_grad(j,k);
81  if (verbose)
82  std::cout << "\t\tanalytic gradient = " << ana_grad << std::endl;
83  Scalar param = params_values(j,k);
84  params_values(j,k) = param + step_size;
85  params.set_values(params_values);
86  /* Compute the numerical gradients in training mode to ensure that the means and standard
87  * deviations used for batch normalization are the same as those used during the analytic
88  * gradient computation. */
89  Scalar loss_inc = loss.function(net.propagate(data_pair.first, true),
90  data_pair.second).mean();
91  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's
92  * parameters is included in the gradient. */
93  Scalar reg_pen_inc = params.get_regularization_penalty();
94  params_values(j,k) = param - step_size;
95  params.set_values(params_values);
96  Scalar loss_dec = loss.function(net.propagate(data_pair.first, true),
97  data_pair.second).mean();
98  Scalar reg_pen_dec = params.get_regularization_penalty();
99  params_values(j,k) = param;
100  params.set_values(params_values);
101  // Include the regularization penalty as well.
102  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
103  if (verbose)
104  std::cout << "\t\tnumerical gradient = " << num_grad;
105  if (!NumericUtils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
106  if (verbose)
107  std::cout << " *****FAIL*****";
108  failure = true;
109  }
110  if (verbose)
111  std::cout << std::endl;
112  }
113  }
114  params.reset_grad();
115  }
116  // Empty the network caches.
117  net.empty_caches();
118  return !failure;
119  }
120 };

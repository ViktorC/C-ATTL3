49  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
50  assert((Dimensions<std::size_t,Base::DATA_RANK>(in.dimensions()).template demote<>()) == Base::dims);
51  assert(in.dimension(0) > 0);
52  std::size_t rows = in.dimension(0);
53  conversion_dims[0] = rows;
54  MatrixMap<Scalar> in_mat(in.data(), rows, in.size() / rows);
55  /* First subtract the value of the greatest coefficient from each element row-wise
56  * to avoid an overflow due to raising e to great powers. */
57  Matrix<Scalar> act_in_mat = (in_mat.array().colwise() - in_mat.array().rowwise().maxCoeff()).exp();
58  act_in_mat = act_in_mat.array().colwise() / (act_in_mat.array().rowwise().sum() + epsilon);
59  out_mat_cache = std::move(act_in_mat);
60  return TensorMap<Scalar,Root::DATA_RANK>(out_mat_cache.data(), conversion_dims);
61  }
62  inline typename Root::Data pass_back(typename Root::Data out_grad) {
63  assert((Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()) == Base::dims);
64  assert(out_grad.dimension(0) > 0 && out_mat_cache.rows() == out_grad.dimension(0));
65  if (Base::is_input_layer())
66  return typename Root::Data();
67  std::size_t rows = out_grad.dimension(0);
68  MatrixMap<Scalar> out_grad_mat(out_grad.data(), rows, out_grad.size() / rows);
69  Matrix<Scalar> prev_out_grad_mat(rows, out_mat_cache.cols());
70  for (int i = 0; i < rows; ++i) {
71  RowVector<Scalar> row_i = out_mat_cache.row(i);
72  // FIXME Do not evaluate the expressions into a temporary variable.
73  Matrix<Scalar> jacobian = row_i.asDiagonal();
74  jacobian -= row_i.transpose() * row_i;
75  prev_out_grad_mat.row(i) = out_grad_mat.row(i) * jacobian;
76  }
77  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grad_mat.data(), conversion_dims);
78  }
79 private:
80  const Scalar epsilon;
81  RankwiseArray conversion_dims;
82  // Staged computation cache matrix.
83  Matrix<Scalar> out_mat_cache;
84 };

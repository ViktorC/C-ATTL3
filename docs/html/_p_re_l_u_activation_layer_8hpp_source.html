80  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
81  assert((Dimensions<std::size_t,Base::DATA_RANK>(in.dimensions()).template demote<>()) == Base::dims);
82  assert(in.dimension(0) > 0);
83  conversion_dims[0] = in.dimension(0);
84  in_mat_cache = MatrixMap<Scalar>(in.data(), conversion_dims[0], in.size() / conversion_dims[0]);
85  Matrix<Scalar> out_mat = in_mat_cache.cwiseMax(in_mat_cache * Base::params->get_values().asDiagonal());
86  return TensorMap<Scalar,Root::DATA_RANK>(out_mat.data(), conversion_dims);

88  inline typename Root::Data pass_back(typename Root::Data out_grad) {
89  assert((Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()) == Base::dims);
90  assert(out_grad.dimension(0) > 0 && conversion_dims[0] == out_grad.dimension(0));
91  MatrixMap<Scalar> out_grad_mat(out_grad.data(), conversion_dims[0], out_grad.size() / conversion_dims[0]);
92  Matrix<Scalar> prev_out_grad_mat = Matrix<Scalar>(in_mat_cache.rows(), in_mat_cache.cols());
93  const Matrix<Scalar>& alphas = Base::params->get_values();
94  Matrix<Scalar> alphas_grad = Matrix<Scalar>::Zero(1, out_grad_mat.cols());
95  for (int i = 0; i < in_mat_cache.cols(); ++i) {
96  for (int j = 0; j < in_mat_cache.rows(); ++j) {
97  Scalar in_mat_ji = in_mat_cache(j,i);
98  Scalar out_mat_ji = out_grad_mat(j,i);
99  if (in_mat_ji >= 0)
100  prev_out_grad_mat(j,i) = out_mat_ji;
101  else {
102  prev_out_grad_mat(j,i) = alphas(0,i) * out_mat_ji;
103  alphas_grad(0,i) += in_mat_ji * out_mat_ji;
104  }
105  }
106  }
107  Base::params->accumulate_grad(alphas_grad);
108  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grad_mat.data(), conversion_dims);
109  }
110 private:
111  RankwiseArray conversion_dims;
112  // Staged computation cache.

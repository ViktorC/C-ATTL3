75  inline bool verify_gradients(Net& net, Provider& provider, Scalar step_size = (internal::Utils<Scalar>::EPSILON2 +
76  internal::Utils<Scalar>::EPSILON3) / 2, Scalar abs_epsilon = internal::Utils<Scalar>::EPSILON2,
77  Scalar rel_epsilon = internal::Utils<Scalar>::EPSILON3) const {
78  assert(net.get_input_dims() == provider.get_obs_dims());
79  assert(net.get_output_dims() == provider.get_obj_dims());
80  assert(step_size > 0);
81  assert(abs_epsilon >= 0 && rel_epsilon > 0);
82  bool failure = false;
83  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
84  std::size_t instances = data_pair.first.dimension(0);
85  provider.reset();
86  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
87  * back-propagate is to be divided by the number of observations in the batch. */
88  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);
89  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
90  for (std::size_t i = 0; i < layers.size(); ++i) {
91  Layer<Scalar,Rank>& layer = *layers[i];
92  if (layer.is_parametric()) {
93  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) <<
94  std::string(28, '-') << std::endl;
95  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
96  * parameters' gradient. */
97  layer.regularize();
98  Matrix<Scalar>& params = layer.get_params();
99  const Matrix<Scalar>& params_grad = layer.get_params_grad();
100  for (int j = 0; j < params.rows(); ++j) {
101  for (int k = 0; k < params.cols(); ++k) {
102  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
103  Scalar ana_grad = params_grad(j,k);
104  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
105  Scalar param = params(j,k);
106  params(j,k) = param + step_size;
107  /* Compute the numerical gradients in training mode to ensure that the means and standard
108  * deviations used for batch normalization are the same as those used during the analytic
109  * gradient computation. */
110  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
111  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's parameters
112  * is included in the gradient. */
113  Scalar reg_pen_inc = layer.get_regularization_penalty();
114  params(j,k) = param - step_size;
115  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
116  Scalar reg_pen_dec = layer.get_regularization_penalty();
117  params(j,k) = param;
118  // Include the regularization penalty as well.
119  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
120  std::cout << "\t\tNumerical gradient = " << num_grad;
121  if (!internal::Utils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
122  std::cout << " *****FAIL*****";
123  failure = true;
124  }
125  std::cout << std::endl;
126  }
127  }
128  }
129  }
130  // Empty the network caches.
131  net.empty_caches();
132  return !failure;
133  }
147  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0) {
148  assert(net.get_input_dims() == training_prov.get_obs_dims());
149  assert(net.get_output_dims() == training_prov.get_obj_dims());
150  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());

153  // Fit the optimizer parameters to the network.
154  fit(net);
155  Scalar prev_test_loss = internal::Utils<Scalar>::MAX;
156  unsigned cons_loss_inc = 0;
157  // Start the optimization iterations.
158  for (unsigned i = 0; i <= epochs; ++i) {
159  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
160  // Train.
161  if (i != 0) {
162  training_prov.reset();
163  std::cout << "\ttraining loss: " << std::to_string(train(net, training_prov, i)) << std::endl;
164  }
165  // Validate.
166  test_prov.reset();
167  Scalar test_loss = test(net, test_prov, i);
168  std::cout << "\ttest loss: " << std::to_string(test_loss);
169  if (test_loss >= prev_test_loss) {
170  cons_loss_inc++;
171  std::cout << " *****INCREASED LOSS*****";
172  if (early_stop > 0 && cons_loss_inc >= early_stop)
173  return prev_test_loss;
174  } else
175  cons_loss_inc = 0;
176  std::cout << std::endl << std::endl;
177  prev_test_loss = test_loss;
178  }
179  // Empty the network caches.
180  net.empty_caches();
181  return prev_test_loss;
182  }

347 template<typename Scalar, std::size_t Rank, bool Sequential>
348 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
349  typedef Optimizer<Scalar,Rank,Sequential> Base;
350 public:
351  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size) :
352  Base::Optimizer(loss),
353  batch_size(batch_size) {
354  assert(batch_size > 0);
355  }
356  virtual ~SGDOptimizer() = default;
357 protected:
358  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch) {
359  Scalar training_loss = 0;
360  Scalar instances = 0;
361  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
362  // Perform an entire training epoch.
363  while (training_prov.has_more()) {
364  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
365  instances += data_pair.first.dimension(0);
366  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
367  training_loss += Base::loss->function(out, data_pair.second).sum();
368  /* Divide the gradient by the batch size to decouple the learning rate and the batch
369  * size hyper-parameters. Use the nominal batch size as the denominator even if the
370  * actual batch size is different (in case the number of samples in the data set is
371  * not divisible by the batch size and the last batch of the epoch contains fewer
372  * instances than the others) to make sure that the magnitude of the gradient is
373  * proportional to the batch size (just like its 'accuracy' is). */
374  Base::backpropagate(net, Base::loss->d_function(std::move(out), std::move(data_pair.second)) / (Scalar) batch_size);
375  for (unsigned k = 0; k < layers.size(); ++k) {
376  Layer<Scalar,Rank>& layer = *(layers[k]);
377  if (Base::is_parametric(layer)) {
378  Base::regularize(layer);
379  update_params(layer, k, epoch - 1);
380  Base::enforce_constraints(layer);
381  }
382  }
383  }
384  return training_loss / instances;
385  }
386  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch) {
387  Scalar obj_loss = 0;
388  Scalar instances = 0;
389  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
390  // Perform an entire test epoch.
391  while (test_prov.has_more()) {
392  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
393  instances += data_pair.first.dimension(0);
394  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), std::move(data_pair.second)).sum();
395  }
396  Scalar mean_obj_loss = obj_loss / instances;
397  Scalar reg_loss = 0;
398  for (unsigned j = 0; j < layers.size(); ++j)
399  reg_loss += Base::get_regularization_penalty(*layers[j]);
400  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
401  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
402  return mean_obj_loss + reg_loss;
403  }
412  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
413  const unsigned batch_size;
414 };

562  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
563  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
564  params_grad_sqrs_vec = std::vector<Matrix<Scalar>>(layers.size());
565  for (unsigned i = 0; i < params_grad_sqrs_vec.size(); ++i) {
566  Layer<Scalar,Rank>& layer = *(layers[i]);
567  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
568  Matrix<Scalar> params_grad_sqrs = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
569  params_grad_sqrs_vec[i] = params_grad_sqrs;
570  }
571  }
578  inline virtual void update_acc_params_grad_sqrs(Matrix<Scalar>& acc_params_grad_sqrs,
579  const Matrix<Scalar>& params_grad) {
580  // Accumulate the squares of the gradients.
581  acc_params_grad_sqrs += params_grad.cwiseProduct(params_grad);
582  }
583  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
584  Matrix<Scalar>& params_grad_sqrs = params_grad_sqrs_vec[i];
585  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
586  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
587  update_acc_params_grad_sqrs(params_grad_sqrs, params_grad);
588  params -= (learning_rate * params_grad.array() / (params_grad_sqrs.array().sqrt() + epsilon)).matrix();
589  }
590  const Scalar learning_rate;
591  const Scalar epsilon;
592  std::vector<Matrix<Scalar>> params_grad_sqrs_vec;
593 };

629 template<typename Scalar, std::size_t Rank, bool Sequential>
630 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
631 public:
641  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
642  Scalar decay = 5e-2, Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
643  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
644  decay(decay),
645  epsilon(epsilon) {
646  assert(decay >= 0 && decay <= 1);
647  assert(epsilon > 0);
648  }
649 protected:
650  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
651  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
652  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
653  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
654  Layer<Scalar,Rank>& layer = *(layers[i]);
655  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
656  ParamGradAndUpdateSqrs pgus;
657  pgus.params_grad = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
658  pgus.params_update = Matrix<Scalar>::Zero(pgus.params_grad.rows(), pgus.params_grad.cols());
659  pgus_vec[i] = pgus;
660  }
661  }
662  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
663  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
664  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
665  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
666  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
667  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
668  (pgus.params_grad.array() + epsilon).sqrt();
669  params += weight_updates;
670  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
671  }
672  const Scalar decay;
673  const Scalar epsilon;
677  struct ParamGradAndUpdateSqrs {
678  Matrix<Scalar> params_grad;
679  Matrix<Scalar> params_update;
680  };
681  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
682 };

687 template<typename Scalar, std::size_t Rank, bool Sequential>
688 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
689 public:
704  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
705  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
706  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
707  learning_rate(learning_rate),
708  l1_decay(l1_decay),
709  l2_decay(l2_decay),
710  epsilon(epsilon) {
711  assert(learning_rate > 0);
712  assert(l1_decay >= 0 && l1_decay <= 1);
713  assert(l2_decay >= 0 && l2_decay <= 1);
714  assert(epsilon > 0);
715  }
716  virtual ~AdamOptimizer() = default;
717 protected:
718  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
719  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
720  pgn_vec = std::vector<ParamGradNorms>(layers.size());
721  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
722  Layer<Scalar,Rank>& layer = *(layers[i]);
723  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
724  ParamGradNorms vel;
725  vel.params_grad_l1 = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
726  vel.params_grad_l2 = Matrix<Scalar>::Zero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
727  pgn_vec[i] = vel;
728  }
729  }
730  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
731  ParamGradNorms& grad_norms = pgn_vec[i];
732  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - l1_decay, epoch + 1) + epsilon);
733  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - l2_decay, epoch + 1) + epsilon);
734  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
735  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
736  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
737  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
738  l2_decay * params_grad.cwiseProduct(params_grad);
739  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
740  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
741  }
742  const Scalar learning_rate;
743  const Scalar l1_decay;
744  const Scalar l2_decay;
745  const Scalar epsilon;
750  struct ParamGradNorms {

761 class AdaMaxOptimizer : public AdamOptimizer<Scalar,Rank,Sequential> {
762  typedef AdamOptimizer<Scalar,Rank,Sequential> Base;
763 public:
778  inline AdaMaxOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
779  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
780  Base::AdamOptimizer(loss, batch_size, learning_rate, l1_decay, l2_decay, epsilon) { }
781 protected:
782  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
783  typename Base::ParamGradNorms& grad_norms = Base::pgn_vec[i];
784  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 1) + Base::epsilon);
785  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
786  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
787  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
788  grad_norms.params_grad_l2 = ((1 - Base::l2_decay) * grad_norms.params_grad_l2).cwiseMax(params_grad.cwiseAbs());
789  params -= (Base::learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
790  (grad_norms.params_grad_l2.array() + Base::epsilon)).matrix();
791  }
792 };

798 class NadamOptimizer : public AdamOptimizer<Scalar,Rank,Sequential> {
799  typedef AdamOptimizer<Scalar,Rank,Sequential> Base;
800 public:
815  inline NadamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
816  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
817  Base::AdamOptimizer(loss, batch_size, learning_rate, l1_decay, l2_decay, epsilon) { }
818 protected:
819  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
820  typename Base::ParamGradNorms& grad_norms = Base::pgn_vec[i];
821  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 1) + Base::epsilon);
822  Scalar l1_next_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 2) + Base::epsilon);
823  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - Base::l2_decay, epoch + 1) + Base::epsilon);
824  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
825  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
826  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
827  grad_norms.params_grad_l2 = (1 - Base::l2_decay) * grad_norms.params_grad_l2 + Base::l2_decay *
828  params_grad.cwiseProduct(params_grad);
829  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * params_grad +
830  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.params_grad_l1).array() /
831  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();
832  }
833 };

73  inline bool verify_gradients(Net& net, Provider& provider, Scalar step_size = 1e-5, Scalar abs_epsilon = Utils<Scalar>::EPSILON2,
74  Scalar rel_epsilon = Utils<Scalar>::EPSILON3) const {
75  assert(net.get_input_dims() == provider.get_obs_dims());
76  assert(net.get_output_dims() == provider.get_obj_dims());
77  assert(step_size > 0);
78  assert(abs_epsilon >= 0 && rel_epsilon > 0);
79  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(provider.instances());
80  provider.reset();
81  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
82  * back-propagate is to be divided by the number of observations in the batch. */
83  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) /
84  (Scalar) provider.instances());
85  bool failure = false;
86  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
87  for (unsigned i = 0; i < layers.size(); ++i) {
88  Layer<Scalar,Rank>& layer = *(layers[i]);
89  if (layer.is_parametric()) {
90  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) <<
91  std::string(28, '-') << std::endl;
92  Matrix<Scalar>& params = layer.get_params();
93  const Matrix<Scalar>& param_grads = layer.get_param_grads();
94  for (int j = 0; j < params.rows(); ++j) {
95  for (int k = 0; k < params.cols(); ++k) {
96  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
97  Scalar ana_grad = param_grads(j,k);
98  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
99  Scalar param = params(j,k);
100  params(j,k) = param + step_size;
101  /* Compute the numerical gradients in training mode to ensure that the means
102  * and standard deviations used for batch normalization are the same as those
103  * used during the analytic gradient computation. */
104  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true),
105  data_pair.second).mean();
106  params(j,k) = param - step_size;
107  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true),
108  data_pair.second).mean();
109  params(j,k) = param;
110  Scalar num_grad = (loss_inc - loss_dec) / (2 * step_size);
111  std::cout << "\t\tNumerical gradient = " << num_grad;
112  if (!Utils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
113  std::cout << " *****FAIL*****";
114  failure = true;
115  }
116  std::cout << std::endl;
117  }
118  }
119  }
120  }
121  // Empty the network caches.
122  net.empty_caches();
123  return !failure;
124  }
138  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0) {
139  assert(net.get_input_dims() == training_prov.get_obs_dims());
140  assert(net.get_output_dims() == training_prov.get_obj_dims());
141  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
142  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());

144  // Fit the optimizer parameters to the network.
145  fit(net);

147  unsigned cons_loss_inc = 0;
148  // Start the optimization iterations.
149  for (unsigned i = 0; i <= epochs; ++i) {
150  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
151  // Train.
152  if (i != 0) {
153  training_prov.reset();

159  std::cout << "\ttest loss: " << std::to_string(test_loss);

161  cons_loss_inc++;
162  std::cout << " *****INCREASED LOSS*****";
163  if (early_stop > 0 && cons_loss_inc >= early_stop)
164  return prev_test_loss;
165  } else
166  cons_loss_inc = 0;
167  std::cout << std::endl << std::endl;
168  prev_test_loss = test_loss;
169  }
170  return prev_test_loss;
171  }

315 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
316  typedef Optimizer<Scalar,Rank,Sequential> Base;
317 public:
318  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size) :
319  Base::Optimizer(loss),
320  reg(reg),
321  batch_size(batch_size) {
322  assert(reg != nullptr);
323  assert(batch_size > 0);
324  }
325  virtual ~SGDOptimizer() = default;
326 protected:
327  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch) {
328  // TODO Handle overflow.
329  Scalar training_loss = 0;
330  Scalar instances = (Scalar) training_prov.instances();
331  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
332  // Perform an entire training epoch.
333  while (training_prov.has_more()) {
334  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
335  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
336  training_loss += Base::loss->function(out, data_pair.second).sum();
337  /* Again, the loss on a batch is the mean of the losses on the observations in the batch and not their
338  * sum (see the last line of the function), thus the gradients of the loss function w.r.t the output of
339  * the network have to be divided by the number of instances in the batch. */
340  Base::backpropagate(net, Base::loss->d_function(out, data_pair.second) / instances);
341  for (unsigned k = 0; k < layers.size(); ++k) {
342  Layer<Scalar,Rank>& layer = *(layers[k]);
343  if (Base::is_parametric(layer)) {
344  update_params(layer, k, epoch - 1);
345  Base::enforce_constraints(layer);
346  }
347  }
348  }
349  return training_loss / instances;
350  }
351  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch) {
352  // TODO Handle overflow.
353  Scalar obj_loss = 0;
354  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
355  // Perform an entire test epoch.
356  while (test_prov.has_more()) {
357  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
358  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), data_pair.second).sum();
359  }
360  Scalar mean_obj_loss = obj_loss / test_prov.instances();
361  Scalar reg_loss = 0;
362  for (unsigned j = 0; j < layers.size(); ++j)
363  reg_loss += reg->function(Base::get_params(*(layers[j])));
364  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
365  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
366  return mean_obj_loss + reg_loss;
367  }
376  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
377  const RegPenSharedPtr<Scalar> reg;
378  const unsigned batch_size;
379 };

566  update_acc_param_grad_sqrs(param_grad_sqrs, param_grads);

610 template<typename Scalar, std::size_t Rank, bool Sequential>
611 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
612 public:
623  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
624  Scalar decay = 5e-2, Scalar epsilon = Utils<Scalar>::EPSILON2) :
625  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, reg, batch_size),
626  decay(decay),
627  epsilon(epsilon) {
628  assert(decay >= 0 && decay <= 1);
629  assert(epsilon > 0);
630  }
631 protected:
632  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
633  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
634  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
635  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
636  Layer<Scalar,Rank>& layer = *(layers[i]);
637  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
638  pgus.param_grad = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).rows(),
639  Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).cols());
640  pgus.param_grad.setZero(pgus.param_grad.rows(), pgus.param_grad.cols());
641  pgus.param_update = Matrix<Scalar>(pgus.param_grad.rows(), pgus.param_grad.cols());
642  pgus.param_update.setZero(pgus.param_update.rows(), pgus.param_update.cols());
643  }
644  }
645  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
646  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
647  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
648  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
649  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
650  pgus.param_grad = (1 - decay) * pgus.param_grad + decay * param_grads.cwiseProduct(param_grads);
651  Matrix<Scalar> weight_updates = -param_grads.array() * (pgus.param_update.array() + epsilon).sqrt() /
652  (pgus.param_grad.array() + epsilon).sqrt();
653  params += weight_updates;
654  pgus.param_update = (1 - decay) * pgus.param_update + decay * weight_updates.cwiseProduct(weight_updates);
655  }
656  const Scalar decay;
657  const Scalar epsilon;
661  struct ParamGradAndUpdateSqrs {
662  Matrix<Scalar> param_grad;
663  Matrix<Scalar> param_update;
664  };
665  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
666 };

671 template<typename Scalar, std::size_t Rank, bool Sequential>
672 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
673 public:
689  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
690  Scalar learning_rate = 1e-3, Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3,
691  Scalar epsilon = Utils<Scalar>::EPSILON2) :
692  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, reg, batch_size),
693  learning_rate(learning_rate),
694  l1_decay(l1_decay),
695  l2_decay(l2_decay),
696  epsilon(epsilon) {
697  assert(learning_rate > 0);
698  assert(l1_decay >= 0 && l1_decay <= 1);
699  assert(l2_decay >= 0 && l2_decay <= 1);
700  assert(epsilon > 0);
701  }
702  virtual ~AdamOptimizer() = default;
703 protected:
704  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
705  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
706  pgn_vec = std::vector<ParamGradNorms>(layers.size());
707  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
708  Layer<Scalar,Rank>& layer = *(layers[i]);
709  ParamGradNorms& vel = pgn_vec[i];
710  vel.param_grad_l1 = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).rows(),
711  Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer).cols());
712  vel.param_grad_l1.setZero(vel.param_grad_l1.rows(), vel.param_grad_l1.cols());
713  vel.param_grad_l2 = Matrix<Scalar>(vel.param_grad_l1.rows(), vel.param_grad_l1.cols());
714  vel.param_grad_l2.setZero(vel.param_grad_l2.rows(), vel.param_grad_l2.cols());
715  }
716  }
717  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
718  ParamGradNorms& grad_norms = pgn_vec[i];
719  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - l1_decay, epoch + 1) + epsilon);
720  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - l2_decay, epoch + 1) + epsilon);
721  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
722  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
723  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
724  grad_norms.param_grad_l1 = (1 - l1_decay) * grad_norms.param_grad_l1 + l1_decay * param_grads;
725  grad_norms.param_grad_l2 = (1 - l2_decay) * grad_norms.param_grad_l2 +
726  l2_decay * param_grads.cwiseProduct(param_grads);
727  params -= (learning_rate * (grad_norms.param_grad_l1 * l1_corr).array() /
728  ((grad_norms.param_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
729  }
730  const Scalar learning_rate;
731  const Scalar l1_decay;
732  const Scalar l2_decay;
733  const Scalar epsilon;
738  struct ParamGradNorms {
739  Matrix<Scalar> param_grad_l1;
740  Matrix<Scalar> param_grad_l2;
741  };
742  std::vector<ParamGradNorms> pgn_vec;
743 };

776  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
777  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +

780  grad_norms.param_grad_l2 = ((1 - Base::l2_decay) * grad_norms.param_grad_l2).cwiseMax(param_grads.cwiseAbs());
781  params -= (Base::learning_rate * (grad_norms.param_grad_l1 * l1_corr).array() /
782  (grad_norms.param_grad_l2.array() + Base::epsilon)).matrix();
783  }

790 class NadamOptimizer : public AdamOptimizer<Scalar,Rank,Sequential> {
791  typedef AdamOptimizer<Scalar,Rank,Sequential> Base;
792 public:
808  inline NadamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
809  Scalar learning_rate = 1e-3, Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3,
810  Scalar epsilon = Utils<Scalar>::EPSILON2) :
811  Base::AdamOptimizer(loss, reg, batch_size, learning_rate, l1_decay, l2_decay, epsilon) { }
812 protected:
813  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
814  typename Base::ParamGradNorms& grad_norms = Base::pgn_vec[i];
815  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 1) + Base::epsilon);
816  Scalar l1_next_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 2) + Base::epsilon);
817  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - Base::l2_decay, epoch + 1) + Base::epsilon);
818  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
819  Matrix<Scalar> param_grads = Optimizer<Scalar,Rank,Sequential>::get_param_grads(layer) +
820  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
821  grad_norms.param_grad_l1 = (1 - Base::l1_decay) * grad_norms.param_grad_l1 + Base::l1_decay * param_grads;
822  grad_norms.param_grad_l2 = (1 - Base::l2_decay) * grad_norms.param_grad_l2 + Base::l2_decay *
823  param_grads.cwiseProduct(param_grads);
824  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * param_grads +
825  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.param_grad_l1).array() /
826  ((grad_norms.param_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();
827  }
828 };

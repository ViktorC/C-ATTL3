76  inline bool verify_gradients(Net& net, Provider& provider, Scalar step_size = (internal::NumericUtils<Scalar>::EPSILON2 +
77  internal::NumericUtils<Scalar>::EPSILON3) / 2, Scalar abs_epsilon = internal::NumericUtils<Scalar>::EPSILON2,
78  Scalar rel_epsilon = internal::NumericUtils<Scalar>::EPSILON3) const {
79  assert(net.get_input_dims() == provider.get_obs_dims());
80  assert(net.get_output_dims() == provider.get_obj_dims());
81  assert(step_size > 0);
82  assert(abs_epsilon >= 0 && rel_epsilon > 0);
83  bool failure = false;
84  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
85  std::size_t instances = data_pair.first.dimension(0);
86  provider.reset();
87  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
88  * back-propagate is to be divided by the number of observations in the batch. */
89  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);

95  std::string(28, '-') << std::endl;
96  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
97  * parameters' gradient. */
98  layer.regularize();
99  Matrix<Scalar>& params = layer.get_params();
100  const Matrix<Scalar>& params_grad = layer.get_params_grad();
101  for (int j = 0; j < params.rows(); ++j) {
102  for (int k = 0; k < params.cols(); ++k) {
103  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
104  Scalar ana_grad = params_grad(j,k);
105  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
106  Scalar param = params(j,k);
107  params(j,k) = param + step_size;
108  /* Compute the numerical gradients in training mode to ensure that the means and standard
109  * deviations used for batch normalization are the same as those used during the analytic
110  * gradient computation. */
111  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
112  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's parameters
113  * is included in the gradient. */
114  Scalar reg_pen_inc = layer.get_regularization_penalty();
115  params(j,k) = param - step_size;
116  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
117  Scalar reg_pen_dec = layer.get_regularization_penalty();
118  params(j,k) = param;
119  // Include the regularization penalty as well.
120  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
121  std::cout << "\t\tNumerical gradient = " << num_grad;
122  if (!internal::NumericUtils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
123  std::cout << " *****FAIL*****";
124  failure = true;
125  }
126  std::cout << std::endl;
127  }
128  }
129  }
130  }
131  // Empty the network caches.
132  net.empty_caches();
133  return !failure;
134  }
148  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0) {
149  assert(net.get_input_dims() == training_prov.get_obs_dims());
150  assert(net.get_output_dims() == training_prov.get_obj_dims());
151  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
152  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
153  assert(epochs > 0);
154  // Fit the optimizer parameters to the network.
155  fit(net);
156  Scalar prev_test_loss = internal::NumericUtils<Scalar>::MAX;
157  unsigned cons_loss_inc = 0;
158  // Start the optimization iterations.
159  for (unsigned i = 0; i <= epochs; ++i) {
160  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
161  // Train.
162  if (i != 0) {
163  training_prov.reset();
164  std::cout << "\ttraining loss: " << std::to_string(train(net, training_prov, i)) << std::endl;
165  }
166  // Validate.
167  test_prov.reset();
168  Scalar test_loss = test(net, test_prov, i);
169  std::cout << "\ttest loss: " << std::to_string(test_loss);
170  if (test_loss >= prev_test_loss) {
171  cons_loss_inc++;
172  std::cout << " *****INCREASED LOSS*****";
173  if (early_stop > 0 && cons_loss_inc >= early_stop)
174  return prev_test_loss;
175  } else
176  cons_loss_inc = 0;
177  std::cout << std::endl << std::endl;
178  prev_test_loss = test_loss;
179  }
180  // Reset the providers.
181  training_prov.reset();
182  test_prov.reset();
183  // Empty the network caches.
184  net.empty_caches();
185  return prev_test_loss;
186  }

351 template<typename Scalar, std::size_t Rank, bool Sequential>
352 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
353  typedef Optimizer<Scalar,Rank,Sequential> Base;
354 public:
355  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size) :
356  Base::Optimizer(loss),
357  batch_size(batch_size) {
358  assert(batch_size > 0);
359  }
360  virtual ~SGDOptimizer() = default;
361 protected:
362  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch) {
363  Scalar training_loss = 0;
364  Scalar instances = 0;
365  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
366  // Perform an entire training epoch.
367  while (training_prov.has_more()) {
368  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
369  instances += data_pair.first.dimension(0);
370  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
371  training_loss += Base::loss->function(out, data_pair.second).sum();
372  /* Divide the gradient by the batch size to decouple the learning rate and the batch
373  * size hyper-parameters. Use the nominal batch size as the denominator even if the
374  * actual batch size is different (in case the number of samples in the data set is
375  * not divisible by the batch size and the last batch of the epoch contains fewer
376  * instances than the others) to make sure that the magnitude of the gradient is
377  * proportional to the batch size (just like its 'accuracy' is). */
378  Base::backpropagate(net, Base::loss->d_function(std::move(out), std::move(data_pair.second)) / (Scalar) batch_size);
379  for (unsigned k = 0; k < layers.size(); ++k) {
380  Layer<Scalar,Rank>& layer = *(layers[k]);
381  if (Base::is_parametric(layer)) {
382  Base::regularize(layer);
383  update_params(layer, k, epoch - 1);
384  Base::enforce_constraints(layer);
385  }
386  }
387  }
388  return training_loss / instances;
389  }
390  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch) {
391  Scalar obj_loss = 0;
392  Scalar instances = 0;
393  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
394  // Perform an entire test epoch.
395  while (test_prov.has_more()) {
396  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
397  instances += data_pair.first.dimension(0);
398  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), std::move(data_pair.second)).sum();
399  }
400  Scalar mean_obj_loss = obj_loss / instances;
401  Scalar reg_loss = 0;
402  for (unsigned j = 0; j < layers.size(); ++j)
403  reg_loss += Base::get_regularization_penalty(*layers[j]);
404  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
405  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
406  return mean_obj_loss + reg_loss;
407  }
416  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
417  const unsigned batch_size;
418 };

569  for (unsigned i = 0; i < params_grad_sqrs_vec.size(); ++i) {

582  inline virtual void update_acc_params_grad_sqrs(Matrix<Scalar>& acc_params_grad_sqrs,
583  const Matrix<Scalar>& params_grad) {
584  // Accumulate the squares of the gradients.
585  acc_params_grad_sqrs += params_grad.cwiseProduct(params_grad);
586  }

588  Matrix<Scalar>& params_grad_sqrs = params_grad_sqrs_vec[i];

590  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
591  update_acc_params_grad_sqrs(params_grad_sqrs, params_grad);
592  params -= (learning_rate * params_grad.array() / (params_grad_sqrs.array().sqrt() + epsilon)).matrix();
593  }

595  const Scalar epsilon;

633 template<typename Scalar, std::size_t Rank, bool Sequential>
634 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
635 public:
645  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
646  Scalar decay = 5e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
647  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
648  decay(decay),
649  epsilon(epsilon) {
650  assert(decay >= 0 && decay <= 1);
651  assert(epsilon > 0);
652  }
653 protected:
654  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
655  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
656  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
657  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
658  Layer<Scalar,Rank>& layer = *(layers[i]);
659  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
660  ParamGradAndUpdateSqrs pgus;
661  pgus.params_grad = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
662  pgus.params_update = Matrix<Scalar>::Zero(pgus.params_grad.rows(), pgus.params_grad.cols());
663  pgus_vec[i] = pgus;
664  }
665  }
666  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
667  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
668  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
669  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
670  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
671  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
672  (pgus.params_grad.array() + epsilon).sqrt();
673  params += weight_updates;
674  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
675  }
676  const Scalar decay;
677  const Scalar epsilon;
681  struct ParamGradAndUpdateSqrs {
682  Matrix<Scalar> params_grad;
683  Matrix<Scalar> params_update;
684  };
685  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
686 };

691 template<typename Scalar, std::size_t Rank, bool Sequential>
692 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
693 public:
708  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
709  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
710  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
711  learning_rate(learning_rate),
712  l1_decay(l1_decay),
713  l2_decay(l2_decay),
714  epsilon(epsilon) {
715  assert(learning_rate > 0);
716  assert(l1_decay >= 0 && l1_decay <= 1);
717  assert(l2_decay >= 0 && l2_decay <= 1);
718  assert(epsilon > 0);
719  }
720  virtual ~AdamOptimizer() = default;
721 protected:
722  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
723  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
724  pgn_vec = std::vector<ParamGradNorms>(layers.size());
725  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
726  Layer<Scalar,Rank>& layer = *(layers[i]);
727  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
728  ParamGradNorms vel;
729  vel.params_grad_l1 = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
730  vel.params_grad_l2 = Matrix<Scalar>::Zero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
731  pgn_vec[i] = vel;
732  }
733  }
734  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
735  ParamGradNorms& grad_norms = pgn_vec[i];
736  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - l1_decay, epoch + 1) + epsilon);
737  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - l2_decay, epoch + 1) + epsilon);
738  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
739  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
740  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
741  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
742  l2_decay * params_grad.cwiseProduct(params_grad);
743  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
744  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
745  }
746  const Scalar learning_rate;

765 class AdaMaxOptimizer : public AdamOptimizer<Scalar,Rank,Sequential> {
766  typedef AdamOptimizer<Scalar,Rank,Sequential> Base;
767 public:
782  inline AdaMaxOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
783  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
784  Base::AdamOptimizer(loss, batch_size, learning_rate, l1_decay, l2_decay, epsilon) { }
785 protected:
786  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
787  typename Base::ParamGradNorms& grad_norms = Base::pgn_vec[i];
788  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - Base::l1_decay, epoch + 1) + Base::epsilon);
789  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
790  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
791  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
792  grad_norms.params_grad_l2 = ((1 - Base::l2_decay) * grad_norms.params_grad_l2).cwiseMax(params_grad.cwiseAbs());
793  params -= (Base::learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
794  (grad_norms.params_grad_l2.array() + Base::epsilon)).matrix();
795  }
796 };

802 class NadamOptimizer : public AdamOptimizer<Scalar,Rank,Sequential> {
803  typedef AdamOptimizer<Scalar,Rank,Sequential> Base;
804 public:
819  inline NadamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
820  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
821  Base::AdamOptimizer(loss, batch_size, learning_rate, l1_decay, l2_decay, epsilon) { }
822 protected:
823  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
824  typename Base::ParamGradNorms& grad_norms = Base::pgn_vec[i];
825  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - Base::l1_decay, epoch + 1) + Base::epsilon);
826  Scalar l1_next_corr = (Scalar) 1 / (1 - pow(1 - Base::l1_decay, epoch + 2) + Base::epsilon);
827  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - Base::l2_decay, epoch + 1) + Base::epsilon);
828  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
829  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
830  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
831  grad_norms.params_grad_l2 = (1 - Base::l2_decay) * grad_norms.params_grad_l2 + Base::l2_decay *
832  params_grad.cwiseProduct(params_grad);
833  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * params_grad +
834  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.params_grad_l1).array() /
835  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();
836  }
837 };

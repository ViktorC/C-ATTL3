74  inline bool verify_gradients(Net& net, Provider& provider, Scalar step_size = 1e-5, Scalar abs_epsilon = Utils<Scalar>::EPSILON2,
75  Scalar rel_epsilon = Utils<Scalar>::EPSILON3) const {
76  assert(net.get_input_dims() == provider.get_obs_dims());
77  assert(net.get_output_dims() == provider.get_obj_dims());
78  assert(step_size > 0);
79  assert(abs_epsilon >= 0 && rel_epsilon > 0);
80  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
81  std::size_t instances = data_pair.first.dimension(0);
82  provider.reset();
83  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
84  * back-propagate is to be divided by the number of observations in the batch. */
85  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);
86  bool failure = false;
87  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
88  for (unsigned i = 0; i < layers.size(); ++i) {
89  Layer<Scalar,Rank>& layer = *(layers[i]);
90  if (layer.is_parametric()) {
91  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) <<
92  std::string(28, '-') << std::endl;
93  Matrix<Scalar>& params = layer.get_params();
94  const Matrix<Scalar>& params_grad = layer.get_params_grad();
95  for (int j = 0; j < params.rows(); ++j) {
96  for (int k = 0; k < params.cols(); ++k) {
97  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
98  Scalar ana_grad = params_grad(j,k);
99  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
100  Scalar param = params(j,k);
101  params(j,k) = param + step_size;
102  /* Compute the numerical gradients in training mode to ensure that the means
103  * and standard deviations used for batch normalization are the same as those
104  * used during the analytic gradient computation. */
105  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
106  params(j,k) = param - step_size;
107  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
108  params(j,k) = param;
109  Scalar num_grad = (loss_inc - loss_dec) / (2 * step_size);
110  std::cout << "\t\tNumerical gradient = " << num_grad;
111  if (!Utils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
112  std::cout << " *****FAIL*****";
113  failure = true;
114  }
115  std::cout << std::endl;
116  }
117  }
118  }
119  }
120  // Empty the network caches.
121  net.empty_caches();
122  return !failure;

147  // Start the optimization iterations.
148  for (unsigned i = 0; i <= epochs; ++i) {
149  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
150  // Train.

161  std::cout << " *****INCREASED LOSS*****";

163  return prev_test_loss;
164  } else
165  cons_loss_inc = 0;
166  std::cout << std::endl << std::endl;
167  prev_test_loss = test_loss;
168  }
169  // Empty the network caches.
170  net.empty_caches();

172  }

315 template<typename Scalar, std::size_t Rank, bool Sequential>
316 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
317  typedef Optimizer<Scalar,Rank,Sequential> Base;
318 public:
319  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size) :
320  Base::Optimizer(loss),
321  reg(reg),
322  batch_size(batch_size) {
323  assert(reg != nullptr);
324  assert(batch_size > 0);
325  }
326  virtual ~SGDOptimizer() = default;
327 protected:
328  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch) {
329  Scalar training_loss = 0;
330  Scalar instances = 0;
331  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
332  // Perform an entire training epoch.
333  while (training_prov.has_more()) {
334  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
335  instances += data_pair.first.dimension(0);
336  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
337  training_loss += Base::loss->function(out, data_pair.second).sum();
338  /* Divide the gradient by the batch size to decouple the learning rate and the batch
339  * size hyper-parameters. Use the nominal batch size as the denominator even if the
340  * actual batch size is different (in case the number of samples in the data set is
341  * not divisible by the batch size and the last batch of the epoch contains fewer
342  * instances than the others) to make sure that the magnitude of the gradient is
343  * proportional to the batch size (just like its 'accuracy' is). */
344  Base::backpropagate(net, Base::loss->d_function(out, data_pair.second) / (Scalar) batch_size);
345  for (unsigned k = 0; k < layers.size(); ++k) {
346  Layer<Scalar,Rank>& layer = *(layers[k]);
347  if (Base::is_parametric(layer)) {
348  update_params(layer, k, epoch - 1);
349  Base::enforce_constraints(layer);
350  }
351  }
352  }
353  return training_loss / instances;
354  }
355  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch) {
356  Scalar obj_loss = 0;
357  Scalar instances = 0;
358  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
359  // Perform an entire test epoch.
360  while (test_prov.has_more()) {
361  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
362  instances += data_pair.first.dimension(0);
363  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), data_pair.second).sum();
364  }
365  Scalar mean_obj_loss = obj_loss / instances;
366  Scalar reg_loss = 0;
367  for (unsigned j = 0; j < layers.size(); ++j)
368  reg_loss += reg->function(Base::get_params(*(layers[j])));
369  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
370  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
371  return mean_obj_loss + reg_loss;
372  }
381  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
382  const RegPenSharedPtr<Scalar> reg;
383  const unsigned batch_size;
384 };

615 template<typename Scalar, std::size_t Rank, bool Sequential>
616 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
617 public:
628  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
629  Scalar decay = 5e-2, Scalar epsilon = Utils<Scalar>::EPSILON2) :
630  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, reg, batch_size),
631  decay(decay),
632  epsilon(epsilon) {
633  assert(decay >= 0 && decay <= 1);
634  assert(epsilon > 0);
635  }
636 protected:
637  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
638  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
639  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
640  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
641  Layer<Scalar,Rank>& layer = *(layers[i]);
642  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
643  pgus.params_grad = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer).rows(),
644  Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer).cols());
645  pgus.params_grad.setZero(pgus.params_grad.rows(), pgus.params_grad.cols());
646  pgus.params_update = Matrix<Scalar>(pgus.params_grad.rows(), pgus.params_grad.cols());
647  pgus.params_update.setZero(pgus.params_update.rows(), pgus.params_update.cols());
648  }
649  }
650  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
651  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
652  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
653  Matrix<Scalar> params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer) +
654  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
655  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
656  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
657  (pgus.params_grad.array() + epsilon).sqrt();
658  params += weight_updates;
659  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
660  }
661  const Scalar decay;
662  const Scalar epsilon;
666  struct ParamGradAndUpdateSqrs {
667  Matrix<Scalar> params_grad;
668  Matrix<Scalar> params_update;
669  };
670  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
671 };

694  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,
695  Scalar learning_rate = 1e-3, Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3,

708 protected:
709  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
710  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
711  pgn_vec = std::vector<ParamGradNorms>(layers.size());
712  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
713  Layer<Scalar,Rank>& layer = *(layers[i]);
714  ParamGradNorms& vel = pgn_vec[i];
715  vel.params_grad_l1 = Matrix<Scalar>(Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer).rows(),
716  Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer).cols());
717  vel.params_grad_l1.setZero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
718  vel.params_grad_l2 = Matrix<Scalar>(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
719  vel.params_grad_l2.setZero(vel.params_grad_l2.rows(), vel.params_grad_l2.cols());
720  }
721  }
722  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
723  ParamGradNorms& grad_norms = pgn_vec[i];
724  Scalar l1_corr = 1.0 / (1.0 - pow(1.0 - l1_decay, epoch + 1) + epsilon);
725  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - l2_decay, epoch + 1) + epsilon);
726  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
727  Matrix<Scalar> params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer) +
728  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
729  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
730  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
731  l2_decay * params_grad.cwiseProduct(params_grad);
732  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
733  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
734  }

736  const Scalar l1_decay;
737  const Scalar l2_decay;
738  const Scalar epsilon;

744  Matrix<Scalar> params_grad_l1;
745  Matrix<Scalar> params_grad_l2;
746  };

813  inline NadamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, RegPenSharedPtr<Scalar> reg, unsigned batch_size = 1,

821  Scalar l1_next_corr = 1.0 / (1.0 - pow(1.0 - Base::l1_decay, epoch + 2) + Base::epsilon);
822  Scalar l2_corr = 1.0 / (1.0 - pow(1.0 - Base::l2_decay, epoch + 1) + Base::epsilon);
823  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);

825  SGDOptimizer<Scalar,Rank,Sequential>::reg->d_function(params);
826  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
827  grad_norms.params_grad_l2 = (1 - Base::l2_decay) * grad_norms.params_grad_l2 + Base::l2_decay *
828  params_grad.cwiseProduct(params_grad);
829  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * params_grad +
830  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.params_grad_l1).array() /
831  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();
832  }

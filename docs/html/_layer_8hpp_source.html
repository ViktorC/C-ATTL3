248  std::size_t weight_cols, Scalar max_norm_constraint) :
249  input_dims(input_dims),
250  output_dims(output_dims),
251  weight_init(weight_init),
252  weight_reg(weight_reg),
253  max_norm_constraint(max_norm_constraint),
254  max_norm(internal::NumericUtils<Scalar>::decidedly_greater(max_norm_constraint, (Scalar) 0)),
255  input_layer(false),
256  weights(weight_rows, weight_cols),

261  }

263  input_dims(layer.input_dims),
264  output_dims(layer.output_dims),
265  weight_init(layer.weight_init),
266  weight_reg(layer.weight_reg),
267  max_norm_constraint(layer.max_norm_constraint),
268  max_norm(layer.max_norm),
269  input_layer(layer.input_layer),
270  weights(share_params ? Matrix<Scalar>(0, 0) : layer.weights),
271  weights_grad(layer.weights_grad),
272  weights_ref(share_params ? layer.weights : weights) { }
273  inline bool is_input_layer() const {
274  return input_layer;
275  }
276  inline void set_input_layer(bool input_layer) {
277  this->input_layer = input_layer;
278  }
279  inline Matrix<Scalar>& get_params() {
280  return weights_ref;
281  }
282  inline Matrix<Scalar>& get_params_grad() {
283  return weights_grad;
284  }
285  inline void regularize() {
286  weights_grad += weight_reg->d_function(weights_ref);
287  }
288  inline Scalar get_regularization_penalty() {
289  return weight_reg->function(weights_ref);
290  }
291  inline void enforce_constraints() {
292  if (max_norm) {
293  Scalar l2_norm = weights.squaredNorm();
294  if (l2_norm > max_norm_constraint)
295  weights *= (max_norm_constraint / l2_norm);
296  }
297  }
298  const Dimensions<std::size_t,Rank> input_dims;
299  const Dimensions<std::size_t,Rank> output_dims;
300  const WeightInitSharedPtr<Scalar> weight_init;
301  const ParamRegSharedPtr<Scalar> weight_reg;
302  const Scalar max_norm_constraint;
303  const bool max_norm;
304  /* Eigen matrices are backed by arrays allocated on the heap, so these
305  * members do not burden the stack. */
306  Matrix<Scalar> weights_grad;
307  Matrix<Scalar>& weights_ref;
308 private:
309  bool input_layer;
310  mutable Matrix<Scalar> weights;
311 };

317 class DenseLayer : public KernelLayer<Scalar,Rank> {
318  typedef Layer<Scalar,Rank> Root;
319  typedef KernelLayer<Scalar,Rank> Base;
320  typedef std::array<std::size_t,Root::DATA_RANK> RankwiseArray;
321 public:
331  inline DenseLayer(const Dimensions<std::size_t,Rank>& input_dims, std::size_t output_size, WeightInitSharedPtr<Scalar> weight_init,
332  ParamRegSharedPtr<Scalar> weight_reg = Root::NO_PARAM_REG, Scalar max_norm_constraint = 0) :
333  Base::KernelLayer(input_dims, Dimensions<std::size_t,Rank>({ output_size }), weight_init, weight_reg,
334  input_dims.get_volume() + 1, output_size, max_norm_constraint),
335  out_conversion_dims(Base::output_dims.template promote<>()),
336  prev_out_conversion_dims(Base::input_dims.template promote<>()) { }
337  inline Root* clone() const {
338  return new DenseLayer(*this);
339  }
340 protected:
341  inline DenseLayer(const DenseLayer<Scalar,Rank>& layer, bool share_params = false) :
342  Base::KernelLayer(layer, share_params),
343  out_conversion_dims(layer.out_conversion_dims),
344  prev_out_conversion_dims(layer.prev_out_conversion_dims),
345  biased_in_mat(layer.biased_in_mat) { }
346  inline Layer<Scalar,Rank>* clone_with_shared_params() const {
347  return new DenseLayer(*this, true);
348  }
349  inline void empty_cache() {
350  biased_in_mat = Matrix<Scalar>(0, 0);
351  }
352  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
353  assert((Dimensions<std::size_t,Root::DATA_RANK>(in.dimensions()).template demote<>()) == Base::input_dims);
354  assert(in.dimension(0) > 0);
355  unsigned input_size = Base::input_dims.get_volume();
356  // Add a 1-column to the input for the bias trick.
357  biased_in_mat = Matrix<Scalar>(in.dimension(0), input_size + 1);
358  biased_in_mat.leftCols(input_size) = MatrixMap<Scalar>(in.data(), in.dimension(0), input_size);
359  biased_in_mat.col(input_size).setOnes();
360 #ifndef CATTL3_USE_CUBLAS
361  Matrix<Scalar> out = biased_in_mat * Base::weights_ref;
362 #else
363  Matrix<Scalar> out = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_conv_mat, Base::weights_ref, false, false);
364 #endif
365  out_conversion_dims[0] = out.rows();
366  return TensorMap<Scalar,Root::DATA_RANK>(out.data(), out_conversion_dims);
367  }
368  inline typename Root::Data pass_back(typename Root::Data out_grads) {
369  assert((Dimensions<std::size_t,Root::DATA_RANK>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
370  assert(out_grads.dimension(0) > 0 && biased_in_mat.rows() == out_grads.dimension(0));
371  // Compute the gradient of the outputs with respect to the weights.
372 #ifndef CATTL3_USE_CUBLAS
373  MatrixMap<Scalar> out_grads_mat(out_grads.data(), out_grads.dimension(0), Base::output_dims.get_volume());
374  Base::weights_grad = biased_in_mat.transpose() * out_grads_mat;
375  if (Base::is_input_layer())
376  return typename Root::Data();
377  /* Remove the bias row from the weight matrix, transpose it, and compute the derivative w.r.t. the
378  * previous layer's output. */
379  Matrix<Scalar> prev_out_grads = out_grads_mat * Base::weights_ref.topRows(Base::input_dims.get_volume()).transpose();
380 #else
381  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(), out_grads.dimension(0),
382  Base::output_dims.get_volume());
383  Base::weights_grad = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_mat, out_grads_mat, true, false);
384  if (Base::is_input_layer())
385  return typename Root::Data();
386  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(Base::input_dims.get_volume());
387  Matrix<Scalar> prev_out_grads = internal::CuBLASUtils<Scalar>::get_instance().mul(out_grads_mat,
388  weights_without_bias, false, true);
389 #endif
390  prev_out_conversion_dims[0] = prev_out_grads.rows();
391  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grads.data(), prev_out_conversion_dims);
392  }
393 private:
394  RankwiseArray out_conversion_dims;
395  RankwiseArray prev_out_conversion_dims;
396  // Staged computation caches
397  Matrix<Scalar> biased_in_mat;
398 };

404 class ConvolutionLayer : public KernelLayer<Scalar,3> {
405  typedef Layer<Scalar,3> Root;
406  typedef KernelLayer<Scalar,3> Base;
407  typedef std::array<std::size_t,4> RankwiseArray;
408  typedef std::array<std::pair<std::size_t,std::size_t>,4> PaddingsArray;

432  inline ConvolutionLayer(const Dimensions<std::size_t,3>& input_dims, std::size_t filters, WeightInitSharedPtr<Scalar> weight_init,
433  ParamRegSharedPtr<Scalar> weight_reg = Root::NO_PARAM_REG, std::size_t receptor_height = 3, std::size_t receptor_width = 3,
434  std::size_t vertical_padding = 1, std::size_t horizontal_padding = 1, std::size_t vertical_stride = 1,
435  std::size_t horizontal_stride = 1, std::size_t vertical_dilation = 0, std::size_t horizontal_dilation = 0,
436  Scalar max_norm_constraint = 0) :
437  /* For every filter, there is a column in the weight matrix with the same number of
438  * elements as the area of the receptive field (F * F * D) + 1 for the bias row. */
439  Base::KernelLayer(input_dims, Dimensions<std::size_t,3>({
440  calculate_output_dim(input_dims(0), receptor_height, vertical_padding, vertical_dilation, vertical_stride),
441  calculate_output_dim(input_dims(1), receptor_width, horizontal_padding, horizontal_dilation, horizontal_stride),
442  filters }), weight_init, weight_reg, receptor_height * receptor_width * input_dims(2) + 1,
443  filters, max_norm_constraint),
444  filters(filters),
445  receptor_height(receptor_height),
446  receptor_width(receptor_width),
447  vertical_padding(vertical_padding),
448  horizontal_padding(horizontal_padding),
449  vertical_stride(vertical_stride),
450  horizontal_stride(horizontal_stride),
451  vertical_dilation(vertical_dilation),
452  horizontal_dilation(horizontal_dilation),
453  padded_height(input_dims(0) + 2 * vertical_padding),
454  padded_width(input_dims(1) + 2 * horizontal_padding),
455  dil_receptor_height(receptor_height + (receptor_height - 1) * vertical_dilation),
456  dil_receptor_width(receptor_width + (receptor_width - 1) * horizontal_dilation),
457  out_conversion_dims({ 0u, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) }),
458  patch_offsets({ 0u, 0u, 0u, 0u }),
459  patch_extents({ 0u, dil_receptor_height, dil_receptor_width, input_dims(2) }),
460  dil_strides({ 1u, vertical_dilation + 1u, horizontal_dilation + 1u, 1u }),
461  no_padding_offsets({ 0u, vertical_padding, horizontal_padding, 0u }),
462  no_padding_extents({ 0u, input_dims(0), input_dims(1), input_dims(2) }),
463  paddings({ std::make_pair(0, 0), std::make_pair(vertical_padding, vertical_padding),
464  std::make_pair(horizontal_padding, horizontal_padding), std::make_pair(0, 0) }) {
465  assert(filters > 0);
466  assert(receptor_height > 0);
467  assert(receptor_width > 0);
468  assert(vertical_stride > 0 && horizontal_stride > 0);
469  assert(input_dims(1) + 2 * vertical_padding >= dil_receptor_height &&
470  input_dims(2) + 2 * horizontal_padding >= dil_receptor_width);
471  }
472  inline Root* clone() const {
473  return new ConvolutionLayer(*this);
474  }

486  horizontal_dilation(layer.horizontal_dilation),
487  padded_height(layer.padded_height),

489  dil_receptor_height(layer.dil_receptor_height),
490  dil_receptor_width(layer.dil_receptor_width),

493  patch_extents(layer.patch_extents),
494  dil_strides(layer.dil_strides),
495  no_padding_offsets(layer.no_padding_offsets),

497  paddings(layer.paddings),

499  inline Root* clone_with_shared_params() const {

501  }
502  inline void empty_cache() {
503  biased_in_conv_mat = Matrix<Scalar>(0, 0);
504  }
505  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
506  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
507  assert(in.dimension(0) > 0);
508  // Spatial padding.
509  if (vertical_padding > 0 || horizontal_padding > 0)
510  in = typename Root::Data(in.pad(paddings));
511  std::size_t rows = in.dimension(0);
512  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
513  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
514  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
515  * out receptor of one sample. The same receptor location along all samples of the batch is represented
516  * by a contiguous block of these rows. There is one block for each receptor location. */
517  std::size_t patch_ind = 0;
518  patch_extents[0] = rows;
519  biased_in_conv_mat = Matrix<Scalar>(total_patches, receptor_vol + 1);
520  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
521  patch_offsets[2] = i;
522  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
523  patch_offsets[1] = j;
524  typename Root::Data patch;
525  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
526  if (vertical_dilation > 0 || horizontal_dilation > 0)
527  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
528  else
529  patch = in.slice(patch_offsets, patch_extents);
530  biased_in_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
531  patch_ind += rows;
532  }
533  }
534  assert(patch_ind == total_patches);
535  // Bias trick.
536  biased_in_conv_mat.col(receptor_vol).setOnes();
537 #ifndef CATTL3_USE_CUBLAS
538  Matrix<Scalar> out = biased_in_conv_mat * Base::weights_ref;
539 #else
540  Matrix<Scalar> out = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_conv_mat,
541  Base::weights_ref, false, false);
542 #endif
543  out_conversion_dims[0] = rows;
544  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
545  }
546  inline typename Root::Data pass_back(typename Root::Data out_grads) {
547  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
548  assert(out_grads.dimension(0) > 0 && biased_in_conv_mat.rows() / (Base::output_dims(0) *
549  Base::output_dims(1)) == out_grads.dimension(0));
550  std::size_t rows = out_grads.dimension(0);
551  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
552  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
553 #ifndef CATTL3_USE_CUBLAS
554  MatrixMap<Scalar> out_grads_mat(out_grads.data(), total_patches, filters);
555  Base::weights_grad = biased_in_conv_mat.transpose() * out_grads_mat;
556  if (Base::is_input_layer())
557  return typename Root::Data();
558  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
559  * previous layer's output. */
560  Matrix<Scalar> prev_out_grads_conv_mat = out_grads_mat * Base::weights_ref.topRows(receptor_vol).transpose();
561 #else
562  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(),
563  rows * Base::output_dims(0) * Base::output_dims(1), filters);
564  Base::weights_grad = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_conv_mat, out_grads_mat, true, false);
565  if (Base::is_input_layer())
566  return typename Root::Data();
567  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(receptor_vol);
568  Matrix<Scalar> prev_out_grads_conv_mat = internal::CuBLASUtils<Scalar>::get_instance().mul(out_grads_mat,
569  weights_without_bias, false, true);
570 #endif
571  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
572  * to get the derivative w.r.t. the individual input nodes. */
573  typename Root::Data prev_out_grads(rows, padded_height, padded_width, Base::input_dims(2));
574  prev_out_grads.setZero();
575  std::size_t patch_ind = 0;
576  patch_extents[0] = rows;
577  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
578  patch_offsets[2] = i;
579  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
580  patch_offsets[1] = j;
581  // Accumulate the gradients where the receptor-patch-tensors overlap.
582  Matrix<Scalar> prev_out_grads_conv_mat_block = prev_out_grads_conv_mat.block(patch_ind, 0,
583  rows, receptor_vol);
584  TensorMap<Scalar,4> prev_out_grads_patch(prev_out_grads_conv_mat_block.data(), rows,
585  receptor_height, receptor_width, Base::input_dims(2));
586  if (vertical_dilation > 0 || horizontal_dilation > 0)
587  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_patch;
588  else
589  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_patch;
590  patch_ind += rows;
591  }
592  }
593  assert(patch_ind == prev_out_grads_conv_mat.rows());
594  if (vertical_padding > 0 || horizontal_padding > 0) {
595  // Cut off the padding.
596  no_padding_extents[0] = rows;
597  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
598  } else
599  return prev_out_grads;
600  }
611  inline static std::size_t calculate_output_dim(std::size_t input_dim, std::size_t receptor_size, std::size_t padding,
612  std::size_t dilation, std::size_t stride) {
613  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
614  }

743  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
744  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
745  assert(in.dimension(0) > 0);
746  std::size_t rows = in.dimension(0);
747  std::size_t depth = Base::input_dims(2);
748  std::size_t receptor_vol = Base::weights_ref.cols();
749  std::size_t total_patches = rows * Base::input_dims(0) * Base::input_dims(1);
750  biased_in_mat = Matrix<Scalar>(total_patches, depth + 1);
751  biased_in_mat.block(0, 0, total_patches, depth) = MatrixMap<Scalar>(in.data(), total_patches, depth);
752  biased_in_mat.col(depth).setOnes();
753 #ifndef CATTL3_USE_CUBLAS
754  Matrix<Scalar> out_conv_mat = biased_in_mat * Base::weights_ref;
755 #else
756  Matrix<Scalar> out_conv_mat = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_mat,
757  Base::weights_ref, false, false);
758 #endif
759  /* Given the values of the stretched out receptor patches, accumulate them in the output tensor. */
760  typename Root::Data out(rows, padded_height, padded_width, Base::output_dims(2));
761  out.setZero();
762  std::size_t patch_ind = 0;
763  patch_extents[0] = rows;
764  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
765  patch_offsets[2] = i;
766  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
767  patch_offsets[1] = j;
768  // Accumulate the gradients where the receptor-patch-tensors overlap.
769  Matrix<Scalar> out_conv_mat_block = out_conv_mat.block(patch_ind, 0, rows, receptor_vol);
770  TensorMap<Scalar,4> out_patch(out_conv_mat_block.data(), rows, receptor_height,
771  receptor_width, Base::output_dims(2));
772  if (vertical_dilation > 0 || horizontal_dilation > 0)
773  out.slice(patch_offsets, patch_extents).stride(dil_strides) += out_patch;
774  else
775  out.slice(patch_offsets, patch_extents) += out_patch;
776  patch_ind += rows;
777  }
778  }
779  assert(patch_ind == total_patches);
780  if (vertical_padding > 0 || horizontal_padding > 0) {
781  // Cut off the padding.
782  no_padding_extents[0] = rows;
783  return out.slice(no_padding_offsets, no_padding_extents);
784  } else
785  return out;
786  }
787  inline typename Root::Data pass_back(typename Root::Data out_grads) {
788  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
789  assert(out_grads.dimension(0) > 0 && biased_in_mat.rows() / (Base::input_dims(0) *
790  Base::input_dims(1)) == out_grads.dimension(0));
791  // Spatial padding.
792  if (vertical_padding > 0 || horizontal_padding > 0)
793  out_grads = typename Root::Data(out_grads.pad(paddings));
794  std::size_t rows = out_grads.dimension(0);
795  std::size_t depth = Base::input_dims(2);
796  std::size_t receptor_vol = Base::weights_ref.cols();
797  std::size_t total_patches = rows * Base::input_dims(0) * Base::input_dims(1);
798  std::size_t patch_ind = 0;
799  patch_extents[0] = rows;
800  Matrix<Scalar> out_grads_conv_mat(total_patches, receptor_vol);
801  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
802  patch_offsets[2] = i;
803  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
804  patch_offsets[1] = j;
805  typename Root::Data patch;
806  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
807  if (vertical_dilation > 0 || horizontal_dilation > 0)
808  patch = out_grads.slice(patch_offsets, patch_extents).stride(dil_strides);
809  else
810  patch = out_grads.slice(patch_offsets, patch_extents);
811  out_grads_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(),
812  rows, receptor_vol);
813  patch_ind += rows;
814  }
815  }
816  assert(patch_ind == total_patches);
817 #ifndef CATTL3_USE_CUBLAS
818  Base::weights_grad = biased_in_mat.transpose() * out_grads_conv_mat;
819  if (Base::is_input_layer())
820  return typename Root::Data();
821  Matrix<Scalar> prev_out_grads = out_grads_conv_mat * Base::weights_ref.topRows(depth).transpose();
822 #else
823  Base::weights_grad = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in_mat,
824  out_grads_conv_mat, true, false);
825  if (Base::is_input_layer())
826  return typename Root::Data();
827  Matrix<Scalar> weights_without_bias = Base::weights_ref.rows(depth);
828  Matrix<Scalar> prev_out_grads = internal::CuBLASUtils<Scalar>::get_instance().mul(out_grads_conv_mat,
829  weights_without_bias, false, true);
830 #endif
831  prev_out_conversion_dims[0] = rows;
832  return TensorMap<Scalar,4>(prev_out_grads.data(), prev_out_conversion_dims);
833  }
844  inline static std::size_t calculate_output_dim(std::size_t input_dim, std::size_t receptor_size, std::size_t padding,
845  std::size_t dilation, std::size_t stride) {
846  return (input_dim - 1) * stride + receptor_size + (receptor_size - 1) * dilation - 2 * padding;
847  }
848 private:
849  // The defining attributes of the convolutional layer.
850  const std::size_t filters;
851  const std::size_t receptor_height;
852  const std::size_t receptor_width;
853  const std::size_t vertical_padding;
854  const std::size_t horizontal_padding;
855  const std::size_t vertical_stride;
856  const std::size_t horizontal_stride;
857  const std::size_t vertical_dilation;
858  const std::size_t horizontal_dilation;
859  // Pre-computed values to improve propagation-time performance.
860  const std::size_t padded_height;
861  const std::size_t padded_width;
862  const std::size_t dil_receptor_height;
863  const std::size_t dil_receptor_width;
864  RankwiseArray prev_out_conversion_dims;
865  RankwiseArray patch_offsets;
866  RankwiseArray patch_extents;
867  RankwiseArray dil_strides;
868  RankwiseArray no_padding_offsets;
869  RankwiseArray no_padding_extents;
870  PaddingsArray paddings;
871  // Staged computation caches
872  Matrix<Scalar> biased_in_mat;
873 };

385  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
386  assert((Dimensions<std::size_t,Root::DATA_RANK>(in.dimensions()).template demote<>()) == Base::input_dims);
387  assert(in.dimension(0) > 0);
388  unsigned input_size = Base::input_dims.get_volume();
389  // Add a 1-column to the input for the bias trick.
390  biased_in_mat = Matrix<Scalar>(in.dimension(0), input_size + 1);
391  biased_in_mat.leftCols(input_size) = MatrixMap<Scalar>(in.data(), in.dimension(0), input_size);
392  biased_in_mat.col(input_size).setOnes();
393 #ifndef CATTL3_USE_CUBLAS
394  Matrix<Scalar> out = biased_in_mat * Base::weights_ref;
395 #else
396  Matrix<Scalar> out = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat, Base::weights_ref, false, false);
397 #endif
398  out_conversion_dims[0] = out.rows();
399  return TensorMap<Scalar,Root::DATA_RANK>(out.data(), out_conversion_dims);
400  }
401  inline typename Root::Data pass_back(typename Root::Data out_grads) {
402  assert((Dimensions<std::size_t,Root::DATA_RANK>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
403  assert(out_grads.dimension(0) > 0 && biased_in_mat.rows() == out_grads.dimension(0));
404  // Compute the gradient of the outputs with respect to the weights.
405 #ifndef CATTL3_USE_CUBLAS
406  MatrixMap<Scalar> out_grads_mat(out_grads.data(), out_grads.dimension(0), Base::output_dims.get_volume());
407  Base::weights_grad = biased_in_mat.transpose() * out_grads_mat;
408  if (Base::is_input_layer())
409  return typename Root::Data();
410  /* Remove the bias row from the weight matrix, transpose it, and compute the derivative w.r.t. the
411  * previous layer's output. */
412  Matrix<Scalar> prev_out_grads = out_grads_mat * Base::weights_ref.topRows(Base::input_dims.get_volume()).transpose();
413 #else
414  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(), out_grads.dimension(0),
415  Base::output_dims.get_volume());
416  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat, out_grads_mat, true, false);
417  if (Base::is_input_layer())
418  return typename Root::Data();
419  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(Base::input_dims.get_volume());
420  Matrix<Scalar> prev_out_grads = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_mat,
421  weights_without_bias, false, true);
422 #endif
423  prev_out_conversion_dims[0] = prev_out_grads.rows();
424  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grads.data(), prev_out_conversion_dims);
425  }
426 private:
427  RankwiseArray out_conversion_dims;
428  RankwiseArray prev_out_conversion_dims;
429  // Staged computation caches
430  Matrix<Scalar> biased_in_mat;
431 };

513  inline Tensor<Scalar,4> _pass_forward(Tensor<Scalar,4> in, bool training) {
514  // Spatial padding.
515  if (vertical_padding > 0 || horizontal_padding > 0)
516  in = Tensor<Scalar,4>(in.pad(paddings));
517  std::size_t rows = in.dimension(0);
518  std::size_t total_patches = rows * patches_per_sample;
519  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
520  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
521  * out receptor of one sample. The same receptor location along all samples of the batch is represented
522  * by a contiguous block of these rows. */
523  std::size_t patch_ind = 0;
524  patch_extents[0] = rows;
525  biased_in_conv_mat = Matrix<Scalar>(total_patches, receptor_vol + 1);
526  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
527  patch_offsets[2] = i;
528  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
529  patch_offsets[1] = j;
530  Tensor<Scalar,4> patch;
531  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
532  if (vertical_dilation > 0 || horizontal_dilation > 0)
533  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
534  else
535  patch = in.slice(patch_offsets, patch_extents);
536  biased_in_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
537  patch_ind += rows;
538  }
539  }
540  assert(patch_ind == total_patches);
541  // Bias trick.
542  biased_in_conv_mat.col(receptor_vol).setOnes();
543 #ifndef CATTL3_USE_CUBLAS
544  Matrix<Scalar> out = biased_in_conv_mat * Base::weights_ref;
545 #else
546  Matrix<Scalar> out = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_conv_mat,
547  Base::weights_ref, false, false);
548 #endif
549  out_conversion_dims[0] = rows;
550  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
551  }
552  inline Tensor<Scalar,4> _pass_back(Tensor<Scalar,4> out_grads) {
553  std::size_t rows = out_grads.dimension(0);
554  std::size_t total_patches = rows * patches_per_sample;
555  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
556 #ifndef CATTL3_USE_CUBLAS
557  MatrixMap<Scalar> out_grads_mat(out_grads.data(), total_patches, filters);
558  Base::weights_grad = biased_in_conv_mat.transpose() * out_grads_mat;
559  if (Base::is_input_layer())
560  return Tensor<Scalar,4>();
561  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
562  * previous layer's output. */
563  Matrix<Scalar> prev_out_grads_conv_mat = out_grads_mat * Base::weights_ref.topRows(receptor_vol).transpose();
564 #else
565  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(), total_patches, filters);
566  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_conv_mat, out_grads_mat, true, false);
567  if (Base::is_input_layer())
568  return Tensor<Scalar,4>();
569  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(receptor_vol);
570  Matrix<Scalar> prev_out_grads_conv_mat = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_mat,
571  weights_without_bias, false, true);
572 #endif
573  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
574  * to get the derivative w.r.t. the individual input nodes. */
575  Tensor<Scalar,4> prev_out_grads(rows, padded_height, padded_width, ext_input_dims(2));
576  prev_out_grads.setZero();
577  std::size_t patch_ind = 0;
578  patch_extents[0] = rows;
579  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
580  patch_offsets[2] = i;
581  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
582  patch_offsets[1] = j;
583  // Accumulate the gradients where the receptor-patch-tensors overlap.
584  Matrix<Scalar> prev_out_grads_conv_mat_block = prev_out_grads_conv_mat.block(patch_ind, 0,
585  rows, receptor_vol);
586  TensorMap<Scalar,4> prev_out_grads_patch(prev_out_grads_conv_mat_block.data(), rows,
587  receptor_height, receptor_width, ext_input_dims(2));
588  if (vertical_dilation > 0 || horizontal_dilation > 0)
589  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_patch;
590  else
591  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_patch;
592  patch_ind += rows;
593  }
594  }
595  assert(patch_ind == prev_out_grads_conv_mat.rows());
596  if (vertical_padding > 0 || horizontal_padding > 0) {
597  // Cut off the padding.
598  no_padding_extents[0] = rows;
599  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
600  } else
601  return prev_out_grads;
602  }

942  inline Tensor<Scalar,4> _pass_forward(Tensor<Scalar,4> in, bool training) {
943  std::size_t rows = in.dimension(0);
944  std::size_t depth = ext_input_dims(2);
945  std::size_t receptor_vol = Base::weights_ref.cols();
946  std::size_t total_patches = rows * patches_per_sample;
947  biased_in_mat = Matrix<Scalar>(total_patches, depth + 1);
948  biased_in_mat.block(0, 0, total_patches, depth) = MatrixMap<Scalar>(in.data(), total_patches, depth);
949  biased_in_mat.col(depth).setOnes();
950 #ifndef CATTL3_USE_CUBLAS
951  Matrix<Scalar> out_conv_mat = biased_in_mat * Base::weights_ref;
952 #else
953  Matrix<Scalar> out_conv_mat = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat,
954  Base::weights_ref, false, false);
955 #endif
956  /* Given the values of the stretched out receptor patches, accumulate them in the output tensor. */
957  Tensor<Scalar,4> out(rows, padded_height, padded_width, ext_output_dims(2));
958  out.setZero();
959  std::size_t patch_ind = 0;
960  patch_extents[0] = rows;
961  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
962  patch_offsets[2] = i;
963  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
964  patch_offsets[1] = j;
965  // Accumulate the gradients where the receptor-patch-tensors overlap.
966  Matrix<Scalar> out_conv_mat_block = out_conv_mat.block(patch_ind, 0, rows, receptor_vol);
967  TensorMap<Scalar,4> out_patch(out_conv_mat_block.data(), rows, receptor_height,
968  receptor_width, ext_output_dims(2));
969  if (vertical_dilation > 0 || horizontal_dilation > 0)
970  out.slice(patch_offsets, patch_extents).stride(dil_strides) += out_patch;
971  else
972  out.slice(patch_offsets, patch_extents) += out_patch;
973  patch_ind += rows;
974  }
975  }
976  assert(patch_ind == total_patches);
977  if (vertical_padding > 0 || horizontal_padding > 0) {
978  // Cut off the padding.
979  no_padding_extents[0] = rows;
980  return out.slice(no_padding_offsets, no_padding_extents);
981  } else
982  return out;
983  }
984  inline Tensor<Scalar,4> _pass_back(Tensor<Scalar,4> out_grads) {
985  // Spatial padding.
986  if (vertical_padding > 0 || horizontal_padding > 0)
987  out_grads = Tensor<Scalar,4>(out_grads.pad(paddings));
988  std::size_t rows = out_grads.dimension(0);
989  std::size_t depth = ext_input_dims(2);
990  std::size_t receptor_vol = Base::weights_ref.cols();
991  std::size_t total_patches = rows * patches_per_sample;
992  std::size_t patch_ind = 0;
993  patch_extents[0] = rows;
994  Matrix<Scalar> out_grads_conv_mat(total_patches, receptor_vol);
995  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
996  patch_offsets[2] = i;
997  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
998  patch_offsets[1] = j;
999  Tensor<Scalar,4> patch;
1000  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
1001  if (vertical_dilation > 0 || horizontal_dilation > 0)
1002  patch = out_grads.slice(patch_offsets, patch_extents).stride(dil_strides);
1003  else
1004  patch = out_grads.slice(patch_offsets, patch_extents);
1005  out_grads_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(),
1006  rows, receptor_vol);
1007  patch_ind += rows;
1008  }
1009  }
1010  assert(patch_ind == total_patches);
1011 #ifndef CATTL3_USE_CUBLAS
1012  Base::weights_grad = biased_in_mat.transpose() * out_grads_conv_mat;
1013  if (Base::is_input_layer())
1014  return Tensor<Scalar,4>();
1015  Matrix<Scalar> prev_out_grads = out_grads_conv_mat * Base::weights_ref.topRows(depth).transpose();
1016 #else
1017  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat,
1018  out_grads_conv_mat, true, false);
1019  if (Base::is_input_layer())
1020  return Tensor<Scalar,4>();
1021  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(depth);
1022  Matrix<Scalar> prev_out_grads = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_conv_mat,
1023  weights_without_bias, false, true);
1024 #endif
1025  prev_out_conversion_dims[0] = rows;
1026  return TensorMap<Scalar,4>(prev_out_grads.data(), prev_out_conversion_dims);
1027  }

1052  const std::size_t vertical_dilation;

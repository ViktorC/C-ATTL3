376  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
377  assert((Dimensions<std::size_t,Root::DATA_RANK>(in.dimensions()).template demote<>()) == Base::input_dims);
378  assert(in.dimension(0) > 0);
379  unsigned input_size = Base::input_dims.get_volume();
380  // Add a 1-column to the input for the bias trick.
381  biased_in_mat = Matrix<Scalar>(in.dimension(0), input_size + 1);
382  biased_in_mat.leftCols(input_size) = MatrixMap<Scalar>(in.data(), in.dimension(0), input_size);
383  biased_in_mat.col(input_size).setOnes();
384 #ifndef CATTL3_USE_CUBLAS
385  Matrix<Scalar> out = biased_in_mat * Base::weights_ref;
386 #else
387  Matrix<Scalar> out = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat, Base::weights_ref, false, false);
388 #endif
389  out_conversion_dims[0] = out.rows();
390  return TensorMap<Scalar,Root::DATA_RANK>(out.data(), out_conversion_dims);
391  }
392  inline typename Root::Data pass_back(typename Root::Data out_grads) {
393  assert((Dimensions<std::size_t,Root::DATA_RANK>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
394  assert(out_grads.dimension(0) > 0 && biased_in_mat.rows() == out_grads.dimension(0));
395  // Compute the gradient of the outputs with respect to the weights.
396 #ifndef CATTL3_USE_CUBLAS
397  MatrixMap<Scalar> out_grads_mat(out_grads.data(), out_grads.dimension(0), Base::output_dims.get_volume());
398  Base::weights_grad = biased_in_mat.transpose() * out_grads_mat;
399  if (Base::is_input_layer())
400  return typename Root::Data();
401  /* Remove the bias row from the weight matrix, transpose it, and compute the derivative w.r.t. the
402  * previous layer's output. */
403  Matrix<Scalar> prev_out_grads = out_grads_mat * Base::weights_ref.topRows(Base::input_dims.get_volume()).transpose();
404 #else
405  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(), out_grads.dimension(0),
406  Base::output_dims.get_volume());
407  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat, out_grads_mat, true, false);
408  if (Base::is_input_layer())
409  return typename Root::Data();
410  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(Base::input_dims.get_volume());
411  Matrix<Scalar> prev_out_grads = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_mat,
412  weights_without_bias, false, true);
413 #endif
414  prev_out_conversion_dims[0] = prev_out_grads.rows();
415  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grads.data(), prev_out_conversion_dims);
416  }
417 private:
418  RankwiseArray out_conversion_dims;
419  RankwiseArray prev_out_conversion_dims;
420  // Staged computation caches
421  Matrix<Scalar> biased_in_mat;
422 };

456  inline ConvolutionLayer(const Dimensions<std::size_t,3>& input_dims, std::size_t filters, WeightInitSharedPtr<Scalar> weight_init,

471  vertical_padding(vertical_padding),
472  horizontal_padding(horizontal_padding),
473  vertical_stride(vertical_stride),
474  horizontal_stride(horizontal_stride),
475  vertical_dilation(vertical_dilation),
476  horizontal_dilation(horizontal_dilation),
477  padded_height(input_dims(0) + 2 * vertical_padding),
478  padded_width(input_dims(1) + 2 * horizontal_padding),
479  dil_receptor_height(receptor_height + (receptor_height - 1) * vertical_dilation),
480  dil_receptor_width(receptor_width + (receptor_width - 1) * horizontal_dilation),
481  out_conversion_dims({ 0u, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) }),
482  patch_offsets({ 0u, 0u, 0u, 0u }),

486  no_padding_extents({ 0u, input_dims(0), input_dims(1), input_dims(2) }),

488  std::make_pair(horizontal_padding, horizontal_padding), std::make_pair(0, 0) }) {
489  assert(filters > 0);
490  assert(receptor_height > 0);
491  assert(receptor_width > 0);
492  assert(vertical_stride > 0 && horizontal_stride > 0);

529  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
530  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
531  assert(in.dimension(0) > 0);
532  // Spatial padding.
533  if (vertical_padding > 0 || horizontal_padding > 0)
534  in = typename Root::Data(in.pad(paddings));
535  std::size_t rows = in.dimension(0);
536  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
537  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
538  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
539  * out receptor of one sample. The same receptor location along all samples of the batch is represented
540  * by a contiguous block of these rows. There is one block for each receptor location. */
541  std::size_t patch_ind = 0;
542  patch_extents[0] = rows;
543  biased_in_conv_mat = Matrix<Scalar>(total_patches, receptor_vol + 1);
544  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
545  patch_offsets[2] = i;
546  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
547  patch_offsets[1] = j;
548  typename Root::Data patch;
549  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
550  if (vertical_dilation > 0 || horizontal_dilation > 0)
551  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
552  else
553  patch = in.slice(patch_offsets, patch_extents);
554  biased_in_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
555  patch_ind += rows;
556  }
557  }
558  assert(patch_ind == total_patches);
559  // Bias trick.
560  biased_in_conv_mat.col(receptor_vol).setOnes();
561 #ifndef CATTL3_USE_CUBLAS
562  Matrix<Scalar> out = biased_in_conv_mat * Base::weights_ref;
563 #else
564  Matrix<Scalar> out = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_conv_mat,
565  Base::weights_ref, false, false);
566 #endif
567  out_conversion_dims[0] = rows;
568  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
569  }
570  inline typename Root::Data pass_back(typename Root::Data out_grads) {
571  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
572  assert(out_grads.dimension(0) > 0 && biased_in_conv_mat.rows() / (Base::output_dims(0) *
573  Base::output_dims(1)) == out_grads.dimension(0));
574  std::size_t rows = out_grads.dimension(0);
575  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
576  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
577 #ifndef CATTL3_USE_CUBLAS
578  MatrixMap<Scalar> out_grads_mat(out_grads.data(), total_patches, filters);
579  Base::weights_grad = biased_in_conv_mat.transpose() * out_grads_mat;
580  if (Base::is_input_layer())
581  return typename Root::Data();
582  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
583  * previous layer's output. */
584  Matrix<Scalar> prev_out_grads_conv_mat = out_grads_mat * Base::weights_ref.topRows(receptor_vol).transpose();
585 #else
586  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(),
587  rows * Base::output_dims(0) * Base::output_dims(1), filters);
588  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_conv_mat, out_grads_mat, true, false);
589  if (Base::is_input_layer())
590  return typename Root::Data();
591  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(receptor_vol);
592  Matrix<Scalar> prev_out_grads_conv_mat = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_mat,
593  weights_without_bias, false, true);
594 #endif
595  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
596  * to get the derivative w.r.t. the individual input nodes. */
597  typename Root::Data prev_out_grads(rows, padded_height, padded_width, Base::input_dims(2));
598  prev_out_grads.setZero();
599  std::size_t patch_ind = 0;
600  patch_extents[0] = rows;
601  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
602  patch_offsets[2] = i;
603  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
604  patch_offsets[1] = j;
605  // Accumulate the gradients where the receptor-patch-tensors overlap.
606  Matrix<Scalar> prev_out_grads_conv_mat_block = prev_out_grads_conv_mat.block(patch_ind, 0,
607  rows, receptor_vol);
608  TensorMap<Scalar,4> prev_out_grads_patch(prev_out_grads_conv_mat_block.data(), rows,
609  receptor_height, receptor_width, Base::input_dims(2));
610  if (vertical_dilation > 0 || horizontal_dilation > 0)
611  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_patch;
612  else
613  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_patch;
614  patch_ind += rows;
615  }
616  }
617  assert(patch_ind == prev_out_grads_conv_mat.rows());
618  if (vertical_padding > 0 || horizontal_padding > 0) {
619  // Cut off the padding.
620  no_padding_extents[0] = rows;
621  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
622  } else
623  return prev_out_grads;
624  }
635  inline static std::size_t calculate_output_dim(std::size_t input_dim, std::size_t receptor_size, std::size_t padding,
636  std::size_t dilation, std::size_t stride) {
637  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
638  }

769  assert(in.dimension(0) > 0);
770  std::size_t rows = in.dimension(0);
771  std::size_t depth = Base::input_dims(2);
772  std::size_t receptor_vol = Base::weights_ref.cols();
773  std::size_t total_patches = rows * Base::input_dims(0) * Base::input_dims(1);
774  biased_in_mat = Matrix<Scalar>(total_patches, depth + 1);
775  biased_in_mat.block(0, 0, total_patches, depth) = MatrixMap<Scalar>(in.data(), total_patches, depth);
776  biased_in_mat.col(depth).setOnes();

778  Matrix<Scalar> out_conv_mat = biased_in_mat * Base::weights_ref;
779 #else
780  Matrix<Scalar> out_conv_mat = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat,
781  Base::weights_ref, false, false);
782 #endif
783  /* Given the values of the stretched out receptor patches, accumulate them in the output tensor. */
784  typename Root::Data out(rows, padded_height, padded_width, Base::output_dims(2));
785  out.setZero();

787  patch_extents[0] = rows;
788  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
789  patch_offsets[2] = i;
790  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
791  patch_offsets[1] = j;
792  // Accumulate the gradients where the receptor-patch-tensors overlap.
793  Matrix<Scalar> out_conv_mat_block = out_conv_mat.block(patch_ind, 0, rows, receptor_vol);
794  TensorMap<Scalar,4> out_patch(out_conv_mat_block.data(), rows, receptor_height,
795  receptor_width, Base::output_dims(2));
796  if (vertical_dilation > 0 || horizontal_dilation > 0)
797  out.slice(patch_offsets, patch_extents).stride(dil_strides) += out_patch;
798  else
799  out.slice(patch_offsets, patch_extents) += out_patch;
800  patch_ind += rows;
801  }
802  }
803  assert(patch_ind == total_patches);
804  if (vertical_padding > 0 || horizontal_padding > 0) {
805  // Cut off the padding.
806  no_padding_extents[0] = rows;
807  return out.slice(no_padding_offsets, no_padding_extents);
808  } else
809  return out;
810  }
811  inline typename Root::Data pass_back(typename Root::Data out_grads) {
812  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
813  assert(out_grads.dimension(0) > 0 && biased_in_mat.rows() / (Base::input_dims(0) *
814  Base::input_dims(1)) == out_grads.dimension(0));
815  // Spatial padding.
816  if (vertical_padding > 0 || horizontal_padding > 0)
817  out_grads = typename Root::Data(out_grads.pad(paddings));
818  std::size_t rows = out_grads.dimension(0);
819  std::size_t depth = Base::input_dims(2);
820  std::size_t receptor_vol = Base::weights_ref.cols();
821  std::size_t total_patches = rows * Base::input_dims(0) * Base::input_dims(1);
822  std::size_t patch_ind = 0;
823  patch_extents[0] = rows;
824  Matrix<Scalar> out_grads_conv_mat(total_patches, receptor_vol);
825  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
826  patch_offsets[2] = i;
827  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
828  patch_offsets[1] = j;
829  typename Root::Data patch;
830  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
831  if (vertical_dilation > 0 || horizontal_dilation > 0)
832  patch = out_grads.slice(patch_offsets, patch_extents).stride(dil_strides);
833  else
834  patch = out_grads.slice(patch_offsets, patch_extents);
835  out_grads_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(),
836  rows, receptor_vol);
837  patch_ind += rows;
838  }
839  }
840  assert(patch_ind == total_patches);
841 #ifndef CATTL3_USE_CUBLAS
842  Base::weights_grad = biased_in_mat.transpose() * out_grads_conv_mat;
843  if (Base::is_input_layer())
844  return typename Root::Data();
845  Matrix<Scalar> prev_out_grads = out_grads_conv_mat * Base::weights_ref.topRows(depth).transpose();
846 #else
847  Base::weights_grad = internal::CuBLASHandle<Scalar>::get_instance().mul(biased_in_mat,
848  out_grads_conv_mat, true, false);
849  if (Base::is_input_layer())
850  return typename Root::Data();
851  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(depth);
852  Matrix<Scalar> prev_out_grads = internal::CuBLASHandle<Scalar>::get_instance().mul(out_grads_conv_mat,
853  weights_without_bias, false, true);
854 #endif
855  prev_out_conversion_dims[0] = rows;
856  return TensorMap<Scalar,4>(prev_out_grads.data(), prev_out_conversion_dims);
857  }

869  std::size_t dilation, std::size_t stride) {
870  return (input_dim - 1) * stride + receptor_size + (receptor_size - 1) * dilation - 2 * padding;
871  }
872 private:

874  const std::size_t filters;
875  const std::size_t receptor_height;
876  const std::size_t receptor_width;
877  const std::size_t vertical_padding;
878  const std::size_t horizontal_padding;
879  const std::size_t vertical_stride;

886  const std::size_t dil_receptor_height;
887  const std::size_t dil_receptor_width;
888  RankwiseArray prev_out_conversion_dims;
889  RankwiseArray patch_offsets;
890  RankwiseArray patch_extents;
891  RankwiseArray dil_strides;
892  RankwiseArray no_padding_offsets;
893  RankwiseArray no_padding_extents;

1251  Matrix<Scalar> prev_out_grads(out.rows(), out.cols());
1252  for (int i = 0; i < prev_out_grads.rows(); ++i) {
1253  RowVector<Scalar> row_i = out.row(i);
1254  // FIXME Do not evaluate the expressions into a temporary variable.
1255  Matrix<Scalar> jacobian = row_i.asDiagonal();
1256  jacobian -= row_i.transpose() * row_i;
1257  prev_out_grads.row(i) = out_grads_mat.row(i) * jacobian;
1258  }
1259  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grads.data(), conversion_dims);

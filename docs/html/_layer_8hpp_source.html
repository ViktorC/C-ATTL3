320  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
321  assert((Dimensions<std::size_t,Root::DATA_RANK>(in.dimensions()).template demote<>()) == Base::input_dims);
322  assert(in.dimension(0) > 0);
323  unsigned input_size = Base::input_dims.get_volume();
324  // Add a 1-column to the input for the bias trick.
325  biased_in = Matrix<Scalar>(in.dimension(0), input_size + 1);
326  biased_in.leftCols(input_size) = MatrixMap<Scalar>(in.data(), in.dimension(0), input_size);
327  biased_in.col(input_size).setOnes();
328 #ifndef CATTLE_USE_CUBLAS
329  Matrix<Scalar> out = biased_in * Base::weights_ref;
330 #else
331  Matrix<Scalar> out = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in, Base::weights_ref, false, false);
332 #endif

335  }
336  inline typename Root::Data pass_back(typename Root::Data out_grads) {
337  assert((Dimensions<std::size_t,Root::DATA_RANK>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
338  assert(out_grads.dimension(0) > 0 && biased_in.rows() == out_grads.dimension(0));
339  // Compute the gradient of the outputs with respect to the weights.
340 #ifndef CATTLE_USE_CUBLAS
341  MatrixMap<Scalar> out_grads_mat(out_grads.data(), out_grads.dimension(0), Base::output_dims.get_volume());
342  Base::weights_grad = biased_in.transpose() * out_grads_mat;
343  if (Base::is_input_layer())
344  return typename Root::Data();
345  /* Remove the bias row from the weight matrix, transpose it, and compute the derivative w.r.t. the
346  * previous layer's output. */
347  Matrix<Scalar> prev_out_grads = out_grads_mat * Base::weights_ref.topRows(Base::input_dims.get_volume()).transpose();
348 #else
349  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(), out_grads.dimension(0),
350  Base::output_dims.get_volume());
351  Base::weights_grad = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in, out_grads_mat, true, false);
352  if (Base::is_input_layer())
353  return typename Root::Data();
354  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(Base::input_dims.get_volume());
355  Matrix<Scalar> prev_out_grads = internal::CuBLASUtils<Scalar>::get_instance().mul(out_grads_mat,
356  weights_without_bias, false, true);
357 #endif
358  prev_out_conversion_dims[0] = prev_out_grads.rows();
359  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grads.data(), prev_out_conversion_dims);
360  }
361 private:
362  RankwiseArray out_conversion_dims;
363  RankwiseArray prev_out_conversion_dims;
364  // Staged computation caches
365  Matrix<Scalar> biased_in;
366 };

400  inline ConvLayer(const Dimensions<std::size_t,3>& input_dims, std::size_t filters, WeightInitSharedPtr<Scalar> weight_init,

415  vertical_padding(vertical_padding),
416  horizontal_padding(horizontal_padding),
417  vertical_stride(vertical_stride),
418  horizontal_stride(horizontal_stride),
419  vertical_dilation(vertical_dilation),
420  horizontal_dilation(horizontal_dilation),
421  padded_height(input_dims(0) + 2 * vertical_padding),
422  padded_width(input_dims(1) + 2 * horizontal_padding),
423  dil_receptor_height(receptor_height + (receptor_height - 1) * vertical_dilation),
424  dil_receptor_width(receptor_width + (receptor_width - 1) * horizontal_dilation),
425  out_conversion_dims({ 0u, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) }),
426  patch_offsets({ 0u, 0u, 0u, 0u }),
427  patch_extents({ 0u, dil_receptor_height, dil_receptor_width, input_dims(2) }),
428  dil_strides({ 1u, vertical_dilation + 1u, horizontal_dilation + 1u, 1u }),
429  no_padding_offsets({ 0u, vertical_padding, horizontal_padding, 0u }),
430  no_padding_extents({ 0u, input_dims(0), input_dims(1), input_dims(2) }),
431  paddings({ std::make_pair(0, 0), std::make_pair(vertical_padding, vertical_padding),
432  std::make_pair(horizontal_padding, horizontal_padding), std::make_pair(0, 0) }) {
433  assert(filters > 0);
434  assert(receptor_height > 0);
435  assert(receptor_width > 0);
436  assert(vertical_stride > 0 && horizontal_stride > 0);
437  assert(input_dims(1) + 2 * vertical_padding >= dil_receptor_height &&
438  input_dims(2) + 2 * horizontal_padding >= dil_receptor_width);

469  inline void empty_cache() {

472  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
473  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
474  assert(in.dimension(0) > 0);
475  // Spatial padding.
476  if (vertical_padding > 0 || horizontal_padding > 0)
477  in = typename Root::Data(in.pad(paddings));
478  std::size_t rows = in.dimension(0);
479  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
480  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
481  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
482  * out receptor of one sample. The same receptor location along all samples of the batch is represented
483  * by a contiguous block of these rows. There is one block for each receptor location. */
484  std::size_t patch_ind = 0;
485  patch_extents[0] = rows;
486  biased_in = Matrix<Scalar>(total_patches, receptor_vol + 1);
487  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
488  patch_offsets[2] = i;
489  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
490  patch_offsets[1] = j;
491  typename Root::Data patch;
492  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
493  if (vertical_dilation > 0 || horizontal_dilation > 0)
494  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
495  else
496  patch = in.slice(patch_offsets, patch_extents);
497  biased_in.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
498  patch_ind += rows;
499  }
500  }
501  assert(patch_ind == total_patches);
502  // Bias trick.
503  biased_in.col(receptor_vol).setOnes();
504 #ifndef CATTLE_USE_CUBLAS
505  Matrix<Scalar> out = biased_in * Base::weights_ref;
506 #else
507  Matrix<Scalar> out = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in,
508  Base::weights_ref, false, false);
509 #endif
510  out_conversion_dims[0] = rows;
511  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
512  }
513  inline typename Root::Data pass_back(typename Root::Data out_grads) {
514  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
515  assert(out_grads.dimension(0) > 0 && biased_in.rows() / (Base::output_dims(0) *
516  Base::output_dims(1)) == out_grads.dimension(0));
517  std::size_t rows = out_grads.dimension(0);
518  std::size_t patches = Base::output_dims(0) * Base::output_dims(1);
519  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
520 #ifndef CATTLE_USE_CUBLAS
521  MatrixMap<Scalar> out_grads_mat(out_grads.data(), rows * Base::output_dims(0) * Base::output_dims(1), filters);
522  Base::weights_grad = biased_in.transpose() * out_grads_mat;
523  if (Base::is_input_layer())
524  return typename Root::Data();
525  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
526  * previous layer's output. */
527  Matrix<Scalar> prev_out_grads_mat = out_grads_mat * Base::weights_ref.topRows(receptor_vol).transpose();
528 #else
529  Matrix<Scalar> out_grads_mat = MatrixMap<Scalar>(out_grads.data(),
530  rows * Base::output_dims(0) * Base::output_dims(1), filters);
531  Base::weights_grad = internal::CuBLASUtils<Scalar>::get_instance().mul(biased_in, out_grads_mat, true, false);
532  if (Base::is_input_layer())
533  return typename Root::Data();
534  Matrix<Scalar> weights_without_bias = Base::weights_ref.topRows(receptor_vol);
535  Matrix<Scalar> prev_out_grads_mat = internal::CuBLASUtils<Scalar>::get_instance().mul(out_grads_mat,
536  weights_without_bias, false, true);
537 #endif
538  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
539  * to get the derivative w.r.t. the individual input nodes. */
540  typename Root::Data prev_out_grads(rows, padded_height, padded_width, Base::input_dims(2));
541  prev_out_grads.setZero();
542  std::size_t patch_ind = 0;
543  patch_extents[0] = rows;
544  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
545  patch_offsets[2] = i;
546  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
547  patch_offsets[1] = j;
548  // Accumulate the gradients where the receptor-patch-tensors overlap.
549  Matrix<Scalar> prev_out_grads_block = prev_out_grads_mat.block(patch_ind, 0, rows, receptor_vol);
550  TensorMap<Scalar,4> prev_out_grads_block_map(prev_out_grads_block.data(), patch_extents);
551  if (vertical_dilation > 0 || horizontal_dilation > 0)
552  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_block_map;
553  else
554  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_block_map;
555  patch_ind += rows;
556  }
557  }
558  assert(patch_ind == prev_out_grads_mat.rows());
559  if (vertical_padding > 0 || horizontal_padding > 0) {
560  // Cut off the padding.
561  no_padding_extents[0] = rows;
562  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
563  } else
564  return prev_out_grads;
565  }
576  inline static std::size_t calculate_output_dim(std::size_t input_dim, std::size_t receptor_size, std::size_t padding,
577  std::size_t dilation, std::size_t stride) {
578  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
579  }
580 private:
581  // The defining attributes of the convolutional layer.
582  const std::size_t filters;
583  const std::size_t receptor_height;
584  const std::size_t receptor_width;
585  const std::size_t vertical_padding;
586  const std::size_t horizontal_padding;
587  const std::size_t vertical_stride;
588  const std::size_t horizontal_stride;
589  const std::size_t vertical_dilation;
590  const std::size_t horizontal_dilation;
591  // Pre-computed values to improve propagation-time performance.
592  const std::size_t padded_height;
593  const std::size_t padded_width;
594  const std::size_t dil_receptor_height;
595  const std::size_t dil_receptor_width;
596  RankwiseArray out_conversion_dims;
597  RankwiseArray patch_offsets;
598  RankwiseArray patch_extents;
599  RankwiseArray dil_strides;
600  RankwiseArray no_padding_offsets;
601  RankwiseArray no_padding_extents;
602  PaddingsArray paddings;
603  // Staged computation caches
604  Matrix<Scalar> biased_in;

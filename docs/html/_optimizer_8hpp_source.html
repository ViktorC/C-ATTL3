45 protected:

71  inline bool verify_gradients(Net& net, Provider& provider, bool verbose = true,
72  Scalar step_size = internal::NumericUtils<Scalar>::EPSILON2 / 2,
73  Scalar abs_epsilon = internal::NumericUtils<Scalar>::EPSILON2,
74  Scalar rel_epsilon = internal::NumericUtils<Scalar>::EPSILON2) const {
75  assert(net.get_input_dims() == provider.get_obs_dims());
76  assert(net.get_output_dims() == provider.get_obj_dims());
77  assert(step_size > 0);
78  assert(abs_epsilon >= 0 && rel_epsilon > 0);
79  bool failure = false;
80  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
81  std::size_t instances = data_pair.first.dimension(0);
82  provider.reset();
83  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
84  * back-propagate is to be divided by the number of observations in the batch. */
85  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);
86  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
87  for (std::size_t i = 0; i < layers.size(); ++i) {
88  Layer<Scalar,Rank>& layer = *layers[i];
89  if (layer.is_parametric()) {
90  if (verbose)
91  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) << std::string(28, '-') << std::endl;
92  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
93  * parameters' gradient. */
94  layer.regularize();
95  Matrix<Scalar>& params = layer.get_params();
96  const Matrix<Scalar>& params_grad = layer.get_params_grad();
97  for (int j = 0; j < params.rows(); ++j) {
98  for (int k = 0; k < params.cols(); ++k) {
99  if (verbose)
100  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
101  Scalar ana_grad = params_grad(j,k);
102  if (verbose)
103  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
104  Scalar param = params(j,k);
105  params(j,k) = param + step_size;
106  /* Compute the numerical gradients in training mode to ensure that the means and standard
107  * deviations used for batch normalization are the same as those used during the analytic
108  * gradient computation. */
109  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
110  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's parameters
111  * is included in the gradient. */
112  Scalar reg_pen_inc = layer.get_regularization_penalty();
113  params(j,k) = param - step_size;
114  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
115  Scalar reg_pen_dec = layer.get_regularization_penalty();
116  params(j,k) = param;
117  // Include the regularization penalty as well.
118  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
119  if (verbose)
120  std::cout << "\t\tNumerical gradient = " << num_grad;
121  if (!internal::NumericUtils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
122  if (verbose)
123  std::cout << " *****FAIL*****";
124  failure = true;
125  }
126  if (verbose)
127  std::cout << std::endl;
128  }
129  }
130  }
131  }
132  // Empty the network caches.
133  net.empty_caches();
134  return !failure;
135  }
152  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0,
153  bool verbose = true) {
154  assert(net.get_input_dims() == training_prov.get_obs_dims());
155  assert(net.get_output_dims() == training_prov.get_obj_dims());
156  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
157  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
158  // Fit the optimizer parameters to the network.
159  fit(net);
160  Scalar prev_test_loss = internal::NumericUtils<Scalar>::MAX;
161  unsigned cons_loss_inc = 0;
162  if (verbose)
163  std::cout << "<Optimization>" << std::endl;
164  // Start the optimization iterations.
165  for (unsigned i = 0; i <= epochs; ++i) {
166  if (verbose)
167  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
168  // Train.
169  if (i != 0) {
170  training_prov.reset();
171  if (verbose)
172  std::cout << "\ttraining loss: " << std::to_string(_train(net, training_prov, i, verbose)) << std::endl;
173  }
174  // Validate.
175  test_prov.reset();
176  Scalar test_loss = _test(net, test_prov, i, verbose);
177  if (verbose)
178  std::cout << "\ttest loss: " << std::to_string(test_loss);
179  if (test_loss >= prev_test_loss) {
180  cons_loss_inc++;
181  if (verbose)
182  std::cout << " *****INCREASED LOSS*****";
183  if (early_stop > 0 && cons_loss_inc >= early_stop)
184  return prev_test_loss;
185  } else
186  cons_loss_inc = 0;
187  if (verbose)
188  std::cout << std::endl << std::endl;
189  prev_test_loss = test_loss;
190  }
191  // Reset the providers.
192  training_prov.reset();
193  test_prov.reset();
194  // Empty the network caches.
195  net.empty_caches();
196  return prev_test_loss;
197  }

217  if (verbose)
218  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
219  prov.reset();

222  std::cout << "\ttraining loss: " << std::to_string(train_loss) << std::endl;

227  }

420 template<typename Scalar, std::size_t Rank, bool Sequential>
421 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
422  typedef Optimizer<Scalar,Rank,Sequential> Base;
423 public:
424  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size) :
425  Base::Optimizer(loss),
426  batch_size(batch_size) {
427  assert(batch_size > 0);
428  }
429  virtual ~SGDOptimizer() = default;
430 protected:
431  inline Scalar _train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch, bool verbose) {
432  Scalar training_loss = 0;
433  Scalar instances = 0;
434  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
435  // Perform an entire training epoch.
436  while (training_prov.has_more()) {
437  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
438  instances += data_pair.first.dimension(0);
439  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
440  training_loss += Base::loss->function(out, data_pair.second).sum();
441  /* Divide the gradient by the batch size to decouple the learning rate and the batch
442  * size hyper-parameters. Use the nominal batch size as the denominator even if the
443  * actual batch size is different (in case the number of samples in the data set is
444  * not divisible by the batch size and the last batch of the epoch contains fewer
445  * instances than the others) to make sure that the magnitude of the gradient is
446  * proportional to the batch size (just like its 'accuracy' is). */
447  Base::backpropagate(net, Base::loss->d_function(std::move(out), std::move(data_pair.second)) / (Scalar) batch_size);
448  for (unsigned k = 0; k < layers.size(); ++k) {
449  Layer<Scalar,Rank>& layer = *(layers[k]);
450  if (Base::is_parametric(layer) && !layer.is_frozen()) {
451  Base::regularize(layer);
452  update_params(layer, k, epoch - 1);
453  Base::enforce_constraints(layer);
454  }
455  }
456  }
457  return training_loss / instances;
458  }
459  inline Scalar _test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch, bool verbose) {
460  Scalar obj_loss = 0;
461  Scalar instances = 0;
462  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
463  // Perform an entire test epoch.
464  while (test_prov.has_more()) {
465  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
466  instances += data_pair.first.dimension(0);
467  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), std::move(data_pair.second)).sum();
468  }
469  Scalar mean_obj_loss = obj_loss / instances;
470  Scalar reg_loss = 0;
471  for (unsigned j = 0; j < layers.size(); ++j)
472  reg_loss += Base::get_regularization_penalty(*layers[j]);
473  if (verbose) {
474  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
475  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
476  }
477  return mean_obj_loss + reg_loss;
478  }
487  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
488  const unsigned batch_size;
489 };

522 template<typename Scalar, std::size_t Rank, bool Sequential>
523 class MomentumAcceleratedSGDOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
524 public:
537  inline MomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
538  Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
539  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
540  init_learning_rate(init_learning_rate),
541  annealing_rate(annealing_rate),
542  momentum(momentum) {
543  assert(init_learning_rate > 0);
544  assert(annealing_rate >= 0);
545  assert(momentum > 0 && momentum < 1);
546  }
547  virtual ~MomentumAcceleratedSGDOptimizer() = default;
548  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
549  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
550  params_grad_vec = std::vector<Matrix<Scalar>>(layers.size());
551  for (unsigned i = 0; i < params_grad_vec.size(); ++i) {
552  Layer<Scalar,Rank>& layer = *(layers[i]);
553  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
554  Matrix<Scalar> acc_params_grad = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
555  params_grad_vec[i] = acc_params_grad;
556  }
557  }
558 protected:
559  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
560  Scalar learning_rate = calculate_learning_rate(epoch);
561  Matrix<Scalar>& params_grad = params_grad_vec[i];
562  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
563  params += momentum * params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
564  }
571  Scalar calculate_learning_rate(unsigned epoch) {
572  return init_learning_rate / (1 + annealing_rate * epoch);
573  }
574  const Scalar init_learning_rate;
575  const Scalar annealing_rate;
576  const Scalar momentum;
577  std::vector<Matrix<Scalar>> params_grad_vec;
578 };

583 template<typename Scalar, std::size_t Rank, bool Sequential>
584 class NesterovMomentumAcceleratedSGDOptimizer : public MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> {
585  typedef MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> Base;
586 public:
599  inline NesterovMomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss,
600  unsigned batch_size = 1, Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
601  Base::MomentumAcceleratedSGDOptimizer(loss, batch_size, init_learning_rate, annealing_rate, momentum) { };
602 protected:
603  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
604  Scalar learning_rate = Base::calculate_learning_rate(epoch);
605  Matrix<Scalar>& acc_params_grad = Base::params_grad_vec[i];
606  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
607  Matrix<Scalar> params_grad_bak = acc_params_grad;
608  acc_params_grad = Base::momentum * acc_params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
609  params += -Base::momentum * params_grad_bak + (1 + Base::momentum) * acc_params_grad;
610  }
611 };

616 template<typename Scalar, std::size_t Rank, bool Sequential>
617 class AdagradOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
618 public:
627  inline AdagradOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
628  Scalar learning_rate = 1e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
629  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
630  learning_rate(learning_rate),
631  epsilon(epsilon) {
632  assert(learning_rate > 0);
633  assert(epsilon > 0);
634  }
635  virtual ~AdagradOptimizer() = default;
636 protected:
637 public:
638  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
639  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
640  params_grad_sqrs_vec = std::vector<Matrix<Scalar>>(layers.size());
641  for (unsigned i = 0; i < params_grad_sqrs_vec.size(); ++i) {
642  Layer<Scalar,Rank>& layer = *(layers[i]);
643  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
644  Matrix<Scalar> params_grad_sqrs = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
645  params_grad_sqrs_vec[i] = params_grad_sqrs;
646  }
647  }
654  inline virtual void update_acc_params_grad_sqrs(Matrix<Scalar>& acc_params_grad_sqrs,
655  const Matrix<Scalar>& params_grad) {
656  // Accumulate the squares of the gradients.
657  acc_params_grad_sqrs += params_grad.cwiseProduct(params_grad);
658  }
659  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
660  Matrix<Scalar>& params_grad_sqrs = params_grad_sqrs_vec[i];
661  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
662  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
663  update_acc_params_grad_sqrs(params_grad_sqrs, params_grad);
664  params -= (learning_rate * params_grad.array() / (params_grad_sqrs.array().sqrt() + epsilon)).matrix();
665  }
666  const Scalar learning_rate;
667  const Scalar epsilon;
668  std::vector<Matrix<Scalar>> params_grad_sqrs_vec;
669 };

705 template<typename Scalar, std::size_t Rank, bool Sequential>
706 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
707 public:
717  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
718  Scalar decay = 5e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
719  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
720  decay(decay),
721  epsilon(epsilon) {
722  assert(decay >= 0 && decay <= 1);
723  assert(epsilon > 0);
724  }
725  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
726  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
727  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
728  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
729  Layer<Scalar,Rank>& layer = *(layers[i]);
730  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
731  ParamGradAndUpdateSqrs pgus;
732  pgus.params_grad = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
733  pgus.params_update = Matrix<Scalar>::Zero(pgus.params_grad.rows(), pgus.params_grad.cols());
734  pgus_vec[i] = pgus;
735  }
736  }
737 protected:
738  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
739  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
740  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
741  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
742  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
743  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
744  (pgus.params_grad.array() + epsilon).sqrt();
745  params += weight_updates;
746  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
747  }
748  const Scalar decay;
749  const Scalar epsilon;
753  struct ParamGradAndUpdateSqrs {
754  Matrix<Scalar> params_grad;
755  Matrix<Scalar> params_update;
756  };
757  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
758 };

763 template<typename Scalar, std::size_t Rank, bool Sequential>
764 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
765 public:
780  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
781  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
782  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
783  learning_rate(learning_rate),
784  l1_decay(l1_decay),
785  l2_decay(l2_decay),
786  epsilon(epsilon) {
787  assert(learning_rate > 0);
788  assert(l1_decay >= 0 && l1_decay <= 1);
789  assert(l2_decay >= 0 && l2_decay <= 1);
790  assert(epsilon > 0);
791  }
792  virtual ~AdamOptimizer() = default;
793  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
794  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
795  pgn_vec = std::vector<ParamGradNorms>(layers.size());
796  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
797  Layer<Scalar,Rank>& layer = *(layers[i]);
798  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
799  ParamGradNorms vel;
800  vel.params_grad_l1 = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
801  vel.params_grad_l2 = Matrix<Scalar>::Zero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
802  pgn_vec[i] = vel;
803  }
804  }
805 protected:
806  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
807  ParamGradNorms& grad_norms = pgn_vec[i];
808  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - l1_decay, epoch + 1) + epsilon);
809  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - l2_decay, epoch + 1) + epsilon);
810  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
811  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
812  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
813  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
814  l2_decay * params_grad.cwiseProduct(params_grad);
815  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
816  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
817  }
818  const Scalar learning_rate;
819  const Scalar l1_decay;
820  const Scalar l2_decay;
821  const Scalar epsilon;
826  struct ParamGradNorms {
827  Matrix<Scalar> params_grad_l1;
828  Matrix<Scalar> params_grad_l2;
829  };
830  std::vector<ParamGradNorms> pgn_vec;
831 };

864  grad_norms.params_grad_l2 = ((1 - Base::l2_decay) * grad_norms.params_grad_l2).cwiseMax(params_grad.cwiseAbs());
865  params -= (Base::learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /

891  inline NadamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,

897  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - Base::l1_decay, epoch + 1) + Base::epsilon);
898  Scalar l1_next_corr = (Scalar) 1 / (1 - pow(1 - Base::l1_decay, epoch + 2) + Base::epsilon);
899  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - Base::l2_decay, epoch + 1) + Base::epsilon);
900  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
901  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
902  grad_norms.params_grad_l1 = (1 - Base::l1_decay) * grad_norms.params_grad_l1 + Base::l1_decay * params_grad;
903  grad_norms.params_grad_l2 = (1 - Base::l2_decay) * grad_norms.params_grad_l2 + Base::l2_decay *
904  params_grad.cwiseProduct(params_grad);
905  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * params_grad +
906  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.params_grad_l1).array() /
907  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + Base::epsilon)).matrix();

69  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, std::size_t epochs,
70  std::size_t early_stop = 0, Scalar target_loss = NumericUtils<Scalar>::MIN, bool verbose = true) {
71  assert(net.get_input_dims() == training_prov.get_obs_dims());
72  assert(net.get_output_dims() == training_prov.get_obj_dims());
73  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
74  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
75  // Fit the optimizer parameters to the network.
76  fit(net);
77  Scalar prev_test_loss = NumericUtils<Scalar>::MAX;
78  std::size_t cons_loss_inc = 0;
79  if (verbose)
80  std::cout << "<Optimization>" << std::endl;
81  // Start the optimization iterations.
82  for (std::size_t i = 0; i <= epochs; ++i) {
83  if (verbose)
84  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
85  // Train.
86  if (i != 0) {
87  training_prov.reset();
88  Scalar train_loss = _train(net, training_prov, i, verbose);
89  if (verbose) {
90  std::cout << std::left << std::setw(20) << "\ttraining loss: " << std::right <<
91  std::to_string(train_loss) << std::endl;
92  }
93  }
94  // Validate.
95  test_prov.reset();
96  Scalar test_loss = _test(net, test_prov, i, verbose);
97  if (verbose) {
98  std::cout << std::left << std::setw(20) << "\ttest loss: " << std::right <<
99  std::to_string(test_loss);
100  }
101  if (test_loss >= prev_test_loss) {
102  cons_loss_inc++;
103  if (verbose)
104  std::cout << " *****INCREASED LOSS*****";
105  if (early_stop > 0 && cons_loss_inc >= early_stop)
106  break;
107  } else
108  cons_loss_inc = 0;
109  if (verbose)
110  std::cout << std::endl << std::endl;
111  prev_test_loss = test_loss;
112  if (prev_test_loss <= target_loss)
113  break;
114  }
115  // Reset the providers.
116  training_prov.reset();
117  test_prov.reset();
118  // Empty the network caches.
119  net.empty_caches();
120  return prev_test_loss;
121  }

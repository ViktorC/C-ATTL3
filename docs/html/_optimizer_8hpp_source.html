48 template<typename Scalar, std::size_t Rank, bool Sequential>
49 class Optimizer {
50  static_assert(std::is_floating_point<Scalar>::value, "non floating-point scalar type");
51  static_assert(Rank > 0 && Rank < 4, "illegal optimizer rank");
52 protected:
53  typedef NeuralNetwork<Scalar,Rank,Sequential> Net;
54  typedef DataProvider<Scalar,Rank,Sequential> Provider;
55  typedef Tensor<Scalar,Rank + Sequential + 1> Data;
56 public:
57  inline Optimizer(LossSharedPtr<Scalar,Rank,Sequential> loss) :
58  loss(loss) {
59  assert(loss != nullptr);
60  }
61  virtual ~Optimizer() = default;
78  inline bool verify_gradients(Net& net, Provider& provider, bool verbose = true,
79  Scalar step_size = (internal::NumericUtils<Scalar>::EPSILON2 + internal::NumericUtils<Scalar>::EPSILON3) / 2,
80  Scalar abs_epsilon = internal::NumericUtils<Scalar>::EPSILON2,
81  Scalar rel_epsilon = internal::NumericUtils<Scalar>::EPSILON3) const {
82  assert(net.get_input_dims() == provider.get_obs_dims());
83  assert(net.get_output_dims() == provider.get_obj_dims());
84  assert(step_size > 0);
85  assert(abs_epsilon >= 0 && rel_epsilon > 0);
86  bool failure = false;
87  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
88  std::size_t instances = data_pair.first.dimension(0);
89  provider.reset();
90  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
91  * back-propagate is to be divided by the number of observations in the batch. */
92  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);
93  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
94  for (std::size_t i = 0; i < layers.size(); ++i) {
95  Layer<Scalar,Rank>& layer = *layers[i];
96  if (layer.is_parametric()) {
97  if (verbose)
98  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) << std::string(28, '-') << std::endl;
99  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
100  * parameters' gradient. */
101  layer.regularize();
102  Matrix<Scalar>& params = layer.get_params();
103  const Matrix<Scalar>& params_grad = layer.get_params_grad();
104  for (int j = 0; j < params.rows(); ++j) {
105  for (int k = 0; k < params.cols(); ++k) {
106  if (verbose)
107  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
108  Scalar ana_grad = params_grad(j,k);
109  if (verbose)
110  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
111  Scalar param = params(j,k);
112  params(j,k) = param + step_size;
113  /* Compute the numerical gradients in training mode to ensure that the means and standard
114  * deviations used for batch normalization are the same as those used during the analytic
115  * gradient computation. */
116  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
117  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's parameters
118  * is included in the gradient. */
119  Scalar reg_pen_inc = layer.get_regularization_penalty();
120  params(j,k) = param - step_size;
121  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
122  Scalar reg_pen_dec = layer.get_regularization_penalty();
123  params(j,k) = param;
124  // Include the regularization penalty as well.
125  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
126  if (verbose)
127  std::cout << "\t\tNumerical gradient = " << num_grad;
128  if (!internal::NumericUtils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
129  if (verbose)
130  std::cout << " *****FAIL*****";
131  failure = true;
132  }
133  if (verbose)
134  std::cout << std::endl;
135  }
136  }
137  }
138  }
139  // Empty the network caches.
140  net.empty_caches();
141  return !failure;
142  }
158  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0,
159  bool verbose = true) {
160  assert(net.get_input_dims() == training_prov.get_obs_dims());
161  assert(net.get_output_dims() == training_prov.get_obj_dims());
162  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
163  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
164  assert(epochs > 0);
165  // Fit the optimizer parameters to the network.
166  fit(net);
167  Scalar prev_test_loss = internal::NumericUtils<Scalar>::MAX;
168  unsigned cons_loss_inc = 0;
169  // Start the optimization iterations.
170  for (unsigned i = 0; i <= epochs; ++i) {
171  if (verbose)
172  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
173  // Train.
174  if (i != 0) {
175  training_prov.reset();
176  if (verbose)
177  std::cout << "\ttraining loss: " << std::to_string(train(net, training_prov, i, verbose)) << std::endl;
178  }
179  // Validate.
180  test_prov.reset();
181  Scalar test_loss = test(net, test_prov, i, verbose);
182  if (verbose)
183  std::cout << "\ttest loss: " << std::to_string(test_loss);
184  if (test_loss >= prev_test_loss) {
185  cons_loss_inc++;
186  if (verbose)
187  std::cout << " *****INCREASED LOSS*****";
188  if (early_stop > 0 && cons_loss_inc >= early_stop)
189  return prev_test_loss;
190  } else
191  cons_loss_inc = 0;
192  if (verbose)
193  std::cout << std::endl << std::endl;
194  prev_test_loss = test_loss;
195  }
196  // Reset the providers.
197  training_prov.reset();
198  test_prov.reset();
199  // Empty the network caches.
200  net.empty_caches();
201  return prev_test_loss;
202  }
203 protected:
210  virtual void fit(Net& net) = 0;
222  virtual Scalar train(Net& net, Provider& training_prov, unsigned epoch, bool verbose) = 0;
234  virtual Scalar test(Net& net, Provider& test_prov, unsigned epoch, bool verbose) = 0;
246  inline static std::vector<Layer<Scalar,Rank>*> get_layers(Net& net) {
247  return net.get_layers();
248  }
261  inline static Data propagate(Net& net, Data in) {
262  return net.propagate(std::move(in), true);
263  }
276  inline static void backpropagate(Net& net, Data out_grads) {
277  net.backpropagate(std::move(out_grads));
278  }
290  inline static bool is_parametric(Layer<Scalar,Rank>& layer) {
291  return layer.is_parametric();
292  }
305  inline static Matrix<Scalar>& get_params(Layer<Scalar,Rank>& layer) {
306  return layer.get_params();
307  }
321  inline static const Matrix<Scalar>& get_params_grad(Layer<Scalar,Rank>& layer) {
322  return layer.get_params_grad();
323  }
335  inline static void regularize(Layer<Scalar,Rank>& layer) {
336  return layer.regularize();
337  }
349  inline static Scalar get_regularization_penalty(Layer<Scalar,Rank>& layer) {
350  return layer.get_regularization_penalty();
351  }
362  inline static void enforce_constraints(Layer<Scalar,Rank>& layer) {
363  layer.enforce_constraints();
364  }
365  const LossSharedPtr<Scalar,Rank,Sequential> loss;
366 };

371 template<typename Scalar, std::size_t Rank, bool Sequential>
372 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
373  typedef Optimizer<Scalar,Rank,Sequential> Base;
374 public:
375  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size) :
376  Base::Optimizer(loss),
377  batch_size(batch_size) {
378  assert(batch_size > 0);
379  }
380  virtual ~SGDOptimizer() = default;
381 protected:
382  inline Scalar train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch, bool verbose) {
383  Scalar training_loss = 0;
384  Scalar instances = 0;
385  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
386  // Perform an entire training epoch.
387  while (training_prov.has_more()) {
388  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
389  instances += data_pair.first.dimension(0);
390  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
391  training_loss += Base::loss->function(out, data_pair.second).sum();
392  /* Divide the gradient by the batch size to decouple the learning rate and the batch
393  * size hyper-parameters. Use the nominal batch size as the denominator even if the
394  * actual batch size is different (in case the number of samples in the data set is
395  * not divisible by the batch size and the last batch of the epoch contains fewer
396  * instances than the others) to make sure that the magnitude of the gradient is
397  * proportional to the batch size (just like its 'accuracy' is). */
398  Base::backpropagate(net, Base::loss->d_function(std::move(out), std::move(data_pair.second)) / (Scalar) batch_size);
399  for (unsigned k = 0; k < layers.size(); ++k) {
400  Layer<Scalar,Rank>& layer = *(layers[k]);
401  if (Base::is_parametric(layer)) {
402  Base::regularize(layer);
403  update_params(layer, k, epoch - 1);
404  Base::enforce_constraints(layer);
405  }
406  }
407  }
408  return training_loss / instances;
409  }
410  inline Scalar test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch, bool verbose) {
411  Scalar obj_loss = 0;
412  Scalar instances = 0;
413  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
414  // Perform an entire test epoch.
415  while (test_prov.has_more()) {
416  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
417  instances += data_pair.first.dimension(0);
418  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), std::move(data_pair.second)).sum();
419  }
420  Scalar mean_obj_loss = obj_loss / instances;
421  Scalar reg_loss = 0;
422  for (unsigned j = 0; j < layers.size(); ++j)
423  reg_loss += Base::get_regularization_penalty(*layers[j]);
424  if (verbose) {
425  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
426  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
427  }
428  return mean_obj_loss + reg_loss;
429  }
438  virtual void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
439  const unsigned batch_size;
440 };

474 class MomentumAcceleratedSGDOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
475 public:
488  inline MomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
489  Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
490  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
491  init_learning_rate(init_learning_rate),
492  annealing_rate(annealing_rate),
493  momentum(momentum) {
494  assert(init_learning_rate > 0);
495  assert(annealing_rate >= 0);
496  assert(momentum > 0 && momentum < 1);
497  }
498  virtual ~MomentumAcceleratedSGDOptimizer() = default;
499 protected:
500  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
501  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
502  params_grad_vec = std::vector<Matrix<Scalar>>(layers.size());
503  for (unsigned i = 0; i < params_grad_vec.size(); ++i) {
504  Layer<Scalar,Rank>& layer = *(layers[i]);
505  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
506  Matrix<Scalar> acc_params_grad = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
507  params_grad_vec[i] = acc_params_grad;
508  }
509  }
510  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
511  Scalar learning_rate = calculate_learning_rate(epoch);
512  Matrix<Scalar>& params_grad = params_grad_vec[i];
513  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
514  params += momentum * params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
515  }
522  Scalar calculate_learning_rate(unsigned epoch) {
523  return init_learning_rate / (1 + annealing_rate * epoch);
524  }
525  const Scalar init_learning_rate;
526  const Scalar annealing_rate;
527  const Scalar momentum;
528  std::vector<Matrix<Scalar>> params_grad_vec;
529 };

535 class NesterovMomentumAcceleratedSGDOptimizer : public MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> {
536  typedef MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> Base;
537 public:
550  inline NesterovMomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss,
551  unsigned batch_size = 1, Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
552  Base::MomentumAcceleratedSGDOptimizer(loss, batch_size, init_learning_rate, annealing_rate, momentum) { };

555  Scalar learning_rate = Base::calculate_learning_rate(epoch);
556  Matrix<Scalar>& acc_params_grad = Base::params_grad_vec[i];
557  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
558  Matrix<Scalar> params_grad_bak = acc_params_grad;
559  acc_params_grad = Base::momentum * acc_params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
560  params += -Base::momentum * params_grad_bak + (1 + Base::momentum) * acc_params_grad;
561  }

567 template<typename Scalar, std::size_t Rank, bool Sequential>
568 class AdagradOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
569 public:
578  inline AdagradOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
579  Scalar learning_rate = 1e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
580  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
581  learning_rate(learning_rate),
582  epsilon(epsilon) {
583  assert(learning_rate > 0);
584  assert(epsilon > 0);
585  }
586  virtual ~AdagradOptimizer() = default;
587 protected:
588  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
589  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
590  params_grad_sqrs_vec = std::vector<Matrix<Scalar>>(layers.size());
591  for (unsigned i = 0; i < params_grad_sqrs_vec.size(); ++i) {
592  Layer<Scalar,Rank>& layer = *(layers[i]);
593  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
594  Matrix<Scalar> params_grad_sqrs = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
595  params_grad_sqrs_vec[i] = params_grad_sqrs;
596  }
597  }
604  inline virtual void update_acc_params_grad_sqrs(Matrix<Scalar>& acc_params_grad_sqrs,
605  const Matrix<Scalar>& params_grad) {
606  // Accumulate the squares of the gradients.
607  acc_params_grad_sqrs += params_grad.cwiseProduct(params_grad);
608  }
609  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
610  Matrix<Scalar>& params_grad_sqrs = params_grad_sqrs_vec[i];
611  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
612  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
613  update_acc_params_grad_sqrs(params_grad_sqrs, params_grad);
614  params -= (learning_rate * params_grad.array() / (params_grad_sqrs.array().sqrt() + epsilon)).matrix();
615  }
616  const Scalar learning_rate;
617  const Scalar epsilon;
618  std::vector<Matrix<Scalar>> params_grad_sqrs_vec;
619 };

655 template<typename Scalar, std::size_t Rank, bool Sequential>
656 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
657 public:
667  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
668  Scalar decay = 5e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
669  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
670  decay(decay),
671  epsilon(epsilon) {
672  assert(decay >= 0 && decay <= 1);
673  assert(epsilon > 0);
674  }
675 protected:
676  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
677  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
678  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
679  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
680  Layer<Scalar,Rank>& layer = *(layers[i]);
681  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
682  ParamGradAndUpdateSqrs pgus;
683  pgus.params_grad = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
684  pgus.params_update = Matrix<Scalar>::Zero(pgus.params_grad.rows(), pgus.params_grad.cols());
685  pgus_vec[i] = pgus;
686  }
687  }
688  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
689  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
690  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
691  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
692  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
693  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
694  (pgus.params_grad.array() + epsilon).sqrt();
695  params += weight_updates;
696  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
697  }
698  const Scalar decay;
699  const Scalar epsilon;
703  struct ParamGradAndUpdateSqrs {
704  Matrix<Scalar> params_grad;
705  Matrix<Scalar> params_update;
706  };
707  std::vector<ParamGradAndUpdateSqrs> pgus_vec;
708 };

713 template<typename Scalar, std::size_t Rank, bool Sequential>
714 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
715 public:
730  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
731  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
732  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
733  learning_rate(learning_rate),
734  l1_decay(l1_decay),
735  l2_decay(l2_decay),
736  epsilon(epsilon) {
737  assert(learning_rate > 0);
738  assert(l1_decay >= 0 && l1_decay <= 1);
739  assert(l2_decay >= 0 && l2_decay <= 1);
740  assert(epsilon > 0);
741  }
742  virtual ~AdamOptimizer() = default;
743 protected:
744  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
745  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
746  pgn_vec = std::vector<ParamGradNorms>(layers.size());
747  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
748  Layer<Scalar,Rank>& layer = *(layers[i]);
749  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
750  ParamGradNorms vel;
751  vel.params_grad_l1 = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
752  vel.params_grad_l2 = Matrix<Scalar>::Zero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
753  pgn_vec[i] = vel;
754  }
755  }
756  inline void update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
757  ParamGradNorms& grad_norms = pgn_vec[i];
758  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - l1_decay, epoch + 1) + epsilon);
759  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - l2_decay, epoch + 1) + epsilon);
760  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
761  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
762  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
763  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
764  l2_decay * params_grad.cwiseProduct(params_grad);
765  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
766  ((grad_norms.params_grad_l2 * l2_corr).array().sqrt() + epsilon)).matrix();
767  }

769  const Scalar l1_decay;
770  const Scalar l2_decay;
771  const Scalar epsilon;
776  struct ParamGradNorms {
777  Matrix<Scalar> params_grad_l1;
778  Matrix<Scalar> params_grad_l2;
779  };
780  std::vector<ParamGradNorms> pgn_vec;
781 };

71  inline bool verify_gradients(Net& net, Provider& provider, bool verbose = true,
72  Scalar step_size = internal::NumericUtils<Scalar>::EPSILON2 / 2,
73  Scalar abs_epsilon = internal::NumericUtils<Scalar>::EPSILON2,
74  Scalar rel_epsilon = internal::NumericUtils<Scalar>::EPSILON2) const {
75  assert(net.get_input_dims() == provider.get_obs_dims());
76  assert(net.get_output_dims() == provider.get_obj_dims());
77  assert(step_size > 0);
78  assert(abs_epsilon >= 0 && rel_epsilon > 0);
79  bool failure = false;
80  DataPair<Scalar,Rank,Sequential> data_pair = provider.get_data(std::numeric_limits<std::size_t>::max());
81  std::size_t instances = data_pair.first.dimension(0);
82  provider.reset();
83  /* As the loss to minimize is the mean of the losses for all the training observations, the gradient to
84  * back-propagate is to be divided by the number of observations in the batch. */
85  net.backpropagate(loss->d_function(net.propagate(data_pair.first, true), data_pair.second) / (Scalar) instances);
86  std::vector<Layer<Scalar,Rank>*> layers = net.get_layers();
87  for (std::size_t i = 0; i < layers.size(); ++i) {
88  Layer<Scalar,Rank>& layer = *layers[i];
89  if (layer.is_parametric()) {
90  if (verbose)
91  std::cout << "Layer " << std::setw(3) << std::to_string(i + 1) << std::string(28, '-') << std::endl;
92  /* Add the derivative of the regularization function w.r.t. to the parameters of the layer to the
93  * parameters' gradient. */
94  layer.regularize();
95  Matrix<Scalar>& params = layer.get_params();
96  const Matrix<Scalar>& params_grad = layer.get_params_grad();
97  for (int j = 0; j < params.rows(); ++j) {
98  for (int k = 0; k < params.cols(); ++k) {
99  if (verbose)
100  std::cout << "\tParam[" << i << "," << j << "," << k << "]:" << std::endl;
101  Scalar ana_grad = params_grad(j,k);
102  if (verbose)
103  std::cout << "\t\tAnalytic gradient = " << ana_grad << std::endl;
104  Scalar param = params(j,k);
105  params(j,k) = param + step_size;
106  /* Compute the numerical gradients in training mode to ensure that the means and standard
107  * deviations used for batch normalization are the same as those used during the analytic
108  * gradient computation. */
109  Scalar loss_inc = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
110  /* Calculate the new regularization penalty as its derivative w.r.t. the layer's parameters
111  * is included in the gradient. */
112  Scalar reg_pen_inc = layer.get_regularization_penalty();
113  params(j,k) = param - step_size;
114  Scalar loss_dec = loss->function(net.propagate(data_pair.first, true), data_pair.second).mean();
115  Scalar reg_pen_dec = layer.get_regularization_penalty();
116  params(j,k) = param;
117  // Include the regularization penalty as well.
118  Scalar num_grad = (loss_inc + reg_pen_inc - (loss_dec + reg_pen_dec)) / (2 * step_size);
119  if (verbose)
120  std::cout << "\t\tNumerical gradient = " << num_grad;
121  if (!internal::NumericUtils<Scalar>::almost_equal(ana_grad, num_grad, abs_epsilon, rel_epsilon)) {
122  if (verbose)
123  std::cout << " *****FAIL*****";
124  failure = true;
125  }
126  if (verbose)
127  std::cout << std::endl;
128  }
129  }
130  }
131  }
132  // Empty the network caches.
133  net.empty_caches();
134  return !failure;
135  }
154  inline Scalar optimize(Net& net, Provider& training_prov, Provider& test_prov, unsigned epochs, unsigned early_stop = 0,
155  Scalar target_loss = internal::NumericUtils<Scalar>::MIN, bool verbose = true) {
156  assert(net.get_input_dims() == training_prov.get_obs_dims());
157  assert(net.get_output_dims() == training_prov.get_obj_dims());
158  assert(training_prov.get_obs_dims() == test_prov.get_obs_dims());
159  assert(training_prov.get_obj_dims() == test_prov.get_obj_dims());
160  // Fit the optimizer parameters to the network.
161  fit(net);
162  Scalar prev_test_loss = internal::NumericUtils<Scalar>::MAX;
163  unsigned cons_loss_inc = 0;
164  if (verbose)
165  std::cout << "<Optimization>" << std::endl;
166  // Start the optimization iterations.
167  for (unsigned i = 0; i <= epochs; ++i) {
168  if (verbose)
169  std::cout << "Epoch " << std::setw(3) << i << std::string(28, '-') << std::endl;
170  // Train.
171  if (i != 0) {
172  training_prov.reset();
173  if (verbose)
174  std::cout << "\ttraining loss: " << std::to_string(_train(net, training_prov, i, verbose)) << std::endl;
175  }
176  // Validate.
177  test_prov.reset();
178  Scalar test_loss = _test(net, test_prov, i, verbose);
179  if (verbose)
180  std::cout << "\ttest loss: " << std::to_string(test_loss);
181  if (test_loss >= prev_test_loss) {
182  cons_loss_inc++;
183  if (verbose)
184  std::cout << " *****INCREASED LOSS*****";
185  if (early_stop > 0 && cons_loss_inc >= early_stop)
186  break;
187  } else
188  cons_loss_inc = 0;
189  if (verbose)
190  std::cout << std::endl << std::endl;
191  prev_test_loss = test_loss;
192  if (prev_test_loss <= target_loss)
193  break;
194  }
195  // Reset the providers.
196  training_prov.reset();
197  test_prov.reset();
198  // Empty the network caches.
199  net.empty_caches();
200  return prev_test_loss;
201  }

476 template<typename Scalar, std::size_t Rank, bool Sequential>
477 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
478  typedef Optimizer<Scalar,Rank,Sequential> Base;
479 public:
480  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size) :
481  Base::Optimizer(loss),
482  batch_size(batch_size) {
483  assert(batch_size > 0);
484  }
485  virtual ~SGDOptimizer() = default;
486 protected:
487  inline Scalar _train(typename Base::Net& net, typename Base::Provider& training_prov, unsigned epoch, bool verbose) {
488  Scalar training_loss = 0;
489  Scalar instances = 0;
490  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
491  // Perform an entire training epoch.
492  while (training_prov.has_more()) {
493  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
494  instances += data_pair.first.dimension(0);
495  typename Base::Data out = Base::propagate(net, std::move(data_pair.first));
496  training_loss += Base::loss->function(out, data_pair.second).sum();
497  /* Divide the gradient by the batch size to decouple the learning rate and the batch
498  * size hyper-parameters. Use the nominal batch size as the denominator even if the
499  * actual batch size is different (in case the number of samples in the data set is
500  * not divisible by the batch size and the last batch of the epoch contains fewer
501  * instances than the others) to make sure that the magnitude of the gradient is
502  * proportional to the batch size (just like its 'accuracy' is). */
503  Base::backpropagate(net, Base::loss->d_function(std::move(out), std::move(data_pair.second)) / (Scalar) batch_size);
504  for (unsigned k = 0; k < layers.size(); ++k) {
505  Layer<Scalar,Rank>& layer = *(layers[k]);
506  if (Base::is_parametric(layer) && !layer.is_frozen()) {
507  Base::regularize(layer);
508  _update_params(layer, k, epoch - 1);
509  Base::enforce_constraints(layer);
510  }
511  }
512  }
513  return training_loss / instances;
514  }
515  inline Scalar _test(typename Base::Net& net, typename Base::Provider& test_prov, unsigned epoch, bool verbose) {
516  Scalar obj_loss = 0;
517  Scalar instances = 0;
518  std::vector<Layer<Scalar,Rank>*> layers = Base::get_layers(net);
519  // Perform an entire test epoch.
520  while (test_prov.has_more()) {
521  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
522  instances += data_pair.first.dimension(0);
523  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)), std::move(data_pair.second)).sum();
524  }
525  Scalar mean_obj_loss = obj_loss / instances;
526  Scalar reg_loss = 0;
527  for (unsigned j = 0; j < layers.size(); ++j) {
528  Layer<Scalar,Rank>& layer = *(layers[j]);
529  if (Base::is_parametric(layer) && !layer.is_frozen())
530  reg_loss += Base::get_regularization_penalty(layer);
531  }
532  if (verbose) {
533  std::cout << "\tobj loss: " << std::to_string(mean_obj_loss) << std::endl;
534  std::cout << "\treg loss: " << std::to_string(reg_loss) << std::endl;
535  }
536  return mean_obj_loss + reg_loss;
537  }
546  virtual void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) = 0;
547  const unsigned batch_size;
548 };

581 template<typename Scalar, std::size_t Rank, bool Sequential>
582 class MomentumAcceleratedSGDOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
583 public:
596  inline MomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
597  Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
598  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
599  init_learning_rate(init_learning_rate),
600  annealing_rate(annealing_rate),
601  momentum(momentum) {
602  assert(init_learning_rate > 0);
603  assert(annealing_rate >= 0);
604  assert(momentum > 0 && momentum < 1);
605  }
606  virtual ~MomentumAcceleratedSGDOptimizer() = default;
607  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
608  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
609  params_grad_vec = std::vector<Matrix<Scalar>>(layers.size());
610  for (unsigned i = 0; i < params_grad_vec.size(); ++i) {
611  Layer<Scalar,Rank>& layer = *(layers[i]);
612  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
613  Matrix<Scalar> acc_params_grad = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
614  params_grad_vec[i] = acc_params_grad;
615  }
616  }
617 protected:
618  inline void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
619  Scalar learning_rate = calculate_learning_rate(epoch);
620  Matrix<Scalar>& params_grad = params_grad_vec[i];
621  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
622  params += momentum * params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
623  }
630  Scalar calculate_learning_rate(unsigned epoch) {
631  return init_learning_rate / (1 + annealing_rate * epoch);
632  }
633  const Scalar init_learning_rate;
634  const Scalar annealing_rate;
635  const Scalar momentum;
636  std::vector<Matrix<Scalar>> params_grad_vec;
637 };

642 template<typename Scalar, std::size_t Rank, bool Sequential>
643 class NesterovMomentumAcceleratedSGDOptimizer : public MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> {
644  typedef MomentumAcceleratedSGDOptimizer<Scalar,Rank,Sequential> Base;
645 public:
658  inline NesterovMomentumAcceleratedSGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss,
659  unsigned batch_size = 1, Scalar init_learning_rate = 1e-3, Scalar annealing_rate = 1e-3, Scalar momentum = .9) :
660  Base::MomentumAcceleratedSGDOptimizer(loss, batch_size, init_learning_rate, annealing_rate, momentum) { };
661 protected:
662  inline void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
663  Scalar learning_rate = Base::calculate_learning_rate(epoch);
664  Matrix<Scalar>& acc_params_grad = Base::params_grad_vec[i];
665  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
666  Matrix<Scalar> params_grad_bak = acc_params_grad;
667  acc_params_grad = Base::momentum * acc_params_grad - learning_rate * Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
668  params += -Base::momentum * params_grad_bak + (1 + Base::momentum) * acc_params_grad;
669  }
670 };

675 template<typename Scalar, std::size_t Rank, bool Sequential>
676 class AdagradOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
677 public:
686  inline AdagradOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
687  Scalar learning_rate = 1e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
688  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
689  learning_rate(learning_rate),
690  epsilon(epsilon) {
691  assert(learning_rate > 0);
692  assert(epsilon > 0);
693  }
694  virtual ~AdagradOptimizer() = default;
695 protected:
696 public:
697  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
698  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
699  params_grad_sqrs_vec = std::vector<Matrix<Scalar>>(layers.size());
700  for (unsigned i = 0; i < params_grad_sqrs_vec.size(); ++i) {
701  Layer<Scalar,Rank>& layer = *(layers[i]);
702  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
703  Matrix<Scalar> params_grad_sqrs = Matrix<Scalar>::Zero(params_grad.rows(), params_grad.cols());
704  params_grad_sqrs_vec[i] = params_grad_sqrs;
705  }
706  }
713  inline virtual void _update_acc_params_grad_sqrs(Matrix<Scalar>& acc_params_grad_sqrs,
714  const Matrix<Scalar>& params_grad) {
715  // Accumulate the squares of the gradients.
716  acc_params_grad_sqrs += params_grad.cwiseProduct(params_grad);
717  }
718  inline void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
719  Matrix<Scalar>& params_grad_sqrs = params_grad_sqrs_vec[i];
720  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
721  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
722  _update_acc_params_grad_sqrs(params_grad_sqrs, params_grad);
723  params -= (learning_rate * params_grad.array() / (params_grad_sqrs.array().sqrt() + epsilon)).matrix();
724  }
725  const Scalar learning_rate;
726  const Scalar epsilon;
727  std::vector<Matrix<Scalar>> params_grad_sqrs_vec;
728 };

733 template<typename Scalar, std::size_t Rank, bool Sequential>
734 class RMSPropOptimizer : public AdagradOptimizer<Scalar,Rank,Sequential> {
735 public:
747  inline RMSPropOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
748  Scalar l2_decay = 1e-1, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :

757  }
758  const Scalar l2_decay;
759 };

764 template<typename Scalar, std::size_t Rank, bool Sequential>
765 class AdadeltaOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
766 public:
776  inline AdadeltaOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1,
777  Scalar decay = 5e-2, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
778  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
779  decay(decay),
780  epsilon(epsilon) {
781  assert(decay >= 0 && decay <= 1);
782  assert(epsilon > 0);
783  }
784  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
785  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
786  pgus_vec = std::vector<ParamGradAndUpdateSqrs>(layers.size());
787  for (unsigned i = 0; i < pgus_vec.size(); ++i) {
788  Layer<Scalar,Rank>& layer = *(layers[i]);
789  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
790  ParamGradAndUpdateSqrs pgus;
791  pgus.params_grad = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
792  pgus.params_update = Matrix<Scalar>::Zero(pgus.params_grad.rows(), pgus.params_grad.cols());
793  pgus_vec[i] = pgus;
794  }
795  }
796 protected:
797  inline void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
798  ParamGradAndUpdateSqrs& pgus = pgus_vec[i];
799  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
800  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
801  pgus.params_grad = (1 - decay) * pgus.params_grad + decay * params_grad.cwiseProduct(params_grad);
802  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
803  (pgus.params_grad.array() + epsilon).sqrt();
804  params += weight_updates;
805  pgus.params_update = (1 - decay) * pgus.params_update + decay * weight_updates.cwiseProduct(weight_updates);
806  }
807  const Scalar decay;
808  const Scalar epsilon;

822 template<typename Scalar, std::size_t Rank, bool Sequential>
823 class AdamOptimizer : public SGDOptimizer<Scalar,Rank,Sequential> {
824 public:
839  inline AdamOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, unsigned batch_size = 1, Scalar learning_rate = 1e-3,
840  Scalar l1_decay = 1e-1, Scalar l2_decay = 1e-3, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
841  SGDOptimizer<Scalar,Rank,Sequential>::SGDOptimizer(loss, batch_size),
842  learning_rate(learning_rate),
843  l1_decay(l1_decay),
844  l2_decay(l2_decay),
845  epsilon(epsilon) {
846  assert(learning_rate > 0);
847  assert(l1_decay >= 0 && l1_decay <= 1);
848  assert(l2_decay >= 0 && l2_decay <= 1);
849  assert(epsilon > 0);
850  }
851  virtual ~AdamOptimizer() = default;
852  inline void fit(NeuralNetwork<Scalar,Rank,Sequential>& net) {
853  std::vector<Layer<Scalar,Rank>*> layers = Optimizer<Scalar,Rank,Sequential>::get_layers(net);
854  pgn_vec = std::vector<ParamGradNorms>(layers.size());
855  for (unsigned i = 0; i < pgn_vec.size(); ++i) {
856  Layer<Scalar,Rank>& layer = *(layers[i]);
857  const Matrix<Scalar>& param_grads = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
858  ParamGradNorms vel;
859  vel.params_grad_l1 = Matrix<Scalar>::Zero(param_grads.rows(), param_grads.cols());
860  vel.params_grad_l2 = Matrix<Scalar>::Zero(vel.params_grad_l1.rows(), vel.params_grad_l1.cols());
861  pgn_vec[i] = vel;
862  }
863  }
864 protected:
865  inline void _update_params(Layer<Scalar,Rank>& layer, unsigned i, unsigned epoch) {
866  ParamGradNorms& grad_norms = pgn_vec[i];
867  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - l1_decay, epoch + 1) + epsilon);
868  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - l2_decay, epoch + 1) + epsilon);
869  Matrix<Scalar>& params = Optimizer<Scalar,Rank,Sequential>::get_params(layer);
870  const Matrix<Scalar>& params_grad = Optimizer<Scalar,Rank,Sequential>::get_params_grad(layer);
871  grad_norms.params_grad_l1 = (1 - l1_decay) * grad_norms.params_grad_l1 + l1_decay * params_grad;
872  grad_norms.params_grad_l2 = (1 - l2_decay) * grad_norms.params_grad_l2 +
873  l2_decay * params_grad.cwiseProduct(params_grad);
874  params -= (learning_rate * (grad_norms.params_grad_l1 * l1_corr).array() /
875  ((grad_norms.params_grad_l2 * l2_corr).array() + epsilon).sqrt()).matrix();
876  }
877  const Scalar learning_rate;
878  const Scalar l1_decay;
879  const Scalar l2_decay;
880  const Scalar epsilon;
885  struct ParamGradNorms {
886  Matrix<Scalar> params_grad_l1;
887  Matrix<Scalar> params_grad_l2;
888  };
889  std::vector<ParamGradNorms> pgn_vec;
890 };

962  grad_norms.params_grad_l2 = (1 - Base::l2_decay) * grad_norms.params_grad_l2 + Base::l2_decay *
963  params_grad.cwiseProduct(params_grad);
964  params -= (Base::learning_rate * (Base::l1_decay * l1_corr * params_grad +
965  (1.0 - Base::l1_decay) * l1_next_corr * grad_norms.params_grad_l1).array() /

1525  assert(input_dims == data_dims.template demote<2>());
1526  int samples = data_dims(0);
1527  int input_seq_length = data_dims(1);
1528  // Calculate the output sequence length and delay based on the input sequence length.
1529  std::pair<std::size_t,std::size_t> output_seq_info = output_seq_size_func((std::size_t) input_seq_length);
1530  int output_seq_length = (int) output_seq_info.first;
1531  int output_seq_delay = (int) output_seq_info.second;
1532  assert(output_seq_length > 0);
1533  if (reversed) {
1534  RankwiseBoolArray reverse;
1535  reverse.fill(false);
1536  reverse[1] = true;
1537  input = input.reverse(reverse);
1538  }
1539  int output_end = output_seq_length + output_seq_delay;
1540  int time_steps = std::max(input_seq_length, output_end);
1541  // If in training mode, unroll the network (unless it has already been unrolled for the same alignment).
1542  if (training && (input_seq_length != this->input_seq_length ||
1543  output_seq_length != this->output_seq_length || output_seq_delay != this->output_seq_delay)) {
1544  if (time_steps > 1) {
1545  // Empty the caches of the main cell to reduce the amount of data to copy.
1546  empty_caches();
1547  // Emptying the caches also clears the cell vector, thus it has to be recreated afterwards.
1548  cells = std::vector<Cell>(time_steps - 1);
1549  // Unroll the network by creating n -1 copies of the main cell;
1550  for (int j = 1; j < time_steps; ++j) {
1551  Cell& cell = cells[j - 1];
1552  cell.state_kernel = KernelPtr<Scalar,Rank>((KernelLayer<Scalar,Rank>*)

1554  cell.state_act = ActivationPtr<Scalar,Rank>((ActivationLayer<Scalar,Rank>*)
1555  Root::clone_with_shared_params(*main_cell.state_act));
1556  // Only copy the kernels and activations that will actually be used.
1557  if (j < input_seq_length)
1558  cell.input_kernel = KernelPtr<Scalar,Rank>((KernelLayer<Scalar,Rank>*)
1559  Root::clone_with_shared_params(*main_cell.input_kernel));
1560  if (j >= output_seq_delay && j < output_end) {
1561  cell.output_kernel = KernelPtr<Scalar,Rank>((KernelLayer<Scalar,Rank>*)
1562  Root::clone_with_shared_params(*main_cell.output_kernel));
1563  cell.output_act = ActivationPtr<Scalar,Rank>((ActivationLayer<Scalar,Rank>*)
1564  Root::clone_with_shared_params(*main_cell.output_act));
1565  }
1566  }
1567  } else

1569  }
1570  // If the network is stateful, retain the state.
1571  if (!Stateful || batch_size == -1) {

1573  dims(0) = samples;
1574  state = Tensor<Scalar,Rank + 1>(dims);
1575  state.setZero();
1576  } else if (samples != batch_size) {
1577  std::array<std::size_t,Rank + 1> offsets;
1578  std::array<std::size_t,Rank + 1> extents = main_cell.input_kernel->get_output_dims().template promote<>();
1579  offsets.fill(0);
1580  extents[0] = samples;

1582  if (samples > batch_size) {
1583  new_state = Tensor<Scalar,Rank + 1>(extents);
1584  new_state.setZero();
1585  extents[0] = batch_size;
1586  new_state.slice(offsets, extents) = state;
1587  } else
1588  new_state = state.slice(offsets, extents);
1589  state = std::move(new_state);
1590  }
1591  RankwiseIntArray input_offsets;
1592  RankwiseIntArray input_extents = data_dims;
1593  RankwiseIntArray output_offsets;
1594  RankwiseIntArray output_extents = output_dims.template promote<2>();
1595  input_offsets.fill(0);
1596  output_offsets.fill(0);
1597  input_extents[1] = 1;
1598  output_extents[0] = samples;
1599  typename Root::Data out;
1600  // If the output is a single time step prediction, there is no need to create an output tensor.
1601  if (output_seq_length > 1) {
1602  output_extents[1] = output_seq_length;
1603  out = typename Root::Data(output_extents);
1604  }
1605  output_extents[1] = 1;
1606  Dimensions<std::size_t,Rank + 1> input_time_step_dims = input_dims.template promote<>();
1607  input_time_step_dims(0) = samples;
1608  for (int i = 0; i < time_steps; ++i) {
1609  // In inference mode, do not unroll the network.
1610  Cell& cell = !training || i == 0 ? main_cell : cells[i - 1];
1611  // Always apply the state kernel.
1612  state = Root::pass_forward(*cell.state_kernel, std::move(state), training);
1613  // If in inference mode, empty the caches after passing the data through each layer.
1614  if (!training)
1615  Root::empty_cache(*cell.state_kernel);
1616  // If there is an input for the time step...
1617  if (i < input_seq_length) {
1618  typename Root::Data in_i_seq;
1619  if (input_seq_length == 1)
1620  in_i_seq = std::move(input);
1621  else {
1622  in_i_seq = input.slice(input_offsets, input_extents);
1623  input_offsets[1] += 1;
1624  }
1625  if (MulInt) {
1626  if (training) {
1627  /* If multiplicative integration is enabled, cache the factors of the multiplication so that
1628  * the function can be differentiated in the backward pass. */
1629  cell.state_kernel_cache = state;
1630  cell.input_kernel_cache = Root::pass_forward(*cell.input_kernel,
1631  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1632  input_time_step_dims), training);
1633  state *= cell.input_kernel_cache;
1634  } else
1635  state *= Root::pass_forward(*cell.input_kernel,
1636  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1637  input_time_step_dims), training);
1638  } else
1639  state += Root::pass_forward(*cell.input_kernel,
1640  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(std::move(in_i_seq),
1641  input_time_step_dims), training);
1642  if (!training)
1643  Root::empty_cache(*cell.input_kernel);
1644  }
1645  state = Root::pass_forward(*cell.state_act, std::move(state), training);
1646  if (!training)
1647  Root::empty_cache(*cell.state_act);
1648  // If there is an output for the time step...
1649  if (i >= output_seq_delay && i < output_end) {
1650  TimeStepData out_i = Root::pass_forward(*cell.output_kernel, state, training);
1651  if (!training)
1652  Root::empty_cache(*cell.output_kernel);
1653  // If the output is a single time step prediction, just return it.
1654  if (output_seq_length == 1)
1655  out = Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1656  Root::pass_forward(*cell.output_act, std::move(out_i), training),
1657  output_extents);
1658  else {
1659  out.slice(output_offsets, output_extents) = Utils<Scalar>::template map_tensor_to_tensor<
1660  Rank + 1,Root::DATA_RANKS>(Root::pass_forward(*cell.output_act, std::move(out_i),
1661  training), output_extents);
1662  output_offsets[1] += 1;
1663  }
1664  if (!training)
1665  Root::empty_cache(*cell.output_act);
1666  }
1667  }
1668  batch_size = samples;
1669  this->input_seq_length = input_seq_length;
1670  this->output_seq_length = output_seq_length;
1671  this->output_seq_delay = output_seq_delay;
1672  return out;
1673  }
1674  inline typename Root::Data backpropagate(typename Root::Data out_grads) {
1675  Utils<Scalar>::template check_dim_validity<Root::DATA_RANKS>(out_grads);
1676  Dimensions<std::size_t,Root::DATA_RANKS> data_dims = Utils<Scalar>::template get_dims<Root::DATA_RANKS>(out_grads);
1677  assert(output_dims == data_dims.template demote<2>() && batch_size == data_dims(0) &&
1678  output_seq_length == data_dims(1));
1679  TimeStepData null_tensor;
1680  RankwiseIntArray output_offsets;
1681  RankwiseIntArray output_extents = data_dims;
1682  RankwiseIntArray input_offsets;
1683  RankwiseIntArray input_extents = input_dims.template promote<2>();
1684  output_offsets.fill(0);
1685  output_offsets[1] = output_seq_length - 1;
1686  input_offsets.fill(0);
1687  input_offsets[1] = input_seq_length - 1;
1688  output_extents[1] = 1;
1689  input_extents[0] = batch_size;
1690  typename Root::Data prev_out_grads;
1691  if (input_seq_length > 1) {
1692  input_extents[1] = input_seq_length;
1693  prev_out_grads = typename Root::Data(input_extents);
1694  }
1695  input_extents[1] = 1;

1697  state_grads.setZero();

1699  out_time_step_dims(0) = batch_size;

1702  for (int i = time_steps - 1; i >= 0; --i) {

1706  typename Root::Data out_grads_seq_i;
1707  if (output_seq_length == 1)
1708  out_grads_seq_i = std::move(out_grads);
1709  else {

1711  output_offsets[1] -= 1;
1712  }

1714  Utils<Scalar>::template map_tensor_to_tensor<Root::DATA_RANKS,Rank + 1>(
1715  std::move(out_grads_seq_i), out_time_step_dims));
1716  Root::empty_cache(*cell.output_act);
1717  state_grads += Root::pass_back(*cell.output_kernel, std::move(out_grads_i));
1718  Root::empty_cache(*cell.output_kernel);
1719  }

1721  state_grads = Root::pass_back(*cell.state_act, std::move(state_grads));
1722  Root::empty_cache(*cell.state_act);
1723  // If there was an input at the time step...
1724  if (i < input_seq_length) {

1726  if (foremost) {

1728  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads);
1729  cell.state_kernel_cache = null_tensor;
1730  } else // Additive integration.
1731  Root::pass_back(*cell.input_kernel, state_grads);

1735  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads), input_extents);
1736  cell.state_kernel_cache = null_tensor;
1737  } else
1738  prev_out_grads = Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1739  Root::pass_back(*cell.input_kernel, state_grads), input_extents);
1740  } else {
1741  if (MulInt) {
1742  prev_out_grads.slice(input_offsets, input_extents) =
1743  Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1744  Root::pass_back(*cell.input_kernel, cell.state_kernel_cache * state_grads), input_extents);
1745  cell.state_kernel_cache = null_tensor;
1746  } else
1747  prev_out_grads.slice(input_offsets, input_extents) =
1748  Utils<Scalar>::template map_tensor_to_tensor<Rank + 1,Root::DATA_RANKS>(
1749  Root::pass_back(*cell.input_kernel, state_grads), input_extents);
1750  input_offsets[1] -= 1;
1751  }
1752  Root::empty_cache(*cell.input_kernel);
1753  }
1754  // Compute the the state kernel's gradient.
1755  if (MulInt) {
1756  state_grads = Root::pass_back(*cell.state_kernel, cell.input_kernel_cache * state_grads);
1757  cell.input_kernel_cache = null_tensor;
1758  } else
1759  state_grads = Root::pass_back(*cell.state_kernel, std::move(state_grads));
1760  Root::empty_cache(*cell.state_kernel);
1761  }
1762  // Roll the network up and accumulate the gradients.
1763  // FIXME Single evaluation.

1765  Matrix<Scalar>& w_params_grad = Root::get_params_grad(*main_cell.state_kernel);
1766  Matrix<Scalar>& v_params_grad = Root::get_params_grad(*main_cell.output_kernel);
1767  Matrix<Scalar>& state_act_params_grad = Root::get_params_grad(*main_cell.state_act);
1768  Matrix<Scalar>& output_act_params_grad = Root::get_params_grad(*main_cell.output_act);
1769  for (int i = 1; i < time_steps; ++i) {
1770  Cell& cell = cells[i - 1];
1771  w_params_grad += Root::get_params_grad(*cell.state_kernel);
1772  state_act_params_grad += Root::get_params_grad(*cell.state_act);
1773  if (i < input_seq_length)
1774  u_params_grad += Root::get_params_grad(*cell.input_kernel);
1775  if (i >= output_seq_delay && i < output_end) {
1776  v_params_grad += Root::get_params_grad(*cell.output_kernel);
1777  output_act_params_grad += Root::get_params_grad(*cell.output_act);
1778  }
1779  }
1780  return prev_out_grads;

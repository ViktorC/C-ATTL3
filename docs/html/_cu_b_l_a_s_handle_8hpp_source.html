78  inline void matrix_mul(Scalar* a, std::size_t a_orig_rows, std::size_t a_orig_cols, bool transpose_a,
79  Scalar* b, std::size_t b_orig_rows, std::size_t b_orig_cols, bool transpose_b,
80  /* out */ Scalar* c) {
81  std::size_t a_rows, a_cols, b_rows, b_cols;
82  if (transpose_a) {
83  a_rows = a_orig_cols;
84  a_cols = a_orig_rows;
85  } else {
86  a_rows = a_orig_rows;
87  a_cols = a_orig_cols;
88  }
89  if (transpose_b) {
90  b_rows = b_orig_cols;
91  b_cols = b_orig_rows;
92  } else {
93  b_rows = b_orig_rows;
94  b_cols = b_orig_cols;
95  }
96  assert(a_cols == b_rows);
97  // Device arrays.
98  Scalar* d_a;
99  Scalar* d_b;
100  Scalar* d_c;
101  // Allocate the memory for the arrays on the device.
102  cudaAssert(cudaMalloc(&d_a, a_rows * a_cols * sizeof(Scalar)));
103  cudaAssert(cudaMalloc(&d_b, b_rows * b_cols * sizeof(Scalar)));
104  cudaAssert(cudaMalloc(&d_c, a_rows * b_cols * sizeof(Scalar)));
105  // Copy the contents of the host arrays to the device arrays.
106  cublasAssert(cublasSetMatrix(a_orig_rows, a_orig_cols, sizeof(Scalar), a, a_orig_rows,
107  d_a, a_orig_rows));
108  cublasAssert(cublasSetMatrix(b_orig_rows, b_orig_cols, sizeof(Scalar), b, b_orig_rows,
109  d_b, b_orig_rows));
110  cublasOperation_t a_op = transpose_a ? CUBLAS_OP_T : CUBLAS_OP_N;
111  cublasOperation_t b_op = transpose_b ? CUBLAS_OP_T : CUBLAS_OP_N;
112  Scalar alpha = 1;
113  Scalar beta = 0;
114  // Resolve the GEMM precision based on the scalar type.
115  GemmRoutine<Scalar> gemm = resolve_gemm_routine<Scalar>();
116  // Perform the matrix multiplication.
117  cublasAssert(gemm(handle, a_op, b_op, a_rows, b_cols, a_cols, &alpha, d_a, a_orig_rows,
118  d_b, b_orig_rows, &beta, d_c, a_rows));
119  /* Copy the contents of the device array holding the results of the matrix
120  * multiplication back to the host. */
121  cublasAssert(cublasGetMatrix(a_rows, b_cols, sizeof(Scalar), d_c, a_rows, c, a_rows));
122  cudaAssert(cudaFree(d_a));
123  cudaAssert(cudaFree(d_b));
124  cudaAssert(cudaFree(d_c));
125  }
126 private:

128  inline CuBLASHandle() :

131  cublasAssert(cublasCreate(&handle));
132  }

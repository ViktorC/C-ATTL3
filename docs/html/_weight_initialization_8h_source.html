52 template<typename Scalar>
53 class GaussianWeightInitialization : public WeightInitialization<Scalar> {
54 public:
55  inline GaussianWeightInitialization(Scalar bias) :
56  bias(bias) { }
57  virtual ~GaussianWeightInitialization() = default;
58  inline virtual void apply(Matrix<Scalar>& weights) const {
59  int rows = weights.rows();
60  int cols = weights.cols();
61  std::default_random_engine gen;
62  std::normal_distribution<Scalar> dist(0, sd(rows - 1, cols));
63  for (int i = 0; i < rows; ++i) {
64  if (i == rows - 1) { // Bias row.
65  weights.row(i).setConstant(bias);
66  } else {
67  for (int j = 0; j < cols; ++j)
68  weights(i,j) = dist(gen);
69  }
70  }
71  }
72 protected:
73  virtual Scalar sd(unsigned fan_ins, unsigned fan_outs) const = 0;
74  const Scalar bias;
75 };

54  inline DataPair<Scalar,3,false> _get_data(const std::string& obs_file,
55  std::ifstream& obs_file_stream, const std::string& obj_file,
56  std::ifstream& obj_file_stream, std::size_t batch_size) {
57  typename Root::Data obs(batch_size, Base::obs_dims(0), Base::obs_dims(1), Base::obs_dims(2));
58  typename Root::Data obj(batch_size, Base::obj_dims(0), Base::obj_dims(1), Base::obj_dims(2));
59  obj.setZero();
60  std::size_t i;
61  for (i = 0; i < batch_size && obs_file_stream.read(obs_buffer, OBS_INSTANCE_LENGTH); ++i) {
62  // Read and set the label.
63  char label;
64  obj_file_stream.read(&label, LABEL_INSTANCE_LENGTH);
65  obj(i,static_cast<std::size_t>(label),0u,0u) = (Scalar) 1;
66  // Set the image.
67  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(obs_buffer);
68  std::size_t buffer_ind = 0;
69  for (std::size_t height = 0; height < Base::obs_dims(0); ++height) {
70  for (std::size_t width = 0; width < Base::obs_dims(1); ++width)
71  obs(i,height,width,0u) = (Scalar) u_buffer[buffer_ind++];
72  }
73  assert(buffer_ind == OBS_INSTANCE_LENGTH);
74  }
75  if (i == batch_size)
76  return std::make_pair(obs, obj);
77  obs_extents[0] = i;
78  obj_extents[0] = i;
79  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
80  }
81  inline std::size_t _skip(std::ifstream& obs_file_stream, std::ifstream& obj_file_stream,
82  std::size_t instances) {
83  // Skip observations.
84  std::streampos curr_obs_pos = obs_file_stream.tellg();
85  obs_file_stream.seekg(0, std::ios::end);
86  std::size_t obs_skip_extent = obs_file_stream.tellg() - curr_obs_pos;
87  obs_file_stream.seekg(curr_obs_pos);
88  obs_file_stream.ignore(instances * OBS_INSTANCE_LENGTH);
89  std::size_t skipped_obs = std::min(instances, obs_skip_extent / OBS_INSTANCE_LENGTH);
90  // Skip labels.
91  std::streampos curr_label_pos = obj_file_stream.tellg();
92  obj_file_stream.seekg(0, std::ios::end);
93  std::size_t label_skip_extent = obj_file_stream.tellg() - curr_label_pos;
94  obj_file_stream.seekg(curr_label_pos);
95  obj_file_stream.ignore(instances * LABEL_INSTANCE_LENGTH);
96  std::size_t skipped_labels = std::min(instances, label_skip_extent / LABEL_INSTANCE_LENGTH);
97  assert(skipped_obs == skipped_labels);
98  return skipped_obs;
99  }
100 private:
101  char obs_buffer[OBS_INSTANCE_LENGTH];
102  RankwiseArray offsets, obs_extents, obj_extents;
103 };

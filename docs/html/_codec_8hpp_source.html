58 template<typename Scalar, PPMFormatType Type = P6>
59 class PPMCodec : public Codec<Scalar,3> {
60  static_assert(Type >= P2 && Type <= P6, "illegal ppm format type argument");
61  static constexpr int MAX_SINGLE_BYTE_VAL = 255;
62  static constexpr int MAX_DOUBLE_BYTE_VAL = 65535;
63  static constexpr int MAX_LINE_LENGTH = 70;
64  static constexpr int MAX_VAL_STRING_LENGTH = 5;
65  static constexpr int BUFFER_SIZE = 3072;
66  static constexpr bool GRAY_SCALE = Type == P2 || Type == P5;
67  static constexpr bool BINARY = Type == P5 || Type == P6;
68 public:
69  PPMCodec() : type(resolve_type_string()) { }
70  inline void encode(const Tensor<Scalar,3>& data, const std::string& file_path) const {
71  assert(data.dimension(0) > 0 && data.dimension(1) > 0 &&
72  data.dimension(2) == (GRAY_SCALE ? 1 : 3));
73  std::ofstream file_stream(file_path, BINARY ? std::ios::binary : std::ios::out);
74  assert(file_stream.is_open());
75  Tensor<Scalar,0> max_tensor = data.maximum();
76  const int max_val = (int) max_tensor(0u);
77  assert(max_val >= 0 && max_val <= MAX_DOUBLE_BYTE_VAL);
78  const bool single_byte = max_val <= MAX_SINGLE_BYTE_VAL;
79  std::string header = type + "\n" + std::to_string(data.dimension(1)) + " " +
80  std::to_string(data.dimension(0)) + "\n" + std::to_string(max_val) + "\n";
81  file_stream.write(header.c_str(), header.length());
82  int ind = 0;
83  // For non-binary formats.
84  int last_line_break = 0;
85  unsigned char buffer[+BUFFER_SIZE];
86  for (std::size_t i = 0; i < data.dimension(0); ++i) {
87  for (std::size_t j = 0; j < data.dimension(1); ++j) {
88  for (std::size_t k = 0; k < data.dimension(2); ++k) {
89  int val = data(i,j,k);
90  assert(val >= 0);
91  if (BINARY) {
92  if (ind == +BUFFER_SIZE) {
93  file_stream.write(reinterpret_cast<char*>(buffer), ind);
94  ind = 0;
95  }
96  // The buffer size is divisible by 2; no need to worry about buffer overflow.
97  if (!single_byte)
98  buffer[ind++] = (unsigned char) (val >> 8);
99  buffer[ind++] = (unsigned char) val;
100  } else {
101  if (ind >= +BUFFER_SIZE - (MAX_VAL_STRING_LENGTH + 1)) {
102  file_stream.write(reinterpret_cast<char*>(buffer), ind);
103  last_line_break -= ind;
104  ind = 0;
105  }
106  std::string val_string = std::to_string(val);
107  for (int l = 0; l < val_string.length(); ++l)
108  buffer[ind++] = *(val_string.c_str() + l);
109  if ((ind + 1 - last_line_break >= (MAX_LINE_LENGTH - MAX_VAL_STRING_LENGTH))) {
110  buffer[ind++] = '\n';
111  last_line_break = ind;
112  } else
113  buffer[ind++] = ' ';
114  }
115  }
116  }
117  }
118  if (ind != 0)
119  file_stream.write(reinterpret_cast<char*>(buffer), ind);
120  }
121  inline Tensor<Scalar,3> decode(const std::string& file_path) const {
122  std::ifstream file_stream(file_path, BINARY ? std::ios::binary : std::ios::in);
123  assert(file_stream.is_open());
124  std::string format_type, dims, max_val_string;
125  int width, height, max_val;
126  std::getline(file_stream, format_type);
127  assert(type == format_type);
128  std::getline(file_stream, dims);
129  std::istringstream dims_stream(dims);
130  dims_stream >> width;
131  dims_stream >> height;
132  assert(width > 0 && height > 0);
133  std::getline(file_stream, max_val_string);
134  std::istringstream max_val_stream(max_val_string);
135  max_val_stream >> max_val;
136  assert(max_val >= 0 && max_val <= MAX_DOUBLE_BYTE_VAL);
137  const std::size_t depth = GRAY_SCALE ? 1u : 3u;
138  const int total_values = height * width * depth;
139  Tensor<Scalar,3> data((std::size_t) height, (std::size_t) width, depth);
140  unsigned char buffer[+BUFFER_SIZE];
141  int ind = 0;
142  if (BINARY) {
143  const bool single_byte = max_val <= MAX_SINGLE_BYTE_VAL;
144  int values_in_buffer = std::min(+BUFFER_SIZE, (2 - single_byte) * total_values);
145  int read_values = values_in_buffer;
146  file_stream.read(reinterpret_cast<char*>(&buffer), values_in_buffer);
147  assert(file_stream.gcount() == values_in_buffer);
148  for (std::size_t i = 0; i < height; ++i) {
149  for (std::size_t j = 0; j < width; ++j) {
150  for (std::size_t k = 0; k < depth; ++k) {
151  if (ind == values_in_buffer) {
152  values_in_buffer = std::min(+BUFFER_SIZE,
153  (2 - single_byte) * total_values - read_values);
154  file_stream.read(reinterpret_cast<char*>(&buffer), values_in_buffer);
155  assert(file_stream.gcount() == values_in_buffer);
156  read_values += values_in_buffer;
157  ind = 0;
158  }
159  unsigned val;
160  if (single_byte)
161  val = (unsigned) buffer[ind++];
162  else { // No buffer overflow possible due to the even buffer size.
163  val = (unsigned) buffer[ind++];
164  val |= ((unsigned) buffer[ind++]) << 8;
165  }
166  data(i,j,k) = (Scalar) val;
167  }
168  }
169  }
170  } else {
171  file_stream.read(reinterpret_cast<char*>(&buffer), +BUFFER_SIZE);
172  for (std::size_t i = 0; i < height; ++i) {
173  for (std::size_t j = 0; j < width; ++j) {
174  for (std::size_t k = 0; k < depth; ++k) {
175  std::vector<unsigned char> chars;
176  bool found_num = false;
177  for (;;) {
178  if (ind == +BUFFER_SIZE) {
179  file_stream.read(reinterpret_cast<char*>(&buffer), +BUFFER_SIZE);
180  ind = 0;
181  } else if (ind == file_stream.gcount())
182  break;
183  unsigned char curr_char = buffer[ind++];
184  if (curr_char >= '0' && curr_char <= '9') {
185  chars.push_back(curr_char);
186  found_num = true;
187  } else if (found_num)
188  break;
189  }
190  assert(found_num);
191  std::string val_string(chars.begin(), chars.end());
192  data(i,j,k) = (Scalar) std::stoi(val_string);
193  }
194  }
195  }
196  }
197  return data;
198  }
199 private:
203  inline static std::string resolve_type_string() {

205  case P2:
206  return "P2";
207  case P3:
208  return "P3";
209  case P5:
210  return "P5";
211  case P6:
212  return "P6";
213  default:
214  return "";
215  }
216  }
217  const std::string type;
218 };

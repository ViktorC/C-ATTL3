18 template<typename Scalar, std::size_t Rank>
19 class SimpleActivationGPULayer : public ActivationGPULayer<Scalar,Rank> {
20  typedef Layer<Scalar,Rank> Root;
21  typedef ActivationGPULayer<Scalar,Rank> Base;
22 public:
23  inline SimpleActivationGPULayer(const typename Root::Dims& dims, cudnnActivationMode_t act_mode,
24  Scalar coeff = 0) :
25  Base(dims),
26  coeff(coeff),
27  alpha(1),
28  beta(0),
29  act_desc() {
30  cudnnAssert(cudnnCreateActivationDescriptor(&act_desc));
31  cudnnAssert(cudnnSetActivationDescriptor(act_desc, act_mode, CuDNNTensor<Scalar>::NAN_PROP, coeff));
32  }
33  inline virtual ~SimpleActivationGPULayer() {
34  cudnnAssert(cudnnDestroyActivationDescriptor(act_desc));
35  }
36  inline void empty_cache() {
37  in_cache = CuDNNTensor<Scalar>();
38  out_cache = CuDNNTensor<Scalar>();
39  }
40  inline CuDNNTensor<Scalar> pass_forward(CuDNNTensor<Scalar> in, bool training) {
41  assert(in.height() == Base::gpu_dims(0) && in.width() == Base::gpu_dims(1) &&
42  in.channels() == Base::gpu_dims(2));
43  assert(in.samples() > 0);
44  in_cache = std::move(in);
45  CuDNNTensor<Scalar> out(in_cache.samples(), in_cache.height(), in_cache.width(),
46  in_cache.channels());
47  cudnnAssert(cudnnActivationForward(CuDNNHandle::get_instance(), act_desc, &alpha, in_cache.desc(),
48  in_cache.data(), &beta, out.desc(), out.data()));
49  if (training)
50  out_cache = out;
51  return out;
52  }
53  inline CuDNNTensor<Scalar> pass_back(CuDNNTensor<Scalar> out_grad) {
54  assert(out_grad.height() == Base::gpu_dims(0) && out_grad.width() == Base::gpu_dims(1) &&
55  out_grad.channels() == Base::gpu_dims(2));
56  assert(out_grad.samples() == in_cache.samples());
57  if (Base::is_input_layer())
58  return CuDNNTensor<Scalar>();
59  cudnnAssert(cudnnActivationBackward(CuDNNHandle::get_instance(), act_desc, &alpha, out_cache.desc(),
60  out_cache.data(), out_grad.desc(), out_grad.data(), in_cache.desc(), in_cache.data(), &beta,
61  out_grad.desc(), out_grad.data()));
62  return out_grad;
63  }
64 private:
65  const Scalar coeff, alpha, beta;
66  cudnnActivationDescriptor_t act_desc;
67  CuDNNTensor<Scalar> in_cache, out_cache;
68 };

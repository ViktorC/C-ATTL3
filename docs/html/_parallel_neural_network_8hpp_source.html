50  inline ParallelNeuralNetwork(std::vector<Lane>&& lanes, bool foremost = true) :
51  lanes(std::move(lanes)),
52  foremost(foremost),
53  outputs(this->lanes.size()) {
54  assert(this->lanes.size() > 0 && "lanes must contain at least 1 element");
55  assert(this->lanes[0] != nullptr && "lanes contains null pointers");
56  Base& first_lane = *this->lanes[0];
57  const typename Base::Dims& input_dims = first_lane.get_input_dims();
58  typename Base::Dims output_dims = first_lane.get_output_dims();
59  for (std::size_t i = 1; i < this->lanes.size(); ++i) {
60  assert(this->lanes[i] != nullptr && "lanes contains null pointers");
61  Base& lane = *this->lanes[i];
62  assert(input_dims == lane.get_input_dims());
63  const typename Base::Dims& lane_output_dims = lane.get_output_dims();
64  if (MergeType == PARALLEL_CONCAT_HI_RANK || MergeType == PARALLEL_CONCAT_LO_RANK) {
65  if (MergeType == PARALLEL_CONCAT_HI_RANK) {
66  for (std::size_t i = 0; i < +CONCAT_RANK; ++i)
67  assert(output_dims(i) == lane_output_dims(i));
68  } else {
69  for (std::size_t i = Rank - 1; i > +CONCAT_RANK; --i)
70  assert(output_dims(i) == lane_output_dims(i));
71  }
72  output_dims(+CONCAT_RANK) += lane_output_dims(+CONCAT_RANK);
73  } else
74  assert(output_dims == lane_output_dims);
75  }
76  set_foremost(foremost);
77  this->input_dims = first_lane.get_input_dims();
78  this->output_dims = output_dims;
79  }

142  inline typename Base::Data propagate(typename Base::Data input, bool training) {
143  assert(input_dims == (Dimensions<std::size_t,Base::DATA_RANK>(input.dimensions()).template demote<>()));
144  std::size_t rows = input.dimension(0);
145  int pthread_state;
146  typename Base::Data out;
147  std::size_t lane_num = lanes.size();
148  std::size_t helper_thread_num = lane_num - 1;
149  pthread_t threads[helper_thread_num];
150  pthread_attr_t attr;
151  if (helper_thread_num > 0) {
152  pthread_state = pthread_attr_init(&attr);
153  assert(!pthread_state);
154  pthread_state = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
155  assert(!pthread_state);
156  }
157  PropArgs args_arr[lane_num];
158  for (int i = helper_thread_num; i >= 0; --i) {
159  PropArgs args;
160  args.obj = this;
161  args.lane_id = i;
162  args.training = training;
163  args.in = &input;
164  args_arr[i] = args;
165  // Leave the first lane to the main thread.
166  if (i == 0)
167  propagate(&args_arr[i]);
168  else {
169  pthread_state = pthread_create(&threads[i - 1], &attr, propagate, &args_arr[i]);
170  assert(!pthread_state);
171  }
172  }
173  for (std::size_t i = 0; i < lane_num; ++i) {
174  if (i == 0) {
175  out = std::move(args_arr[i].out);
176  if (MergeType == PARALLEL_MUL)
177  outputs[i] = out;
178  } else {
179  pthread_state = pthread_join(threads[i - 1], nullptr);
180  assert(!pthread_state);
181  if (MergeType == PARALLEL_SUM)
182  out += args_arr[i].out;
183  else if (MergeType == PARALLEL_MUL) {
184  outputs[i] = std::move(args_arr[i].out);
185  out *= outputs[i];
186  } else {
187  // Must be evaluated first due to the dimension difference.
188  typename Base::Data concat = out.concatenate(std::move(args_arr[i].out), +CONCAT_BATCH_RANK);
189  out = std::move(concat);
190  }
191  }
192  }
193  if (helper_thread_num > 0) {
194  pthread_state = pthread_attr_destroy(&attr);
195  assert(!pthread_state);
196  }
197  return out;
198  }
199  inline typename Base::Data backpropagate(typename Base::Data out_grad) {
200  assert(output_dims == (Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()));
201  typename Base::Data prev_out_grad;
202  if (foremost)
203  prev_out_grad = typename Base::Data();
204  else {
205  RankwiseArray dims = input_dims.template promote<>();
206  dims[0] = out_grad.dimension(0);
207  prev_out_grad = typename Base::Data(dims);
208  prev_out_grad.setZero();
209  }
210  int pthread_state;
211  std::size_t lane_num = lanes.size();
212  std::size_t helper_thread_num = lane_num - 1;
213  pthread_t threads[helper_thread_num];
214  pthread_attr_t attr;
215  if (helper_thread_num > 0) {
216  pthread_state = pthread_attr_init(&attr);
217  assert(!pthread_state);
218  pthread_state = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
219  assert(!pthread_state);
220  }
221  BackpropArgs args_arr[lane_num];
222  int concat_rank_offset = out_grad.dimension(+CONCAT_BATCH_RANK);
223  for (int i = helper_thread_num; i >= 0; --i) {
224  concat_rank_offset -= lanes[i]->get_output_dims()(+CONCAT_RANK);
225  BackpropArgs args;
226  args.obj = this;
227  args.lane_id = i;
228  args.concat_rank_offset = concat_rank_offset;
229  args.out_grad = &out_grad;
230  args_arr[i] = args;
231  // Leave the first lane to the main thread.
232  if (i == 0)
233  backpropagate(&args_arr[i]);
234  else {
235  pthread_state = pthread_create(&threads[i - 1], &attr, backpropagate, &args_arr[i]);
236  assert(!pthread_state);
237  }
238  }
239  for (std::size_t i = 0; i < lanes.size(); ++i) {
240  if (i != 0) {
241  pthread_state = pthread_join(threads[i - 1], nullptr);
242  assert(!pthread_state);
243  }
244  if (!foremost)
245  prev_out_grad += args_arr[i].prev_out_grad;
246  }
247  if (helper_thread_num > 0) {
248  pthread_state = pthread_attr_destroy(&attr);
249  assert(!pthread_state);
250  }
251  return prev_out_grad;
252  }

286  inline static void* backpropagate(void* args_ptr) {
287  BackpropArgs& args = *((BackpropArgs*) args_ptr);
288  Base& lane = *args.obj->lanes[args.lane_id];
289  if (MergeType == PARALLEL_SUM)
290  args.prev_out_grad = lane.backpropagate(*args.out_grad);
291  else if (MergeType == PARALLEL_MUL) {
292  typename Base::Data out_grad = *args.out_grad;
293  for (std::size_t i = 0; i < args.obj->lanes.size(); ++i) {
294  if (i != (std::size_t) args.lane_id)
295  out_grad *= args.obj->outputs[i];
296  }
297  args.prev_out_grad = lane.backpropagate(std::move(out_grad));
298  } else {
299  RankwiseArray offsets;
300  RankwiseArray extents = lane.get_output_dims().template promote<>();
301  offsets.fill(0);
302  offsets[+CONCAT_BATCH_RANK] = args.concat_rank_offset;
303  extents[0] = args.out_grad->dimension(0);
304  typename Base::Data out_grad_slice = args.out_grad->slice(offsets, extents);
305  args.prev_out_grad = lane.backpropagate(std::move(out_grad_slice));
306  }
307  return nullptr;
308  }

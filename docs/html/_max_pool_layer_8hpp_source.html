22 template<typename Scalar, std::size_t Rank>
23 class MaxPoolLayerBase : public PoolLayer<Scalar,Rank> {
24  typedef Layer<Scalar,Rank> Root;
25  typedef PoolLayer<Scalar,Rank> Base;
26 public:
27  inline void empty_cache() {
28  max_inds = std::vector<std::vector<unsigned>>(0);
29  }
30 protected:
31  inline MaxPoolLayerBase(const typename Root::Dims& input_dims, std::size_t receptor_height,
32  std::size_t receptor_width, std::size_t vertical_stride, std::size_t horizontal_stride) :
33  Base::PoolLayer(input_dims, receptor_height, receptor_width, vertical_stride,
34  horizontal_stride) { }
35  inline void _init_cache() {
36  max_inds = std::vector<std::vector<unsigned>>(Base::ext_output_dims(0) * Base::ext_output_dims(1));
37  }
38  inline Tensor<Scalar,4> _reduce(const Tensor<Scalar,4>& patch, std::size_t patch_ind) {
39  std::size_t rows = patch.dimension(0);
40  std::size_t depth = patch.dimension(3);
41  std::vector<unsigned> inds(rows * depth);
42  Tensor<Scalar,4> reduced_patch(rows, 1u, 1u, depth);
43  for (std::size_t i = 0; i < depth; ++i) {
44  for (std::size_t j = 0; j < rows; ++j) {
45  Scalar max = NumericUtils<Scalar>::MIN;
46  unsigned max_height = 0;
47  unsigned max_width = 0;
48  for (std::size_t k = 0; k < Base::receptor_width; ++k) {
49  for (std::size_t l = 0; l < Base::receptor_height; ++l) {
50  Scalar val = patch(j,l,k,i);
51  if (val > max) {
52  max = val;
53  max_height = l;
54  max_width = k;
55  }
56  }
57  }
58  inds[i * rows + j] = max_width * Base::receptor_height + max_height;
59  reduced_patch(j,0u,0u,i) = max;
60  }
61  }
62  max_inds[patch_ind] = inds;
63  return reduced_patch;
64  }
65  inline Tensor<Scalar,4> _d_reduce(const Tensor<Scalar,4>& grad, std::size_t patch_ind) {
66  std::size_t rows = grad.dimension(0);
67  std::size_t depth = grad.dimension(3);
68  Tensor<Scalar,4> patch(rows, Base::receptor_height, Base::receptor_width, depth);
69  patch.setZero();
70  std::vector<unsigned>& inds = max_inds[patch_ind];
71  for (std::size_t i = 0; i < depth; ++i) {
72  for (std::size_t j = 0; j < rows; ++j) {
73  unsigned max_ind = inds[i * rows + j];
74  unsigned max_height = max_ind % Base::receptor_height;
75  unsigned max_width = max_ind / Base::receptor_height;
76  patch(j,max_height,max_width,i) = grad(j,0u,0u,i);
77  }
78  }
79  return patch;
80  }
81 private:
82  std::vector<std::vector<unsigned>> max_inds;
83 };

90 template<typename Scalar, std::size_t Rank = 3>
91 class MaxPoolLayer : public MaxPoolLayerBase<Scalar,Rank> {
92  typedef Layer<Scalar,3> Root;
93  typedef PoolLayer<Scalar,3> PoolBase;
94  typedef MaxPoolLayerBase<Scalar,3> MaxPoolBase;
95 public:
105  inline MaxPoolLayer(const typename Root::Dims& input_dims, std::size_t receptor_height = 2,
106  std::size_t receptor_width = 2, std::size_t vertical_stride = 2, std::size_t horizontal_stride = 2) :
107  MaxPoolBase::MaxPoolLayerBase(input_dims, receptor_height, receptor_width, vertical_stride,
108  horizontal_stride) { }
109  inline Root* clone() const {
110  return new MaxPoolLayer(*this);
111  }
112  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
113  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == PoolBase::input_dims);
114  assert(in.dimension(0) > 0);
115  batch_size = in.dimension(0);
116  return PoolBase::_pass_forward(std::move(in), training);
117  }
118  inline typename Root::Data pass_back(typename Root::Data out_grad) {
119  assert((Dimensions<std::size_t,4>(out_grad.dimensions()).template demote<>()) == PoolBase::output_dims);
120  assert(out_grad.dimension(0) > 0 && batch_size == out_grad.dimension(0));
121  if (PoolBase::is_input_layer())
122  return typename Root::Data();
123  return PoolBase::_pass_back(std::move(out_grad));
124  }
125 private:
126  std::size_t batch_size;
127 };

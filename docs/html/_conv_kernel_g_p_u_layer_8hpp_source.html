83  inline CuDNNTensor<Scalar> pass_forward(CuDNNTensor<Scalar> in, bool training) {
84  assert(in.height() == GPUBase::gpu_input_dims(0) && in.width() == GPUBase::gpu_input_dims(1) &&
85  in.channels() == GPUBase::gpu_input_dims(2));
86  assert(in.samples() > 0);
87  StandardGPUParameters<Scalar>& weights = static_cast<StandardGPUParameters<Scalar>&>(*Base::weights);
88  StandardGPUParameters<Scalar>& bias = static_cast<StandardGPUParameters<Scalar>&>(*Base::bias);
89  in_cache = std::move(in);
90  CuDNNTensor<Scalar> out(in_cache.samples(), GPUBase::gpu_output_dims(0), GPUBase::gpu_output_dims(1),
91  GPUBase::gpu_output_dims(2));
92  // Create and set up the convolution descriptor.
93  cudnnConvolutionDescriptor_t conv_desc;
94  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
95  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
96  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
97  CuDNNTensor<Scalar>::DATA_TYPE));
98  // Have cuDNN find the most performant algorithm given the convolution parameters.
99  cudnnConvolutionFwdAlgo_t conv_algo;
100  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(CuDNNHandle::get_instance(), in_cache.desc(),
101  weights.get_gpu_values().filter_desc(), conv_desc, out.desc(), CUDNN_CONVOLUTION_FWD_PREFER_FASTEST,
102  0, &conv_algo));
103  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
104  * the convolution parameters. */
105  std::size_t workspace_size;
106  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(CuDNNHandle::get_instance(), in_cache.desc(),
107  weights.get_gpu_values().filter_desc(), conv_desc, out.desc(), conv_algo, &workspace_size));
108  // Allocate the memory for the workspace required for the convolution.
109  CUDAArray<Scalar> workspace(static_cast<std::size_t>(ceil(static_cast<Scalar>(workspace_size) /
110  sizeof(Scalar))));
111  // Perform the convolution.
112  cudnnAssert(cudnnConvolutionForward(CuDNNHandle::get_instance(), &alpha, in_cache.desc(), in_cache.data(),
113  weights.get_gpu_values().filter_desc(), weights.get_values().data(), conv_desc, conv_algo,
114  workspace.data(), workspace_size, &beta, out.desc(), out.data()));
115  // Free the convolution resources.
116  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
117  // Apply the bias to the output tensor.
118  out += bias.get_gpu_values();
119  return out;
120  }
121  inline CuDNNTensor<Scalar> pass_back(CuDNNTensor<Scalar> out_grad) {
122  assert(out_grad.height() == GPUBase::gpu_output_dims(0) && out_grad.width() == GPUBase::gpu_output_dims(1) &&
123  out_grad.channels() == GPUBase::gpu_output_dims(2));
124  assert(out_grad.samples() == in_cache.samples());
125  StandardGPUParameters<Scalar>& weights = static_cast<StandardGPUParameters<Scalar>&>(*Base::weights);
126  StandardGPUParameters<Scalar>& bias = static_cast<StandardGPUParameters<Scalar>&>(*Base::bias);
127  CuDNNTensor<Scalar> weights_grad(weights.samples(), weights.height(), weights.width(), weights.channels());
128  CuDNNTensor<Scalar> bias_grad(bias.samples(), bias.height(), bias.width(), bias.channels());
129  // Create and set up the backward convolution descriptor.
130  cudnnConvolutionDescriptor_t dconv_desc;
131  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
132  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding,
133  vertical_stride, horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
134  CuDNNTensor<Scalar>::DATA_TYPE));
135  // Have cuDNN find the most performant algorithm given the convolution parameters.
136  cudnnConvolutionBwdFilterAlgo_t dconv_filter_algo;
137  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(CuDNNHandle::get_instance(), in_cache.desc(),
138  out_grad.desc(), dconv_desc, weights_grad.filter_desc(), CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST,
139  0, &dconv_filter_algo));
140  /* Have cuDNN compute the data_workspace memory required for the selected backward convolution algorithms given
141  * the convolution parameters. */
142  std::size_t filter_workspace_size;
143  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(CuDNNHandle::get_instance(), in_cache.desc(),
144  out_grad.desc(), dconv_desc, weights_grad.filter_desc(), dconv_filter_algo, &filter_workspace_size));
145  // Allocate the memory required for the backwards filter convolution on the device.
146  CUDAArray<Scalar> filter_workspace(static_cast<std::size_t>(ceil(static_cast<Scalar>(filter_workspace_size) /
147  sizeof(Scalar))));
148  // Perform the backwards filter convolution.
149  cudnnAssert(cudnnConvolutionBackwardFilter(CuDNNHandle::get_instance(), &alpha, in_cache.desc(),
150  in_cache.data(), out_grad.desc(), out_grad.data(), dconv_desc, dconv_filter_algo,
151  filter_workspace.data(), filter_workspace_size, &beta, weights_grad.filter_desc(),
152  weights_grad.data()));
153  // Free up resources.
154  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
155  // Perform the backwards bias convolution.
156  cudnnAssert(cudnnConvolutionBackwardBias(CuDNNHandle::get_instance(), &alpha, out_grad.desc(),
157  out_grad.data(), &beta, bias_grad.desc(), bias_grad.data()));
158  weights.accumulate_grad(weights_grad);
159  bias.accumulate_grad(bias_grad);
160  if (Base::is_input_layer())
161  return CuDNNTensor<Scalar>();
162  CuDNNTensor<Scalar> prev_out_grad(out_grad.samples(), GPUBase::gpu_input_dims(0), GPUBase::gpu_input_dims(1),
163  GPUBase::gpu_input_dims(2));
164  // Get the backward data convolution algorithm.
165  cudnnConvolutionBwdDataAlgo_t dconv_data_algo;
166  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(CuDNNHandle::get_instance(),
167  weights.get_gpu_values().filter_desc(), out_grad.desc(), dconv_desc, prev_out_grad.desc(),
168  CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
169  // Calculate the workspace size needed.
170  std::size_t data_workspace_size;
171  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(CuDNNHandle::get_instance(),
172  weights.get_gpu_values().filter_desc(), out_grad.desc(), dconv_desc, prev_out_grad.desc(),
173  dconv_data_algo, &data_workspace_size));
174  // Allocate the memory required for the backwards data convolution on the device.
175  CUDAArray<Scalar> data_workspace(static_cast<std::size_t>(ceil(static_cast<Scalar>(data_workspace_size) /
176  sizeof(Scalar))));
177  // Perform the backwards data convolution.
178  cudnnAssert(cudnnConvolutionBackwardData(CuDNNHandle::get_instance(), &alpha,
179  weights.get_gpu_values().filter_desc(), weights.get_gpu_values().data(), out_grad.desc(),
180  out_grad.data(), dconv_desc, dconv_data_algo, data_workspace.data(), data_workspace_size, &beta,
181  prev_out_grad.desc(), prev_out_grad.data()));
182  return prev_out_grad;
183  }

185  inline static Dimensions<std::size_t,3> calculate_output_dims(const Dimensions<std::size_t,3>& input_dims,
186  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width,
187  std::size_t vertical_padding, std::size_t horizontal_padding, std::size_t vertical_stride,
188  std::size_t horizontal_stride, std::size_t vertical_dilation, std::size_t horizontal_dilation) {
189  // Create and set the input tensor descriptor.
190  cudnnTensorDescriptor_t input_desc;
191  CuDNNTensor<Scalar>::create_tensor_descriptor(input_desc, 1, input_dims(0), input_dims(1), input_dims(2));
192  // Create and set up the filter descriptor.
193  cudnnFilterDescriptor_t filter_desc;
194  CuDNNTensor<Scalar>::create_filter_descriptor(filter_desc, filters, receptor_height,
195  receptor_width, input_dims(2));
196  // Create and set up the convolution descriptor.
197  cudnnConvolutionDescriptor_t conv_desc;
198  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
199  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding,
200  vertical_stride, horizontal_stride, vertical_dilation, horizontal_dilation,
201  CUDNN_CROSS_CORRELATION, CuDNNTensor<Scalar>::DATA_TYPE));
202  // Compute the dimensions.
203  int n, h, w, c;
204  cudnnAssert(cudnnGetConvolution2dForwardOutputDim(conv_desc, input_desc, filter_desc, &n, &c, &h, &w));
205  // Free the resources.
206  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
207  CuDNNTensor<Scalar>::destroy_tensor_descriptor(input_desc);
208  CuDNNTensor<Scalar>::destroy_filter_descriptor(filter_desc);
209  return { (std::size_t) h, (std::size_t) w, (std::size_t) c };
210  }
211  inline static Dimensions<std::size_t,Rank> calculate_adjusted_output_dims(
212  const Dimensions<std::size_t,Rank>& input_dims, std::size_t filters, std::size_t receptor_height,
213  std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
214  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
215  std::size_t horizontal_dilation) {
216  auto ext_input_dims = input_dims.template extend<3 - Rank>();
217  auto ext_output_dims = calculate_output_dims(ext_input_dims, filters, receptor_height, receptor_width,
218  vertical_padding, horizontal_padding, vertical_stride, horizontal_stride, vertical_dilation,
219  horizontal_dilation);
220  ext_output_dims(2) /= filters;
221  ext_output_dims(Rank - 1) *= filters;
222  return ext_output_dims.template contract<3 - Rank>();
223  }
224  const std::size_t filters, receptor_height, receptor_width, vertical_padding, horizontal_padding,
225  vertical_stride, horizontal_stride, vertical_dilation, horizontal_dilation;
226  const Scalar alpha, beta;
227  CuDNNTensor<Scalar> in_cache;
228 };

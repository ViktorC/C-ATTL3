99 template<typename Scalar, std::size_t Rank>
100 class UniversalLoss<Scalar,Rank,true> : public Loss<Scalar,Rank,true> {
101  typedef Loss<Scalar,Rank,true> Base;
102 protected:
103  typedef std::array<std::size_t,Base::DATA_RANK> RankwiseArray;
112  virtual ColVector<Scalar> _function(typename Base::Data out, typename Base::Data obj) const = 0;
121  virtual typename Base::Data _d_function(typename Base::Data out, typename Base::Data obj,
122  const RankwiseArray& grad_dims) const = 0;
123 public:
124  virtual ~UniversalLoss() = default;
125  inline ColVector<Scalar> function(typename Base::Data out, typename Base::Data obj) const {
126  assert(out.dimensions() == obj.dimensions());
127  int time_steps = out.dimension(1);
128  if (time_steps == 1)
129  return _function(std::move(out), std::move(obj));
130  RankwiseArray offsets;
131  RankwiseArray extents = out.dimensions();
132  offsets.fill(0);
133  extents[1] = 1;
134  ColVector<Scalar> loss = ColVector<Scalar>::Zero(out.dimension(0), 1);
135  for (int i = 0; i < time_steps; ++i) {
136  offsets[1] = i;
137  typename Base::Data out_i = out.slice(offsets, extents);
138  typename Base::Data obj_i = obj.slice(offsets, extents);
139  loss += _function(std::move(out_i), std::move(obj_i));
140  }
141  return loss;
142  }
143  inline typename Base::Data d_function(const typename Base::Data out, const typename Base::Data obj) const {
144  assert(out.dimensions() == obj.dimensions());
145  int time_steps = out.dimension(1);
146  if (time_steps == 1)
147  return _d_function(std::move(out), std::move(obj), out.dimensions());
148  RankwiseArray offsets;
149  RankwiseArray extents = out.dimensions();
150  offsets.fill(0);
151  typename Base::Data grads(extents);
152  extents[1] = 1;
153  grads.setZero();
154  for (int i = 0; i < time_steps; ++i) {
155  offsets[1] = i;
156  typename Base::Data out_i = out.slice(offsets, extents);
157  typename Base::Data obj_i = obj.slice(offsets, extents);
158  grads.slice(offsets, extents) += _d_function(std::move(out_i), std::move(obj_i), extents);
159  }
160  return grads;
161  }
162 };

167 template<typename Scalar, std::size_t Rank, bool Sequential>
168 class AbsoluteLoss : public UniversalLoss<Scalar,Rank,Sequential> {
169  typedef Loss<Scalar,Rank,Sequential> Root;
170  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
171 protected:

208 template<typename Scalar, std::size_t Rank, bool Sequential, bool Squared = false>
209 class HingeLoss : public UniversalLoss<Scalar,Rank,Sequential> {
210  typedef Loss<Scalar,Rank,Sequential> Root;
211  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
212 protected:
213  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
214  std::size_t rows = out.dimension(0);
215  std::size_t cols = out.size() / rows;
216  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
217  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
218  ColVector<Scalar> loss(out_mat.rows());
219  for (int i = 0; i < obj_mat.rows(); ++i) {
220  unsigned ones = 0;
221  int correct_class_ind = -1;
222  for (int j = 0; j < obj_mat.cols(); ++j) {
223  Scalar obj_ij = obj_mat(i,j);
224  assert((internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
225  internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
226  if (internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
227  ones++;
228  correct_class_ind = j;
229  }
230  }
231  assert(ones == 1);
232  Scalar loss_i = 0;
233  Scalar correct_class_score = out_mat(i,correct_class_ind);
234  for (int j = 0; j < obj_mat.cols(); ++j) {
235  if (j == correct_class_ind)
236  continue;
237  Scalar loss_ij = std::max(.0, out_mat(i,j) - correct_class_score + 1);
238  loss_i += Squared ? loss_ij * loss_ij : loss_ij;
239  }
240  loss(i) = loss_i;
241  }
242  return loss;
243  }
244  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
245  const typename Base::RankwiseArray& grad_dims) const {
246  std::size_t rows = out.dimension(0);
247  std::size_t cols = out.size() / rows;
248  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
249  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
250  Matrix<Scalar> out_grads(out_mat.rows(), out_mat.cols());
251  for (int i = 0; i < out_mat.rows(); ++i) {
252  unsigned ones = 0;
253  int correct_class_ind = -1;
254  for (int j = 0; j < out_mat.cols(); ++j) {
255  Scalar obj_ij = obj_mat(i,j);
256  assert((internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
257  internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
258  if (internal::Utils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
259  ones++;
260  correct_class_ind = j;
261  }
262  }
263  assert(ones == 1);
264  Scalar total_out_grad = 0;
265  Scalar correct_class_score = out_mat(i,correct_class_ind);
266  for (int j = 0; j < out_mat.cols(); ++j) {
267  if (j == correct_class_ind)
268  continue;
269  Scalar out_ij = out_mat(i,j);
270  Scalar margin = out_ij - correct_class_score + 1;
271  if (internal::Utils<Scalar>::decidedly_greater(margin, (Scalar) 0)) {
272  Scalar out_grad = Squared ? 2 * (out_ij - correct_class_score) : 1;
273  total_out_grad += out_grad;
274  out_grads(i,j) = out_grad;
275  } else
276  out_grads(i,j) = 0;
277  }
278  out_grads(i,correct_class_ind) = -total_out_grad;
279  }
280  return TensorMap<Scalar,Root::DATA_RANK>(out_grads.data(), grad_dims);
281  }

383  for (int i = 0; i < out_mat.rows(); ++i) {
384  for (int j = 0; j < out_mat.cols(); ++j) {
385  Scalar obj_ij = obj_mat(i,j);

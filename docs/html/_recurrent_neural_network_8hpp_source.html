172  inline typename Root::Data propagate(typename Root::Data input, bool training) {
173  Dimensions<std::size_t,Root::DATA_RANK> data_dims = input.dimensions();
174  assert(input_dims == data_dims.template demote<2>());
175  int samples = data_dims(0);
176  int input_seq_length = data_dims(1);
177  // Calculate the output sequence length and delay based on the input sequence length.
178  std::pair<std::size_t,std::size_t> output_seq_info = output_seq_size_func((std::size_t) input_seq_length);
179  int output_seq_length = (int) output_seq_info.first;
180  int output_seq_delay = (int) output_seq_info.second;
181  assert(output_seq_length > 0);
182  if (reversed)
183  Base::reverse_along_time_axis(input);
184  int output_end = output_seq_length + output_seq_delay;
185  int time_steps = std::max(input_seq_length, output_end);
186  // If in training mode, unroll the network (unless it has already been unrolled for the same alignment).
187  if (training && (input_seq_length != this->input_seq_length ||
188  output_seq_length != this->output_seq_length || output_seq_delay != this->output_seq_delay))
189  unroll_network(time_steps, input_seq_length, output_seq_delay, output_end);
190  setup_hidden_state(samples);
191  RankwiseArray input_offsets, output_offsets;
192  RankwiseArray input_extents = data_dims;
193  RankwiseArray output_extents = output_dims.template promote<2>();
194  input_offsets.fill(0);
195  output_offsets.fill(0);
196  input_extents[1] = 1;
197  output_extents[0] = samples;
198  typename Root::Data out;
199  // If the output is a single time step prediction, there is no need to create an output tensor.
200  if (output_seq_length > 1) {
201  output_extents[1] = output_seq_length;
202  out = typename Root::Data(output_extents);
203  }
204  output_extents[1] = 1;
205  Dimensions<std::size_t,Rank + 1> input_time_step_dims = input_dims.template promote<>();
206  input_time_step_dims(0) = samples;
207  for (int i = 0; i < time_steps; ++i) {
208  // In inference mode, do not unroll the network.
209  Cell& cell = !training || i == 0 ? main_cell : cells[i - 1];
210  // Always apply the state kernel.
211  state = cell.state_kernel->pass_forward(std::move(state), training);
212  // If there is an input for the time step...
213  if (i < input_seq_length) {
214  typename Root::Data in_i_seq;
215  if (input_seq_length == 1)
216  in_i_seq = std::move(input);
217  else {
218  in_i_seq = input.slice(input_offsets, input_extents);
219  input_offsets[1] += 1;
220  }
221  TensorMap<Scalar,Rank + 1> in_i(in_i_seq.data(), input_time_step_dims);
222  if (MulInt) {
223  if (training) {
224  /* If multiplicative integration is enabled, cache the factors of the multiplication so that
225  * the function can be differentiated in the backward pass. */
226  cell.state_kernel_cache = state;
227  cell.input_kernel_cache = cell.input_kernel->pass_forward(in_i, training);
228  state *= cell.input_kernel_cache;
229  } else
230  state *= cell.input_kernel->pass_forward(in_i, training);
231  } else
232  state += cell.input_kernel->pass_forward(in_i, training);
233  }
234  state = cell.state_act->pass_forward(std::move(state), training);
235  // If there is an output for the time step...
236  if (i >= output_seq_delay && i < output_end) {
237  // If the output is a single time step prediction, just return it.
238  TimeStepData act_out_i = cell.output_act->pass_forward(
239  cell.output_kernel->pass_forward(state, training), training);
240  TensorMap<Scalar,Root::DATA_RANK> out_i_seq(act_out_i.data(), output_extents);
241  if (output_seq_length == 1)
242  out = out_i_seq;
243  else {
244  out.slice(output_offsets, output_extents) = out_i_seq;
245  output_offsets[1] += 1;
246  }
247  }
248  }
249  batch_size = samples;
250  this->input_seq_length = input_seq_length;
251  this->output_seq_length = output_seq_length;
252  this->output_seq_delay = output_seq_delay;
253  return out;
254  }
255  inline typename Root::Data backpropagate(typename Root::Data out_grad) {
256  Dimensions<std::size_t,Root::DATA_RANK> data_dims = out_grad.dimensions();
257  assert(output_dims == data_dims.template demote<2>() && batch_size == data_dims(0) &&
258  output_seq_length == data_dims(1));
259  RankwiseArray output_offsets, input_offsets;
260  RankwiseArray output_extents = data_dims;
261  RankwiseArray input_extents = input_dims.template promote<2>();
262  output_offsets.fill(0);
263  input_offsets.fill(0);
264  output_offsets[1] = output_seq_length - 1;
265  input_offsets[1] = input_seq_length - 1;
266  output_extents[1] = 1;
267  input_extents[0] = batch_size;
268  typename Root::Data prev_out_grad;
269  if (input_seq_length > 1) {
270  input_extents[1] = input_seq_length;
271  prev_out_grad = typename Root::Data(input_extents);
272  }
273  input_extents[1] = 1;
274  TimeStepData state_grad(state.dimensions());
275  state_grad.setZero();
276  Dimensions<std::size_t,Rank + 1> out_time_step_dims = output_dims.template promote<>();
277  out_time_step_dims(0) = batch_size;
278  int output_end = output_seq_length + output_seq_delay;
279  int time_steps = std::max(input_seq_length, output_end);
280  for (int i = time_steps - 1; i >= 0; --i) {
281  Cell& cell = i == 0 ? main_cell : cells[i - 1];
282  // If there was an output at the time step...
283  if (i >= output_seq_delay && i < output_end) {
284  typename Root::Data out_grad_seq_i;
285  if (output_seq_length == 1)
286  out_grad_seq_i = std::move(out_grad);
287  else {
288  out_grad_seq_i = out_grad.slice(output_offsets, output_extents);
289  output_offsets[1] -= 1;
290  }
291  state_grad += cell.output_kernel->pass_back(cell.output_act->pass_back(
292  TensorMap<Scalar,Rank + 1>(out_grad_seq_i.data(), out_time_step_dims)));
293  }
294  // Always back-propagate the state gradient.
295  state_grad = cell.state_act->pass_back(std::move(state_grad));
296  // If there was an input at the time step...
297  if (i < input_seq_length) {
298  // If it is the foremost layer, the gradients do not need to be propagated further back.
299  if (foremost) {
300  if (MulInt) // Multiplicative integration.
301  cell.input_kernel->pass_back(cell.state_kernel_cache * state_grad);
302  else // Additive integration.
303  cell.input_kernel->pass_back(state_grad);
304  } else if (input_seq_length == 1) {
305  TimeStepData input_i;
306  if (MulInt)
307  input_i = cell.input_kernel->pass_back(cell.state_kernel_cache * state_grad);
308  else
309  input_i = cell.input_kernel->pass_back(state_grad);
310  prev_out_grad = TensorMap<Scalar,Root::DATA_RANK>(input_i.data(), input_extents);
311  } else {
312  TimeStepData input_i;
313  if (MulInt)
314  input_i = cell.input_kernel->pass_back(cell.state_kernel_cache * state_grad);
315  else
316  input_i = cell.input_kernel->pass_back(state_grad);
317  prev_out_grad.slice(input_offsets, input_extents) =
318  TensorMap<Scalar,Root::DATA_RANK>(input_i.data(), input_extents);
319  input_offsets[1] -= 1;
320  }
321  // Compute the the state kernel's gradient.
322  if (MulInt)
323  state_grad = cell.state_kernel->pass_back(cell.input_kernel_cache * state_grad);
324  else
325  state_grad = cell.state_kernel->pass_back(std::move(state_grad));
326  } else
327  state_grad = cell.state_kernel->pass_back(std::move(state_grad));
328  }
329  return prev_out_grad;
330  }

356  inline void unroll_network(std::size_t time_steps, std::size_t input_seq_length,
357  std::size_t output_seq_delay, std::size_t output_end) {
358  if (time_steps > 1) {
359  // Empty the caches of the main cell to reduce the amount of data to copy.
360  empty_caches();
361  // Emptying the caches also clears the cell vector, thus it has to be recreated afterwards.
362  cells = std::vector<Cell>(time_steps - 1);
363  // Unroll the network by creating n -1 copies of the main cell;
364  for (int j = 1; j < time_steps; ++j) {
365  Cell& cell = cells[j - 1];
366  cell.state_kernel = KernelPtr<Scalar,Rank>(static_cast<KernelLayer<Scalar,Rank>*>(
367  main_cell.state_kernel->clone_with_shared_params()));
368  cell.state_act = ActivationPtr<Scalar,Rank>(static_cast<ActivationLayer<Scalar,Rank>*>(
369  main_cell.state_act->clone_with_shared_params()));
370  // Only copy the kernels and activations that will actually be used.
371  if (j < input_seq_length) {
372  cell.input_kernel = KernelPtr<Scalar,Rank>(static_cast<KernelLayer<Scalar,Rank>*>(
373  main_cell.input_kernel->clone_with_shared_params()));
374  }
375  if (j >= output_seq_delay && j < output_end) {
376  cell.output_kernel = KernelPtr<Scalar,Rank>(static_cast<KernelLayer<Scalar,Rank>*>(
377  main_cell.output_kernel->clone_with_shared_params()));
378  cell.output_act = ActivationPtr<Scalar,Rank>(static_cast<ActivationLayer<Scalar,Rank>*>(
379  main_cell.output_act->clone_with_shared_params()));
380  }
381  }
382  } else
383  cells = std::vector<Cell>(0);
384  }
385  inline void setup_hidden_state(std::size_t samples) {
386  if (!Stateful || batch_size == -1) {
387  Dimensions<std::size_t,Rank + 1> dims = main_cell.input_kernel->get_output_dims().template promote<>();
388  dims(0) = samples;
389  state = TimeStepData(dims);
390  state.setZero();
391  } else if (samples != batch_size) {
392  // If the network is stateful, retain the state.
393  std::array<std::size_t,Rank + 1> offsets;
394  std::array<std::size_t,Rank + 1> extents = main_cell.input_kernel->get_output_dims().template promote<>();
395  offsets.fill(0);
396  extents[0] = samples;
397  TimeStepData new_state;
398  if (samples > batch_size) {
399  new_state = TimeStepData(extents);
400  new_state.setZero();
401  extents[0] = batch_size;
402  new_state.slice(offsets, extents) = state;
403  } else
404  new_state = state.slice(offsets, extents);
405  state = std::move(new_state);
406  }

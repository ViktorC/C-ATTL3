54 template<typename Scalar, std::size_t Rank, bool Standardize = false>
55 class NormalizationPreprocessor : public Preprocessor<Scalar,Rank,false> {
56 public:
60  inline NormalizationPreprocessor(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
61  epsilon(epsilon) {
62  assert(epsilon > 0 && "epsilon must be greater than 0");
63  }
64  virtual ~NormalizationPreprocessor() = default;
65  inline virtual void fit(const Tensor<Scalar,Rank + 1>& data) {
66  std::size_t rows = data.dimension(0);
67  assert(rows > 0);
68  dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
69  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
70  means = data_mat.colwise().mean();
71  if (Standardize)
72  sd = (data_mat.rowwise() - means).array().square().colwise().mean().sqrt();
73  }
74  inline virtual void transform(Tensor<Scalar,Rank + 1>& data) const {
75  std::size_t rows = data.dimension(0);
76  assert(rows > 0);
77  Dimensions<std::size_t,Rank> demoted_dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
78  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
79  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
80  data_mat = data_mat.rowwise() - means;
81  if (Standardize)
82  data_mat *= (sd.array() + epsilon).inverse().matrix().asDiagonal();
83  }
84 protected:
85  const Scalar epsilon;
86  Dimensions<std::size_t,Rank> dims;
87  RowVector<Scalar> means;
88  RowVector<Scalar> sd;
89 };

94 template<typename Scalar, bool Standardize>
95 class NormalizationPreprocessor<Scalar,3,Standardize> {
96 public:
100  inline NormalizationPreprocessor(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
101  epsilon(epsilon) {
102  assert(epsilon > 0 && "epsilon must be greater than 0");
103  }
104  virtual ~NormalizationPreprocessor() = default;
105  inline virtual void fit(const Tensor<Scalar,4>& data) {
106  std::size_t rows = data.dimension(0);
107  assert(rows > 0);
108  dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
109  std::size_t depth = dims(2);
110  means = Matrix<Scalar>(depth, dims.get_volume() / depth);
111  sd = Matrix<Scalar>(means.rows(), means.cols());
112  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
113  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
114  for (std::size_t i = 0; i < depth; ++i) {
115  offsets[3] = i;
116  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
117  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
118  means.row(i) = data_slice_mat.colwise().mean();
119  if (Standardize)
120  sd.row(i) = (data_slice_mat.rowwise() - means.row(i)).array().square().colwise().mean().sqrt();
121  }
122  }
123  inline virtual void transform(Tensor<Scalar,4>& data) const {
124  std::size_t rows = data.dimension(0);
125  Dimensions<std::size_t,3> demoted_dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
126  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
127  assert(rows > 0);
128  std::size_t depth = data.dimension(3);
129  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
130  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
131  for (std::size_t i = 0; i < depth; ++i) {
132  offsets[3] = i;
133  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
134  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
135  data_slice_mat = data_slice_mat.rowwise() - means.row(i);
136  if (Standardize)
137  data_slice_mat *= (sd.row(i).array() + epsilon).inverse().matrix().asDiagonal();
138  data.slice(offsets, extents) = std::move(data_slice);
139  }
140  }
141 protected:
142  const Scalar epsilon;
143  Dimensions<std::size_t,3> dims;
144  Matrix<Scalar> means;
145  Matrix<Scalar> sd;
146 };

155 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool Whiten = false>
156 class PCAPreprocessorBase : public NormalizationPreprocessor<Scalar,Rank,Standardize> {
157 protected:
158  typedef NormalizationPreprocessor<Scalar,Rank,Standardize> Base;
159  inline PCAPreprocessorBase(Scalar min_rel_var_to_retain, Scalar epsilon) :
160  Base::NormalizationPreprocessor(epsilon),
161  min_rel_var_to_retain(min_rel_var_to_retain),
162  reduce_dims(internal::NumericUtils<Scalar>::decidedly_lesser(min_rel_var_to_retain, (Scalar) 1)) {
163  assert(min_rel_var_to_retain > 0 && min_rel_var_to_retain <= 1 &&
164  "the minimum relative variance to be retained must be greater "
165  "then 0 and less than or equal to 1");
166  }
167  inline void _fit(Tensor<Scalar,Rank + 1> data, int i) {
168  std::size_t rows = data.dimension(0);
169  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
170  Matrix<Scalar> normalized_data = data_mat.rowwise() - Base::means.row(i);
171  if (Standardize)
172  normalized_data *= (Base::sd.row(i).array() + Base::epsilon).inverse().matrix().asDiagonal();
173  // Compute the covariance matrix.
174  Matrix<Scalar> cov = normalized_data.transpose() * normalized_data / normalized_data.rows();
175  // Eigen decomposition.
176  internal::EigenSolver<Scalar> eigen_solver(cov);
177  // Determine the number of components to retain.
178  const ColVector<Scalar>& eigen_values = eigen_solver.eigenvalues();
179  int dims_to_retain = 0;
180  if (reduce_dims) {
181  const Scalar min_var_to_retain = eigen_values.sum() * min_rel_var_to_retain;
182  Scalar var = 0;
183  for (; dims_to_retain < eigen_values.rows(); ++dims_to_retain) {
184  // The eigen values are in ascending order.
185  var += eigen_values(eigen_values.rows() - (1 + dims_to_retain));
186  if (internal::NumericUtils<Scalar>::decidedly_greater(var, min_var_to_retain))
187  break;
188  }
189  } else
190  dims_to_retain = eigen_values.rows();
191  // The eigen vectors are sorted by the magnitude of their corresponding eigen values.
192  ed_vec[i].eigen_basis = eigen_solver.eigenvectors().rightCols(dims_to_retain);
193  if (Whiten) // The eigen values are only needed if whitening is enabled.
194  ed_vec[i].eigen_values = eigen_values.bottomRows(dims_to_retain).transpose();
195  }
196  inline Tensor<Scalar,Rank + 1> _transform(Tensor<Scalar,Rank + 1> data, int i) const {
197  std::size_t rows = data.dimension(0);
198  Dimensions<std::size_t,Rank + 1> output_dims;
199  output_dims(0) = rows;
200  if (!reduce_dims)
201  output_dims = data.dimensions();
202  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
203  data_mat *= ed_vec[i].eigen_basis;
204  if (Whiten)
205  data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
206  if (reduce_dims)
207  output_dims(1) = data_mat.cols();
208  return TensorMap<Scalar,Rank + 1>(data.data(), output_dims);
209  }
210  Scalar min_rel_var_to_retain;
211  bool reduce_dims;
212  struct EigenDecomposition {
213  Matrix<Scalar> eigen_basis;
214  RowVector<Scalar> eigen_values;
215  };
216  std::vector<EigenDecomposition> ed_vec;
217 };

225 class PCAPreprocessor : public PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> {
226  typedef PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> Base;
227  typedef typename Base::Base Root;
228 public:
236  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :

238  inline void fit(const Tensor<Scalar,Rank + 1>& data) {

253 class PCAPreprocessor<Scalar,3,Standardize,Whiten> : public PCAPreprocessorBase<Scalar,3,Standardize,Whiten> {
254  typedef PCAPreprocessorBase<Scalar,3,Standardize,Whiten> Base;
255  typedef typename Base::Base Root;
256 public:
265  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
266  Base::PCAPreprocessorBase(min_rel_var_to_retain, epsilon) { }
267  inline void fit(const Tensor<Scalar,4>& data) {
268  assert((!Base::reduce_dims || data.dimension(3) == 1) && "cannot reduce the dimensionality of multi-channel tensors");
269  Root::fit(data);
270  std::size_t channels = Root::dims(2);
271  Base::ed_vec = std::vector<typename Base::EigenDecomposition>(channels);
272  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
273  std::array<std::size_t,4> extents({ data.dimension(0), Root::dims(0), Root::dims(1), 1u });
274  for (std::size_t i = 0; i < channels; ++i) {
275  offsets[3] = i;
276  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
277  Base::_fit(std::move(data_slice_i), i);
278  }
279  }
280  inline void transform(Tensor<Scalar,4>& data) const {
281  Root::transform(data);
282  if (Base::reduce_dims) {
283  Dimensions<std::size_t,3> output_dims({ data.dimension(1) * data.dimension(2), 1u, 1u });
284  data = Base::_transform(std::move(data), 0);
285  } else {
286  std::size_t rows = data.dimension(0);
287  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
288  std::array<std::size_t,4> extents({ rows, Root::dims(0), Root::dims(1), 1u });
289  for (std::size_t i = 0; i < Root::dims(2); ++i) {
290  offsets[3] = i;
291  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);

293  }
294  }

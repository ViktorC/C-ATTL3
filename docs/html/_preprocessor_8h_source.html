53 template<typename Scalar, std::size_t Rank, bool Standardize = false>
54 class NormalizationPreprocessor : public Preprocessor<Scalar,Rank,false> {
55 public:
59  inline NormalizationPreprocessor(Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
60  epsilon(epsilon) {
61  assert(epsilon > 0 && "epsilon must be greater than 0");
62  }
63  virtual ~NormalizationPreprocessor() = default;
64  inline virtual void fit(const Tensor<Scalar,Rank + 1>& data) {
65  std::size_t rows = data.dimension(0);
66  assert(rows > 0);
67  dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
68  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
69  means = data_mat.colwise().mean();
70  if (Standardize)
71  sd = (data_mat.rowwise() - means).array().square().colwise().mean().sqrt();
72  }
73  inline virtual void transform(Tensor<Scalar,Rank + 1>& data) const {
74  std::size_t rows = data.dimension(0);
75  assert(rows > 0);
76  Dimensions<std::size_t,Rank> demoted_dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
77  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
78  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
79  data_mat = data_mat.rowwise() - means;
80  if (Standardize)
81  data_mat *= (sd.array() + epsilon).inverse().matrix().asDiagonal();
82  }
83 protected:
84  const Scalar epsilon;
85  Dimensions<std::size_t,Rank> dims;
86  RowVector<Scalar> means;
87  RowVector<Scalar> sd;
88 };

93 template<typename Scalar, bool Standardize>
94 class NormalizationPreprocessor<Scalar,3,Standardize> {
95 public:
99  inline NormalizationPreprocessor(Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
100  epsilon(epsilon) {
101  assert(epsilon > 0 && "epsilon must be greater than 0");
102  }
103  virtual ~NormalizationPreprocessor() = default;
104  inline virtual void fit(const Tensor<Scalar,4>& data) {
105  std::size_t rows = data.dimension(0);
106  assert(rows > 0);
107  dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
108  std::size_t depth = dims(2);
109  means = Matrix<Scalar>(depth, dims.get_volume() / depth);
110  sd = Matrix<Scalar>(means.rows(), means.cols());
111  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
112  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
113  for (std::size_t i = 0; i < depth; ++i) {
114  offsets[3] = i;
115  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
116  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
117  means.row(i) = data_slice_mat.colwise().mean();
118  if (Standardize)
119  sd.row(i) = (data_slice_mat.rowwise() - means.row(i)).array().square().colwise().mean().sqrt();
120  }
121  }
122  inline virtual void transform(Tensor<Scalar,4>& data) const {
123  std::size_t rows = data.dimension(0);
124  Dimensions<std::size_t,3> demoted_dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
125  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
126  assert(rows > 0);
127  std::size_t depth = data.dimension(3);
128  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
129  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
130  for (std::size_t i = 0; i < depth; ++i) {
131  offsets[3] = i;
132  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
133  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
134  data_slice_mat = data_slice_mat.rowwise() - means.row(i);
135  if (Standardize)
136  data_slice_mat *= (sd.row(i).array() + epsilon).inverse().matrix().asDiagonal();
137  data.slice(offsets, extents) = std::move(data_slice);
138  }
139  }
140 protected:
141  const Scalar epsilon;
142  Dimensions<std::size_t,3> dims;
143  Matrix<Scalar> means;
144  Matrix<Scalar> sd;
145 };

151 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool Whiten = false>
152 class PCAPreprocessorBase : public NormalizationPreprocessor<Scalar,Rank,Standardize> {
153 protected:
154  typedef NormalizationPreprocessor<Scalar,Rank,Standardize> Base;
155  inline PCAPreprocessorBase(Scalar min_rel_var_to_retain, Scalar epsilon) :
156  Base::NormalizationPreprocessor(epsilon),
157  min_rel_var_to_retain(min_rel_var_to_retain),
158  reduce_dims(internal::Utils<Scalar>::decidedly_lesser(min_rel_var_to_retain, (Scalar) 1)) {
159  assert(min_rel_var_to_retain > 0 && min_rel_var_to_retain <= 1 &&
160  "the minimum relative variance to be retained must be greater "
161  "then 0 and less than or equal to 1");
162  }
163  inline void _fit(Tensor<Scalar,Rank + 1> data, int i) {
164  std::size_t rows = data.dimension(0);
165  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
166  Matrix<Scalar> normalized_data = data_mat.rowwise() - Base::means.row(i);
167  if (Standardize)
168  normalized_data *= (Base::sd.row(i).array() + Base::epsilon).inverse().matrix().asDiagonal();
169  // Compute the covariance matrix.
170  Matrix<Scalar> cov = normalized_data.transpose() * normalized_data / normalized_data.rows();
171  // Eigen decomposition.
172  internal::EigenSolver<Scalar> eigen_solver(cov);
173  // Determine the number of components to retain.
174  const ColVector<Scalar>& eigen_values = eigen_solver.eigenvalues();
175  int dims_to_retain = 0;
176  if (reduce_dims) {
177  const Scalar min_var_to_retain = eigen_values.sum() * min_rel_var_to_retain;
178  Scalar var = 0;
179  for (; dims_to_retain < eigen_values.rows(); ++dims_to_retain) {
180  // The eigen values are in ascending order.
181  var += eigen_values(eigen_values.rows() - (1 + dims_to_retain));
182  if (internal::Utils<Scalar>::decidedly_greater(var, min_var_to_retain))
183  break;
184  }
185  } else
186  dims_to_retain = eigen_values.rows();
187  // The eigen vectors are sorted by the magnitude of their corresponding eigen values.
188  ed_vec[i].eigen_basis = eigen_solver.eigenvectors().rightCols(dims_to_retain);
189  if (Whiten) // The eigen values are only needed if whitening is enabled.
190  ed_vec[i].eigen_values = eigen_values.bottomRows(dims_to_retain).transpose();
191  }
192  inline Tensor<Scalar,Rank + 1> _transform(Tensor<Scalar,Rank + 1> data, int i) const {
193  std::size_t rows = data.dimension(0);
194  Dimensions<std::size_t,Rank + 1> output_dims;
195  output_dims(0) = rows;
196  if (!reduce_dims)
197  output_dims = data.dimensions();
198  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
199  data_mat *= ed_vec[i].eigen_basis;
200  if (Whiten)
201  data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
202  if (reduce_dims)
203  output_dims(1) = data_mat.cols();
204  return TensorMap<Scalar,Rank + 1>(data.data(), output_dims);
205  }
206  Scalar min_rel_var_to_retain;
207  bool reduce_dims;
208  struct EigenDecomposition {
209  Matrix<Scalar> eigen_basis;
210  RowVector<Scalar> eigen_values;
211  };
212  std::vector<EigenDecomposition> ed_vec;
213 };

219 class PCAPreprocessor : public PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> {
220  typedef PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> Base;

246 template<typename Scalar, bool Standardize, bool Whiten>
247 class PCAPreprocessor<Scalar,3,Standardize,Whiten> : public PCAPreprocessorBase<Scalar,3,Standardize,Whiten> {
248  typedef PCAPreprocessorBase<Scalar,3,Standardize,Whiten> Base;
249  typedef typename Base::Base Root;
250 public:
259  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = internal::Utils<Scalar>::EPSILON2) :
260  Base::PCAPreprocessorBase(min_rel_var_to_retain, epsilon) { }
261  inline void fit(const Tensor<Scalar,4>& data) {
262  assert((!Base::reduce_dims || data.dimension(3) == 1) && "cannot reduce the dimensionality of multi-channel tensors");
263  Root::fit(data);
264  std::size_t channels = Root::dims(2);
265  Base::ed_vec = std::vector<typename Base::EigenDecomposition>(channels);
266  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
267  std::array<std::size_t,4> extents({ data.dimension(0), Root::dims(0), Root::dims(1), 1u });
268  for (std::size_t i = 0; i < channels; ++i) {
269  offsets[3] = i;
270  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
271  Base::_fit(std::move(data_slice_i), i);
272  }
273  }
274  inline void transform(Tensor<Scalar,4>& data) const {
275  Root::transform(data);
276  if (Base::reduce_dims) {
277  Dimensions<std::size_t,3> output_dims({ data.dimension(1) * data.dimension(2), 1u, 1u });
278  data = Base::_transform(std::move(data), 0);
279  } else {
280  std::size_t rows = data.dimension(0);
281  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
282  std::array<std::size_t,4> extents({ rows, Root::dims(0), Root::dims(1), 1u });
283  for (std::size_t i = 0; i < Root::dims(2); ++i) {
284  offsets[3] = i;
285  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
286  data.slice(offsets, extents) = Base::_transform(std::move(data_slice_i), i);
287  }
288  }
289  }
290 };

54 template<typename Scalar, std::size_t Rank>
55 class NormalizationPreprocessor : public Preprocessor<Scalar,Rank,false> {
56 public:
61  inline NormalizationPreprocessor(bool standardize = false, Scalar epsilon = Utils<Scalar>::EPSILON2) :
62  standardize(standardize),
63  epsilon(epsilon) {
64  assert(epsilon > 0 && "epsilon must be greater than 0");
65  }
66  virtual ~NormalizationPreprocessor() = default;
67  inline virtual void fit(const Tensor<Scalar,Rank + 1>& data) {
68  int rows = data.dimension(0);
69  assert(rows > 0);
70  dims = Utils<Scalar>::template get_dims<Rank + 1>(data).demote();
71  Matrix<Scalar> data_mat = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(data);
72  means = data_mat.colwise().mean();
73  if (standardize)
74  sd = (data_mat.rowwise() - means).array().square().colwise().mean().sqrt();
75  }
76  inline virtual void transform(Tensor<Scalar,Rank + 1>& data) const {
77  int rows = data.dimension(0);
78  assert(rows > 0);
79  Dimensions<int,Rank> demoted_dims = Utils<Scalar>::template get_dims<Rank + 1>(data).demote();
80  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
81  Matrix<Scalar> data_mat = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(data));
82  data_mat = data_mat.rowwise() - means;
83  if (standardize)
84  data_mat *= (sd.array() + epsilon).inverse().matrix().asDiagonal();
85  data = Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(data_mat, demoted_dims);
86  }
87 protected:
88  const bool standardize;
89  const Scalar epsilon;
90  Dimensions<int,Rank> dims;
91  RowVector<Scalar> means;
92  RowVector<Scalar> sd;
93 };

98 template<typename Scalar>
99 class NormalizationPreprocessor<Scalar,3> {
100 public:
105  inline NormalizationPreprocessor(bool standardize = false, Scalar epsilon = Utils<Scalar>::EPSILON2) :
106  standardize(standardize),
107  epsilon(epsilon) {
108  assert(epsilon > 0 && "epsilon must be greater than 0");
109  }
110  virtual ~NormalizationPreprocessor() = default;
111  inline virtual void fit(const Tensor<Scalar,4>& data) {
112  int rows = data.dimension(0);
113  assert(rows > 0);
114  dims = Utils<Scalar>::template get_dims<4>(data).demote();
115  int depth = dims(2);
116  means = Matrix<Scalar>(depth, dims.get_volume() / depth);
117  sd = Matrix<Scalar>(means.rows(), means.cols());
118  std::array<int,4> offsets({ 0, 0, 0, 0 });
119  std::array<int,4> extents({ rows, dims(0), dims(1), 1 });
120  for (int i = 0; i < depth; ++i) {
121  offsets[3] = i;
122  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
123  Matrix<Scalar> data_mat = Utils<Scalar>::template map_tensor_to_mat<4>(std::move(data_slice_i));
124  means.row(i) = data_mat.colwise().mean();
125  if (standardize)
126  sd.row(i) = (data_mat.rowwise() - means.row(i)).array().square().colwise().mean().sqrt();
127  }
128  }
129  inline virtual void transform(Tensor<Scalar,4>& data) const {
130  int rows = data.dimension(0);
131  Dimensions<int,3> demoted_dims = Utils<Scalar>::template get_dims<4>(data).demote();
132  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
133  assert(rows > 0);
134  int depth = data.dimension(3);
135  Dimensions<int,3> slice_dims({ dims(0), dims(1), 1 });
136  std::array<int,4> offsets = { 0, 0, 0, 0 };
137  std::array<int,4> extents = { rows, slice_dims(0), slice_dims(1), slice_dims(2) };
138  for (int i = 0; i < depth; ++i) {
139  offsets[3] = i;
140  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
141  Matrix<Scalar> data_ch_i = Utils<Scalar>::template map_tensor_to_mat<4>(std::move(data_slice_i));
142  data_ch_i = data_ch_i.rowwise() - means.row(i);
143  if (standardize)
144  data_ch_i *= (sd.row(i).array() + epsilon).inverse().matrix().asDiagonal();
145  data.slice(offsets, extents) = Utils<Scalar>::template map_mat_to_tensor<4>(data_ch_i, slice_dims);
146  }
147  }
148 protected:
149  const bool standardize;
150  const Scalar epsilon;
151  Dimensions<int,3> dims;
152  Matrix<Scalar> means;
153  Matrix<Scalar> sd;
154 };

160 template<typename Scalar, std::size_t Rank>
161 class PCAPreprocessorBase : public NormalizationPreprocessor<Scalar,Rank> {
162 protected:
163  typedef NormalizationPreprocessor<Scalar,Rank> Base;
164  inline PCAPreprocessorBase(bool standardize, bool whiten, Scalar min_rel_var_to_retain, Scalar epsilon) :
165  Base::NormalizationPreprocessor(standardize, epsilon),
166  whiten(whiten),
167  min_rel_var_to_retain(min_rel_var_to_retain),
168  reduce_dims(Utils<Scalar>::decidedly_lesser(min_rel_var_to_retain, (Scalar) 1)) {
169  assert(min_rel_var_to_retain > 0 && min_rel_var_to_retain <= 1 &&
170  "the minimum relative variance to be retained must be greater "
171  "then 0 and less than or equal to 1");
172  }
173  inline void _fit(Tensor<Scalar,Rank + 1> data, int i) {
174  Matrix<Scalar> normalized_data = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(data)).rowwise() -
175  Base::means.row(i);
176  if (Base::standardize)
177  normalized_data *= (Base::sd.row(i).array() + Base::epsilon).inverse().matrix().asDiagonal();
178  // Compute the covariance matrix.
179  Matrix<Scalar> cov = normalized_data.transpose() * normalized_data / normalized_data.rows();
180  // Eigen decomposition.
181  Eigen::SelfAdjointEigenSolver<Matrix<Scalar>> eigen_solver(cov);
182  // Determine the number of components to retain.
183  const ColVector<Scalar>& eigen_values = eigen_solver.eigenvalues();
184  int dims_to_retain = 0;
185  if (reduce_dims) {
186  const Scalar min_var_to_retain = eigen_values.sum() * min_rel_var_to_retain;
187  Scalar var = 0;
188  for (; dims_to_retain < eigen_values.rows(); ++dims_to_retain) {
189  // The eigen values are in ascending order.
190  var += eigen_values(eigen_values.rows() - (1 + dims_to_retain));
191  if (Utils<Scalar>::decidedly_greater(var, min_var_to_retain))
192  break;
193  }
194  } else
195  dims_to_retain = eigen_values.rows();
196  // The eigen vectors are sorted by the magnitude of their corresponding eigen values.
197  ed_vec[i].eigen_basis = eigen_solver.eigenvectors().rightCols(dims_to_retain);
198  if (whiten) // The eigen values are only needed if whitening is enabled.
199  ed_vec[i].eigen_values = eigen_values.bottomRows(dims_to_retain).transpose();
200  }
201  inline Tensor<Scalar,Rank + 1> _transform(Tensor<Scalar,Rank + 1> data, int i) const {
202  Dimensions<int,Rank> output_dims;
203  if (!reduce_dims)
204  output_dims = Utils<Scalar>::template get_dims<Rank + 1>(data).demote();
205  Matrix<Scalar> data_mat = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(data));
206  data_mat *= ed_vec[i].eigen_basis;
207  if (whiten)
208  data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
209  if (reduce_dims)
210  output_dims(0) = data_mat.cols();
211  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(data_mat, output_dims);
212  }
213  bool whiten;
214  Scalar min_rel_var_to_retain;
215  bool reduce_dims;
216  struct EigenDecomposition {
217  Matrix<Scalar> eigen_basis;
218  RowVector<Scalar> eigen_values;
219  };
220  std::vector<EigenDecomposition> ed_vec;
221 };

226 template<typename Scalar, std::size_t Rank>
227 class PCAPreprocessor : public PCAPreprocessorBase<Scalar,Rank> {
228  typedef PCAPreprocessorBase<Scalar,Rank> Base;
229  typedef typename Base::Base Root;
230 public:
240  inline PCAPreprocessor(bool standardize = false, bool whiten = false, Scalar min_rel_var_to_retain = 1,
241  Scalar epsilon = Utils<Scalar>::EPSILON2) :
242  Base::PCAPreprocessorBase(standardize, whiten, min_rel_var_to_retain, epsilon) { }
243  inline void fit(const Tensor<Scalar,Rank + 1>& data) {
244  Root::fit(data);

246  Base::_fit(data, 0);
247  }

249  Root::transform(data);
250  data = Base::_transform(std::move(data), 0);
251  }

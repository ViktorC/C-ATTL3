72  inline Matrix<Scalar> mul(Matrix<Scalar>& a, Matrix<Scalar>& b, bool transpose_a,
73  bool transpose_b) {
74  std::size_t a_orig_rows = a.rows();
75  std::size_t a_orig_cols = a.cols();
76  std::size_t b_orig_rows = b.rows();
77  std::size_t b_orig_cols = b.cols();
78  std::size_t a_rows, a_cols, b_rows, b_cols;
79  if (transpose_a) {
80  a_rows = a_orig_cols;
81  a_cols = a_orig_rows;
82  } else {
83  a_rows = a_orig_rows;
84  a_cols = a_orig_cols;
85  }
86  if (transpose_b) {
87  b_rows = b_orig_cols;
88  b_cols = b_orig_rows;
89  } else {
90  b_rows = b_orig_rows;
91  b_cols = b_orig_cols;
92  }
93  assert(a_cols == b_rows);
94  Matrix<Scalar> c(a_rows, b_cols);
95  cudaError_t cuda_stat;
96  cublasStatus_t cublas_stat;
97  // Device arrays.
98  Scalar* d_a;
99  Scalar* d_b;
100  Scalar* d_c;
101  // Allocate the memory for the arrays on the device.
102  cuda_stat = cudaMalloc(&d_a, a_rows * a_cols * sizeof(Scalar));
103  if (cuda_stat != cudaSuccess)
104  throw std::runtime_error("cuda malloc failure: " +
105  std::to_string(cuda_stat));
106  cuda_stat = cudaMalloc(&d_b, b_rows * b_cols * sizeof(Scalar));
107  if (cuda_stat != cudaSuccess) {
108  cudaFree(d_a);
109  throw std::runtime_error("cuda malloc failure: " +
110  std::to_string(cuda_stat));
111  }
112  cuda_stat = cudaMalloc(&d_c, a_rows * b_cols * sizeof(Scalar));
113  if (cuda_stat != cudaSuccess) {
114  cudaFree(d_a);
115  cudaFree(d_b);
116  throw std::runtime_error("cuda malloc failure: " +
117  std::to_string(cuda_stat));
118  }
119  // Copy the contents of the host arrays to the device arrays.
120  cublas_stat = cublasSetMatrix(a_orig_rows, a_orig_cols, sizeof(Scalar), a.data(),
121  a_orig_rows, d_a, a_orig_rows);
122  if (cublas_stat != CUBLAS_STATUS_SUCCESS) {
123  cudaFree(d_a);
124  cudaFree(d_b);
125  cudaFree(d_c);
126  throw std::runtime_error("cublas matrix mapping failure: " +
127  std::to_string(cublas_stat));
128  }
129  cublas_stat = cublasSetMatrix(b_orig_rows, b_orig_cols, sizeof(Scalar), b.data(),
130  b_orig_rows, d_b, b_orig_rows);
131  if (cublas_stat != CUBLAS_STATUS_SUCCESS) {
132  cudaFree(d_a);
133  cudaFree(d_b);
134  cudaFree(d_c);
135  throw std::runtime_error("cublas matrix mapping failure: " +
136  std::to_string(cublas_stat));
137  }
138  cublasOperation_t a_op = transpose_a ? CUBLAS_OP_T : CUBLAS_OP_N;
139  cublasOperation_t b_op = transpose_b ? CUBLAS_OP_T : CUBLAS_OP_N;
140  Scalar alpha = 1;
141  Scalar beta = 0;
142  // Resolve the GEMM precision based on the scalar type.
143  GemmRoutine<Scalar> gemm = get_gemm_routine<Scalar>();
144  // Perform the matrix multiplication.
145  cublas_stat = gemm(handle, a_op, b_op, a_rows, b_cols, a_cols, &alpha, d_a,
146  a_orig_rows, d_b, b_orig_rows, &beta, d_c, a_rows);
147  if (cublas_stat != CUBLAS_STATUS_SUCCESS) {
148  cudaFree(d_a);
149  cudaFree(d_b);
150  cudaFree(d_c);
151  throw std::runtime_error("cublas gemm failure: " +
152  std::to_string(cublas_stat));
153  }
154  /* Copy the contents of the device array holding the results of the matrix
155  * multiplication back to the host. */
156  cublas_stat = cublasGetMatrix(a_rows, b_cols, sizeof(Scalar), d_c, a_rows,
157  c.data(), a_rows);
158  cudaFree(d_a);
159  cudaFree(d_b);
160  cudaFree(d_c);
161  if (cublas_stat != CUBLAS_STATUS_SUCCESS)
162  throw std::runtime_error("cublas matrix retrieval failure: " +
163  std::to_string(cublas_stat));
164  return c;
165  }
166 private:
167  cublasHandle_t handle;
168  CuBLASUtils() :
169  handle() {
170  // Create the cuBLAS handle.
171  cublasStatus_t cublas_stat = cublasCreate(&handle);
172  assert(cublas_stat == CUBLAS_STATUS_SUCCESS);
173  }

77  inline Tensor<Scalar,4> _pass_forward(Tensor<Scalar,4> in, bool training) {
78  std::size_t rows = in.dimension(0);
79  patch_extents[0] = rows;
80  reduced_patch_extents[0] = rows;
81  Tensor<Scalar,4> out(rows, ext_output_dims(0), ext_output_dims(1), ext_output_dims(2));
82  _init_cache();
83  std::size_t patch_ind = 0;
84  std::size_t out_i = 0;
85  for (std::size_t i = 0; i <= width_rem; i += horizontal_stride, ++out_i) {
86  patch_offsets[2] = i;
87  reduced_patch_offsets[2] = out_i;
88  std::size_t out_j = 0;
89  for (std::size_t j = 0; j <= height_rem; j += vertical_stride, ++out_j) {
90  patch_offsets[1] = j;
91  reduced_patch_offsets[1] = out_j;
92  Tensor<Scalar,4> patch = in.slice(patch_offsets, patch_extents);
93  out.slice(reduced_patch_offsets, reduced_patch_extents) = _reduce(patch, patch_ind++);
94  }
95  }
96  return out;
97  }
98  inline Tensor<Scalar,4> _pass_back(Tensor<Scalar,4> out_grad) {
99  Tensor<Scalar,4> prev_out_grad(patch_extents[0], ext_input_dims(0), ext_input_dims(1), ext_input_dims(2));
100  prev_out_grad.setZero();
101  std::size_t patch_ind = 0;
102  std::size_t out_grad_i = 0;
103  for (std::size_t i = 0; i <= width_rem; i += horizontal_stride, ++out_grad_i) {
104  patch_offsets[2] = i;
105  reduced_patch_offsets[2] = out_grad_i;
106  std::size_t out_grad_j = 0;
107  for (std::size_t j = 0; j <= height_rem; j += vertical_stride, ++out_grad_j) {
108  patch_offsets[1] = j;
109  reduced_patch_offsets[1] = out_grad_j;
110  Tensor<Scalar,4> reduced_patch_grad = out_grad.slice(reduced_patch_offsets, reduced_patch_extents);
111  // Accumulate the gradients where the patches overlap.
112  prev_out_grad.slice(patch_offsets, patch_extents) += _d_reduce(reduced_patch_grad, patch_ind++);
113  }
114  }
115  return prev_out_grad;
116  }

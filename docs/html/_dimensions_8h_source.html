36 template<typename IndexType, std::size_t Rank>
37 class Dimensions : public DimExpression<Dimensions<IndexType,Rank>,IndexType,Rank> {
38  template<typename OtherIndexType, std::size_t OtherRank>
39  friend class Dimensions;
40 public:
41  inline Dimensions() :
42  values(Rank, 1) { }
43  inline Dimensions(const std::initializer_list<IndexType>& values) :
44  Dimensions() {
45  assert(values.size() <= Rank);
46  std::copy(values.begin(), values.end(), this->values.begin());
47  }
48  inline Dimensions(const std::array<IndexType,Rank>& array) :
49  Dimensions() {
50  assert(array.size() <= Rank);
51  std::copy(array.begin(), array.end(), values.begin());
52  }
53  template<typename OtherDerived>
54  inline Dimensions(const DimExpression<OtherDerived,IndexType,Rank>& dims) :
55  Dimensions() {
56  for (std::size_t i = 0; i < Rank; ++i)
57  values[i] = dims(i);
58  }
65  template<std::size_t Ranks = 1>
66  inline Dimensions<IndexType,Rank + Ranks> promote() const {
67  Dimensions<IndexType,Rank + Ranks> promoted;
68  std::copy(values.begin(), values.end(), promoted.values.begin() + Ranks);
69  return promoted;
70  }
76  template<std::size_t Ranks = 1>
77  inline Dimensions<IndexType,Rank - Ranks> demote() const {
78  static_assert(Rank > Ranks, "rank must be greater than the number of ranks to demote by");
79  Dimensions<IndexType,Rank - Ranks> demoted;
80  std::copy(values.begin() + Ranks, values.end(), demoted.values.begin());
81  return demoted;
82  }
90  inline IndexType operator()(std::size_t i) const {
91  if (i < 0 || i >= Rank)
92  throw std::invalid_argument("illegal index value: " + i);
93  return values[i];
94  }
102  inline IndexType& operator()(std::size_t i) {
103  if (i < 0 || i >= Rank)
104  throw std::invalid_argument("illegal index value: " + i);
105  return values[i];
106  }
111  inline operator std::array<IndexType,Rank>() const {
112  std::array<IndexType,Rank> array;
113  std::copy(values.begin(), values.end(), array.begin());
114  return array;
115  }
116 private:
117  std::vector<IndexType> values;
118 };

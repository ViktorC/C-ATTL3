36 template<typename IndexType, std::size_t Rank>
37 class Dimensions : public DimExpression<Dimensions<IndexType,Rank>,IndexType,Rank> {
38  template<typename OtherIndexType, std::size_t OtherRank>
39  friend class Dimensions;
40 public:
41  inline Dimensions() :
42  values(Rank, 1) { }
43  inline Dimensions(const std::initializer_list<IndexType>& values) :
44  Dimensions() {
45  assert(values.size() <= Rank);
46  std::copy(values.begin(), values.end(), this->values.begin());
47  }
48  inline Dimensions(const std::array<IndexType,Rank>& array) :
49  Dimensions() {
50  std::copy(array.begin(), array.end(), values.begin());
51  }
52  template<typename OtherDerived>
53  inline Dimensions(const DimExpression<OtherDerived,IndexType,Rank>& dims) :
54  Dimensions() {
55  for (std::size_t i = 0; i < Rank; ++i)
56  values[i] = dims(i);
57  }
64  template<std::size_t Ranks = 1>
65  inline Dimensions<IndexType,Rank + Ranks> promote() const {
66  Dimensions<IndexType,Rank + Ranks> promoted;
67  std::copy(values.begin(), values.end(), promoted.values.begin() + Ranks);
68  return promoted;
69  }
75  template<std::size_t Ranks = 1>
76  inline Dimensions<IndexType,Rank - Ranks> demote() const {
77  static_assert(Rank > Ranks, "rank must be greater than the number of ranks to demote by");
78  Dimensions<IndexType,Rank - Ranks> demoted;
79  std::copy(values.begin() + Ranks, values.end(), demoted.values.begin());
80  return demoted;
81  }
89  inline IndexType operator()(std::size_t i) const {
90  if (i < 0 || i >= Rank)
91  throw std::invalid_argument("illegal index value: " + i);
92  return values[i];
93  }
101  inline IndexType& operator()(std::size_t i) {
102  if (i < 0 || i >= Rank)
103  throw std::invalid_argument("illegal index value: " + i);
104  return values[i];
105  }
110  inline operator std::array<IndexType,Rank>() const {
111  std::array<IndexType,Rank> array;
112  std::copy(values.begin(), values.end(), array.begin());
113  return array;
114  }
115 private:
116  std::vector<IndexType> values;
117 };

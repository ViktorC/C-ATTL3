22 template<typename Scalar, std::size_t Rank, bool Standardize, bool Whiten, bool PerLastRank>
23 class PCAPreprocessorBase : public NormalizationPreprocessor<Scalar,Rank,Standardize,PerLastRank> {
24 protected:
25  typedef NormalizationPreprocessor<Scalar,Rank,Standardize,PerLastRank> Base;
26  inline PCAPreprocessorBase(Scalar min_rel_var_to_retain, Scalar epsilon) :
27  Base::NormalizationPreprocessor(epsilon),
28  min_rel_var_to_retain(min_rel_var_to_retain),
29  reduce_dims(NumericUtils<Scalar>::decidedly_lesser(min_rel_var_to_retain, (Scalar) 1)) {
30  assert(min_rel_var_to_retain > 0 && min_rel_var_to_retain <= 1 &&
31  "the minimum relative variance to be retained must be greater "
32  "then 0 and less than or equal to 1");
33  }
34  inline void _fit(Tensor<Scalar,Rank + 1> data, int i) {
35  std::size_t rows = data.dimension(0);
36  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
37  Matrix<Scalar> normalized_data = data_mat.rowwise() - Base::means.row(i);
38  if (Standardize)
39  normalized_data *= (Base::sd.row(i).array() + Base::epsilon).inverse().matrix().asDiagonal();
40  // Compute the covariance matrix.
41  Matrix<Scalar> cov = normalized_data.transpose() * normalized_data / normalized_data.rows();
42  // Eigen decomposition.
43  EigenSolver<Scalar> eigen_solver(cov);
44  // Determine the number of components to retain.
45  const ColVector<Scalar>& eigen_values = eigen_solver.eigenvalues();
46  int dims_to_retain = 0;
47  if (reduce_dims) {
48  const Scalar min_var_to_retain = eigen_values.sum() * min_rel_var_to_retain;
49  Scalar var = 0;
50  for (; dims_to_retain < eigen_values.rows(); ++dims_to_retain) {
51  // The eigen values are in ascending order.
52  var += eigen_values(eigen_values.rows() - (1 + dims_to_retain));
53  if (NumericUtils<Scalar>::decidedly_greater(var, min_var_to_retain))
54  break;
55  }
56  } else
57  dims_to_retain = eigen_values.rows();
58  // The eigen vectors are sorted by the magnitude of their corresponding eigen values.
59  ed_vec[i].eigen_basis = eigen_solver.eigenvectors().rightCols(dims_to_retain);
60  if (Whiten) // The eigen values are only needed if whitening is enabled.
61  ed_vec[i].eigen_values = eigen_values.bottomRows(dims_to_retain).transpose();
62  }
63  inline Tensor<Scalar,Rank + 1> _transform(Tensor<Scalar,Rank + 1> data, int i) const {
64  std::size_t rows = data.dimension(0);
65  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
66  if (reduce_dims) {
67  Dimensions<std::size_t,Rank + 1> output_dims;
68  output_dims(0) = rows;
69  Matrix<Scalar> transformed_data_mat = data_mat * ed_vec[i].eigen_basis;
70  if (Whiten)
71  transformed_data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
72  output_dims(1) = transformed_data_mat.cols();
73  return TensorMap<Scalar,Rank + 1>(transformed_data_mat.data(), output_dims);
74  } else {
75  Dimensions<std::size_t,Rank + 1> output_dims = data.dimensions();
76  data_mat *= ed_vec[i].eigen_basis;
77  if (Whiten)
78  data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
79  return TensorMap<Scalar,Rank + 1>(data_mat.data(), output_dims);
80  }
81  }
82  Scalar min_rel_var_to_retain;
83  bool reduce_dims;
84  struct EigenDecomposition {
85  Matrix<Scalar> eigen_basis;
86  RowVector<Scalar> eigen_values;
87  };
88  std::vector<EigenDecomposition> ed_vec;
89 };

95 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool Whiten = false,
96  bool PerLastRank = (Rank == 3)>
97 class PCAPreprocessor : public PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten,PerLastRank> {
98  typedef PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten,PerLastRank> Base;
99  typedef typename Base::Base Root;
100 public:
109  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = NumericUtils<Scalar>::EPSILON2) :
110  Base::PCAPreprocessorBase(min_rel_var_to_retain, epsilon) { }
111  inline void fit(const Tensor<Scalar,Rank + 1>& data) {
112  assert((!Base::reduce_dims || data.dimension(Rank) == 1) && "cannot reduce the dimensionality of multi-channel data");
113  Root::fit(data);
114  channels = data.dimension(Rank);
115  Base::ed_vec = std::vector<typename Base::EigenDecomposition>(channels);
116  if (channels == 1)
117  Base::_fit(data, 0);
118  else {
119  std::array<std::size_t,Rank + 1> offsets;
120  offsets.fill(0);
121  auto extents = data.dimensions();
122  extents[Rank] = 1;
123  for (std::size_t i = 0; i < channels; ++i) {
124  offsets[Rank] = i;
125  Tensor<Scalar,Rank + 1> data_slice_i = data.slice(offsets, extents);
126  Base::_fit(std::move(data_slice_i), i);
127  }
128  }
129  }
130  inline void transform(Tensor<Scalar,Rank + 1>& data) const {
131  Root::transform(data);
132  if (Base::reduce_dims || channels == 1)
133  data = Base::_transform(std::move(data), 0);
134  else {
135  std::array<std::size_t,Rank + 1> offsets;
136  offsets.fill(0);
137  auto extents = data.dimensions();
138  extents[Rank] = 1;
139  for (std::size_t i = 0; i < channels; ++i) {
140  offsets[Rank] = i;
141  Tensor<Scalar,Rank + 1> data_slice_i = data.slice(offsets, extents);
142  data.slice(offsets, extents) = Base::_transform(std::move(data_slice_i), i);
143  }
144  }
145  }
146 private:
147  std::size_t channels;
148 };

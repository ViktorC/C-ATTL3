50  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
51  if (!has_more())
52  throw std::out_of_range("no more data left to fetch");
53  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(files[current_file_ind], current_file_stream,
54  batch_size);
55  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
56  /* If the data contains fewer batches than expected, the end of the file has been reached and the
57  * rest of the data should be read from the next file. */
58  while (data_pair.first.dimension(0) < batch_size && has_more()) {
59  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(files[current_file_ind],
60  current_file_stream, batch_size - data_pair.first.dimension(0));
61  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
62  // It has to be evaluated into a temporary due to the dimension incompatibility.
63  typename Base::Data obs_concat = data_pair.first.concatenate(std::move(add_data_pair.first), 0);
64  data_pair.first = std::move(obs_concat);
65  typename Base::Data obj_concat = data_pair.second.concatenate(std::move(add_data_pair.second), 0);
66  data_pair.second = std::move(obj_concat);
67  }
68  return data_pair;
69  }

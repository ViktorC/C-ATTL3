115  inline Tensor<Scalar,4> _pass_forward(Tensor<Scalar,4> in, bool training) {
116  // Spatial padding.
117  if (vertical_padding > 0 || horizontal_padding > 0)
118  in = Tensor<Scalar,4>(in.pad(paddings));
119  std::size_t rows = in.dimension(0);
120  std::size_t total_patches = rows * patches_per_sample;
121  std::size_t receptor_vol = Base::weights->get_values().rows();
122  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one
123  * stretched out receptor of one sample. The same receptor location along all samples of the
124  * batch is represented by a contiguous block of these rows. */
125  std::size_t patch_ind = 0;
126  patch_extents[0] = rows;
127  in_conv_mat_cache = Matrix<Scalar>(total_patches, receptor_vol);
128  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
129  patch_offsets[2] = i;
130  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
131  patch_offsets[1] = j;
132  Tensor<Scalar,4> patch;
133  // If the patch is dilated, skip the spatial gaps when flattening it into a matrix.
134  if (vertical_dilation > 0 || horizontal_dilation > 0)
135  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
136  else
137  patch = in.slice(patch_offsets, patch_extents);
138  in_conv_mat_cache.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(),
139  rows, receptor_vol);
140  patch_ind += rows;
141  }
142  }
143  assert(patch_ind == total_patches);
144  Matrix<Scalar> out_mat = (in_conv_mat_cache * Base::weights->get_values()).rowwise() +
145  Base::bias->get_values().row(0);
146  out_conversion_dims[0] = rows;
147  return TensorMap<Scalar,4>(out_mat.data(), out_conversion_dims);
148  }
149  inline Tensor<Scalar,4> _pass_back(Tensor<Scalar,4> out_grad) {
150  std::size_t rows = out_grad.dimension(0);
151  std::size_t total_patches = rows * patches_per_sample;
152  std::size_t receptor_vol = Base::weights->get_values().rows();
153  MatrixMap<Scalar> out_grad_mat(out_grad.data(), total_patches, filters);
154  Base::weights->accumulate_grad(in_conv_mat_cache.transpose() * out_grad_mat);
155  Base::bias->accumulate_grad(out_grad_mat.colwise().sum());
156  if (Base::is_input_layer())
157  return Tensor<Scalar,4>();
158  // Compute the gradient of the previous layer's output.
159  Matrix<Scalar> prev_out_grad_conv_mat = out_grad_mat * Base::weights->get_values().transpose();
160  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
161  * to get the derivative w.r.t. the individual input nodes. */
162  Tensor<Scalar,4> prev_out_grad(rows, padded_height, padded_width, ext_input_dims(2));
163  prev_out_grad.setZero();
164  std::size_t patch_ind = 0;
165  patch_extents[0] = rows;
166  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
167  patch_offsets[2] = i;
168  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
169  patch_offsets[1] = j;
170  // Accumulate the gradients where the receptor-patch-tensors overlap.
171  Matrix<Scalar> prev_out_grad_conv_mat_block = prev_out_grad_conv_mat.block(patch_ind, 0,
172  rows, receptor_vol);
173  TensorMap<Scalar,4> prev_out_grad_patch(prev_out_grad_conv_mat_block.data(), rows,
174  receptor_height, receptor_width, ext_input_dims(2));
175  if (vertical_dilation > 0 || horizontal_dilation > 0)
176  prev_out_grad.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grad_patch;
177  else
178  prev_out_grad.slice(patch_offsets, patch_extents) += prev_out_grad_patch;
179  patch_ind += rows;
180  }
181  }
182  assert(patch_ind == prev_out_grad_conv_mat.rows());
183  if (vertical_padding > 0 || horizontal_padding > 0) {
184  // Cut off the padding.
185  no_padding_extents[0] = rows;
186  return prev_out_grad.slice(no_padding_offsets, no_padding_extents);
187  } else
188  return prev_out_grad;
189  }
190  // The defining attributes of the convolutional layer.
191  const std::size_t filters, receptor_height, receptor_width, vertical_padding, horizontal_padding,
192  vertical_stride, horizontal_stride, vertical_dilation, horizontal_dilation;
193 private:
194  inline static std::size_t calculate_spatial_output_dim(std::size_t input_dim, std::size_t receptor_size,
195  std::size_t padding, std::size_t dilation, std::size_t stride) {
196  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
197  }
198  inline static Dimensions<std::size_t,3> calculate_output_dims(const Dimensions<std::size_t,3>& input_dims,
199  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding,
200  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
201  std::size_t vertical_dilation, std::size_t horizontal_dilation) {
202  return { calculate_spatial_output_dim(input_dims(0), receptor_height, vertical_padding, vertical_dilation,
203  vertical_stride), calculate_spatial_output_dim(input_dims(1), receptor_width, horizontal_padding,
204  horizontal_dilation, horizontal_stride), filters };
205  }
206  inline static Dimensions<std::size_t,Rank> calculate_adjusted_output_dims(
207  const Dimensions<std::size_t,Rank>& input_dims, std::size_t filters, std::size_t receptor_height,
208  std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
209  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
210  std::size_t horizontal_dilation) {
211  auto output_dims = calculate_output_dims(input_dims.template extend<3 - Rank>(), filters, receptor_height,
212  receptor_width, vertical_padding, horizontal_padding, vertical_stride, horizontal_stride,
213  vertical_dilation, horizontal_dilation);
214  output_dims(2) /= filters;
215  output_dims(Rank - 1) *= filters;
216  return output_dims.template contract<3 - Rank>();
217  }
218  const Dimensions<std::size_t,3> ext_input_dims, ext_output_dims;
219  // Pre-computed values to improve propagation-time performance.
220  const std::size_t padded_height, padded_width, dil_receptor_height, dil_receptor_width, patches_per_sample;
221  Array4 out_conversion_dims, patch_offsets, patch_extents, dil_strides, no_padding_offsets, no_padding_extents;
222  PaddingsArray4 paddings;
223  // Staged computation caches
224  Matrix<Scalar> in_conv_mat_cache;
225 };

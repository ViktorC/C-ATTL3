101 template<typename Scalar, std::size_t Rank>
102 class UniversalLoss<Scalar,Rank,true> : public Loss<Scalar,Rank,true> {
103  typedef Loss<Scalar,Rank,true> Base;
104 protected:
105  typedef std::array<std::size_t,Base::DATA_RANK> RankwiseArray;
114  virtual ColVector<Scalar> _function(typename Base::Data out, typename Base::Data obj) const = 0;
123  virtual typename Base::Data _d_function(typename Base::Data out, typename Base::Data obj,
124  const RankwiseArray& grad_dims) const = 0;
125 public:
126  virtual ~UniversalLoss() = default;
127  inline ColVector<Scalar> function(typename Base::Data out, typename Base::Data obj) const {
128  assert(out.dimensions() == obj.dimensions());
129  int time_steps = out.dimension(1);
130  if (time_steps == 1)
131  return _function(std::move(out), std::move(obj));
132  RankwiseArray offsets;
133  RankwiseArray extents = out.dimensions();
134  offsets.fill(0);
135  extents[1] = 1;
136  ColVector<Scalar> loss = ColVector<Scalar>::Zero(out.dimension(0), 1);
137  for (int i = 0; i < time_steps; ++i) {
138  offsets[1] = i;
139  typename Base::Data out_i = out.slice(offsets, extents);
140  typename Base::Data obj_i = obj.slice(offsets, extents);
141  loss += _function(std::move(out_i), std::move(obj_i));
142  }
143  return loss;
144  }
145  inline typename Base::Data d_function(const typename Base::Data out, const typename Base::Data obj) const {
146  assert(out.dimensions() == obj.dimensions());
147  int time_steps = out.dimension(1);
148  if (time_steps == 1)
149  return _d_function(std::move(out), std::move(obj), out.dimensions());
150  RankwiseArray offsets;
151  RankwiseArray extents = out.dimensions();
152  offsets.fill(0);
153  typename Base::Data grads(extents);
154  extents[1] = 1;
155  grads.setZero();
156  for (int i = 0; i < time_steps; ++i) {
157  offsets[1] = i;
158  typename Base::Data out_i = out.slice(offsets, extents);
159  typename Base::Data obj_i = obj.slice(offsets, extents);
160  grads.slice(offsets, extents) += _d_function(std::move(out_i), std::move(obj_i), extents);
161  }
162  return grads;
163  }
164 };

169 template<typename Scalar, std::size_t Rank, bool Sequential>
170 class AbsoluteLoss : public UniversalLoss<Scalar,Rank,Sequential> {
171  typedef Loss<Scalar,Rank,Sequential> Root;
172  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
173 protected:
174  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
175  std::size_t rows = out.dimension(0);
176  std::size_t cols = out.size() / rows;
177  return (MatrixMap<Scalar>(out.data(), rows, cols) - MatrixMap<Scalar>(obj.data(), rows, cols))
178  .array().abs().rowwise().sum();
179  }
180  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
181  const typename Base::RankwiseArray& grad_dims) const {
182  typename Root::Data diff = out - obj;
183  return diff.unaryExpr([this](Scalar i) { return (Scalar) (i >= 0 ? 1 : -1); });
184  }
185 };

211 template<typename Scalar, std::size_t Rank, bool Sequential, bool Squared = false>
212 class HingeLoss : public UniversalLoss<Scalar,Rank,Sequential> {
213  typedef Loss<Scalar,Rank,Sequential> Root;
214  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
215 protected:
216  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
217  std::size_t rows = out.dimension(0);
218  std::size_t cols = out.size() / rows;
219  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
220  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
221  ColVector<Scalar> loss(rows);
222  for (int i = 0; i < rows; ++i) {
223  unsigned ones = 0;
224  int correct_class_ind = -1;
225  for (int j = 0; j < cols; ++j) {
226  Scalar obj_ij = obj_mat(i,j);
227  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
228  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
229  if (internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
230  ones++;
231  correct_class_ind = j;
232  }
233  }
234  assert(ones == 1);
235  Scalar loss_i = 0;
236  Scalar correct_class_score = out_mat(i,correct_class_ind);
237  for (int j = 0; j < cols; ++j) {
238  if (j == correct_class_ind)
239  continue;
240  Scalar loss_ij = std::max((Scalar) 0, (Scalar) (out_mat(i,j) - correct_class_score + 1));
241  loss_i += Squared ? loss_ij * loss_ij : loss_ij;
242  }
243  loss(i) = loss_i;
244  }
245  return loss;
246  }
247  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
248  const typename Base::RankwiseArray& grad_dims) const {
249  std::size_t rows = out.dimension(0);
250  std::size_t cols = out.size() / rows;
251  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
252  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
253  Matrix<Scalar> out_grads(rows, cols);
254  for (int i = 0; i < rows; ++i) {
255  unsigned ones = 0;
256  int correct_class_ind = -1;
257  for (int j = 0; j < cols; ++j) {
258  Scalar obj_ij = obj_mat(i,j);
259  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
260  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
261  if (internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
262  ones++;
263  correct_class_ind = j;
264  }
265  }
266  assert(ones == 1);
267  Scalar total_out_grad = 0;
268  Scalar correct_class_score = out_mat(i,correct_class_ind);
269  for (int j = 0; j < cols; ++j) {
270  if (j == correct_class_ind)
271  continue;
272  Scalar out_ij = out_mat(i,j);
273  Scalar margin = out_ij - correct_class_score + 1;
274  if (internal::NumericUtils<Scalar>::decidedly_greater(margin, (Scalar) 0)) {
275  Scalar out_grad = Squared ? 2 * margin : 1;
276  total_out_grad += out_grad;
277  out_grads(i,j) = out_grad;
278  } else
279  out_grads(i,j) = 0;
280  }
281  out_grads(i,correct_class_ind) = -total_out_grad;
282  }
283  return TensorMap<Scalar,Root::DATA_RANK>(out_grads.data(), grad_dims);
284  }
285 };

387  for (int j = 0; j < rows; ++j) {

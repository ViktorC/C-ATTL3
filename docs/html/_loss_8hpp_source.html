95 template<typename Scalar, std::size_t Rank, bool Sequential>
96 class UniversalLoss : public Loss<Scalar,Rank,Sequential> {
97  typedef Loss<Scalar,Rank,Sequential> Base;
98 protected:
99  typedef std::array<std::size_t,Base::DATA_RANK> RankwiseArray;
107  virtual ColVector<Scalar> _function(typename Base::Data out, typename Base::Data obj) const = 0;
116  virtual typename Base::Data _d_function(typename Base::Data out, typename Base::Data obj,
117  const RankwiseArray& grad_dims) const = 0;
118 public:
119  virtual ~UniversalLoss() = default;
120  inline ColVector<Scalar> function(typename Base::Data out, typename Base::Data obj) const {
121  assert(out.dimensions() == obj.dimensions());
122  return _function(std::move(out), std::move(obj));
123  }
124  inline typename Base::Data d_function(typename Base::Data out, typename Base::Data obj) const {
125  assert(out.dimensions() == obj.dimensions());
126  RankwiseArray dims = out.dimensions();
127  return _d_function(std::move(out), std::move(obj), dims);
128  }
129 };

135 template<typename Scalar, std::size_t Rank>
136 class UniversalLoss<Scalar,Rank,true> : public Loss<Scalar,Rank,true> {
137  typedef Loss<Scalar,Rank,true> Base;
138 protected:
139  typedef std::array<std::size_t,Base::DATA_RANK> RankwiseArray;
148  virtual ColVector<Scalar> _function(typename Base::Data out, typename Base::Data obj) const = 0;
157  virtual typename Base::Data _d_function(typename Base::Data out, typename Base::Data obj,
158  const RankwiseArray& grad_dims) const = 0;
159 public:
160  virtual ~UniversalLoss() = default;
161  inline ColVector<Scalar> function(typename Base::Data out, typename Base::Data obj) const {
162  assert(out.dimensions() == obj.dimensions());
163  int time_steps = out.dimension(1);
164  if (time_steps == 1)
165  return _function(std::move(out), std::move(obj));
166  RankwiseArray offsets;
167  RankwiseArray extents = out.dimensions();
168  offsets.fill(0);
169  extents[1] = 1;
170  ColVector<Scalar> loss = ColVector<Scalar>::Zero(out.dimension(0), 1);
171  for (int i = 0; i < time_steps; ++i) {
172  offsets[1] = i;
173  typename Base::Data out_i = out.slice(offsets, extents);
174  typename Base::Data obj_i = obj.slice(offsets, extents);
175  loss += _function(std::move(out_i), std::move(obj_i));
176  }
177  return loss;
178  }
179  inline typename Base::Data d_function(const typename Base::Data out, const typename Base::Data obj) const {
180  assert(out.dimensions() == obj.dimensions());
181  int time_steps = out.dimension(1);
182  if (time_steps == 1)
183  return _d_function(std::move(out), std::move(obj), out.dimensions());
184  RankwiseArray offsets;
185  RankwiseArray extents = out.dimensions();
186  offsets.fill(0);
187  typename Base::Data grads(extents);
188  extents[1] = 1;
189  grads.setZero();
190  for (int i = 0; i < time_steps; ++i) {
191  offsets[1] = i;
192  typename Base::Data out_i = out.slice(offsets, extents);
193  typename Base::Data obj_i = obj.slice(offsets, extents);
194  grads.slice(offsets, extents) += _d_function(std::move(out_i), std::move(obj_i), extents);
195  }
196  return grads;
197  }
198 };

205 template<typename Scalar, std::size_t Rank, bool Sequential>
206 class AbsoluteLoss : public UniversalLoss<Scalar,Rank,Sequential> {
207  typedef Loss<Scalar,Rank,Sequential> Root;
208  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
209 protected:
210  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
211  std::size_t rows = out.dimension(0);
212  std::size_t cols = out.size() / rows;
213  return (MatrixMap<Scalar>(out.data(), rows, cols) - MatrixMap<Scalar>(obj.data(), rows, cols))
214  .array().abs().rowwise().sum();
215  }
216  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
217  const typename Base::RankwiseArray& grad_dims) const {
218  typename Root::Data diff = out - obj;
219  return diff.unaryExpr([this](Scalar i) { return (Scalar) (i >= 0 ? 1 : -1); });
220  }
221 };

252 template<typename Scalar, std::size_t Rank, bool Sequential, bool Squared = false>
253 class HingeLoss : public UniversalLoss<Scalar,Rank,Sequential> {
254  typedef Loss<Scalar,Rank,Sequential> Root;
255  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
256 protected:
257  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
258  std::size_t rows = out.dimension(0);
259  std::size_t cols = out.size() / rows;
260  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
261  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
262  ColVector<Scalar> loss(rows);
263  for (int i = 0; i < rows; ++i) {
264  unsigned ones = 0;
265  int correct_class_ind = -1;
266  for (int j = 0; j < cols; ++j) {
267  Scalar obj_ij = obj_mat(i,j);
268  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
269  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
270  if (internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
271  ones++;
272  correct_class_ind = j;
273  }
274  }
275  assert(ones == 1);
276  Scalar loss_i = 0;
277  Scalar correct_class_score = out_mat(i,correct_class_ind);
278  for (int j = 0; j < cols; ++j) {
279  if (j == correct_class_ind)
280  continue;
281  Scalar loss_ij = std::max((Scalar) 0, (Scalar) (out_mat(i,j) - correct_class_score + 1));
282  loss_i += Squared ? loss_ij * loss_ij : loss_ij;
283  }
284  loss(i) = loss_i;
285  }
286  return loss;
287  }
288  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
289  const typename Base::RankwiseArray& grad_dims) const {
290  std::size_t rows = out.dimension(0);
291  std::size_t cols = out.size() / rows;
292  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
293  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
294  Matrix<Scalar> out_grad(rows, cols);
295  for (int i = 0; i < rows; ++i) {
296  unsigned ones = 0;
297  int correct_class_ind = -1;
298  for (int j = 0; j < cols; ++j) {
299  Scalar obj_ij = obj_mat(i,j);
300  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
301  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
302  if (internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)) {
303  ones++;
304  correct_class_ind = j;
305  }
306  }
307  assert(ones == 1);
308  Scalar total_out_grad = 0;
309  Scalar correct_class_score = out_mat(i,correct_class_ind);
310  for (int j = 0; j < cols; ++j) {
311  if (j == correct_class_ind)
312  continue;
313  Scalar out_ij = out_mat(i,j);
314  Scalar margin = out_ij - correct_class_score + 1;
315  if (internal::NumericUtils<Scalar>::decidedly_greater(margin, (Scalar) 0)) {
316  Scalar out_grad_ij = Squared ? 2 * margin : 1;
317  total_out_grad += out_grad_ij;
318  out_grad(i,j) = out_grad_ij;
319  } else
320  out_grad(i,j) = 0;
321  }
322  out_grad(i,correct_class_ind) = -total_out_grad;
323  }
324  return TensorMap<Scalar,Root::DATA_RANK>(out_grad.data(), grad_dims);
325  }
326 };

334 template<typename Scalar, std::size_t Rank, bool Sequential>
335 class BinaryCrossEntropyLoss : public UniversalLoss<Scalar,Rank,Sequential> {
336  typedef Loss<Scalar,Rank,Sequential> Root;
337  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
338 public:
342  BinaryCrossEntropyLoss(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
343  epsilon(epsilon) { };
344 protected:
345  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
346  assert(out.size() == out.dimension(0));
347  typename Root::Data loss = -(obj * (out + out.constant(epsilon)).log() +
348  (obj.constant(1) - obj) * (out.constant(1 + epsilon) - out).log());
349  return MatrixMap<Scalar>(loss.data(), out.dimension(0), 1);
350  }
351  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
352  const typename Base::RankwiseArray& grad_dims) const {
353  assert(out.size() == out.dimension(0));
354  return -(obj / (out + out.constant(epsilon)) -
355  (obj.constant(1) - obj) / (out.constant(1 + epsilon) - out));
356  }
357 private:
358  Scalar epsilon;
359 };

471 class MultiLabelHingeLoss : public UniversalLoss<Scalar,Rank,Sequential> {
472  typedef Loss<Scalar,Rank,Sequential> Root;
473  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
474 protected:
475  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
476  std::size_t rows = out.dimension(0);
477  std::size_t cols = out.size() / rows;
478  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
479  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
480  ColVector<Scalar> loss(rows);
481  for (int i = 0; i < rows; ++i) {
482  Scalar loss_i = 0;
483  for (int j = 0; j < cols; ++j) {
484  Scalar obj_ij = obj_mat(i,j);
485  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) -1) ||
486  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1)));
487  Scalar loss_ij = std::max((Scalar) 0, (Scalar) 1 - obj_ij * out_mat(i,j));
488  loss_i += Squared ? loss_ij * loss_ij : loss_ij;
489  }
490  loss(i) = loss_i;
491  }
492  return loss;
493  }
494  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
495  const typename Base::RankwiseArray& grad_dims) const {
496  std::size_t rows = out.dimension(0);
497  std::size_t cols = out.size() / rows;
498  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
499  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
500  Matrix<Scalar> out_grad(rows, cols);
501  for (int i = 0; i < cols; ++i) {
502  for (int j = 0; j < rows; ++j) {
503  Scalar obj_ji = obj_mat(j,i);
504  assert((internal::NumericUtils<Scalar>::almost_equal(obj_ji, (Scalar) -1) ||
505  internal::NumericUtils<Scalar>::almost_equal(obj_ji, (Scalar) 1)));
506  Scalar out_ji = out_mat(j,i);
507  Scalar margin = 1 - obj_ji * out_ji;
508  if (internal::NumericUtils<Scalar>::decidedly_greater(margin, (Scalar) 0))
509  out_grad(j,i) = Squared ? 2 * out_ji - 2 * obj_ji : -obj_ji;
510  else
511  out_grad(j,i) = 0;
512  }
513  }
514  return TensorMap<Scalar,Root::DATA_RANK>(out_grad.data(), grad_dims);
515  }

527  typedef Loss<Scalar,Rank,Sequential> Root;
528  typedef UniversalLoss<Scalar,Rank,Sequential> Base;
529 public:
533  MultiLabelLogLoss(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
534  epsilon(epsilon) { };
535 protected:
536  inline ColVector<Scalar> _function(typename Root::Data out, typename Root::Data obj) const {
537  std::size_t rows = out.dimension(0);
538  std::size_t cols = out.size() / rows;
539  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
540  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
541  ColVector<Scalar> loss(rows);
542  for (int i = 0; i < rows; ++i) {
543  Scalar loss_i = 0;
544  for (int j = 0; j < cols; ++j) {
545  Scalar obj_ij = obj_mat(i,j);
546  assert(internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 0) ||
547  internal::NumericUtils<Scalar>::almost_equal(obj_ij, (Scalar) 1));
548  Scalar out_ij = out_mat(i,j);
549  loss_i += obj_ij * log(out_ij + epsilon) + (1 - obj_ij) * log(1 + epsilon - out_ij);
550  }
551  loss(i) = loss_i;
552  }
553  return loss;
554  }
555  inline typename Root::Data _d_function(typename Root::Data out, typename Root::Data obj,
556  const typename Base::RankwiseArray& grad_dims) const {
557  std::size_t rows = out.dimension(0);
558  std::size_t cols = out.size() / rows;
559  MatrixMap<Scalar> out_mat(out.data(), rows, cols);
560  MatrixMap<Scalar> obj_mat(obj.data(), rows, cols);
561  Matrix<Scalar> out_grad(rows, cols);
562  for (int i = 0; i < cols; ++i) {
563  for (int j = 0; j < rows; ++j) {
564  Scalar obj_ji = obj_mat(j,i);
565  assert(internal::NumericUtils<Scalar>::almost_equal(obj_ji, (Scalar) 0) ||
566  internal::NumericUtils<Scalar>::almost_equal(obj_ji, (Scalar) 1));
567  Scalar out_ji = out_mat(j,i);
568  Scalar denominator = out_ji * (1 - out_ji);
569  if (out_ji == 0)
570  out_ji += epsilon;
571  out_grad(j,i) = (obj_ji - out_ji) / denominator;
572  }
573  }
574  return TensorMap<Scalar,Root::DATA_RANK>(out_grad.data(), grad_dims);
575  }
576 private:
577  Scalar epsilon;

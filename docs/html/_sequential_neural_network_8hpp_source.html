95  inline typename Base::Data propagate(typename Base::Data input, bool training) {
96  assert(input_dims == (Dimensions<std::size_t,Base::DATA_RANK>(input.dimensions()).template demote<2>()));
97  std::size_t batch_size = input.dimension(0);
98  std::size_t seq_length = input.dimension(1);
99  joint_input_dims[0] = batch_size * seq_length;
100  split_output_dims[0] = batch_size;
101  split_output_dims[1] = seq_length;
102  TensorMap<Scalar,Rank + 1> joint_input(input.data(), joint_input_dims);
103  Tensor<Scalar,Rank + 1> out = net->propagate(joint_input, training);
104  return TensorMap<Scalar,Rank + 2>(out.data(), split_output_dims);
105  }
106  inline typename Base::Data backpropagate(typename Base::Data out_grad) {
107  assert(output_dims == (Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<2>()));
108  assert(split_output_dims[0] == out_grad.dimension(0));
109  std::size_t batch_size = out_grad.dimension(0);
110  std::size_t seq_length = out_grad.dimension(1);
111  joint_output_dims[0] = batch_size * seq_length;
112  TensorMap<Scalar,Rank + 1> joint_out_grad(out_grad.data(), joint_output_dims);
113  if (foremost) {
114  net->backpropagate(joint_out_grad);
115  return typename Base::Data();
116  } else {
117  Tensor<Scalar,Rank + 1> prev_out_grad = net->backpropagate(joint_out_grad);
118  split_input_dims[0] = batch_size;
119  split_input_dims[1] = seq_length;
120  return TensorMap<Scalar,Rank + 2>(prev_out_grad.data(), split_input_dims);
121  }
122  }

131  swap(network1.split_input_dims, network2.split_input_dims);
132  swap(network1.split_output_dims, network2.split_output_dims);
133  }

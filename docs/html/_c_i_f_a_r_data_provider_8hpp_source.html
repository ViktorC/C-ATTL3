53  inline DataPair<Scalar,3,false> _get_data(const std::string& file_name, std::ifstream& file_stream,
54  std::size_t batch_size) {
55  typename Root::Data obs(batch_size, Base::obs_dims(0), Base::obs_dims(1), Base::obs_dims(2));
56  typename Root::Data obj(batch_size, Base::obj_dims(0), Base::obj_dims(1), Base::obj_dims(2));
57  obj.setZero();
58  std::size_t i;
59  for (i = 0; i < batch_size && file_stream.read(buffer, INSTANCE_LENGTH); ++i) {
60  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(buffer);
61  std::size_t buffer_ind = 0;
62  // Set the label.
63  if (CIFARType == CIFAR_100)
64  buffer_ind++;
65  obj(i,u_buffer[buffer_ind++],0u,0u) = (Scalar) 1;
66  // Set the image.
67  for (std::size_t channel = 0; channel < Base::obs_dims(2); ++channel) {
68  for (std::size_t height = 0; height < Base::obs_dims(1); ++height) {
69  for (std::size_t width = 0; width < Base::obj_dims(0); ++width)
70  obs(i,height,width,channel) = (Scalar) u_buffer[buffer_ind++];
71  }
72  }
73  assert(buffer_ind == INSTANCE_LENGTH);
74  }
75  if (i == batch_size)
76  return std::make_pair(obs, obj);
77  obs_extents[0] = i;
78  obj_extents[0] = i;
79  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
80  }
81  inline std::size_t _skip(std::ifstream& file_stream, std::size_t instances) {
82  std::streampos curr_pos = file_stream.tellg();
83  file_stream.seekg(0, std::ios::end);
84  std::size_t skip_extent = file_stream.tellg() - curr_pos;
85  file_stream.seekg(curr_pos);
86  file_stream.ignore(instances * INSTANCE_LENGTH);
87  return std::min(instances, skip_extent / INSTANCE_LENGTH);
88  }
89 private:

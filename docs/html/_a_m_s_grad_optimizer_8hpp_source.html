50  inline void _update_params(const std::vector<Parameters<Scalar>*>& params_vec, std::size_t epoch, std::size_t timestep) {
51  std::size_t i = 0;
52  for (auto params_ptr : params_vec) {
53  if (!params_ptr->are_optimizable() || params_ptr->are_frozen())
54  continue;
55  typename Base::ParamsGradNorms& grad_norms = Base::pgn_vec[i];
56  const Matrix<Scalar>& params_grad = params_ptr->get_grad();
57  grad_norms.params_grad_l1 = grad_norms.params_grad_l1 * (1 - Base::l1_decay) +
58  params_grad * Base::l1_decay;
59  grad_norms.params_grad_l2 = grad_norms.params_grad_l2 * (1 - Base::l2_decay) +
60  params_grad.cwiseProduct(params_grad) * Base::l2_decay;
61  Matrix<Scalar>& params_grad_l2_max = this->params_grad_l2_max[i++];
62  params_grad_l2_max = grad_norms.params_grad_l2.cwiseMax(params_grad_l2_max);
63  params_ptr->set_values(params_ptr->get_values() -
64  (grad_norms.params_grad_l1.array() * Base::learning_rate /
65  (params_grad_l2_max.array() + Base::epsilon).sqrt()).matrix());
66  }
67  }
68 private:

33  inline StackedNeuralNetwork(std::vector<Block>&& blocks, bool foremost = true) :
34  blocks(std::move(blocks)),
35  foremost(foremost) {
36  assert(this->blocks.size() > 0 && "blocks must contain at least 1 element");
37  assert(this->blocks[0] != nullptr && "blocks contains null pointers");
38  Base& first_block = *this->blocks[0];
39  input_dims = first_block.get_input_dims();
40  output_dims = this->blocks[this->blocks.size() - 1]->get_output_dims();
41  typename Base::Dims prev_dims = first_block.get_output_dims();
42  for (std::size_t i = 1; i < this->blocks.size(); ++i) {
43  assert(this->blocks[i] != nullptr && "blocks contains null pointers");
44  Base& block = *this->blocks[i];
45  assert(prev_dims == block.get_input_dims() && "incompatible network dimensions");
46  block.set_foremost(false);
47  prev_dims = block.get_output_dims();
48  }
49  first_block.set_foremost(foremost);
50  }

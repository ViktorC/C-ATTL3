71  Matrix<Scalar> prev_out_grad_mat(in_mat_cache.rows(), in_mat_cache.cols());
72  for (int i = 0; i < in_mat_cache.cols(); ++i) {
73  for (int j = 0; j < in_mat_cache.rows(); ++j)
74  prev_out_grad_mat(j,i) = (Scalar) ((in_mat_cache(j,i) >= 0 ?
75  1 : (out_mat_cache(j,i) + alpha)) * out_grad_mat(j,i));

78  }

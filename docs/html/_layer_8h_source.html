365  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
366  assert(Utils<Scalar>::template get_dims<4>(in).template demote<>() == Base::input_dims);
367  assert(in.dimension(0) > 0);
368  // Spatial padding.
369  std::array<std::pair<std::size_t,std::size_t>,4> paddings;
370  paddings[0] = std::make_pair(0, 0);
371  paddings[1] = std::make_pair(padding, padding);
372  paddings[2] = std::make_pair(padding, padding);
373  paddings[3] = std::make_pair(0, 0);
374  typename Root::Data padded_in = in.pad(paddings);
375  // Free the memory occupied by the now-expendable input tensor.
376  in = typename Root::Data();
377  std::size_t rows = padded_in.dimension(0);
378  std::size_t patches = Base::output_dims(0) * Base::output_dims(1);
379  std::size_t receptor_vol = receptor_size * receptor_size * Base::input_dims(2);
380  std::size_t height_rem = padded_height - dil_receptor_size;
381  std::size_t width_rem = padded_width - dil_receptor_size;
382  // Prepare the base offsets and extents for slicing and dilation.
383  RankwiseArray row_offsets({ 0, 0, 0, 0 });
384  RankwiseArray row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
385  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, padded_in.dimension(3) });
386  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
387  RankwiseArray dil_strides({ 1, dilation + 1, dilation + 1, 1 });
388  typename Root::Data out(padded_in.dimension(0), Base::output_dims(0), Base::output_dims(1), Base::output_dims(2));
389  biased_in_vec = std::vector<Matrix<Scalar>>(padded_in.dimension(0));
390  /* 'Tensor-row' by 'tensor-row', stretch the receptor locations into row vectors, form a matrix out of
391  * them, and multiply it by the weight matrix. */
392  for (std::size_t i = 0; i < rows; ++i) {
393  row_offsets[0] = i;
394  patch_offsets[0] = i;
395  std::size_t patch_ind = 0;
396  {
397  Matrix<Scalar> in_mat_i(patches, receptor_vol);
398  for (std::size_t j = 0; j <= height_rem; j += stride) {
399  patch_offsets[1] = j;
400  for (std::size_t k = 0; k <= width_rem; k += stride) {
401  patch_offsets[2] = k;
402  typename Root::Data patch;
403  // If the patch is dilated, skip the 'internal padding' when stretching it into a row.
404  if (dilation > 0)
405  patch = padded_in.slice(patch_offsets, patch_extents).stride(dil_strides);
406  else
407  patch = padded_in.slice(patch_offsets, patch_extents);
408  in_mat_i.row(patch_ind++) = Utils<Scalar>::template map_tensor_to_mat<4>(std::move(patch));
409  }
410  }
411  assert(patch_ind == patches);
412  // Set the appended column's elements to 1.
413  biased_in_vec[i] = Matrix<Scalar>(patches, receptor_vol + 1);
414  Matrix<Scalar>& biased_in = biased_in_vec[i];
415  biased_in.col(receptor_vol).setOnes();
416  biased_in.leftCols(receptor_vol) = in_mat_i;
417  }
418  /* Flatten the matrix product into a row vector, reshape it into a 'single-row' sub-tensor, and
419  * assign it to the output tensor's corresponding 'row'. */
420  Matrix<Scalar> out_i = biased_in_vec[i] * Base::weights_ref;
421  out.slice(row_offsets, row_extents) = Utils<Scalar>::template map_mat_to_tensor<4>(MatrixMap<Scalar>(out_i.data(),
422  1, out_i.rows() * out_i.cols()).matrix(), Base::output_dims);
423  }
424  return out;
425  }
426  inline typename Root::Data pass_back(typename Root::Data out_grads) {
427  assert(Utils<Scalar>::template get_dims<4>(out_grads).template demote<>() == Base::output_dims);
428  assert(out_grads.dimension(0) > 0 && biased_in_vec.size() == (unsigned) out_grads.dimension(0));
429  std::size_t rows = out_grads.dimension(0);
430  std::size_t depth = Base::input_dims(2);
431  std::size_t height_rem = padded_height - dil_receptor_size;
432  std::size_t width_rem = padded_width - dil_receptor_size;
433  Dimensions<std::size_t,3> comp_patch_dims({ receptor_size, receptor_size, Base::input_dims(2) });
434  RankwiseArray out_grads_row_offsets({ 0, 0, 0, 0 });
435  RankwiseArray out_grads_row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
436  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, depth });
437  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
438  RankwiseArray strides({ 1, dilation + 1, dilation + 1, 1 });
439  typename Root::Data prev_out_grads(rows, padded_height, padded_width, depth);
440  prev_out_grads.setZero();
441  Base::weights_grad.setZero(Base::weights_grad.rows(), Base::weights_grad.cols());
442  for (std::size_t i = 0; i < rows; ++i) {
443  out_grads_row_offsets[0] = i;
444  Matrix<Scalar> prev_out_grads_mat_i;
445  {
446  typename Root::Data slice_i = out_grads.slice(out_grads_row_offsets, out_grads_row_extents).eval();
447  MatrixMap<Scalar> out_grads_mat_map_i = MatrixMap<Scalar>(slice_i.data(), Base::output_dims(0) *
448  Base::output_dims(1), filters);
449  // Accumulate the gradients across the observations.
450  Base::weights_grad += biased_in_vec[i].transpose() * out_grads_mat_map_i;
451  if (Base::is_input_layer())
452  continue;
453  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
454  * previous layer's output. */
455  prev_out_grads_mat_i = out_grads_mat_map_i * Base::weights_ref.topRows(Base::weights_ref.rows() - 1).transpose();
456  }
457  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
458  * to get the derivative w.r.t. the individual input nodes. */
459  std::size_t patch_ind = 0;
460  patch_offsets[0] = i;
461  for (std::size_t j = 0; j <= height_rem; j += stride) {
462  patch_offsets[1] = j;
463  for (std::size_t k = 0; k <= width_rem; k += stride) {
464  patch_offsets[2] = k;
465  // Accumulate the gradients where the receptor-patch-tensors overlap.
466  if (dilation > 0) {
467  prev_out_grads.slice(patch_offsets, patch_extents).stride(strides) +=
468  Utils<Scalar>::template map_mat_to_tensor<4>(prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
469  } else {
470  prev_out_grads.slice(patch_offsets, patch_extents) += Utils<Scalar>::template map_mat_to_tensor<4>(
471  prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
472  }
473  }
474  }
475  assert(patch_ind == prev_out_grads_mat_i.rows());
476  }
477  // Cut off the padding.
478  RankwiseArray no_padding_offsets({ 0, padding, padding, 0 });
479  RankwiseArray no_padding_extents({ rows, Base::input_dims(0), Base::input_dims(1), Base::input_dims(2) });
480  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
481  }
492  static std::size_t calculate_output_dim(std::size_t input_dim, std::size_t receptor_size, std::size_t padding,
493  std::size_t dilation, std::size_t stride) {
494  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
495  }
496 private:

498  const std::size_t receptor_size;
499  const std::size_t padding;
500  const std::size_t stride;
501  const std::size_t dilation;
502  const std::size_t padded_height;
503  const std::size_t padded_width;
504  const std::size_t dil_receptor_size;
505  // Staged computation caches
506  std::vector<Matrix<Scalar>> biased_in_vec;
507 };

1316  .rowwise() + params_ref.row(2 * i + 1)).eval(), output_dims);
1317  }
1318  inline typename Base::Data _pass_back(typename Base::Data out_grads, const Dimensions<std::size_t,Rank>& output_dims,
1319  int i) {
1320  std::size_t rows = out_grads.dimension(0);
1321  Cache& cache = cache_vec[i];

1323  /* Back-propagate the gradient through the batch normalization 'function' and also calculate the
1324  * gradients of the betas and gammas. */
1325  { // Manage memory by scope restriction.
1326  Matrix<Scalar> out_grads_ch_map_i = Utils<Scalar>::template map_tensor_to_mat<Base::DATA_RANK>(std::move(out_grads));
1327  params_grad.row(2 * i) = out_grads_ch_map_i.cwiseProduct(cache.std_in).colwise().sum();
1328  params_grad.row(2 * i + 1) = out_grads_ch_map_i.colwise().sum();
1329  if (input_layer)
1330  return typename Base::Data();
1331  std_in_grads_i = out_grads_ch_map_i * params_ref.row(2 * i).asDiagonal();
1332  }
1333  return Utils<Scalar>::template map_mat_to_tensor<Base::DATA_RANK>(((((rows * std_in_grads_i).rowwise() -
1334  std_in_grads_i.colwise().sum()) - cache.std_in *
1335  (cache.std_in.cwiseProduct(std_in_grads_i).colwise().sum().asDiagonal())) *
1336  ((1.0 / rows) * cache.inv_in_sd).asDiagonal()).eval(), output_dims);
1337  }

1339  const std::size_t depth;
1340  const Scalar norm_avg_decay;
1341  const Scalar epsilon;

1343  // Dynamic batch normalization parameters.
1344  Matrix<Scalar> avg_means;
1345  Matrix<Scalar> avg_inv_sds;
1346  bool avgs_init;
1347  // Betas and gammas
1348  Matrix<Scalar> params_grad;
1349  Matrix<Scalar>& params_ref;

1351  struct Cache {
1352  RowVector<Scalar> inv_in_sd;
1353  Matrix<Scalar> std_in;
1354  };
1355  std::vector<Cache> cache_vec;

1455  if (Base::is_input_layer())

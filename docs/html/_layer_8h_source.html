401  assert(receptor_height > 0);
402  assert(receptor_width > 0);

434  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
435  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
436  assert(in.dimension(0) > 0);
437  // Spatial padding.
438  if (padding > 0)
439  in = typename Root::Data(in.pad(paddings));
440  std::size_t rows = in.dimension(0);
441  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
442  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
443  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
444  * out receptor of one sample. The same receptor location along all samples of the batch is represented
445  * by a contiguous block of these rows. There is one block for each receptor location. */
446  std::size_t patch_ind = 0;
447  patch_extents[0] = rows;
448  biased_in = Matrix<Scalar>(total_patches, receptor_vol + 1);
449  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += stride) {
450  patch_offsets[1] = j;
451  for (std::size_t k = 0; k <= padded_width - dil_receptor_width; k += stride) {
452  patch_offsets[2] = k;
453  typename Root::Data patch;
454  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
455  if (dilation > 0)
456  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
457  else
458  patch = in.slice(patch_offsets, patch_extents);
459  biased_in.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
460  patch_ind += rows;
461  }
462  }
463  assert(patch_ind == total_patches);
464  // Bias trick.
465  biased_in.col(receptor_vol).setOnes();
466  Matrix<Scalar> out = biased_in * Base::weights_ref;
467  out_conversion_dims[0] = rows;
468  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
469  }
470  inline typename Root::Data pass_back(typename Root::Data out_grads) {
471  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
472  assert(out_grads.dimension(0) > 0 && biased_in.rows() / (Base::output_dims(0) *
473  Base::output_dims(1)) == out_grads.dimension(0));
474  std::size_t rows = out_grads.dimension(0);
475  std::size_t patches = Base::output_dims(0) * Base::output_dims(1);
476  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
477  MatrixMap<Scalar> out_grads_mat_map = MatrixMap<Scalar>(out_grads.data(),
478  rows * Base::output_dims(0) * Base::output_dims(1), filters);
479  Base::weights_grad = biased_in.transpose() * out_grads_mat_map;
480  if (Base::is_input_layer())
481  return typename Root::Data();
482  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
483  * previous layer's output. */
484  Matrix<Scalar> prev_out_grads_mat = out_grads_mat_map * Base::weights_ref.topRows(receptor_vol).transpose();
485  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
486  * to get the derivative w.r.t. the individual input nodes. */
487  typename Root::Data prev_out_grads(rows, padded_height, padded_width, Base::input_dims(2));
488  prev_out_grads.setZero();
489  std::size_t patch_ind = 0;
490  patch_extents[0] = rows;
491  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += stride) {
492  patch_offsets[1] = j;
493  for (std::size_t k = 0; k <= padded_width - dil_receptor_width; k += stride) {
494  patch_offsets[2] = k;
495  // Accumulate the gradients where the receptor-patch-tensors overlap.
496  Matrix<Scalar> prev_out_grads_block = prev_out_grads_mat.block(patch_ind, 0, rows, receptor_vol);
497  TensorMap<Scalar,4> prev_out_grads_block_map(prev_out_grads_block.data(), patch_extents);
498  if (dilation > 0)
499  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_block_map;
500  else
501  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_block_map;
502  patch_ind += rows;
503  }
504  }
505  assert(patch_ind == prev_out_grads_mat.rows());
506  if (padding > 0) {
507  // Cut off the padding.
508  RankwiseArray no_padding_offsets({ 0, padding, padding, 0 });
509  RankwiseArray no_padding_extents({ rows, Base::input_dims(0), Base::input_dims(1), Base::input_dims(2) });
510  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
511  } else
512  return prev_out_grads;
513  }

526  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
527  }

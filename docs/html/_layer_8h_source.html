244  assert(in.dimension(0) > 0);
245  unsigned input_size = Base::input_dims.get_volume();

247  biased_in = Matrix<Scalar>(in.dimension(0), input_size + 1);
248  biased_in.leftCols(input_size) = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(in));
249  biased_in.col(input_size).setOnes();
250  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>((biased_in * Base::weights).eval(), Base::output_dims);
251  }
252  inline Data pass_back(Data out_grads) {
253  assert(Utils<Scalar>::template get_dims<Rank + 1>(out_grads).template demote<>() == Base::output_dims);
254  assert(out_grads.dimension(0) > 0 && biased_in.rows() == out_grads.dimension(0));
255  Matrix<Scalar> out_grads_mat = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(out_grads));
256  // Compute the gradient of the outputs with respect to the weights.
257  Base::weights_grad = biased_in.transpose() * out_grads_mat;
258  if (Base::is_input_layer())
259  return Data();
260  /* Remove the bias row from the weight matrix, transpose it, and compute the derivative w.r.t. the
261  * previous layer's output. */
262  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>((out_grads_mat *
263  Base::weights.topRows(Base::input_dims.get_volume()).transpose()).eval(), Base::input_dims);
264  }
265 private:
266  // Staged computation caches
267  Matrix<Scalar> biased_in;
268 };

299  * elements as the size of the receptive field (F * F * D) + 1 for the bias row. */

309  padded_width(input_dims(1) + 2 * padding),
310  dil_receptor_size(receptor_size + (receptor_size - 1) * dilation) {
311  assert(filters > 0);
312  assert(receptor_size > 0);
313  assert(stride > 0);
314  assert(input_dims(1) + 2 * padding >= receptor_size + (receptor_size - 1) * dilation &&
315  input_dims(2) + 2 * padding >= receptor_size + (receptor_size - 1) * dilation);
316  }
317  inline Layer<Scalar,3>* clone() const {
318  return new ConvLayer(*this);
319  }
320 protected:
321  inline void empty_cache() {
322  biased_in_vec = std::vector<Matrix<Scalar>>(0);
323  }
324  inline Data pass_forward(Data in, bool training) {
325  assert(Utils<Scalar>::template get_dims<4>(in).template demote<>() == Base::input_dims);
326  assert(in.dimension(0) > 0);
327  // Spatial padding.
328  std::array<std::pair<int,int>,4> paddings;
329  paddings[0] = std::make_pair(0, 0);
330  paddings[1] = std::make_pair(padding, padding);
331  paddings[2] = std::make_pair(padding, padding);
332  paddings[3] = std::make_pair(0, 0);
333  Data padded_in = in.pad(paddings);
334  // Free the memory occupied by the now-expendable input tensor.
335  in = Data();
336  int rows = padded_in.dimension(0);
337  int patches = Base::output_dims(0) * Base::output_dims(1);
338  int receptor_vol = receptor_size * receptor_size * Base::input_dims(2);
339  int height_rem = padded_height - dil_receptor_size;
340  int width_rem = padded_width - dil_receptor_size;
341  // Prepare the base offsets and extents for slicing and dilation.
342  RankwiseArray row_offsets({ 0, 0, 0, 0 });
343  RankwiseArray row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
344  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, padded_in.dimension(3) });
345  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
346  RankwiseArray dil_strides({ 1, (int) dilation + 1, (int) dilation + 1, 1 });
347  Data out(padded_in.dimension(0), Base::output_dims(0), Base::output_dims(1), Base::output_dims(2));
348  biased_in_vec = std::vector<Matrix<Scalar>>(padded_in.dimension(0));
349  /* 'Tensor-row' by 'tensor-row', stretch the receptor locations into row vectors, form a matrix out of
350  * them, and multiply it by the weight matrix. */
351  for (int i = 0; i < rows; ++i) {
352  row_offsets[0] = i;
353  patch_offsets[0] = i;
354  int patch_ind = 0;
355  {
356  Matrix<Scalar> in_mat_i(patches, receptor_vol);
357  for (int j = 0; j <= height_rem; j += stride) {
358  patch_offsets[1] = j;
359  for (int k = 0; k <= width_rem; k += stride) {
360  patch_offsets[2] = k;
361  Data patch;
362  // If the patch is dilated, skip the 'internal padding' when stretching it into a row.
363  if (dilation > 0)
364  patch = padded_in.slice(patch_offsets, patch_extents).stride(dil_strides);
365  else
366  patch = padded_in.slice(patch_offsets, patch_extents);
367  in_mat_i.row(patch_ind++) = Utils<Scalar>::template map_tensor_to_mat<4>(std::move(patch));
368  }
369  }
370  assert(patch_ind == patches);
371  // Set the appended column's elements to 1.
372  biased_in_vec[i] = Matrix<Scalar>(patches, receptor_vol + 1);
373  Matrix<Scalar>& biased_in = biased_in_vec[i];
374  biased_in.col(receptor_vol).setOnes();
375  biased_in.leftCols(receptor_vol) = in_mat_i;
376  }
377  /* Flatten the matrix product into a row vector, reshape it into a 'single-row' sub-tensor, and
378  * assign it to the output tensor's corresponding 'row'. */
379  Matrix<Scalar> out_i = biased_in_vec[i] * Base::weights;
380  out.slice(row_offsets, row_extents) = Utils<Scalar>::template map_mat_to_tensor<4>(MatrixMap<Scalar>(out_i.data(),
381  1, out_i.rows() * out_i.cols()).matrix(), Base::output_dims);
382  }
383  return out;
384  }
385  inline Data pass_back(Data out_grads) {
386  assert(Utils<Scalar>::template get_dims<4>(out_grads).template demote<>() == Base::output_dims);
387  assert(out_grads.dimension(0) > 0 && biased_in_vec.size() == (unsigned) out_grads.dimension(0));
388  int rows = out_grads.dimension(0);
389  int depth = Base::input_dims(2);
390  int height_rem = padded_height - dil_receptor_size;
391  int width_rem = padded_width - dil_receptor_size;
392  Dimensions<int,3> comp_patch_dims({ (int) receptor_size, (int) receptor_size, Base::input_dims(2) });
393  RankwiseArray out_grads_row_offsets({ 0, 0, 0, 0 });
394  RankwiseArray out_grads_row_extents({ 1, Base::output_dims(0), Base::output_dims(1), Base::output_dims(2) });
395  RankwiseArray patch_extents({ 1, dil_receptor_size, dil_receptor_size, depth });
396  RankwiseArray patch_offsets({ 0, 0, 0, 0 });
397  RankwiseArray strides({ 1, (int) dilation + 1, (int) dilation + 1, 1 });
398  Data prev_out_grads(rows, padded_height, padded_width, depth);
399  prev_out_grads.setZero();
400  Base::weights_grad.setZero(Base::weights_grad.rows(), Base::weights_grad.cols());
401  for (int i = 0; i < rows; ++i) {
402  out_grads_row_offsets[0] = i;
403  Matrix<Scalar> prev_out_grads_mat_i;
404  {
405  Data slice_i = out_grads.slice(out_grads_row_offsets, out_grads_row_extents).eval();
406  MatrixMap<Scalar> out_grads_mat_map_i = MatrixMap<Scalar>(slice_i.data(), Base::output_dims(0) *
407  Base::output_dims(1), filters);
408  // Accumulate the gradients across the observations.
409  Base::weights_grad += biased_in_vec[i].transpose() * out_grads_mat_map_i;
410  if (Base::is_input_layer())
411  continue;
412  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
413  * previous layer's output. */
414  prev_out_grads_mat_i = out_grads_mat_map_i * Base::weights.topRows(Base::weights.rows() - 1).transpose();
415  }
416  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
417  * to get the derivative w.r.t. the individual input nodes. */
418  int patch_ind = 0;
419  patch_offsets[0] = i;
420  for (int j = 0; j <= height_rem; j += stride) {
421  patch_offsets[1] = j;
422  for (int k = 0; k <= width_rem; k += stride) {
423  patch_offsets[2] = k;
424  // Accumulate the gradients where the receptor-patch-tensors overlap.
425  if (dilation > 0) {
426  prev_out_grads.slice(patch_offsets, patch_extents).stride(strides) +=
427  Utils<Scalar>::template map_mat_to_tensor<4>(prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
428  } else {
429  prev_out_grads.slice(patch_offsets, patch_extents) += Utils<Scalar>::template map_mat_to_tensor<4>(
430  prev_out_grads_mat_i.row(patch_ind++), comp_patch_dims);
431  }
432  }
433  }
434  assert(patch_ind == prev_out_grads_mat_i.rows());
435  }
436  // Cut off the padding.
437  RankwiseArray no_padding_offsets({ 0, (int) padding, (int) padding, 0 });
438  RankwiseArray no_padding_extents({ rows, Base::input_dims(0), Base::input_dims(1), Base::input_dims(2) });
439  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
440  }

452  return (input_dim - receptor_size - (receptor_size - 1) * dilation + 2 * padding) / stride + 1;
453  }

456  const std::size_t receptor_size;

458  const std::size_t stride;
459  const std::size_t dilation;
460  const int padded_height;
461  const int padded_width;
462  const int dil_receptor_size;

464  std::vector<Matrix<Scalar>> biased_in_vec;

680 template<typename Scalar, std::size_t Rank>
681 class SoftmaxActivationLayer : public ActivationLayer<Scalar,Rank> {
682 public:
687  inline SoftmaxActivationLayer(const Dimensions<int,Rank>& dims, Scalar epsilon = Utils<Scalar>::EPSILON2) :
688  ActivationLayer<Scalar,Rank>::ActivationLayer(dims),
689  epsilon(epsilon) { }
690  inline Layer<Scalar,Rank>* clone() const {
691  return new SoftmaxActivationLayer(*this);
692  }
693 protected:
694  inline Matrix<Scalar> activate(const Matrix<Scalar>& in) {
695  /* First subtract the value of the greatest coefficient from each element row-wise
696  * to avoid an overflow due to raising e to great powers. */
697  Matrix<Scalar> out = (in.array().colwise() - in.array().rowwise().maxCoeff()).exp();
698  return out.array().colwise() / (out.array().rowwise().sum() + epsilon);
699  }
700  inline Matrix<Scalar> d_activate(const Matrix<Scalar>& in, const Matrix<Scalar>& out,
701  const Matrix<Scalar>& out_grads) {
702  Matrix<Scalar> d_in(in.rows(), in.cols());
703  for (int i = 0; i < d_in.rows(); ++i) {
704  RowVector<Scalar> row_i = out.row(i);
705  auto jacobian = (row_i.asDiagonal() - (row_i.transpose() * row_i));

1163  params.row(i + 1).setZero();

1185  Matrix<Scalar> in_ch_i = Utils<Scalar>::template map_tensor_to_mat<Rank + 1>(std::move(in));
1186  if (training) {
1187  Cache& cache = cache_vec[i];
1188  RowVector<Scalar> means = in_ch_i.colwise().mean();
1189  Matrix<Scalar> norm_in = in_ch_i.rowwise() - means;
1190  cache.inv_in_sd = (norm_in.array().square().colwise().mean() + epsilon).sqrt().inverse();
1191  cache.std_in = norm_in * cache.inv_in_sd.asDiagonal();
1192  in_ch_i = cache.std_in;
1193  // Maintain a moving average of means and variances for testing.
1194  if (avgs_init) {

1197  cache.inv_in_sd;
1198  } else {
1199  avg_means.row(i) = means;
1200  avg_inv_sds.row(i) = cache.inv_in_sd;
1201  avgs_init = true;
1202  }
1203  } else // For testing, use the moving averages.
1204  in_ch_i = (in_ch_i.rowwise() - avg_means.row(i)) * avg_inv_sds.row(i).asDiagonal();
1205  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(((in_ch_i * params.row(2 * i).asDiagonal())

1216  params_grad.row(2 * i) = out_grads_ch_map_i.cwiseProduct(cache.std_in).colwise().sum();
1217  params_grad.row(2 * i + 1) = out_grads_ch_map_i.colwise().sum();
1218  if (input_layer)
1219  return Data();
1220  std_in_grads_i = out_grads_ch_map_i * params.row(2 * i).asDiagonal();
1221  }
1222  return Utils<Scalar>::template map_mat_to_tensor<Rank + 1>(((((rows * std_in_grads_i).rowwise() -
1223  std_in_grads_i.colwise().sum()) - cache.std_in *
1224  (cache.std_in.cwiseProduct(std_in_grads_i).colwise().sum().asDiagonal())) *
1225  ((1.0 / rows) * cache.inv_in_sd).asDiagonal()).eval(), output_dims);
1226  }

1228  const int depth;
1229  const Scalar norm_avg_decay;

1237  Matrix<Scalar> params;

470  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
471  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == Base::input_dims);
472  assert(in.dimension(0) > 0);
473  // Spatial padding.
474  if (vertical_padding > 0 || horizontal_padding > 0)
475  in = typename Root::Data(in.pad(paddings));
476  std::size_t rows = in.dimension(0);
477  std::size_t total_patches = rows * Base::output_dims(0) * Base::output_dims(1);
478  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
479  /* Flatten the receptor cuboids into row vectors and concatenate them. Each row stands for one stretched
480  * out receptor of one sample. The same receptor location along all samples of the batch is represented
481  * by a contiguous block of these rows. There is one block for each receptor location. */
482  std::size_t patch_ind = 0;
483  patch_extents[0] = rows;
484  biased_in = Matrix<Scalar>(total_patches, receptor_vol + 1);
485  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
486  patch_offsets[2] = i;
487  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
488  patch_offsets[1] = j;
489  typename Root::Data patch;
490  // If the patch is dilated, skip the 'internal padding' when flattening it into a matrix.
491  if (dilation > 0)
492  patch = in.slice(patch_offsets, patch_extents).stride(dil_strides);
493  else
494  patch = in.slice(patch_offsets, patch_extents);
495  biased_in.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(), rows, receptor_vol);
496  patch_ind += rows;
497  }
498  }
499  assert(patch_ind == total_patches);
500  // Bias trick.
501  biased_in.col(receptor_vol).setOnes();
502 #ifndef CATTLE_USE_CUBLAS
503  Matrix<Scalar> out = biased_in * Base::weights_ref;
504 #else
505  Matrix<Scalar> out = internal::CuBLASUtils<Scalar>::mul(biased_in, Base::weights_ref, false, false);
506 #endif
507  out_conversion_dims[0] = rows;
508  return TensorMap<Scalar,4>(out.data(), out_conversion_dims);
509  }

511  assert((Dimensions<std::size_t,4>(out_grads.dimensions()).template demote<>()) == Base::output_dims);
512  assert(out_grads.dimension(0) > 0 && biased_in.rows() / (Base::output_dims(0) *
513  Base::output_dims(1)) == out_grads.dimension(0));
514  std::size_t rows = out_grads.dimension(0);
515  std::size_t patches = Base::output_dims(0) * Base::output_dims(1);
516  std::size_t receptor_vol = Base::weights_ref.rows() - 1;
517 #ifndef CATTLE_USE_CUBLAS
518  MatrixMap<Scalar> out_grads_mat(out_grads.data(), rows * Base::output_dims(0) * Base::output_dims(1), filters);
519  Base::weights_grad = biased_in.transpose() * out_grads_mat;
520  if (Base::is_input_layer())
521  return typename Root::Data();
522  /* Remove the bias row from the weight matrix, transpose it, and compute the gradient of the
523  * previous layer's output. */
524  Matrix<Scalar> prev_out_grads_mat = out_grads_mat * Base::weights_ref.topRows(receptor_vol).transpose();
525 #else

528  Base::weights_grad = internal::CuBLASUtils<Scalar>::mul(biased_in, out_grads_mat, true, false);
529  if (Base::is_input_layer())
530  return typename Root::Data();

532  Matrix<Scalar> prev_out_grads_mat = internal::CuBLASUtils<Scalar>::mul(out_grads_mat,
533  weights_without_bias, false, true);
534 #endif
535  /* Given the gradient of the stretched out receptor patches, perform a 'backwards' convolution
536  * to get the derivative w.r.t. the individual input nodes. */
537  typename Root::Data prev_out_grads(rows, padded_height, padded_width, Base::input_dims(2));
538  prev_out_grads.setZero();
539  std::size_t patch_ind = 0;
540  patch_extents[0] = rows;
541  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
542  patch_offsets[2] = i;
543  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
544  patch_offsets[1] = j;
545  // Accumulate the gradients where the receptor-patch-tensors overlap.
546  Matrix<Scalar> prev_out_grads_block = prev_out_grads_mat.block(patch_ind, 0, rows, receptor_vol);
547  TensorMap<Scalar,4> prev_out_grads_block_map(prev_out_grads_block.data(), patch_extents);
548  if (dilation > 0)
549  prev_out_grads.slice(patch_offsets, patch_extents).stride(dil_strides) += prev_out_grads_block_map;
550  else
551  prev_out_grads.slice(patch_offsets, patch_extents) += prev_out_grads_block_map;
552  patch_ind += rows;
553  }
554  }
555  assert(patch_ind == prev_out_grads_mat.rows());

557  // Cut off the padding.
558  no_padding_extents[0] = rows;
559  return prev_out_grads.slice(no_padding_offsets, no_padding_extents);
560  } else
561  return prev_out_grads;
562  }

925  }

931  /* First subtract the value of the greatest coefficient from each element row-wise

949  RowVector<Scalar> row_i = out.row(i);
950  // FIXME Do not evaluate the expressions into a temporary variable.
951  Matrix<Scalar> jacobian = row_i.asDiagonal();
952  jacobian -= row_i.transpose() * row_i;
953  prev_out_grads.row(i) = out_grads_mat.row(i) * jacobian;

53  inline void _update_params(const std::vector<Parameters<Scalar>*>& params_vec, std::size_t epoch) {
54  std::size_t i = 0;
55  for (auto params_ptr : params_vec) {
56  if (!params_ptr->are_optimizable() || params_ptr->are_frozen())
57  continue;
58  ParamsGradAndUpdateSqrs& pgus = pgus_vec[i++];
59  const Matrix<Scalar>& params_grad = params_ptr->get_grad();
60  pgus.params_grad = pgus.params_grad * (1 - decay) + params_grad.cwiseProduct(params_grad) * decay;
61  Matrix<Scalar> weight_updates = -params_grad.array() * (pgus.params_update.array() + epsilon).sqrt() /
62  (pgus.params_grad.array() + epsilon).sqrt();
63  params_ptr->set_values(params_ptr->get_values() + weight_updates);
64  pgus.params_update = pgus.params_update * (1 - decay) +
65  weight_updates.cwiseProduct(weight_updates) * decay;
66  }
67  }

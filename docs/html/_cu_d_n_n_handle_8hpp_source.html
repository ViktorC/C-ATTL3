73  inline Array4 conv2d_output_dims(const Array4& input_dims, std::size_t filters, std::size_t receptor_height,
74  std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
75  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
76  std::size_t horizontal_dilation) {
77  // Create and set the input tensor descriptor.
78  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
79  // Create and set up the filter descriptor.
80  cudnnFilterDescriptor_t filter_desc;
81  cudnnAssert(cudnnCreateFilterDescriptor(&filter_desc));
82  cudnnAssert(cudnnSetFilter4dDescriptor(filter_desc, DATA_TYPE, TENSOR_FORMAT, filters, input_dims[3],
83  receptor_height, receptor_width));
84  // Create and set up the convolution descriptor.
85  cudnnConvolutionDescriptor_t conv_desc;
86  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
87  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
88  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION, DATA_TYPE));
89  // Compute the dimensions.
90  int n, c, h, w;
91  cudnnAssert(cudnnGetConvolution2dForwardOutputDim(conv_desc, input_desc, filter_desc, &n, &c, &h, &w));
92  // Free the resources.
93  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
94  cudnnAssert(cudnnDestroyFilterDescriptor(filter_desc));
95  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
96  return { (std::size_t) n, (std::size_t) h, (std::size_t) w, (std::size_t) c };
97  }
118  inline void convolution2d_fwd(Scalar* input, Scalar* filter, Scalar* bias, const Array4& input_dims, const Array4& output_dims,
119  std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
120  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation, std::size_t horizontal_dilation,
121  /* out */ Scalar* output) {
122  std::size_t depth = input_dims[3];
123  std::size_t filters = output_dims[3];
124  // Create and set up the input tensor descriptor.
125  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
126  // Create and set up the filter descriptor.
127  cudnnFilterDescriptor_t filter_desc;
128  cudnnAssert(cudnnCreateFilterDescriptor(&filter_desc));
129  cudnnAssert(cudnnSetFilter4dDescriptor(filter_desc, DATA_TYPE, TENSOR_FORMAT, filters, depth,
130  receptor_height, receptor_width));
131  // Create and set up the bias descriptor.
132  cudnnTensorDescriptor_t bias_desc = setup_tens_desc(1, filters, 1, 1);
133  // Create and set up the convolution descriptor.
134  cudnnConvolutionDescriptor_t conv_desc;
135  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
136  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
137  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION, DATA_TYPE));
138  // Create and set up the output tensor descriptor.
139  cudnnTensorDescriptor_t output_desc = setup_tens_desc(output_dims);
140  // Have cuDNN find the most performant algorithm given the convolution parameters.
141  cudnnConvolutionFwdAlgo_t conv_algo;
142  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(handle, input_desc, filter_desc, conv_desc, output_desc,
143  CUDNN_CONVOLUTION_FWD_PREFER_FASTEST, 0, &conv_algo));
144  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
145  * the convolution parameters. */
146  std::size_t workspace_size;
147  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(handle, input_desc, filter_desc, conv_desc, output_desc,
148  conv_algo, &workspace_size));
149  // Allocate the memory for the device arrays required for the convolution.
150  Scalar* dev_input;
151  Scalar* dev_filter;
152  Scalar* dev_workspace;
153  Scalar* dev_output;
154  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * depth * SCALAR_SIZE;
155  cudaAssert(cudaMalloc(&dev_input, input_size));
156  const std::size_t filter_size = receptor_height * receptor_width * depth * filters * SCALAR_SIZE;
157  cudaAssert(cudaMalloc(&dev_filter, filter_size));
158  cudaAssert(cudaMalloc(&dev_workspace, workspace_size));
159  const std::size_t output_size = output_dims[0] * output_dims[1] * output_dims[2] * filters * SCALAR_SIZE;
160  cudaAssert(cudaMalloc(&dev_output, output_size));
161  // Copy the contents of the input tensor and the filter to the respective device arrays.
162  cudaAssert(cudaMemcpy(dev_input, input, input_size, cudaMemcpyHostToDevice));
163  cudaAssert(cudaMemcpy(dev_filter, filter, filter_size, cudaMemcpyHostToDevice));
164  // Perform the convolution.
165  cudnnAssert(cudnnConvolutionForward(handle, &alpha, input_desc, dev_input, filter_desc, dev_filter,
166  conv_desc, conv_algo, dev_workspace, workspace_size, &beta, output_desc, dev_output));
167  // Free the convolution resources.
168  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
169  cudnnAssert(cudnnDestroyFilterDescriptor(filter_desc));
170  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
171  cudaAssert(cudaFree(dev_input));
172  cudaAssert(cudaFree(dev_workspace));
173  cudaAssert(cudaFree(dev_filter));
174  // Allocate the memory for the bias array on the device and fill it.
175  Scalar* dev_bias;
176  const std::size_t bias_size = filters * SCALAR_SIZE;
177  cudaAssert(cudaMalloc(&dev_bias, bias_size));
178  cudaAssert(cudaMemcpy(dev_bias, bias, bias_size, cudaMemcpyHostToDevice));
179  // Apply the bias to the output tensor.
180  cudnnAssert(cudnnAddTensor(handle, &alpha, bias_desc, dev_bias, &beta, output_desc, dev_output));
181  // Copy the output tensor to the host.
182  cudaAssert(cudaMemcpy(output, dev_output, output_size, cudaMemcpyDeviceToHost));
183  // Free up the remaining resources.
184  cudnnAssert(cudnnDestroyTensorDescriptor(bias_desc));
185  cudnnAssert(cudnnDestroyTensorDescriptor(output_desc));
186  cudaAssert(cudaFree(dev_bias));
187  cudaAssert(cudaFree(dev_output));
188  }
213  inline void convolution2d_bwd(Scalar* input, Scalar* out_grad, Scalar* filter, Scalar* bias, const Array4& input_dims,
214  const Array4& output_dims, std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding,
215  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
216  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ Scalar* prev_out_grad,
217  /* out */ Scalar* filter_grad, /* out */ Scalar* bias_grad) {
218  std::size_t depth = input_dims[3];
219  std::size_t filters = output_dims[3];
220  // Create and set up the input tensor descriptor.
221  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
222  // Create and set up the output gradient descriptor.
223  cudnnTensorDescriptor_t out_grad_desc = setup_tens_desc(output_dims);
224  // Create and set up the filter descriptor.
225  cudnnFilterDescriptor_t filter_desc;
226  cudnnAssert(cudnnCreateFilterDescriptor(&filter_desc));
227  cudnnAssert(cudnnSetFilter4dDescriptor(filter_desc, DATA_TYPE, TENSOR_FORMAT, filters, depth,
228  receptor_height, receptor_width));
229  // Create and set up the bias descriptor.
230  cudnnTensorDescriptor_t bias_desc = setup_tens_desc(1, filters, 1, 1);
231  // Create and set up the backward convolution descriptor.
232  cudnnConvolutionDescriptor_t dconv_desc;
233  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
234  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding, vertical_stride,
235  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION, DATA_TYPE));
236  // Have cuDNN find the most performant algorithm given the convolution parameters.
237  cudnnConvolutionFwdAlgo_t dconv_data_algo;
238  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(handle, filter_desc, out_grad_desc, dconv_desc,
239  input_desc, CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
240  cudnnConvolutionFwdAlgo_t dconv_filter_algo;
241  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(handle, input_desc, out_grad_desc, dconv_desc,
242  filter_desc, CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST, 0, &dconv_filter_algo));
243  /* Have cuDNN compute the workspace memory required for the selected backward convolution algorithms given
244  * the convolution parameters. */
245  std::size_t data_workspace_size;
246  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(handle, filter_desc, out_grad_desc, dconv_desc,
247  input_desc, dconv_data_algo, &data_workspace_size));
248  std::size_t filter_workspace_size;
249  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(handle, input_desc, out_grad_desc, dconv_desc,
250  filter_desc, dconv_filter_algo, &filter_workspace_size));
251  // Allocate the memory required for the backwards data convolution on the device.
252  Scalar* dev_filter;
253  Scalar* dev_out_grad;
254  Scalar* dev_data_workspace;
255  Scalar* dev_prev_out_grad;
256  const std::size_t filter_size = receptor_height * receptor_width * filters * SCALAR_SIZE;
257  cudaAssert(cudaMalloc(&dev_filter, filter_size));
258  const std::size_t out_grad_size = output_dims[0] * output_dims[1] * output_dims[2] * SCALAR_SIZE;
259  cudaAssert(cudaMalloc(&dev_out_grad, out_grad_size));
260  cudaAssert(cudaMalloc(&dev_data_workspace, data_workspace_size));
261  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * SCALAR_SIZE;
262  cudaAssert(cudaMalloc(&dev_prev_out_grad, input_size));
263  // Copy the contents of the filter and the output gradient from the host to the device.
264  cudaAssert(cudaMemcpy(dev_filter, filter, filter_size, cudaMemcpyHostToDevice));
265  cudaAssert(cudaMemcpy(dev_out_grad, out_grad, out_grad_size, cudaMemcpyHostToDevice));
266  // Perform the backwards data convolution.
267  cudnnAssert(cudnnConvolutionBackwardData(handle, &alpha, filter_desc, dev_filter, out_grad_desc, dev_out_grad,
268  dconv_desc, dconv_data_algo, dev_data_workspace, data_workspace_size, &beta, input_desc, dev_prev_out_grad));
269  // Copy the previous layer's output gradient from the device to the host.
270  cudaAssert(cudaMemcpy(prev_out_grad, dev_prev_out_grad, input_size, cudaMemcpyDeviceToHost));
271  /* Free the resources not needed anymore (reuse the filter array for the gradient and the
272  * previous layer's output gradient array for the input). */
273  cudaAssert(cudaFree(dev_data_workspace));
274  // Allocate the memory required for the backwards filter convolution on the device.
275  Scalar* dev_filter_workspace;
276  cudaAssert(cudaMalloc(&dev_prev_out_grad, input_size));
277  cudaAssert(cudaMalloc(&dev_filter_workspace, filter_workspace_size));
278  // Populate the input array.
279  cudaAssert(cudaMemcpy(dev_prev_out_grad, input, input_size, cudaMemcpyHostToDevice));
280  // Perform the backwards filter convolution.
281  cudnnAssert(cudnnConvolutionBackwardFilter(handle, &alpha, input_desc, dev_prev_out_grad, out_grad_desc, dev_out_grad,
282  dconv_desc, dconv_filter_algo, dev_filter_workspace, filter_workspace_size, &beta, filter_desc, dev_filter));
283  // Copy the filter gradient to the host.
284  cudaAssert(cudaMemcpy(filter_grad, dev_filter, filter_size, cudaMemcpyDeviceToHost));
285  // Free up resources.
286  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
287  cudnnAssert(cudnnDestroyFilterDescriptor(filter_desc));
288  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
289  cudaAssert(cudaFree(dev_filter));
290  cudaAssert(cudaFree(dev_filter_workspace));
291  cudaAssert(cudaFree(dev_prev_out_grad));
292  // Allocate the memory needed for the bias gradient on the device.
293  Scalar* dev_bias_grad;
294  const std::size_t bias_size = filters * SCALAR_SIZE;
295  cudaAssert(cudaMalloc(&dev_bias_grad, bias_size));
296  // Perform the backwards bias convolution.
297  cudnnAssert(cudnnConvolutionBackwardBias(handle, &alpha, out_grad_desc, dev_out_grad, &beta, bias_desc,
298  dev_bias_grad));
299  // Copy the bias gradient from the device to the host.
300  cudaAssert(cudaMemcpy(bias_grad, dev_bias_grad, bias_size, cudaMemcpyDeviceToHost));
301  // Free up the remaining resources.
302  cudnnAssert(cudnnDestroyTensorDescriptor(out_grad_desc));
303  cudnnAssert(cudnnDestroyTensorDescriptor(bias_desc));
304  cudaAssert(cudaFree(dev_out_grad));
305  cudaAssert(cudaFree(dev_bias_grad));
306  }
316  inline void activation_fwd(Scalar* input, const Array4& dims, cudnnActivationMode_t act_mode,
317  Scalar coeff, /* out */ Scalar* output) {
318  // Create and set the activation descriptor.
319  cudnnActivationDescriptor_t act_desc;
320  cudnnAssert(cudnnCreateActivationDescriptor(&act_desc));
321  cudnnAssert(cudnnSetActivationDescriptor(act_desc, act_mode, NAN_PROP, (double) coeff));
322  // Create and set the input/output tensor descriptor.
323  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
324  // Allocate the necessary device memory and copy the input tensor from the host to the device.
325  Scalar* dev_input;
326  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
327  cudaAssert(cudaMalloc(&dev_input, size));
328  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
329  // Perform the activation.
330  cudnnAssert(cudnnActivationForward(handle, act_desc, &alpha, tens_desc, dev_input, &beta, tens_desc, dev_input));
331  // Copy the output tensor from the device to the host.
332  cudaAssert(cudaMemcpy(output, dev_input, size, cudaMemcpyDeviceToHost));
333  // Free up the resources.
334  cudnnAssert(cudnnDestroyActivationDescriptor(act_desc));
335  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
336  cudaAssert(cudaFree(dev_input));
337  }
349  inline void activation_bwd(Scalar* input, Scalar* output, Scalar* out_grad, const Array4& dims,
350  cudnnActivationMode_t act_mode, Scalar coeff, /* out */ Scalar* prev_out_grad) {
351  // Create and set the activation and tensor descriptors.
352  cudnnActivationDescriptor_t act_desc;
353  cudnnAssert(cudnnCreateActivationDescriptor(&act_desc));
354  cudnnAssert(cudnnSetActivationDescriptor(act_desc, act_mode, NAN_PROP, (double) coeff));
355  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
356  // Allocate and populate the required memory on the device.
357  Scalar* dev_input;
358  Scalar* dev_output;
359  Scalar* dev_out_grad;
360  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
361  cudaAssert(cudaMalloc(&dev_input, size));
362  cudaAssert(cudaMalloc(&dev_output, size));
363  cudaAssert(cudaMalloc(&dev_out_grad, size));
364  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
365  cudaAssert(cudaMemcpy(dev_output, output, size, cudaMemcpyHostToDevice));
366  cudaAssert(cudaMemcpy(dev_out_grad, out_grad, size, cudaMemcpyHostToDevice));
367  // Apply the backwards activation function.
368  cudnnAssert(cudnnActivationBackward(handle, act_desc, &alpha, tens_desc, dev_output, tens_desc, dev_out_grad,
369  tens_desc, dev_input, &beta, tens_desc, dev_out_grad));
370  // Copy the result to the host.
371  cudaAssert(cudaMemcpy(prev_out_grad, dev_out_grad, size, cudaMemcpyDeviceToHost));
372  // Free up the resources.
373  cudnnAssert(cudnnDestroyActivationDescriptor(act_desc));
374  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
375  cudaAssert(cudaFree(dev_input));
376  cudaAssert(cudaFree(dev_output));
377  cudaAssert(cudaFree(dev_out_grad));
378  }

541  inline Array4 pooling2d_output_dims(const Array4& input_dims, cudnnPoolingMode_t pool_mode, std::size_t window_height,
542  std::size_t window_width, std::size_t vertical_padding, std::size_t horizontal_padding, std::size_t vertical_stride,
543  std::size_t horizontal_stride) {
544  // Create and set the input tensor descriptor.
545  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
546  // Create and set the pooling descriptor.
547  cudnnPoolingDescriptor_t pool_desc;
548  cudnnAssert(cudnnCreatePoolingDescriptor(&pool_desc));
549  cudnnAssert(cudnnSetPooling2dDescriptor(pool_desc, pool_mode, NAN_PROP, window_height, window_width,
550  vertical_padding, horizontal_padding, vertical_stride, horizontal_stride));
551  // Compute the dimensions.
552  int n, c, h, w;
553  cudnnAssert(cudnnGetPooling2dForwardOutputDim(pool_desc, input_desc, &n, &c, &h, &w));
554  // Free the resources.
555  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
556  cudnnAssert(cudnnDestroyPoolingDescriptor(pool_desc));
557  return { (std::size_t) n, (std::size_t) h, (std::size_t) w, (std::size_t) c };
558  }
573  inline Array4 max_pooling2d_output_dims(const Array4& input_dims, std::size_t window_height, std::size_t window_width,

616  inline void pooling2d_fwd(Scalar* input, const Array4& input_dims, const Array4& output_dims, cudnnPoolingMode_t pool_mode,
617  std::size_t window_height, std::size_t window_width, std::size_t vertical_padding, std::size_t horizontal_padding,
618  std::size_t vertical_stride, std::size_t horizontal_stride, /* out */ Scalar* output) {
619  // Create and set the input tensor descriptor.
620  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
621  // Create and set the output tensor descriptor.
622  cudnnTensorDescriptor_t output_desc = setup_tens_desc(output_dims);
623  // Create and set the pooling descriptor.
624  cudnnPoolingDescriptor_t pool_desc;
625  cudnnAssert(cudnnCreatePoolingDescriptor(&pool_desc));
626  cudnnAssert(cudnnSetPooling2dDescriptor(pool_desc, pool_mode, NAN_PROP, window_height, window_width,
627  vertical_padding, horizontal_padding, vertical_stride, horizontal_stride));
628  // Allocate the necessary device memory and copy the input tensor from the host to the device.
629  Scalar* dev_input;
630  Scalar* dev_output;
631  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * input_dims[3] * SCALAR_SIZE;
632  cudaAssert(cudaMalloc(&dev_input, input_size));
633  const std::size_t output_size = output_dims[0] * output_dims[1] * output_dims[2] * output_dims[3] * SCALAR_SIZE;
634  cudaAssert(cudaMalloc(&dev_output, output_size));
635  cudaAssert(cudaMemcpy(dev_input, input, input_size, cudaMemcpyHostToDevice));
636  // Perform the pooling.
637  cudnnAssert(cudnnPoolingForward(handle, pool_desc, &alpha, input_desc, dev_input, &beta, output_desc, dev_output));
638  // Copy the output tensor from the device to the host.
639  cudaAssert(cudaMemcpy(output, dev_output, output_size, cudaMemcpyDeviceToHost));
640  // Free up the resources.
641  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
642  cudnnAssert(cudnnDestroyTensorDescriptor(output_desc));
643  cudnnAssert(cudnnDestroyPoolingDescriptor(pool_desc));
644  cudaAssert(cudaFree(dev_input));
645  cudaAssert(cudaFree(dev_output));
646  }
666  inline void pooling2d_bwd(Scalar* input, Scalar* output, Scalar* out_grad, const Array4& input_dims, const Array4& output_dims,
667  cudnnPoolingMode_t pool_mode, std::size_t window_height, std::size_t window_width, std::size_t vertical_padding,
668  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
669  /* out */ Scalar* prev_out_grad) {
670  // Create and set the input tensor descriptor.
671  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
672  // Create and set the output tensor descriptor.
673  cudnnTensorDescriptor_t output_desc = setup_tens_desc(output_dims);
674  // Create and set the pooling descriptor.
675  cudnnPoolingDescriptor_t pool_desc;
676  cudnnAssert(cudnnCreatePoolingDescriptor(&pool_desc));
677  cudnnAssert(cudnnSetPooling2dDescriptor(pool_desc, pool_mode, NAN_PROP, window_height, window_width,
678  vertical_padding, horizontal_padding, vertical_stride, horizontal_stride));
679  // Allocate the necessary device memory.
680  Scalar* dev_input;
681  Scalar* dev_output;
682  Scalar* dev_out_grad;
683  Scalar* dev_prev_out_grad;
684  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * input_dims[3] * SCALAR_SIZE;
685  cudaAssert(cudaMalloc(&dev_input, input_size));
686  const std::size_t output_size = output_dims[0] * output_dims[1] * output_dims[2] * output_dims[3] * SCALAR_SIZE;
687  cudaAssert(cudaMalloc(&dev_output, output_size));

689  cudaAssert(cudaMalloc(&dev_prev_out_grad, input_size));
690  // Copy the contents of the input, output, and output gradient tensors to the device.
691  cudaAssert(cudaMemcpy(dev_input, input, input_size, cudaMemcpyHostToDevice));
692  cudaAssert(cudaMemcpy(dev_output, output, input_size, cudaMemcpyHostToDevice));
693  cudaAssert(cudaMemcpy(dev_prev_out_grad, out_grad, input_size, cudaMemcpyHostToDevice));
694  // Perform the backwards pooling.
695  cudnnAssert(cudnnPoolingBackward(handle, pool_desc, &alpha, output_desc, dev_output, output_desc, dev_out_grad,
696  input_desc, dev_input, &beta, input_desc, dev_prev_out_grad));
697  // Copy the input gradient tensor from the device to the host.
698  cudaAssert(cudaMemcpy(prev_out_grad, dev_prev_out_grad, input_size, cudaMemcpyDeviceToHost));
699  // Free up the resources.
700  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
701  cudnnAssert(cudnnDestroyTensorDescriptor(output_desc));
702  cudnnAssert(cudnnDestroyPoolingDescriptor(pool_desc));
703  cudaAssert(cudaFree(dev_input));
704  cudaAssert(cudaFree(dev_output));
705  cudaAssert(cudaFree(dev_out_grad));
706  cudaAssert(cudaFree(dev_prev_out_grad));
707  }

809  inline void dropout_fwd(Scalar* input, const Array4& dims, Scalar dropout, /* out */ Scalar* output,
810  /* out */ std::vector<Scalar>& reserve) {
811  // Create and set the input/output/reserve tensor descriptor.
812  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
813  // Create and set the dropout descriptor.
814  cudnnDropoutDescriptor_t dropout_desc;
815  cudnnAssert(cudnnCreateDropoutDescriptor(&dropout_desc));
816  cudnnAssert(cudnnSetDropoutDescriptor(dropout_desc, handle, (float) dropout, nullptr, 0, 0));
817  // Calculate the required reserve size.
818  std::size_t reserve_size;
819  cudnnAssert(cudnnDropoutGetReserveSpaceSize(tens_desc, &reserve_size));
820  // Allocate the memory for the device arrays and reserve space.
821  Scalar* dev_input;
822  Scalar* dev_output;
823  Scalar* dev_reserve;
824  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
825  cudaAssert(cudaMalloc(&dev_input, size));
826  cudaAssert(cudaMalloc(&dev_output, size));
827  cudaAssert(cudaMalloc(&dev_reserve, reserve_size));
828  // Copy the contents of the input tensor from the host to the device.
829  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
830  // Perform the dropout.
831  cudnnAssert(cudnnDropoutForward(handle, dropout_desc, tens_desc, dev_input, tens_desc, dev_output,
832  dev_reserve, reserve_size));
833  // Copy the results to the host.
834  cudaAssert(cudaMemcpy(output, dev_output, size, cudaMemcpyDeviceToHost));
835  reserve = std::vector<Scalar>(reserve_size / SCALAR_SIZE);
836  cudaAssert(cudaMemcpy(reserve.data(), dev_reserve, size, cudaMemcpyDeviceToHost));
837  // Free resources.
838  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
839  cudnnAssert(cudnnDestroyDropoutDescriptor(dropout_desc));
840  cudaAssert(cudaFree(dev_input));
841  cudaAssert(cudaFree(dev_output));
842  cudaAssert(cudaFree(dev_reserve));
843  }
853  inline void dropout_bwd(Scalar* out_grad, std::vector<Scalar>& reserve, const Array4& dims, Scalar dropout,
854  /* out */ Scalar* prev_out_grad) {
855  // Create and set the input/output/reserve tensor descriptor.
856  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
857  // Create and set the dropout descriptor.
858  cudnnDropoutDescriptor_t dropout_desc;
859  cudnnAssert(cudnnCreateDropoutDescriptor(&dropout_desc));
860  cudnnAssert(cudnnSetDropoutDescriptor(dropout_desc, handle, (float) dropout, nullptr, 0, 0));
861  // Allocate the memory for the device arrays and reserve space.
862  Scalar* dev_out_grad;
863  Scalar* dev_reserve;
864  Scalar* dev_prev_out_grad;
865  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
866  cudaAssert(cudaMalloc(&dev_out_grad, size));
867  cudaAssert(cudaMalloc(&dev_reserve, size));
868  cudaAssert(cudaMalloc(&dev_prev_out_grad, size));
869  // Copy the contents of the input tensor and the reserve vector from the host to the device.
870  cudaAssert(cudaMemcpy(dev_out_grad, out_grad, size, cudaMemcpyHostToDevice));

874  dev_reserve, reserve.size() * SCALAR_SIZE));
875  // Copy the result to the host.
876  cudaAssert(cudaMemcpy(prev_out_grad, dev_prev_out_grad, size, cudaMemcpyDeviceToHost));
877  // Free the device resources.
878  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
879  cudnnAssert(cudnnDestroyDropoutDescriptor(dropout_desc));
880  cudaAssert(cudaFree(dev_out_grad));
881  cudaAssert(cudaFree(dev_reserve));
882  cudaAssert(cudaFree(dev_prev_out_grad));
883  }

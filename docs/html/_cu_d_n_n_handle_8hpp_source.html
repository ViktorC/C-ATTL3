105  inline void conv2d_output_dims(std::size_t input_height, std::size_t input_width, std::size_t input_channels,
106  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding,
107  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
108  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ std::size_t& output_height,
109  /* out */ std::size_t& output_width, /* out */ std::size_t& output_channels) const {
110  // Create and set the input tensor descriptor.
111  cudnnTensorDescriptor_t input_desc;
112  CuDNNTensorDescriptorManager<>::create_descriptor(input_desc, CuDNNTensor<Scalar>::DATA_TYPE,
113  CuDNNTensor<Scalar>::TENSOR_FORMAT, 1, input_height, input_width, input_channels);
114  // Create and set up the filter descriptor.
115  cudnnFilterDescriptor_t filter_desc;
116  CuDNNTensorDescriptorManager<true>::create_descriptor(filter_desc, CuDNNTensor<Scalar>::DATA_TYPE,
117  CuDNNTensor<Scalar>::TENSOR_FORMAT, filters, receptor_height, receptor_width, input_channels);
118  // Create and set up the convolution descriptor.
119  cudnnConvolutionDescriptor_t conv_desc;
120  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
121  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
122  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
123  CuDNNTensor<Scalar>::DATA_TYPE));
124  // Compute the dimensions.
125  int n, h, w, c;
126  cudnnAssert(cudnnGetConvolution2dForwardOutputDim(conv_desc, input_desc, filter_desc, &n, &c, &h, &w));
127  // Free the resources.
128  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
129  CuDNNTensorDescriptorManager<>::destroy_descriptor(input_desc);
130  CuDNNTensorDescriptorManager<true>::destroy_descriptor(filter_desc);
131  output_height = (std::size_t) h;
132  output_width = (std::size_t) w;
133  output_channels = (std::size_t) c;
134  }
151  inline void convolution2d_fwd(const CuDNNTensor<Scalar>& input, const CuDNNTensor<Scalar,true>& filter,
152  const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding, std::size_t horizontal_padding,
153  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
154  std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& output) const {
155  // Create and set up the convolution descriptor.
156  cudnnConvolutionDescriptor_t conv_desc;
157  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
158  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
159  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
160  CuDNNTensor<Scalar>::DATA_TYPE));
161  // Have cuDNN find the most performant algorithm given the convolution parameters.
162  cudnnConvolutionFwdAlgo_t conv_algo;
163  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(handle, input.desc(), filter.desc(), conv_desc,
164  output.desc(), CUDNN_CONVOLUTION_FWD_PREFER_FASTEST, 0, &conv_algo));
165  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
166  * the convolution parameters. */
167  std::size_t workspace_size;
168  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(handle, input.desc(), filter.desc(), conv_desc,
169  output.desc(), conv_algo, &workspace_size));
170  // Allocate the memory for the workspace required for the convolution.
171  Scalar* workspace;
172  cudaAssert(cudaMalloc(&workspace, workspace_size));
173  // Perform the convolution.
174  cudnnAssert(cudnnConvolutionForward(handle, &alpha, input.desc(), input.data(), filter.desc(),
175  filter.data(), conv_desc, conv_algo, workspace, workspace_size, &beta, output.desc(),
176  output.data()));
177  // Free the convolution resources.
178  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
179  cudaAssert(cudaFree(workspace));
180  // Apply the bias to the output tensor.
181  add_bias(bias, output);
182  }
203  inline void convolution2d_bwd(const CuDNNTensor<Scalar>& input, const CuDNNTensor<Scalar>& out_grad,
204  const CuDNNTensor<Scalar,true>& filter, const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding,
205  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
206  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& prev_out_grad,
207  /* out */ CuDNNTensor<Scalar,true>& filter_grad, /* out */ CuDNNTensor<Scalar>& bias_grad) const {
208  // Create and set up the backward convolution descriptor.
209  cudnnConvolutionDescriptor_t dconv_desc;
210  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
211  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding, vertical_stride,
212  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
213  CuDNNTensor<Scalar>::DATA_TYPE));
214  // Have cuDNN find the most performant algorithm given the convolution parameters.
215  cudnnConvolutionBwdDataAlgo_t dconv_data_algo;
216  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(handle, filter.desc(), out_grad.desc(), dconv_desc,
217  prev_out_grad.desc(), CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
218  cudnnConvolutionBwdFilterAlgo_t dconv_filter_algo;
219  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(handle, input.desc(), out_grad.desc(), dconv_desc,
220  filter_grad.desc(), CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST, 0, &dconv_filter_algo));
221  /* Have cuDNN compute the data_workspace memory required for the selected backward convolution algorithms given
222  * the convolution parameters. */
223  std::size_t data_workspace_size;
224  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(handle, filter.desc(), out_grad.desc(), dconv_desc,
225  prev_out_grad.desc(), dconv_data_algo, &data_workspace_size));
226  std::size_t filter_workspace_size;
227  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(handle, input.desc(), out_grad.desc(), dconv_desc,
228  filter_grad.desc(), dconv_filter_algo, &filter_workspace_size));
229  // Allocate the memory required for the backwards data convolution on the device.
230  Scalar* data_workspace;
231  cudaAssert(cudaMalloc(&data_workspace, data_workspace_size));
232  // Perform the backwards data convolution.
233  cudnnAssert(cudnnConvolutionBackwardData(handle, &alpha, filter.desc(), filter.data(), out_grad.desc(),
234  out_grad.data(), dconv_desc, dconv_data_algo, data_workspace, data_workspace_size, &beta,
235  prev_out_grad.desc(), prev_out_grad.data()));
236  // Free the resources.
237  cudaAssert(cudaFree(data_workspace));
238  // Allocate the memory required for the backwards filter convolution on the device.
239  Scalar* filter_workspace;
240  cudaAssert(cudaMalloc(&filter_workspace, filter_workspace_size));
241  // Perform the backwards filter convolution.
242  cudnnAssert(cudnnConvolutionBackwardFilter(handle, &alpha, input.desc(), input.data(), out_grad.desc(),
243  out_grad.data(), dconv_desc, dconv_filter_algo, filter_workspace, filter_workspace_size, &beta,
244  filter_grad.desc(), filter_grad.data()));
245  // Free up resources.
246  cudaAssert(cudaFree(filter_workspace));
247  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
248  // Perform the backwards bias convolution.
249  cudnnAssert(cudnnConvolutionBackwardBias(handle, &alpha, out_grad.desc(), out_grad.data(), &beta,
250  bias_grad.desc(), bias_grad.data()));
251  }

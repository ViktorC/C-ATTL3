74  inline void conv2d_output_dims(std::size_t input_height, std::size_t input_width, std::size_t input_channels,
75  cudnnTensorFormat_t tensor_format, std::size_t filters, std::size_t receptor_height,
76  std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
77  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
78  std::size_t horizontal_dilation, /* out */ std::size_t& output_height, /* out */ std::size_t& output_width,
79  /* out */ std::size_t& output_channels) const {
80  // Create and set the input tensor descriptor.
81  cudnnTensorDescriptor_t input_desc = CuDNNTensorDescriptorManager<>::create_descriptor(
82  CuDNNTensor<Scalar>::DATA_TYPE, tensor_format, 1, input_height, input_width, input_channels);
83  // Create and set up the filter descriptor.
84  cudnnFilterDescriptor_t filter_desc = CuDNNTensorDescriptorManager<true>::create_descriptor(
85  CuDNNTensor<Scalar>::DATA_TYPE, tensor_format, filters, receptor_height, receptor_width,
86  input_channels);
87  // Create and set up the convolution descriptor.
88  cudnnConvolutionDescriptor_t conv_desc;
89  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
90  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
91  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
92  CuDNNTensor<Scalar>::DATA_TYPE));
93  // Compute the dimensions.
94  int n, h, w, c;
95  cudnnAssert(cudnnGetConvolution2dForwardOutputDim(conv_desc, input_desc, filter_desc, &n, &c, &h, &w));
96  // Free the resources.
97  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
98  CuDNNTensorDescriptorManager<>::destroy_descriptor(input_desc);
99  CuDNNTensorDescriptorManager<true>::destroy_descriptor(filter_desc);
100  output_height = (std::size_t) h;
101  output_width = (std::size_t) w;
102  output_channels = (std::size_t) c;
103  }
120  inline void convolution2d_fwd(const CuDNNTensor<Scalar>& input, const CuDNNTensor<Scalar,true>& filter,
121  const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding, std::size_t horizontal_padding,
122  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
123  std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& output) const {
124  // Create and set up the convolution descriptor.
125  cudnnConvolutionDescriptor_t conv_desc;
126  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
127  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
128  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
129  CuDNNTensor<Scalar>::DATA_TYPE));
130  // Have cuDNN find the most performant algorithm given the convolution parameters.
131  cudnnConvolutionFwdAlgo_t conv_algo;
132  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(handle, input.get_desc(), filter.get_desc(), conv_desc, output.get_desc(),
133  CUDNN_CONVOLUTION_FWD_PREFER_FASTEST, 0, &conv_algo));
134  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
135  * the convolution parameters. */
136  std::size_t workspace_size;
137  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(handle, input.get_desc(), filter.get_desc(), conv_desc, output.get_desc(),
138  conv_algo, &workspace_size));
139  // Allocate the memory for the workspace required for the convolution.
140  Scalar* workspace;
141  cudaAssert(cudaMalloc(&workspace, workspace_size));
142  // Perform the convolution.
143  cudnnAssert(cudnnConvolutionForward(handle, &alpha, input.get_desc(), input.get_data(), filter.get_desc(), filter.get_data(),
144  conv_desc, conv_algo, workspace, workspace_size, &beta, output.get_desc(), output.get_data()));
145  // Free the convolution resources.
146  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
147  cudaAssert(cudaFree(workspace));
148  // Apply the bias to the output tensor.
149  cudnnAssert(cudnnAddTensor(handle, &alpha, bias.get_desc(), bias.get_data(), &beta, output.get_desc(), output.get_data()));
150  }
171  inline void convolution2d_bwd(const CuDNNTensor<Scalar>& input, const CuDNNTensor<Scalar>& out_grad,
172  const CuDNNTensor<Scalar,true>& filter, const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding,
173  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
174  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& prev_out_grad,
175  /* out */ CuDNNTensor<Scalar,true>& filter_grad, /* out */ CuDNNTensor<Scalar>& bias_grad) const {
176  // Create and set up the backward convolution descriptor.
177  cudnnConvolutionDescriptor_t dconv_desc;
178  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
179  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding, vertical_stride,
180  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
181  CuDNNTensor<Scalar>::DATA_TYPE));
182  // Have cuDNN find the most performant algorithm given the convolution parameters.
183  cudnnConvolutionBwdDataAlgo_t dconv_data_algo;
184  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(handle, filter.get_desc(), out_grad.get_desc(), dconv_desc,
185  input.get_desc(), CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
186  cudnnConvolutionBwdFilterAlgo_t dconv_filter_algo;
187  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(handle, input.get_desc(), out_grad.get_desc(), dconv_desc,
188  filter.get_desc(), CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST, 0, &dconv_filter_algo));
189  /* Have cuDNN compute the data_workspace memory required for the selected backward convolution algorithms given
190  * the convolution parameters. */
191  std::size_t data_workspace_size;
192  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(handle, filter.get_desc(), out_grad.get_desc(), dconv_desc,
193  input.get_desc(), dconv_data_algo, &data_workspace_size));
194  std::size_t filter_workspace_size;
195  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(handle, input.get_desc(), out_grad.get_desc(), dconv_desc,
196  filter.get_desc(), dconv_filter_algo, &filter_workspace_size));
197  // Allocate the memory required for the backwards data convolution on the device.
198  Scalar* data_workspace;
199  cudaAssert(cudaMalloc(&data_workspace, data_workspace_size));
200  // Perform the backwards data convolution.
201  cudnnAssert(cudnnConvolutionBackwardData(handle, &alpha, filter.get_desc(), filter.get_data(), out_grad.get_desc(),
202  out_grad.get_data(), dconv_desc, dconv_data_algo, data_workspace, data_workspace_size, &beta,
203  prev_out_grad.get_desc(), prev_out_grad.get_data()));
204  // Free the resources.
205  cudaAssert(cudaFree(data_workspace));
206  // Allocate the memory required for the backwards filter convolution on the device.
207  Scalar* filter_workspace;
208  cudaAssert(cudaMalloc(&filter_workspace, filter_workspace_size));
209  // Perform the backwards filter convolution.
210  cudnnAssert(cudnnConvolutionBackwardFilter(handle, &alpha, input.get_desc(), input.get_data(), out_grad.get_desc(),
211  out_grad.get_data(), dconv_desc, dconv_filter_algo, filter_workspace, filter_workspace_size, &beta,
212  filter_grad.get_desc(), filter_grad.get_data()));
213  // Free up resources.
214  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
215  cudaAssert(cudaFree(filter_workspace));
216  // Perform the backwards bias convolution.
217  cudnnAssert(cudnnConvolutionBackwardBias(handle, &alpha, out_grad.get_desc(), out_grad.get_data(), &beta,
218  bias_grad.get_desc(), bias_grad.get_data()));
219  }

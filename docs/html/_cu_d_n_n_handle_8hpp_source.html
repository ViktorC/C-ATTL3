104  inline void conv2d_output_dims(std::size_t input_height, std::size_t input_width, std::size_t input_channels,
105  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding,
106  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
107  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ std::size_t& output_height,
108  /* out */ std::size_t& output_width, /* out */ std::size_t& output_channels) const {
109  // Create and set the input tensor descriptor.
110  cudnnTensorDescriptor_t input_desc;
111  CuDNNTensor<Scalar>::create_tensor_descriptor(input_desc, 1, input_height, input_width, input_channels);
112  // Create and set up the filter descriptor.
113  cudnnFilterDescriptor_t filter_desc;
114  CuDNNFilter<Scalar>::create_filter_descriptor(filter_desc, filters, receptor_height,
115  receptor_width, input_channels);
116  // Create and set up the convolution descriptor.
117  cudnnConvolutionDescriptor_t conv_desc;
118  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
119  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
120  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
121  CuDNNTensor<Scalar>::DATA_TYPE));
122  // Compute the dimensions.
123  int n, h, w, c;
124  cudnnAssert(cudnnGetConvolution2dForwardOutputDim(conv_desc, input_desc, filter_desc, &n, &c, &h, &w));
125  // Free the resources.
126  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
127  CuDNNTensor<Scalar>::destroy_tensor_descriptor(input_desc);
128  CuDNNFilter<Scalar>::destroy_filter_descriptor(filter_desc);
129  output_height = (std::size_t) h;
130  output_width = (std::size_t) w;
131  output_channels = (std::size_t) c;
132  }
149  inline void convolution2d_fwd(const CuDNNTensor<Scalar>& input, const CuDNNFilter<Scalar>& filter,
150  const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding, std::size_t horizontal_padding,
151  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation,
152  std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& output) const {
153  // Create and set up the convolution descriptor.
154  cudnnConvolutionDescriptor_t conv_desc;
155  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
156  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
157  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
158  CuDNNTensor<Scalar>::DATA_TYPE));
159  // Have cuDNN find the most performant algorithm given the convolution parameters.
160  cudnnConvolutionFwdAlgo_t conv_algo;
161  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(handle, input.desc(), filter.filter_desc(), conv_desc,
162  output.desc(), CUDNN_CONVOLUTION_FWD_PREFER_FASTEST, 0, &conv_algo));
163  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
164  * the convolution parameters. */
165  std::size_t workspace_size;
166  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(handle, input.desc(), filter.filter_desc(), conv_desc,
167  output.desc(), conv_algo, &workspace_size));
168  // Allocate the memory for the workspace required for the convolution.
169  CUDAArray<Scalar> workspace(static_cast<std::size_t>(
170  ceil(static_cast<Scalar>(workspace_size) / sizeof(Scalar))));
171  // Perform the convolution.
172  cudnnAssert(cudnnConvolutionForward(handle, &alpha, input.desc(), input.data(), filter.filter_desc(),
173  filter.data(), conv_desc, conv_algo, workspace.data(), workspace_size, &beta, output.desc(),
174  output.data()));
175  // Free the convolution resources.
176  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
177  // Apply the bias to the output tensor.
178  add_bias(bias, output);
179  }
200  inline void convolution2d_bwd(const CuDNNTensor<Scalar>& input, const CuDNNTensor<Scalar>& out_grad,
201  const CuDNNFilter<Scalar>& filter, const CuDNNTensor<Scalar>& bias, std::size_t vertical_padding,
202  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
203  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ CuDNNTensor<Scalar>& prev_out_grad,
204  /* out */ CuDNNFilter<Scalar>& filter_grad, /* out */ CuDNNTensor<Scalar>& bias_grad) const {
205  // Create and set up the backward convolution descriptor.
206  cudnnConvolutionDescriptor_t dconv_desc;
207  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
208  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding, vertical_stride,
209  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION,
210  CuDNNTensor<Scalar>::DATA_TYPE));
211  // Have cuDNN find the most performant algorithm given the convolution parameters.
212  cudnnConvolutionBwdDataAlgo_t dconv_data_algo;
213  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(handle, filter.filter_desc(), out_grad.desc(), dconv_desc,
214  prev_out_grad.desc(), CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
215  cudnnConvolutionBwdFilterAlgo_t dconv_filter_algo;
216  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(handle, input.desc(), out_grad.desc(), dconv_desc,
217  filter_grad.filter_desc(), CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST, 0, &dconv_filter_algo));
218  /* Have cuDNN compute the data_workspace memory required for the selected backward convolution algorithms given
219  * the convolution parameters. */
220  std::size_t data_workspace_size;
221  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(handle, filter.filter_desc(), out_grad.desc(), dconv_desc,
222  prev_out_grad.desc(), dconv_data_algo, &data_workspace_size));
223  std::size_t filter_workspace_size;
224  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(handle, input.desc(), out_grad.desc(), dconv_desc,
225  filter_grad.filter_desc(), dconv_filter_algo, &filter_workspace_size));
226  // Allocate the memory required for the backwards data convolution on the device.
227  CUDAArray<Scalar> data_workspace(static_cast<std::size_t>(
228  ceil(static_cast<Scalar>(data_workspace_size) / sizeof(Scalar))));
229  cudaAssert(cudaMalloc(&data_workspace, data_workspace_size));
230  // Perform the backwards data convolution.
231  cudnnAssert(cudnnConvolutionBackwardData(handle, &alpha, filter.filter_desc(), filter.data(), out_grad.desc(),
232  out_grad.data(), dconv_desc, dconv_data_algo, data_workspace.data(), data_workspace_size, &beta,
233  prev_out_grad.desc(), prev_out_grad.data()));
234  // Allocate the memory required for the backwards filter convolution on the device.
235  CUDAArray<Scalar> filter_workspace(static_cast<std::size_t>(
236  ceil(static_cast<Scalar>(filter_workspace_size) / sizeof(Scalar))));
237  // Perform the backwards filter convolution.
238  cudnnAssert(cudnnConvolutionBackwardFilter(handle, &alpha, input.desc(), input.data(), out_grad.desc(),
239  out_grad.data(), dconv_desc, dconv_filter_algo, filter_workspace.data(), filter_workspace_size, &beta,
240  filter_grad.filter_desc(), filter_grad.data()));
241  // Free up resources.
242  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
243  // Perform the backwards bias convolution.
244  cudnnAssert(cudnnConvolutionBackwardBias(handle, &alpha, out_grad.desc(), out_grad.data(), &beta,
245  bias_grad.desc(), bias_grad.data()));
246  }

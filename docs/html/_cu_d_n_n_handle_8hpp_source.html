115  inline void convolution2d_fwd(Scalar* input, Scalar* filter, Scalar* bias, const Array4& input_dims, const Array4& output_dims,
116  std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding, std::size_t horizontal_padding,
117  std::size_t vertical_stride, std::size_t horizontal_stride, std::size_t vertical_dilation, std::size_t horizontal_dilation,
118  /* out */ Scalar* output) const {
119  std::size_t depth = input_dims[3];
120  std::size_t filters = output_dims[3];
121  // Create and set up the input tensor descriptor.
122  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
123  // Create and set up the filter descriptor.
124  cudnnFilterDescriptor_t filter_desc;
125  cudnnAssert(cudnnCreateFilterDescriptor(&filter_desc));
126  cudnnAssert(cudnnSetFilter4dDescriptor(filter_desc, DATA_TYPE, TENSOR_FORMAT, filters, depth,
127  receptor_height, receptor_width));
128  // Create and set up the bias descriptor.
129  cudnnTensorDescriptor_t bias_desc = setup_tens_desc(1, filters, 1, 1);
130  // Create and set up the convolution descriptor.
131  cudnnConvolutionDescriptor_t conv_desc;
132  cudnnAssert(cudnnCreateConvolutionDescriptor(&conv_desc));
133  cudnnAssert(cudnnSetConvolution2dDescriptor(conv_desc, vertical_padding, horizontal_padding, vertical_stride,
134  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION, DATA_TYPE));
135  // Create and set up the output tensor descriptor.
136  cudnnTensorDescriptor_t output_desc = setup_tens_desc(output_dims);
137  // Have cuDNN find the most performant algorithm given the convolution parameters.
138  cudnnConvolutionFwdAlgo_t conv_algo;
139  cudnnAssert(cudnnGetConvolutionForwardAlgorithm(handle, input_desc, filter_desc, conv_desc, output_desc,
140  CUDNN_CONVOLUTION_FWD_PREFER_FASTEST, 0, &conv_algo));
141  /* Have cuDNN compute the workspace memory required for the selected convolution algorithm given
142  * the convolution parameters. */
143  std::size_t workspace_size;
144  cudnnAssert(cudnnGetConvolutionForwardWorkspaceSize(handle, input_desc, filter_desc, conv_desc, output_desc,
145  conv_algo, &workspace_size));
146  // Allocate the memory for the device arrays required for the convolution.
147  Scalar* dev_input;
148  Scalar* dev_filter;
149  Scalar* dev_workspace;
150  Scalar* dev_output;
151  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * depth * SCALAR_SIZE;
152  cudaAssert(cudaMalloc(&dev_input, input_size));
153  const std::size_t filter_size = receptor_height * receptor_width * depth * filters * SCALAR_SIZE;
154  cudaAssert(cudaMalloc(&dev_filter, filter_size));
155  cudaAssert(cudaMalloc(&dev_workspace, workspace_size));
156  const std::size_t output_size = output_dims[0] * output_dims[1] * output_dims[2] * filters * SCALAR_SIZE;
157  cudaAssert(cudaMalloc(&dev_output, output_size));
158  // Copy the contents of the input tensor and the filter to the respective device arrays.
159  cudaAssert(cudaMemcpy(dev_input, input, input_size, cudaMemcpyHostToDevice));
160  cudaAssert(cudaMemcpy(dev_filter, filter, filter_size, cudaMemcpyHostToDevice));
161  // Perform the convolution.
162  cudnnAssert(cudnnConvolutionForward(handle, &alpha, input_desc, dev_input, filter_desc, dev_filter,
163  conv_desc, conv_algo, dev_workspace, workspace_size, &beta, output_desc, dev_output));
164  // Free the convolution resources.
165  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
166  cudnnAssert(cudnnDestroyFilterDescriptor(filter_desc));
167  cudnnAssert(cudnnDestroyConvolutionDescriptor(conv_desc));
168  cudaAssert(cudaFree(dev_input));
169  cudaAssert(cudaFree(dev_workspace));
170  cudaAssert(cudaFree(dev_filter));
171  // Allocate the memory for the bias array on the device and fill it.
172  Scalar* dev_bias;
173  const std::size_t bias_size = filters * SCALAR_SIZE;
174  cudaAssert(cudaMalloc(&dev_bias, bias_size));
175  cudaAssert(cudaMemcpy(dev_bias, bias, bias_size, cudaMemcpyHostToDevice));
176  // Apply the bias to the output tensor.
177  cudnnAssert(cudnnAddTensor(handle, &alpha, bias_desc, dev_bias, &beta, output_desc, dev_output));
178  // Copy the output tensor to the host.
179  cudaAssert(cudaMemcpy(output, dev_output, output_size, cudaMemcpyDeviceToHost));
180  // Free up the remaining resources.
181  cudnnAssert(cudnnDestroyTensorDescriptor(bias_desc));
182  cudnnAssert(cudnnDestroyTensorDescriptor(output_desc));
183  cudaAssert(cudaFree(dev_bias));
184  cudaAssert(cudaFree(dev_output));
185  }
210  inline void convolution2d_bwd(Scalar* input, Scalar* out_grad, Scalar* filter, Scalar* bias, const Array4& input_dims,
211  const Array4& output_dims, std::size_t receptor_height, std::size_t receptor_width, std::size_t vertical_padding,
212  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,
213  std::size_t vertical_dilation, std::size_t horizontal_dilation, /* out */ Scalar* prev_out_grad,
214  /* out */ Scalar* filter_grad, /* out */ Scalar* bias_grad) const {
215  std::size_t depth = input_dims[3];
216  std::size_t filters = output_dims[3];
217  // Create and set up the input tensor descriptor.
218  cudnnTensorDescriptor_t input_desc = setup_tens_desc(input_dims);
219  // Create and set up the output gradient descriptor.
220  cudnnTensorDescriptor_t out_grad_desc = setup_tens_desc(output_dims);
221  // Create and set up the filter descriptor.
222  cudnnFilterDescriptor_t filter_desc;
223  cudnnAssert(cudnnCreateFilterDescriptor(&filter_desc));
224  cudnnAssert(cudnnSetFilter4dDescriptor(filter_desc, DATA_TYPE, TENSOR_FORMAT, filters, depth,
225  receptor_height, receptor_width));
226  // Create and set up the bias descriptor.
227  cudnnTensorDescriptor_t bias_desc = setup_tens_desc(1, filters, 1, 1);
228  // Create and set up the backward convolution descriptor.
229  cudnnConvolutionDescriptor_t dconv_desc;
230  cudnnAssert(cudnnCreateConvolutionDescriptor(&dconv_desc));
231  cudnnAssert(cudnnSetConvolution2dDescriptor(dconv_desc, vertical_padding, horizontal_padding, vertical_stride,
232  horizontal_stride, vertical_dilation, horizontal_dilation, CUDNN_CROSS_CORRELATION, DATA_TYPE));
233  // Have cuDNN find the most performant algorithm given the convolution parameters.
234  cudnnConvolutionBwdDataAlgo_t dconv_data_algo;
235  cudnnAssert(cudnnGetConvolutionBackwardDataAlgorithm(handle, filter_desc, out_grad_desc, dconv_desc,
236  input_desc, CUDNN_CONVOLUTION_BWD_DATA_PREFER_FASTEST, 0, &dconv_data_algo));
237  cudnnConvolutionBwdFilterAlgo_t dconv_filter_algo;
238  cudnnAssert(cudnnGetConvolutionBackwardFilterAlgorithm(handle, input_desc, out_grad_desc, dconv_desc,
239  filter_desc, CUDNN_CONVOLUTION_BWD_FILTER_PREFER_FASTEST, 0, &dconv_filter_algo));
240  /* Have cuDNN compute the workspace memory required for the selected backward convolution algorithms given
241  * the convolution parameters. */
242  std::size_t data_workspace_size;
243  cudnnAssert(cudnnGetConvolutionBackwardDataWorkspaceSize(handle, filter_desc, out_grad_desc, dconv_desc,
244  input_desc, dconv_data_algo, &data_workspace_size));
245  std::size_t filter_workspace_size;
246  cudnnAssert(cudnnGetConvolutionBackwardFilterWorkspaceSize(handle, input_desc, out_grad_desc, dconv_desc,
247  filter_desc, dconv_filter_algo, &filter_workspace_size));
248  // Allocate the memory required for the backwards data convolution on the device.
249  Scalar* dev_filter;
250  Scalar* dev_out_grad;
251  Scalar* dev_data_workspace;
252  Scalar* dev_prev_out_grad;
253  const std::size_t filter_size = receptor_height * receptor_width * filters * SCALAR_SIZE;
254  cudaAssert(cudaMalloc(&dev_filter, filter_size));
255  const std::size_t out_grad_size = output_dims[0] * output_dims[1] * output_dims[2] * SCALAR_SIZE;
256  cudaAssert(cudaMalloc(&dev_out_grad, out_grad_size));
257  cudaAssert(cudaMalloc(&dev_data_workspace, data_workspace_size));
258  const std::size_t input_size = input_dims[0] * input_dims[1] * input_dims[2] * SCALAR_SIZE;
259  cudaAssert(cudaMalloc(&dev_prev_out_grad, input_size));
260  // Copy the contents of the filter and the output gradient from the host to the device.
261  cudaAssert(cudaMemcpy(dev_filter, filter, filter_size, cudaMemcpyHostToDevice));
262  cudaAssert(cudaMemcpy(dev_out_grad, out_grad, out_grad_size, cudaMemcpyHostToDevice));
263  // Perform the backwards data convolution.
264  cudnnAssert(cudnnConvolutionBackwardData(handle, &alpha, filter_desc, dev_filter, out_grad_desc, dev_out_grad,
265  dconv_desc, dconv_data_algo, dev_data_workspace, data_workspace_size, &beta, input_desc, dev_prev_out_grad));
266  // Copy the previous layer's output gradient from the device to the host.
267  cudaAssert(cudaMemcpy(prev_out_grad, dev_prev_out_grad, input_size, cudaMemcpyDeviceToHost));
268  /* Free the resources not needed anymore (reuse the filter array for the gradient and the
269  * previous layer's output gradient array for the input). */
270  cudaAssert(cudaFree(dev_data_workspace));
271  // Allocate the memory required for the backwards filter convolution on the device.
272  Scalar* dev_filter_workspace;
273  cudaAssert(cudaMalloc(&dev_prev_out_grad, input_size));
274  cudaAssert(cudaMalloc(&dev_filter_workspace, filter_workspace_size));
275  // Populate the input array.
276  cudaAssert(cudaMemcpy(dev_prev_out_grad, input, input_size, cudaMemcpyHostToDevice));
277  // Perform the backwards filter convolution.
278  cudnnAssert(cudnnConvolutionBackwardFilter(handle, &alpha, input_desc, dev_prev_out_grad, out_grad_desc, dev_out_grad,
279  dconv_desc, dconv_filter_algo, dev_filter_workspace, filter_workspace_size, &beta, filter_desc, dev_filter));
280  // Copy the filter gradient to the host.
281  cudaAssert(cudaMemcpy(filter_grad, dev_filter, filter_size, cudaMemcpyDeviceToHost));
282  // Free up resources.
283  cudnnAssert(cudnnDestroyTensorDescriptor(input_desc));
284  cudnnAssert(cudnnDestroyFilterDescriptor(filter_desc));
285  cudnnAssert(cudnnDestroyConvolutionDescriptor(dconv_desc));
286  cudaAssert(cudaFree(dev_filter));
287  cudaAssert(cudaFree(dev_filter_workspace));
288  cudaAssert(cudaFree(dev_prev_out_grad));
289  // Allocate the memory needed for the bias gradient on the device.
290  Scalar* dev_bias_grad;
291  const std::size_t bias_size = filters * SCALAR_SIZE;
292  cudaAssert(cudaMalloc(&dev_bias_grad, bias_size));
293  // Perform the backwards bias convolution.
294  cudnnAssert(cudnnConvolutionBackwardBias(handle, &alpha, out_grad_desc, dev_out_grad, &beta, bias_desc,
295  dev_bias_grad));
296  // Copy the bias gradient from the device to the host.
297  cudaAssert(cudaMemcpy(bias_grad, dev_bias_grad, bias_size, cudaMemcpyDeviceToHost));
298  // Free up the remaining resources.
299  cudnnAssert(cudnnDestroyTensorDescriptor(out_grad_desc));
300  cudnnAssert(cudnnDestroyTensorDescriptor(bias_desc));
301  cudaAssert(cudaFree(dev_out_grad));
302  cudaAssert(cudaFree(dev_bias_grad));
303  }
313  inline void activation_fwd(Scalar* input, const Array4& dims, cudnnActivationMode_t act_mode,
314  Scalar coeff, /* out */ Scalar* output) const {
315  // Create and set the activation descriptor.
316  cudnnActivationDescriptor_t act_desc;
317  cudnnAssert(cudnnCreateActivationDescriptor(&act_desc));
318  cudnnAssert(cudnnSetActivationDescriptor(act_desc, act_mode, NAN_PROP, (double) coeff));
319  // Create and set the input/output tensor descriptor.
320  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
321  // Allocate the necessary device memory and copy the input tensor from the host to the device.
322  Scalar* dev_input;
323  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
324  cudaAssert(cudaMalloc(&dev_input, size));
325  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
326  // Perform the activation.
327  cudnnAssert(cudnnActivationForward(handle, act_desc, &alpha, tens_desc, dev_input, &beta, tens_desc, dev_input));
328  // Copy the output tensor from the device to the host.
329  cudaAssert(cudaMemcpy(output, dev_input, size, cudaMemcpyDeviceToHost));
330  // Free up the resources.
331  cudnnAssert(cudnnDestroyActivationDescriptor(act_desc));
332  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
333  cudaAssert(cudaFree(dev_input));
334  }
346  inline void activation_bwd(Scalar* input, Scalar* output, Scalar* out_grad, const Array4& dims,
347  cudnnActivationMode_t act_mode, Scalar coeff, /* out */ Scalar* prev_out_grad) const {
348  // Create and set the activation and tensor descriptors.
349  cudnnActivationDescriptor_t act_desc;
350  cudnnAssert(cudnnCreateActivationDescriptor(&act_desc));
351  cudnnAssert(cudnnSetActivationDescriptor(act_desc, act_mode, NAN_PROP, (double) coeff));
352  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
353  // Allocate and populate the required memory on the device.
354  Scalar* dev_input;
355  Scalar* dev_output;
356  Scalar* dev_out_grad;
357  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
358  cudaAssert(cudaMalloc(&dev_input, size));
359  cudaAssert(cudaMalloc(&dev_output, size));
360  cudaAssert(cudaMalloc(&dev_out_grad, size));
361  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
362  cudaAssert(cudaMemcpy(dev_output, output, size, cudaMemcpyHostToDevice));
363  cudaAssert(cudaMemcpy(dev_out_grad, out_grad, size, cudaMemcpyHostToDevice));
364  // Apply the backwards activation function.
365  cudnnAssert(cudnnActivationBackward(handle, act_desc, &alpha, tens_desc, dev_output, tens_desc, dev_out_grad,
366  tens_desc, dev_input, &beta, tens_desc, dev_out_grad));
367  // Copy the result to the host.
368  cudaAssert(cudaMemcpy(prev_out_grad, dev_out_grad, size, cudaMemcpyDeviceToHost));
369  // Free up the resources.
370  cudnnAssert(cudnnDestroyActivationDescriptor(act_desc));
371  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
372  cudaAssert(cudaFree(dev_input));
373  cudaAssert(cudaFree(dev_output));
374  cudaAssert(cudaFree(dev_out_grad));
375  }

480  inline void pooling2d_fwd(Scalar* input, const Array4& input_dims, const Array4& output_dims, cudnnPoolingMode_t pool_mode,
481  std::size_t window_height, std::size_t window_width, std::size_t vertical_padding, std::size_t horizontal_padding,

489  cudnnAssert(cudnnCreatePoolingDescriptor(&pool_desc));

532  std::size_t horizontal_padding, std::size_t vertical_stride, std::size_t horizontal_stride,

783  inline void dropout_fwd(Scalar* input, const Array4& dims, Scalar dropout, /* out */ Scalar* output,
784  /* out */ std::vector<Scalar>& reserve) const {
785  // Create and set the input/output/reserve tensor descriptor.
786  cudnnTensorDescriptor_t tens_desc = setup_tens_desc(dims);
787  // Create and set the dropout descriptor.
788  cudnnDropoutDescriptor_t dropout_desc;
789  cudnnAssert(cudnnCreateDropoutDescriptor(&dropout_desc));
790  cudnnAssert(cudnnSetDropoutDescriptor(dropout_desc, handle, (float) dropout, nullptr, 0, 0));
791  // Calculate the required reserve size.
792  std::size_t reserve_size;
793  cudnnAssert(cudnnDropoutGetReserveSpaceSize(tens_desc, &reserve_size));
794  // Allocate the memory for the device arrays and reserve space.
795  Scalar* dev_input;
796  Scalar* dev_output;
797  Scalar* dev_reserve;
798  const std::size_t size = dims[0] * dims[1] * dims[2] * dims[3] * SCALAR_SIZE;
799  cudaAssert(cudaMalloc(&dev_input, size));
800  cudaAssert(cudaMalloc(&dev_output, size));
801  cudaAssert(cudaMalloc(&dev_reserve, reserve_size));
802  // Copy the contents of the input tensor from the host to the device.
803  cudaAssert(cudaMemcpy(dev_input, input, size, cudaMemcpyHostToDevice));
804  // Perform the dropout.
805  cudnnAssert(cudnnDropoutForward(handle, dropout_desc, tens_desc, dev_input, tens_desc, dev_output,
806  dev_reserve, reserve_size));
807  // Copy the results to the host.
808  cudaAssert(cudaMemcpy(output, dev_output, size, cudaMemcpyDeviceToHost));
809  reserve = std::vector<Scalar>(reserve_size / SCALAR_SIZE);
810  cudaAssert(cudaMemcpy(reserve.data(), dev_reserve, size, cudaMemcpyDeviceToHost));
811  // Free resources.
812  cudnnAssert(cudnnDestroyTensorDescriptor(tens_desc));
813  cudnnAssert(cudnnDestroyDropoutDescriptor(dropout_desc));
814  cudaAssert(cudaFree(dev_input));
815  cudaAssert(cudaFree(dev_output));
816  cudaAssert(cudaFree(dev_reserve));
817  }

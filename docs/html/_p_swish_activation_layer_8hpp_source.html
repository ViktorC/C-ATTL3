77  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
78  assert((Dimensions<std::size_t,Base::DATA_RANK>(in.dimensions()).template demote<>()) == Base::dims);
79  assert(in.dimension(0) > 0);
80  conversion_dims[0] = in.dimension(0);
81  in_mat_cache = MatrixMap<Scalar>(in.data(), conversion_dims[0], in.size() / conversion_dims[0]);
82  sig_out_mat_cache = ((in_mat_cache * (-Base::params->get_values()).asDiagonal()).array().exp() + 1).inverse();
83  Matrix<Scalar> out_mat = in_mat_cache.cwiseProduct(sig_out_mat_cache);
84  return TensorMap<Scalar,Root::DATA_RANK>(out_mat.data(), conversion_dims);
85  }
86  inline typename Root::Data pass_back(typename Root::Data out_grad) {
87  assert((Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()) == Base::dims);
88  assert(out_grad.dimension(0) > 0 && conversion_dims[0] == out_grad.dimension(0));
89  MatrixMap<Scalar> out_grad_mat(out_grad.data(), conversion_dims[0], out_grad.size() / conversion_dims[0]);
90  Matrix<Scalar> one_min_sig_out_mat = 1 - sig_out_mat_cache.array();
91  Base::params->accumulate_grad(sig_out_mat_cache.cwiseProduct(one_min_sig_out_mat).cwiseProduct(in_mat_cache)
92  .cwiseProduct(in_mat_cache).cwiseProduct(out_grad_mat).colwise().sum());
93  if (Base::is_input_layer())
94  return typename Root::Data();
95  Matrix<Scalar> prev_out_grad_mat = sig_out_mat_cache.cwiseProduct(((one_min_sig_out_mat *
96  Base::params->get_values().asDiagonal()).array() *
97  in_mat_cache.array() + 1).matrix()).cwiseProduct(out_grad_mat);
98  return TensorMap<Scalar,Root::DATA_RANK>(prev_out_grad_mat.data(), conversion_dims);
99  }
100 private:
101  RankwiseArray conversion_dims;
102  // Staged computation caches.
103  Matrix<Scalar> in_mat_cache, sig_out_mat_cache;
104 };

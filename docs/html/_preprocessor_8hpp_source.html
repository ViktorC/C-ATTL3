56 template<typename Scalar, std::size_t Rank, bool Standardize = false>
57 class NormalizationPreprocessor : public Preprocessor<Scalar,Rank,false> {
58 public:
62  inline NormalizationPreprocessor(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
63  epsilon(epsilon) {
64  assert(epsilon > 0 && "epsilon must be greater than 0");
65  }
66  virtual ~NormalizationPreprocessor() = default;
67  inline virtual void fit(const Tensor<Scalar,Rank + 1>& data) {
68  std::size_t rows = data.dimension(0);
69  assert(rows > 0);
70  dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
71  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
72  means = data_mat.colwise().mean();
73  if (Standardize)
74  sd = (data_mat.rowwise() - means).array().square().colwise().mean().sqrt();
75  }
76  inline virtual void transform(Tensor<Scalar,Rank + 1>& data) const {
77  std::size_t rows = data.dimension(0);
78  assert(rows > 0);
79  Dimensions<std::size_t,Rank> demoted_dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
80  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
81  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
82  data_mat = data_mat.rowwise() - means;
83  if (Standardize)
84  data_mat *= (sd.array() + epsilon).inverse().matrix().asDiagonal();
85  }
86 protected:
87  const Scalar epsilon;
88  Dimensions<std::size_t,Rank> dims;
89  RowVector<Scalar> means;
90  RowVector<Scalar> sd;
91 };

96 template<typename Scalar, bool Standardize>
97 class NormalizationPreprocessor<Scalar,3,Standardize> {
98 public:
102  inline NormalizationPreprocessor(Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
103  epsilon(epsilon) {
104  assert(epsilon > 0 && "epsilon must be greater than 0");
105  }
106  virtual ~NormalizationPreprocessor() = default;
107  inline virtual void fit(const Tensor<Scalar,4>& data) {
108  std::size_t rows = data.dimension(0);
109  assert(rows > 0);
110  dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
111  std::size_t depth = dims(2);
112  means = Matrix<Scalar>(depth, dims.get_volume() / depth);
113  sd = Matrix<Scalar>(means.rows(), means.cols());
114  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
115  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
116  for (std::size_t i = 0; i < depth; ++i) {
117  offsets[3] = i;
118  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
119  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
120  means.row(i) = data_slice_mat.colwise().mean();
121  if (Standardize)
122  sd.row(i) = (data_slice_mat.rowwise() - means.row(i)).array().square().colwise().mean().sqrt();
123  }
124  }
125  inline virtual void transform(Tensor<Scalar,4>& data) const {
126  std::size_t rows = data.dimension(0);
127  Dimensions<std::size_t,3> demoted_dims = (Dimensions<std::size_t,4>(data.dimensions()).template demote<>());
128  assert(demoted_dims == dims && "mismatched fit and transform input tensor dimensions");
129  assert(rows > 0);
130  std::size_t depth = data.dimension(3);
131  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
132  std::array<std::size_t,4> extents({ rows, dims(0), dims(1), 1u });
133  for (std::size_t i = 0; i < depth; ++i) {
134  offsets[3] = i;
135  Tensor<Scalar,4> data_slice = data.slice(offsets, extents);
136  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
137  data_slice_mat = data_slice_mat.rowwise() - means.row(i);
138  if (Standardize)
139  data_slice_mat *= (sd.row(i).array() + epsilon).inverse().matrix().asDiagonal();
140  data.slice(offsets, extents) = std::move(data_slice);
141  }
142  }
143 protected:
144  const Scalar epsilon;
145  Dimensions<std::size_t,3> dims;
146  Matrix<Scalar> means;
147  Matrix<Scalar> sd;
148 };

154 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool Whiten = false>
155 class PCAPreprocessorBase : public NormalizationPreprocessor<Scalar,Rank,Standardize> {
156 protected:
157  typedef NormalizationPreprocessor<Scalar,Rank,Standardize> Base;
158  inline PCAPreprocessorBase(Scalar min_rel_var_to_retain, Scalar epsilon) :
159  Base::NormalizationPreprocessor(epsilon),
160  min_rel_var_to_retain(min_rel_var_to_retain),
161  reduce_dims(internal::NumericUtils<Scalar>::decidedly_lesser(min_rel_var_to_retain, (Scalar) 1)) {
162  assert(min_rel_var_to_retain > 0 && min_rel_var_to_retain <= 1 &&
163  "the minimum relative variance to be retained must be greater "
164  "then 0 and less than or equal to 1");
165  }
166  inline void _fit(Tensor<Scalar,Rank + 1> data, int i) {
167  std::size_t rows = data.dimension(0);
168  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
169  Matrix<Scalar> normalized_data = data_mat.rowwise() - Base::means.row(i);
170  if (Standardize)
171  normalized_data *= (Base::sd.row(i).array() + Base::epsilon).inverse().matrix().asDiagonal();
172  // Compute the covariance matrix.
173  Matrix<Scalar> cov = normalized_data.transpose() * normalized_data / normalized_data.rows();
174  // Eigen decomposition.
175  EigenSolver<Scalar> eigen_solver(cov);
176  // Determine the number of components to retain.
177  const ColVector<Scalar>& eigen_values = eigen_solver.eigenvalues();
178  int dims_to_retain = 0;
179  if (reduce_dims) {
180  const Scalar min_var_to_retain = eigen_values.sum() * min_rel_var_to_retain;
181  Scalar var = 0;
182  for (; dims_to_retain < eigen_values.rows(); ++dims_to_retain) {
183  // The eigen values are in ascending order.
184  var += eigen_values(eigen_values.rows() - (1 + dims_to_retain));
185  if (internal::NumericUtils<Scalar>::decidedly_greater(var, min_var_to_retain))
186  break;
187  }
188  } else
189  dims_to_retain = eigen_values.rows();
190  // The eigen vectors are sorted by the magnitude of their corresponding eigen values.
191  ed_vec[i].eigen_basis = eigen_solver.eigenvectors().rightCols(dims_to_retain);
192  if (Whiten) // The eigen values are only needed if whitening is enabled.
193  ed_vec[i].eigen_values = eigen_values.bottomRows(dims_to_retain).transpose();
194  }
195  inline Tensor<Scalar,Rank + 1> _transform(Tensor<Scalar,Rank + 1> data, int i) const {
196  std::size_t rows = data.dimension(0);
197  Dimensions<std::size_t,Rank + 1> output_dims;
198  output_dims(0) = rows;
199  if (!reduce_dims)
200  output_dims = data.dimensions();
201  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
202  data_mat *= ed_vec[i].eigen_basis;
203  if (Whiten)
204  data_mat *= (ed_vec[i].eigen_values.array() + Base::epsilon).sqrt().inverse().matrix().asDiagonal();
205  if (reduce_dims)
206  output_dims(1) = data_mat.cols();
207  return TensorMap<Scalar,Rank + 1>(data.data(), output_dims);
208  }
209  Scalar min_rel_var_to_retain;
210  bool reduce_dims;
211  struct EigenDecomposition {
212  Matrix<Scalar> eigen_basis;
213  RowVector<Scalar> eigen_values;
214  };
215  std::vector<EigenDecomposition> ed_vec;
216 };

221 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool Whiten = false>
222 class PCAPreprocessor : public PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> {
223  typedef PCAPreprocessorBase<Scalar,Rank,Standardize,Whiten> Base;
224  typedef typename Base::Base Root;
225 public:
233  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
234  Base::PCAPreprocessorBase(min_rel_var_to_retain, epsilon) { }
235  inline void fit(const Tensor<Scalar,Rank + 1>& data) {
236  Root::fit(data);
237  Base::ed_vec = std::vector<typename Base::EigenDecomposition>(1);
238  Base::_fit(data, 0);
239  }
240  inline void transform(Tensor<Scalar,Rank + 1>& data) const {
241  Root::transform(data);
242  data = Base::_transform(std::move(data), 0);
243  }
244 };

249 template<typename Scalar, bool Standardize, bool Whiten>
250 class PCAPreprocessor<Scalar,3,Standardize,Whiten> : public PCAPreprocessorBase<Scalar,3,Standardize,Whiten> {
251  typedef PCAPreprocessorBase<Scalar,3,Standardize,Whiten> Base;
252  typedef typename Base::Base Root;
253 public:
262  inline PCAPreprocessor(Scalar min_rel_var_to_retain = 1, Scalar epsilon = internal::NumericUtils<Scalar>::EPSILON2) :
263  Base::PCAPreprocessorBase(min_rel_var_to_retain, epsilon) { }
264  inline void fit(const Tensor<Scalar,4>& data) {
265  assert((!Base::reduce_dims || data.dimension(3) == 1) && "cannot reduce the dimensionality of multi-channel tensors");
266  Root::fit(data);
267  std::size_t channels = Root::dims(2);
268  Base::ed_vec = std::vector<typename Base::EigenDecomposition>(channels);
269  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
270  std::array<std::size_t,4> extents({ data.dimension(0), Root::dims(0), Root::dims(1), 1u });
271  for (std::size_t i = 0; i < channels; ++i) {
272  offsets[3] = i;
273  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
274  Base::_fit(std::move(data_slice_i), i);
275  }
276  }
277  inline void transform(Tensor<Scalar,4>& data) const {
278  Root::transform(data);
279  if (Base::reduce_dims) {
280  Dimensions<std::size_t,3> output_dims({ data.dimension(1) * data.dimension(2), 1u, 1u });
281  data = Base::_transform(std::move(data), 0);
282  } else {
283  std::size_t rows = data.dimension(0);
284  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
285  std::array<std::size_t,4> extents({ rows, Root::dims(0), Root::dims(1), 1u });
286  for (std::size_t i = 0; i < Root::dims(2); ++i) {
287  offsets[3] = i;
288  Tensor<Scalar,4> data_slice_i = data.slice(offsets, extents);
289  data.slice(offsets, extents) = Base::_transform(std::move(data_slice_i), i);
290  }
291  }
292  }
293 };

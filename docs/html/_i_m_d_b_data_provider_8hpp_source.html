82  inline static VocabSharedPtr build_vocab(std::string vocab_path) {
83  std::ifstream vocab_stream(vocab_path);
84  assert(vocab_stream.is_open());
85  std::map<std::string,std::size_t> vocab;
86  // Reserve the first two indices for padding and unknown words.
87  vocab.emplace(std::make_pair("<PAD>", +PAD_IND));
88  vocab.emplace(std::make_pair("<UNK>", +UNK_IND));
89  std::size_t index = 2;
90  std::string word;
91  while (vocab_stream >> word)
92  vocab.emplace(std::make_pair(word, index++));
93  return std::make_shared<const std::map<std::string,std::size_t>>(std::move(vocab));

103  inline static std::string convert_to_text(const Tensor<Scalar,3>& obs, VocabSharedPtr vocab) {
104  std::stringstream text_stream;
105  std::string separator("");
106  for (std::size_t i = 0; i < obs.dimension(0); ++i) {
107  for (std::size_t j = 0; j < obs.dimension(1); ++j) {
108  for (std::size_t k = 0; k < obs.dimension(2); ++k) {
109  if (obs(i,j,k) == 1) {
110  for (auto& entry : *vocab) {
111  if (entry.second == k) {
112  text_stream << separator << entry.first;
113  separator = " ";
114  }
115  }
116  }
117  }
118  }
119  }
120  return text_stream.str();
121  }

156  inline static void clean_document(std::stringstream& document) {
157  std::string doc_string = document.str();
158  std::transform(doc_string.begin(), doc_string.end(), doc_string.begin(),
159  static_cast<int (*)(int)>(std::tolower));
160  // Replace illegal character sequences by white spaces.
161  static std::regex illegal_regex("(<br />)+|([^a-zA-Z-'!\?]+)");
162  doc_string = std::regex_replace(doc_string, illegal_regex, " ");
163  // Add a white space before the supported punctuation marks.
164  static std::regex punct_regex("([!\?]{1})");
165  doc_string = std::regex_replace(doc_string, punct_regex, " $1");
166  document.str(doc_string);
167  }
168  inline DataPair<Scalar,1,true> _get_data(const std::string& file_name, std::ifstream& file_stream,
169  std::size_t batch_size) {
170  assert(batch_size > 0);
171  const bool fixed_seq_length = seq_length != 0;
172  typename Root::Data obs(1, (fixed_seq_length ? seq_length : +PREALLOC_SEQ_LENGTH), Base::obs_dims(0u));
173  typename Root::Data obj(1, 1, Base::obj_dims(0u));
174  obs.setZero();
175  // Parse the rating from the name of the file.
176  std::size_t last_under_score = file_name.find_last_of('_');
177  std::size_t last_period = file_name.find_last_of('.');
178  std::string rating_string = file_name.substr(last_under_score + 1,
179  last_period - last_under_score - 1);
180  unsigned rating = (unsigned) std::stoi(rating_string);
181  switch (ObjType) {
182  case BINARY:
183  obj(0u,0u,0u) = (Scalar) (rating > 5);
184  break;
185  case SMOOTH:
186  obj(0u,0u,0u) = ((Scalar) (rating - 1)) / 9;
187  break;
188  case CATEGORICAL:
189  obj.setZero();
190  obj(0u,0u,rating) = (Scalar) 1;
191  break;
192  default:
193  assert(false);
194  }
195  // Read the document into a string so it can be pre-processed.
196  std::stringstream doc_stream;
197  doc_stream << file_stream.rdbuf();
198  clean_document(doc_stream);
199  // Tokenize the document.
200  std::size_t time_step = 0;
201  std::string word;
202  while (doc_stream >> word && (time_step < seq_length || !fixed_seq_length)) {
203  std::size_t ind;
204  Vocab::const_iterator val = vocab->find(word);
205  ind = (val != vocab->end()) ? val->second : +UNK_IND;
206  if (!fixed_seq_length && time_step >= obs.dimension(1)) {
207  typename Root::Data extra_obs(1, +PREALLOC_SEQ_LENGTH, Base::obs_dims(0u));
208  extra_obs.setZero();
209  obs = typename Root::Data(obs.concatenate(std::move(extra_obs), 1));
210  }
211  obs(0u,time_step++,ind) = (Scalar) 1;
212  }
213  if (fixed_seq_length) {
214  for (; time_step < seq_length; ++time_step)
215  obs(0u,time_step,+PAD_IND) = (Scalar) 1;
216  } else {
217  if (time_step < obs.dimension(1)) {
218  RankwiseArray offsets({ 0u, 0u, 0u });
219  RankwiseArray extents({ 1u, time_step, Base::obs_dims(0u) });
220  obs = typename Root::Data(obs.slice(offsets, extents));
221  }
222  }
223  return std::make_pair(std::move(obs), std::move(obj));
224  }

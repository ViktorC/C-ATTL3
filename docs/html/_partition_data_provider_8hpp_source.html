25 class PartitionDataProvider : public DataProvider<Scalar,Rank,Sequential> {
26  typedef DataProvider<Scalar,Rank,Sequential> Base;
27 public:
28  inline PartitionDataProvider(Base& orig_provider, std::size_t offset, std::size_t length) :
29  orig_provider(orig_provider),
30  offset(offset),
31  length(length) {
32  assert(length > 0);
33  reset();
34  }
35  inline const typename Base::Dims& get_obs_dims() const {
36  return orig_provider.get_obs_dims();
37  }
38  inline const typename Base::Dims& get_obj_dims() const {
39  return orig_provider.get_obj_dims();
40  }
41  inline bool has_more() {
42  return instances_read < length && orig_provider.has_more();
43  }
44  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
45  std::size_t instances_to_read = std::min(batch_size, length - instances_read);
46  instances_read += instances_to_read;
47  return orig_provider.get_data(instances_to_read);
48  }
49  inline void reset() {
50  orig_provider.reset();
51  orig_provider.skip(offset);
52  instances_read = 0;
53  }
54  inline void skip(std::size_t instances) {
55  orig_provider.skip(instances);
56  }
57 private:
58  Base& orig_provider;
59  const std::size_t offset, length;
60  std::size_t instances_read;
61 };

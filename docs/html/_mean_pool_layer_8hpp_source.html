22 template<typename Scalar, std::size_t Rank>
23 class MeanPoolLayerBase : public PoolLayer<Scalar,Rank> {
24  typedef Layer<Scalar,Rank> Root;
25  typedef PoolLayer<Scalar,Rank> Base;
26 public:
27  inline void empty_cache() { }
28 protected:
29  inline MeanPoolLayerBase(const typename Root::Dims& input_dims, std::size_t receptor_height,
30  std::size_t receptor_width, std::size_t vertical_stride, std::size_t horizontal_stride) :
31  Base::PoolLayer(input_dims, receptor_height, receptor_width, vertical_stride,
32  horizontal_stride),
33  receptor_area(receptor_height * receptor_width) { }
34  inline void _init_cache() { }
35  inline Tensor<Scalar,4> _reduce(const Tensor<Scalar,4>& patch, std::size_t patch_ind) {
36  Tensor<Scalar,2> reduced_patch = patch.mean(Base::reduction_ranks);
37  return TensorMap<Scalar,4>(reduced_patch.data(), Base::reduced_patch_extents);
38  }
39  inline Tensor<Scalar,4> _d_reduce(const Tensor<Scalar,4>& grad, std::size_t patch_ind) {
40  return (grad / (Scalar) receptor_area).broadcast(Base::broadcast);
41  }
42 private:
43  std::size_t receptor_area;
44 };

49 template<typename Scalar, std::size_t Rank = 3>
50 class MeanPoolLayer : public MeanPoolLayerBase<Scalar,Rank> {
51  typedef Layer<Scalar,3> Root;
52  typedef PoolLayer<Scalar,3> PoolBase;
53  typedef MeanPoolLayerBase<Scalar,3> MeanPoolBase;
54 public:
64  inline MeanPoolLayer(const typename Root::Dims& input_dims, std::size_t receptor_height = 2,
65  std::size_t receptor_width = 2, std::size_t vertical_stride = 2, std::size_t horizontal_stride = 2) :
66  MeanPoolBase::MeanPoolLayerBase(input_dims, receptor_height, receptor_width, vertical_stride,
67  horizontal_stride) { }
68  inline Root* clone() const {
69  return new MeanPoolLayer(*this);
70  }
71  inline typename Root::Data pass_forward(typename Root::Data in, bool training) {
72  assert((Dimensions<std::size_t,4>(in.dimensions()).template demote<>()) == PoolBase::input_dims);
73  assert(in.dimension(0) > 0);
74  batch_size = in.dimension(0);
75  return PoolBase::_pass_forward(std::move(in), training);
76  }
77  inline typename Root::Data pass_back(typename Root::Data out_grad) {
78  assert((Dimensions<std::size_t,4>(out_grad.dimensions()).template demote<>()) == PoolBase::output_dims);
79  assert(out_grad.dimension(0) > 0 && batch_size == out_grad.dimension(0));
80  if (PoolBase::is_input_layer())
81  return typename Root::Data();
82  return PoolBase::_pass_back(std::move(out_grad));
83  }
84 private:
85  std::size_t batch_size;
86 };

24 template<typename Scalar, std::size_t Rank, bool Standardize = false, bool PerLastRank = (Rank == 3)>
25 class NormalizationPreprocessor : public Preprocessor<Scalar,Rank,false> {
26 public:
30  inline NormalizationPreprocessor(Scalar epsilon = NumericUtils<Scalar>::EPSILON2) :
31  epsilon(epsilon) {
32  assert(epsilon > 0 && "epsilon must be greater than 0");
33  }
34  virtual ~NormalizationPreprocessor() = default;
35  inline virtual void fit(const Tensor<Scalar,Rank + 1>& data) {
36  auto rows = data.dimension(0);
37  assert(rows > 0);
38  dims = (Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>());
39  channels = dims(Rank - 1);
40  means = Matrix<Scalar>(channels, dims.get_volume() / channels);
41  sd = Matrix<Scalar>(means.rows(), means.cols());
42  if (channels == 1) {
43  Tensor<Scalar,Rank + 1> data_copy(data);
44  MatrixMap<Scalar> data_mat(data_copy.data(), rows, data.size() / rows);
45  means.row(0) = data_mat.colwise().mean();
46  if (Standardize)
47  sd.row(0) = (data_mat.rowwise() - means.row(0)).array().square().colwise().mean().sqrt();
48  } else {
49  std::array<std::size_t,Rank + 1> offsets;
50  offsets.fill(0);
51  auto extents = data.dimensions();
52  extents[Rank] = 1;
53  for (std::size_t i = 0; i < channels; ++i) {
54  offsets[Rank] = i;
55  Tensor<Scalar,Rank + 1> data_slice = data.slice(offsets, extents);
56  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
57  means.row(i) = data_slice_mat.colwise().mean();
58  if (Standardize)
59  sd.row(i) = (data_slice_mat.rowwise() - means.row(i)).array().square().colwise().mean().sqrt();
60  }
61  }
62  }
63  inline virtual void transform(Tensor<Scalar,Rank + 1>& data) const {
64  auto rows = data.dimension(0);
65  assert(rows > 0);
66  assert((Dimensions<std::size_t,Rank + 1>(data.dimensions()).template demote<>()) == dims &&
67  "mismatched fit and transform input tensor dimensions");
68  if (channels == 1) {
69  MatrixMap<Scalar> data_mat(data.data(), rows, data.size() / rows);
70  data_mat = data_mat.rowwise() - means.row(0);
71  if (Standardize)
72  data_mat *= (sd.row(0).array() + epsilon).inverse().matrix().asDiagonal();
73  } else {
74  std::array<std::size_t,Rank + 1> offsets;
75  offsets.fill(0);
76  auto extents = data.dimensions();
77  extents[Rank] = 1;
78  for (std::size_t i = 0; i < channels; ++i) {
79  offsets[Rank] = i;
80  Tensor<Scalar,Rank + 1> data_slice = data.slice(offsets, extents);
81  MatrixMap<Scalar> data_slice_mat(data_slice.data(), rows, data_slice.size() / rows);
82  data_slice_mat = data_slice_mat.rowwise() - means.row(i);
83  if (Standardize)
84  data_slice_mat *= (sd.row(i).array() + epsilon).inverse().matrix().asDiagonal();
85  data.slice(offsets, extents) = std::move(data_slice);
86  }
87  }
88  }
89 protected:
90  const Scalar epsilon;
91  Dimensions<std::size_t,Rank> dims;
92  Matrix<Scalar> means;
93  Matrix<Scalar> sd;
94  std::size_t channels;
95 };

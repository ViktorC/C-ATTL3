41  inline bool has_more() {
42  if (current_file_stream_pair_has_more())
43  return true;
44  ++current_file_pair_ind;
45  for (; current_file_pair_ind < file_pairs.size(); ++current_file_pair_ind) {
46  init_current_file_stream_pair();
47  if (current_file_stream_pair_has_more())
48  return true;
49  }
50  return false;
51  }
52  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
53  if (!has_more())
54  throw std::out_of_range("no more data left to fetch");
55  const FilePair& current_file_pair = file_pairs[current_file_pair_ind];
56  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(current_file_pair.first,
57  current_file_stream_pair.first, current_file_pair.second, current_file_stream_pair.second,
58  batch_size);
59  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
60  while (data_pair.first.dimension(0) < batch_size && has_more()) {
61  const FilePair& new_current_file_pair = file_pairs[current_file_pair_ind];
62  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(new_current_file_pair.first,
63  current_file_stream_pair.first, new_current_file_pair.second, current_file_stream_pair.second,
64  batch_size - data_pair.first.dimension(0));
65  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
66  typename Base::Data obs_concat = data_pair.first.concatenate(std::move(add_data_pair.first), 0);
67  data_pair.first = std::move(obs_concat);
68  typename Base::Data obj_concat = data_pair.second.concatenate(std::move(add_data_pair.second), 0);
69  data_pair.second = std::move(obj_concat);
70  }
71  return data_pair;
72  }
73  inline void reset() {
74  current_file_pair_ind = 0;
75  init_current_file_stream_pair();
76  _set_to_beg(current_file_stream_pair.first, current_file_stream_pair.second);
77  }
78  inline void skip(std::size_t instances) {
79  if (!has_more())
80  return;
81  std::size_t skipped = _skip(current_file_stream_pair.first,current_file_stream_pair.second,
82  instances);
83  while (skipped < instances && has_more())
84  skipped += _skip(current_file_stream_pair.first, current_file_stream_pair.second,
85  instances - skipped);
86  }

142  return current_file_stream_pair.first && current_file_stream_pair.first.peek() != EOF &&
143  current_file_stream_pair.second && current_file_stream_pair.second.peek() != EOF;
144  }
145  void init_current_file_stream_pair() {
146  const FilePair& file_pair = file_pairs[current_file_pair_ind];
147  std::ifstream obs_stream(file_pair.first, ObsBinary ? std::ios::binary : std::ios::in);
148  assert(obs_stream.is_open());
149  std::ifstream obj_stream(file_pair.second, ObjBinary ? std::ios::binary : std::ios::in);
150  assert(obj_stream.is_open());
151  current_file_stream_pair = std::make_pair(std::move(obs_stream), std::move(obj_stream));
152  }
153  std::vector<FilePair> file_pairs;

155  FileStreamPair current_file_stream_pair;

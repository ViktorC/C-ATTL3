48  inline DenseNeuralNetwork(std::vector<Module>&& modules, bool foremost = true) :
49  modules(std::move(modules)),
50  foremost(foremost) {
51  assert(this->modules.size() > 0 && "modules must contain at least 1 element");
52  Base& first_module = *this->modules[0];
53  input_dims = first_module.get_input_dims();
54  typename Base::Dims output_dims = first_module.get_output_dims();
55  output_dims(+CONCAT_RANK) += input_dims(+CONCAT_RANK);
56  if (ConcatType == DENSE_LOWEST_RANK) {
57  for (std::size_t i = Rank - 1; i > +CONCAT_RANK; --i)
58  assert(input_dims(i) == output_dims(i));
59  } else {
60  for (std::size_t i = 0; i < +CONCAT_RANK; ++i)
61  assert(input_dims(i) == output_dims(i));
62  }
63  for (std::size_t i = 1; i < this->modules.size(); ++i) {
64  Base& module = *this->modules[i];
65  const typename Base::Dims& module_input_dims = module.get_input_dims();
66  assert(module_input_dims == output_dims && "incompatible module dimensions");
67  output_dims(+CONCAT_RANK) += module.get_output_dims()(+CONCAT_RANK);
68  module.set_foremost(false);
69  }
70  this->output_dims = output_dims;
71  first_module.set_foremost(foremost);
72  }

131  inline typename Base::Data propagate(typename Base::Data input, bool training) {
132  assert(input_dims == (Dimensions<std::size_t,Base::DATA_RANK>(input.dimensions()).template demote<>()));
133  for (std::size_t i = 0; i < modules.size(); ++i) {
134  typename Base::Data concat = input.concatenate(modules[i]->propagate(input, training), +CONCAT_BATCH_RANK);
135  input = std::move(concat);
136  }
137  return input;
138  }
139  inline typename Base::Data backpropagate(typename Base::Data out_grad) {
140  assert(output_dims == (Dimensions<std::size_t,Base::DATA_RANK>(out_grad.dimensions()).template demote<>()));
141  RankwiseArray offsets;
142  RankwiseArray extents = input_dims.template promote<>();
143  offsets.fill(0);
144  extents[0] = out_grad.dimension(0);
145  for (int i = modules.size() - 1; i >= 0; --i) {
146  Base& module = *modules[i];
147  int layer_input_concat_rank_dim = module.get_input_dims()(+CONCAT_RANK);
148  int layer_output_concat_rank_dim = module.get_output_dims()(+CONCAT_RANK);
149  offsets[+CONCAT_BATCH_RANK] = layer_input_concat_rank_dim;
150  extents[+CONCAT_BATCH_RANK] = layer_output_concat_rank_dim;
151  typename Base::Data out_grad_i = out_grad.slice(offsets, extents);
152  offsets[+CONCAT_BATCH_RANK] = 0;
153  extents[+CONCAT_BATCH_RANK] = layer_input_concat_rank_dim;
154  if (foremost && i == 0)
155  module.backpropagate(std::move(out_grad_i));
156  else
157  out_grad = typename Base::Data(out_grad.slice(offsets, extents) +
158  module.backpropagate(std::move(out_grad_i)));
159  }
160  return out_grad;
161  }

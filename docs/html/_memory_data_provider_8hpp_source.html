73  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size = std::numeric_limits<std::size_t>::max()) {
74  if (!has_more())

76  std::size_t max_batch_size = std::min(batch_size, instances - offsets[0]);
77  data_extents[0] = max_batch_size;
78  obj_extents[0] = max_batch_size;

80  typename Base::Data obj_batch = obj->slice(offsets, obj_extents);
81  offsets[0] = std::min(instances, offsets[0] + max_batch_size);
82  return std::make_pair(std::move(data_batch), std::move(obj_batch));
83  }

91  inline void shuffle_tensor_rows() {
92  std::size_t rows = obs->dimension(0);
93  MatrixMap<Scalar> obs_mat(obs->data(), rows, obs->size() / rows);
94  MatrixMap<Scalar> obj_mat(obj->data(), rows, obj->size() / rows);
95  // Create an identity matrix.
96  PermMatrix perm(rows);
97  perm.setIdentity();
98  // Shuffle its indices.
99  std::random_shuffle(perm.indices().data(), perm.indices().data() + perm.indices().size());
100  // And apply the same row permutation transformation to both the observations and the objectives.
101  obs_mat = perm * obs_mat;
102  obj_mat = perm * obj_mat;
103  }

154 class MemoryDataProvider : public DataProvider<Scalar,Rank,Sequential> {
155  typedef DataProvider<Scalar,Rank,Sequential> Base;
156 public:
164  inline MemoryDataProvider(typename Base::DataPtr obs, typename Base::DataPtr obj) :
165  obs(std::move(obs)),
166  obj(std::move(obj)),
167  offsets() {
168  assert(this->obs != nullptr);
169  assert(this->obj != nullptr);
170  assert(this->obs->dimension(0) == this->obj->dimension(0) &&
171  "mismatched data and obj tensor row numbers");
172  Dimensions<std::size_t,Base::DATA_RANK> obs_dims = this->obs->dimensions();
173  Dimensions<std::size_t,Base::DATA_RANK> obj_dims = this->obj->dimensions();
174  this->obs_dims = obs_dims.template demote<Sequential + 1>();
175  this->obj_dims = obj_dims.template demote<Sequential + 1>();
176  instances = (std::size_t) this->obs->dimension(0);
177  offsets.fill(0);
178  data_extents = obs_dims;
179  obj_extents = obj_dims;
180  if (Shuffle)
181  shuffle_tensor_rows();
182  }
183  inline const Dimensions<std::size_t,Rank>& get_obs_dims() const {
184  return obs_dims;
185  }
186  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {
187  return obj_dims;
188  }
189  inline bool has_more() {
190  return offsets[0] < (int) instances;
191  }
192  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
193  if (!has_more())
194  throw std::out_of_range("no more data left to fetch");
195  std::size_t max_batch_size = std::min(batch_size, instances - offsets[0]);
196  data_extents[0] = max_batch_size;
197  obj_extents[0] = max_batch_size;
198  typename Base::Data data_batch = obs->slice(offsets, data_extents);
199  typename Base::Data obj_batch = obj->slice(offsets, obj_extents);
200  offsets[0] = std::min(instances, offsets[0] + max_batch_size);
201  return std::make_pair(std::move(data_batch), std::move(obj_batch));
202  }
203  inline void reset() {
204  offsets[0] = 0;
205  }
206 protected:
207  inline void skip(std::size_t instances) {
208  offsets[0] = std::min((int) this->instances, (int) (offsets[0] + instances));
209  }
210  inline void shuffle_tensor_rows() {
211  std::size_t rows = obs->dimension(0);
212  MatrixMap<Scalar> obs_mat(obs->data(), rows, obs->size() / rows);
213  MatrixMap<Scalar> obj_mat(obj->data(), rows, obj->size() / rows);
214  // Create an identity matrix.
215  internal::PermMatrix perm(rows);
216  perm.setIdentity();
217  // Shuffle its indices.
218  std::random_shuffle(perm.indices().data(), perm.indices().data() + perm.indices().size());
219  // And apply the same row permutation transformation to both the observations and the objectives.
220  obs_mat = perm * obs_mat;
221  obj_mat = perm * obj_mat;
222  }
223 private:
224  typename Base::DataPtr obs;
225  typename Base::DataPtr obj;
226  Dimensions<std::size_t,Rank> obs_dims;
227  Dimensions<std::size_t,Rank> obj_dims;
228  std::size_t instances;
229  std::array<std::size_t,Base::DATA_RANK> offsets;
230  std::array<std::size_t,Base::DATA_RANK> data_extents;
231  std::array<std::size_t,Base::DATA_RANK> obj_extents;
232 };

254  if (!has_more())

257  batch_size);

265  // It has to be evaluated into a temporary due to the dimension incompatibility.
266  typename Base::Data obs_concat =
267  data_pair.first.concatenate(std::move(add_data_pair.first),0);
268  data_pair.first = std::move(obs_concat);
269  typename Base::Data obj_concat =
270  data_pair.second.concatenate(std::move(add_data_pair.second), 0);
271  data_pair.second = std::move(obj_concat);
272  }
273  return data_pair;
274  }

277  std::ifstream& data_stream = data_streams[i];

280  }
281  current_stream = 0;
282  }

491  inline DataPair<Scalar,3,false> _get_data(std::ifstream& obs_input_stream,
492  std::ifstream& obj_input_stream, std::size_t batch_size) {
493  Tensor<Scalar,4> obs(batch_size, 28u, 28u, 1u);
494  Tensor<Scalar,4> obj(batch_size, 10u, 1u, 1u);
495  obj.setZero();
496  std::size_t i;
497  for (i = 0; i < batch_size && obs_input_stream.read(obs_buffer, OBS_INSTANCE_LENGTH); ++i) {
498  // Read and set the label.
499  unsigned char label;
500  obj_input_stream >> label;
501  obj(i,(std::size_t) label,0u,0u) = (Scalar) 1;
502  // Set the image.
503  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(obs_buffer);

511  if (i == batch_size)
512  return std::make_pair(obs, obj);
513  obs_extents[0] = i;
514  obj_extents[0] = i;
515  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
516  }

519  // Skip observations.

521  obs_input_stream.seekg(0, std::ios::end);
522  std::size_t obs_skip_extent = obs_input_stream.tellg() - curr_obs_pos;
523  obs_input_stream.seekg(curr_obs_pos);
524  obs_input_stream.ignore(instances * OBS_INSTANCE_LENGTH);
525  std::size_t skipped_obs = std::min(instances, obs_skip_extent / OBS_INSTANCE_LENGTH);
526  // Skip labels.

528  obj_input_stream.seekg(0, std::ios::end);

530  obj_input_stream.seekg(curr_label_pos);
531  obj_input_stream.ignore(instances * LABEL_INSTANCE_LENGTH);
532  std::size_t skipped_labels = std::min(instances, label_skip_extent / LABEL_INSTANCE_LENGTH);
533  assert(skipped_obs == skipped_labels);
534  return skipped_obs;
535  }

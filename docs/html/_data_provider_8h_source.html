110  inline InMemoryDataProvider(DataPtr obs, DataPtr obj, bool shuffle = true) :
111  obs(std::move(obs)),
112  obj(std::move(obj)),
113  offsets() {
114  assert(this->obs != nullptr);
115  assert(this->obj != nullptr);
116  Utils<Scalar>::template check_dim_validity<DATA_RANKS>(*this->obs);
117  Utils<Scalar>::template check_dim_validity<DATA_RANKS>(*this->obj);
118  assert(this->obs->dimension(0) == this->obj->dimension(0) && "mismatched data and obj tensor row numbers");
119  Dimensions<int,DATA_RANKS> obs_dims = Utils<Scalar>::template get_dims<DATA_RANKS>(*this->obs);
120  Dimensions<int,DATA_RANKS> obj_dims = Utils<Scalar>::template get_dims<DATA_RANKS>(*this->obj);
121  this->obs_dims = obs_dims.template demote<Sequential + 1>();
122  this->obj_dims = obj_dims.template demote<Sequential + 1>();
123  rows = (std::size_t) this->obs->dimension(0);
124  offsets.fill(0);
125  data_extents = obs_dims;
126  obj_extents = obj_dims;
127  if (shuffle) {
128  Utils<Scalar>::template shuffle_tensor_rows<DATA_RANKS>(*this->obs);
129  Utils<Scalar>::template shuffle_tensor_rows<DATA_RANKS>(*this->obj);
130  }
131  }

144  inline DataPair<Scalar,Rank,Sequential> get_data(unsigned batch_size) {
145  int max_batch_size = std::min((int) batch_size, (int) (rows - offsets[0]));
146  data_extents[0] = max_batch_size;
147  obj_extents[0] = max_batch_size;
148  Data data_batch = obs->slice(offsets, data_extents);
149  Data obj_batch = obj->slice(offsets, obj_extents);
150  offsets[0] = std::min((int) rows, (int) offsets[0] + max_batch_size);
151  return std::make_pair(data_batch, obj_batch);
152  }
153  void reset() {

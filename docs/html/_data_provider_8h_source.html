110 class PartitionDataProvider : public DataProvider<Scalar,Rank,Sequential> {
111  typedef DataProvider<Scalar,Rank,Sequential> Base;
112 public:
113  inline PartitionDataProvider(Base& orig_provider, std::size_t offset, std::size_t length) :
114  orig_provider(orig_provider),

116  length(length) {
117  assert(length > 0);
118  reset();

120  inline const Dimensions<std::size_t,Rank>& get_obs_dims() const {
121  return orig_provider.get_obs_dims();

123  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {
124  return orig_provider.get_obj_dims();
125  }
126  inline bool has_more() {
127  return instances_read < length && orig_provider.has_more();
128  }
129  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
130  std::size_t instances_to_read = std::min(batch_size, length - instances_read);
131  instances_read += instances_to_read;
132  return orig_provider.get_data(instances_to_read);
133  }
134  inline void reset() {
135  orig_provider.reset();
136  orig_provider.skip(offset);
137  instances_read = 0;
138  }
139 protected:
140  inline void skip(std::size_t instances) {
141  orig_provider.skip(instances);
142  }
143 private:
144  Base& orig_provider;
145  const std::size_t offset;
146  const std::size_t length;
147  std::size_t instances_read;
148 };

156 class MemoryDataProvider : public DataProvider<Scalar,Rank,Sequential> {
157  typedef DataProvider<Scalar,Rank,Sequential> Base;
158 public:
167  inline MemoryDataProvider(typename Base::DataPtr obs, typename Base::DataPtr obj,
168  bool shuffle = true) :
169  obs(std::move(obs)),
170  obj(std::move(obj)),
171  offsets() {
172  assert(this->obs != nullptr);
173  assert(this->obj != nullptr);
174  Utils<Scalar>::template check_dim_validity<Base::DATA_RANKS>(*this->obs);
175  Utils<Scalar>::template check_dim_validity<Base::DATA_RANKS>(*this->obj);
176  assert(this->obs->dimension(0) == this->obj->dimension(0) &&
177  "mismatched data and obj tensor row numbers");
178  Dimensions<std::size_t,Base::DATA_RANKS> obs_dims =
179  Utils<Scalar>::template get_dims<Base::DATA_RANKS>(*this->obs);
180  Dimensions<std::size_t,Base::DATA_RANKS> obj_dims =
181  Utils<Scalar>::template get_dims<Base::DATA_RANKS>(*this->obj);
182  this->obs_dims = obs_dims.template demote<Sequential + 1>();
183  this->obj_dims = obj_dims.template demote<Sequential + 1>();
184  instances = (std::size_t) this->obs->dimension(0);
185  offsets.fill(0);
186  data_extents = obs_dims;
187  obj_extents = obj_dims;
188  if (shuffle) {
189  Utils<Scalar>::template shuffle_tensor_rows<Base::DATA_RANKS>(*this->obs);
190  Utils<Scalar>::template shuffle_tensor_rows<Base::DATA_RANKS>(*this->obj);
191  }
192  }
193  inline const Dimensions<std::size_t,Rank>& get_obs_dims() const {
194  return obs_dims;
195  }
196  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {
197  return obj_dims;
198  }
199  inline bool has_more() {
200  return offsets[0] < (int) instances;
201  }
202  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
203  if (!has_more())
204  throw std::out_of_range("no more data left to fetch");
205  std::size_t max_batch_size = std::min(batch_size, instances - offsets[0]);
206  data_extents[0] = max_batch_size;
207  obj_extents[0] = max_batch_size;
208  typename Base::Data data_batch = obs->slice(offsets, data_extents);
209  typename Base::Data obj_batch = obj->slice(offsets, obj_extents);
210  offsets[0] = std::min(instances, offsets[0] + max_batch_size);
211  return std::make_pair(data_batch, obj_batch);
212  }
213  inline void reset() {
214  offsets[0] = 0;
215  }
216 protected:
217  inline void skip(std::size_t instances) {
218  offsets[0] = std::min((int) this->instances, (int) (offsets[0] + instances));
219  }
220 private:
221  typename Base::DataPtr obs;
222  typename Base::DataPtr obj;
223  Dimensions<std::size_t,Rank> obs_dims;
224  Dimensions<std::size_t,Rank> obj_dims;
225  std::size_t instances;
226  std::array<std::size_t,Base::DATA_RANKS> offsets;
227  std::array<std::size_t,Base::DATA_RANKS> data_extents;
228  std::array<std::size_t,Base::DATA_RANKS> obj_extents;
229 };

238 class JointFileDataProvider : public DataProvider<Scalar,Rank,Sequential> {
239  typedef DataProvider<Scalar,Rank,Sequential> Base;
240 public:
241  virtual ~JointFileDataProvider() = default;
242  inline bool has_more() {
243  for (; current_stream < data_streams.size(); ++current_stream) {
244  std::ifstream& data_stream = data_streams[current_stream];
245  if (data_stream && data_stream.peek() != EOF)
246  return true;
247  }
248  return false;
249  }
250  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
251  if (!has_more())
252  throw std::out_of_range("no more data left to fetch");
253  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(data_streams[current_stream],
254  batch_size);
255  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
256  /* If the data contains fewer batches than expected, the end of the file has been reached and the
257  * rest of the data should be read from the next file. */
258  while (data_pair.first.dimension(0) < batch_size && has_more()) {
259  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(data_streams[current_stream],
260  batch_size - data_pair.first.dimension(0));
261  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
262  data_pair.first = data_pair.first.concatenate(add_data_pair.first, 0);
263  data_pair.second = data_pair.second.concatenate(add_data_pair.second, 0);
264  }
265  return data_pair;
266  }
267  inline void reset() {
268  std::size_t max_ind = std::min(current_stream, (std::size_t) (data_streams.size() - 1));
269  for (std::size_t i = 0; i <= max_ind; ++i)
270  data_streams[i].seekg(0, std::ios::beg);
271  current_stream = 0;
272  }
273 protected:
274  inline JointFileDataProvider(std::vector<std::string> dataset_paths, bool binary) :
275  data_streams(dataset_paths.size()),
276  current_stream(0) {
277  assert(!dataset_paths.empty());
278  for (std::size_t i = 0; i < dataset_paths.size(); ++i) {
279  data_streams[i] = std::ifstream(dataset_paths[i], binary ? std::ios::binary : std::ios::in);
280  assert(data_streams[i].is_open());
281  }
282  }
283  inline JointFileDataProvider(std::string dataset_path, bool binary) :
284  JointFileDataProvider({ dataset_path }, binary) { }
294  virtual DataPair<Scalar,Rank,Sequential> _get_data(std::ifstream& data_stream,
295  std::size_t batch_size) = 0;
305  virtual std::size_t _skip(std::ifstream& data_stream, std::size_t instances) = 0;
306  inline void skip(std::size_t instances) {
307  if (!has_more())
308  return;
309  std::size_t skipped = _skip(data_streams[current_stream], instances);
310  while (skipped < instances && has_more())
311  skipped += _skip(data_streams[current_stream], instances - skipped);
312  }
313 private:
314  std::vector<std::ifstream> data_streams;
315  std::size_t current_stream;
316 };

430 class CIFAR10DataProvider : public JointFileDataProvider<Scalar,3,false> {
431  typedef DataProvider<Scalar,3,false> Root;
432  typedef JointFileDataProvider<Scalar,3,false> Base;
433  static constexpr std::size_t INSTANCE_LENGTH = 3073;
434 public:
435  inline CIFAR10DataProvider(std::vector<std::string> file_paths) :
436  Base::JointFileDataProvider(file_paths, true),
437  obs({ 32u, 32u, 3u }),
438  obj({ 10u, 1u, 1u }) { }
439  inline const Dimensions<std::size_t,3>& get_obs_dims() const {
440  return obs;
441  }
442  inline const Dimensions<std::size_t,3>& get_obj_dims() const {
443  return obj;
444  }
445 protected:
446  inline DataPair<Scalar,3,false> _get_data(std::ifstream& data_stream,
447  std::size_t batch_size) {
448  Tensor<Scalar,4> obs(batch_size, 32u, 32u, 3u);
449  Tensor<Scalar,4> obj(batch_size, 10u, 1u, 1u);
450  obj.setZero();
451  std::size_t i;
452  for (i = 0; i < batch_size && data_stream.read(buffer, INSTANCE_LENGTH); ++i) {
453  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(buffer);
454  obj(i,u_buffer[0],0u,0u) = (Scalar) 1;
455  std::size_t buffer_ind = 1;
456  for (std::size_t channel = 0; channel < 3; ++channel) {
457  for (std::size_t row = 0; row < 32; ++row) {
458  for (std::size_t column = 0; column < 32; ++column)
459  obs(i,column,row,channel) = (Scalar) u_buffer[buffer_ind++];
460  }
461  }
462  assert(buffer_ind == INSTANCE_LENGTH);
463  }
464  if (i == batch_size)
465  return std::make_pair(obs, obj);
466  std::array<std::size_t,4> offsets({ 0, 0, 0, 0 });
467  std::array<std::size_t,4> obs_extents({ i, 32u, 32u, 3u });
468  std::array<std::size_t,4> obj_extents({ i, 10u, 1u, 1u });
469  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
470  }
471  inline std::size_t _skip(std::ifstream& data_stream, std::size_t instances) {
472  std::streampos curr_pos = data_stream.tellg();
473  data_stream.seekg(0, std::ios::end);
474  std::size_t skip_extent = data_stream.tellg() - curr_pos;
475  data_stream.seekg(curr_pos);
476  data_stream.ignore(instances * INSTANCE_LENGTH);
477  return std::min(instances, skip_extent / INSTANCE_LENGTH);
478  }
479 private:
480  const Dimensions<std::size_t,3> obs;
481  const Dimensions<std::size_t,3> obj;
482  char buffer[INSTANCE_LENGTH];
483 };

50 template<typename Scalar, std::size_t Rank, bool Sequential>
51 class InMemoryDataProvider : public DataProvider<Scalar,Rank,Sequential> {
52  static constexpr std::size_t DATA_RANKS = Rank + Sequential + 1;
53  typedef Tensor<Scalar,DATA_RANKS> Data;
54  typedef TensorPtr<Scalar,Rank,Sequential> DataPtr;
55 public:
56  inline InMemoryDataProvider(DataPtr obs, DataPtr obj, bool shuffle = true) :
57  obs(std::move(obs)),
58  obj(std::move(obj)),
59  offsets() {
60  assert(this->obs != nullptr);
61  assert(this->obj != nullptr);
62  Utils<Scalar>::template check_dim_validity<DATA_RANKS>(*this->obs);
63  Utils<Scalar>::template check_dim_validity<DATA_RANKS>(*this->obj);
64  assert(this->obs->dimension(0) == this->obj->dimension(0) && "mismatched data and obj tensor row numbers");
65  Dimensions<int,DATA_RANKS> obs_dims = Utils<Scalar>::template get_dims<DATA_RANKS>(*this->obs);
66  Dimensions<int,DATA_RANKS> obj_dims = Utils<Scalar>::template get_dims<DATA_RANKS>(*this->obj);
67  this->obs_dims = obs_dims.template demote<Sequential + 1>();
68  this->obj_dims = obj_dims.template demote<Sequential + 1>();
69  rows = (std::size_t) this->obs->dimension(0);
70  offsets.fill(0);
71  data_extents = obs_dims;
72  obj_extents = obj_dims;
73  if (shuffle) {
74  Utils<Scalar>::template shuffle_tensor_rows<DATA_RANKS>(*this->obs);
75  Utils<Scalar>::template shuffle_tensor_rows<DATA_RANKS>(*this->obj);
76  }
77  }

79  return obs_dims;
80  }
81  inline const Dimensions<int,Rank>& get_obj_dims() const {
82  return obj_dims;
83  }
84  inline unsigned instances() const {
85  return rows;
86  }
87  inline bool has_more() const {
88  return offsets[0] < (int) rows;
89  }
90  inline DataPair<Scalar,Rank,Sequential> get_data(unsigned batch_size) {
91  int max_batch_size = std::min((int) batch_size, (int) (rows - offsets[0]));
92  data_extents[0] = max_batch_size;
93  obj_extents[0] = max_batch_size;
94  Data data_batch = obs->slice(offsets, data_extents);
95  Data obj_batch = obj->slice(offsets, obj_extents);
96  offsets[0] = std::min((int) rows, (int) offsets[0] + max_batch_size);
97  return std::make_pair(data_batch, obj_batch);
98  }
99  void reset() {
100  offsets[0] = 0;
101  }

103  DataPtr obs;

106  Dimensions<int,Rank> obj_dims;
107  std::size_t rows;
108  std::array<int,DATA_RANKS> offsets;
109  std::array<int,DATA_RANKS> data_extents;
110  std::array<int,DATA_RANKS> obj_extents;

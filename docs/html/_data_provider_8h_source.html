154 class MemoryDataProvider : public DataProvider<Scalar,Rank,Sequential> {
155  typedef DataProvider<Scalar,Rank,Sequential> Base;
156 public:
165  inline MemoryDataProvider(typename Base::DataPtr obs, typename Base::DataPtr obj,
166  bool shuffle = true) :
167  obs(std::move(obs)),
168  obj(std::move(obj)),
169  offsets() {
170  assert(this->obs != nullptr);
171  assert(this->obj != nullptr);
172  Utils<Scalar>::template check_dim_validity<Base::DATA_RANK>(*this->obs);
173  Utils<Scalar>::template check_dim_validity<Base::DATA_RANK>(*this->obj);
174  assert(this->obs->dimension(0) == this->obj->dimension(0) &&
175  "mismatched data and obj tensor row numbers");
176  Dimensions<std::size_t,Base::DATA_RANK> obs_dims =
177  Utils<Scalar>::template get_dims<Base::DATA_RANK>(*this->obs);
178  Dimensions<std::size_t,Base::DATA_RANK> obj_dims =
179  Utils<Scalar>::template get_dims<Base::DATA_RANK>(*this->obj);
180  this->obs_dims = obs_dims.template demote<Sequential + 1>();
181  this->obj_dims = obj_dims.template demote<Sequential + 1>();
182  instances = (std::size_t) this->obs->dimension(0);
183  offsets.fill(0);
184  data_extents = obs_dims;
185  obj_extents = obj_dims;
186  if (shuffle) {
187  Utils<Scalar>::template shuffle_tensor_rows<Base::DATA_RANK>(*this->obs);
188  Utils<Scalar>::template shuffle_tensor_rows<Base::DATA_RANK>(*this->obj);
189  }
190  }
191  inline const Dimensions<std::size_t,Rank>& get_obs_dims() const {
192  return obs_dims;
193  }
194  inline const Dimensions<std::size_t,Rank>& get_obj_dims() const {
195  return obj_dims;
196  }
197  inline bool has_more() {
198  return offsets[0] < (int) instances;
199  }
200  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
201  if (!has_more())
202  throw std::out_of_range("no more data left to fetch");
203  std::size_t max_batch_size = std::min(batch_size, instances - offsets[0]);
204  data_extents[0] = max_batch_size;
205  obj_extents[0] = max_batch_size;
206  typename Base::Data data_batch = obs->slice(offsets, data_extents);
207  typename Base::Data obj_batch = obj->slice(offsets, obj_extents);
208  offsets[0] = std::min(instances, offsets[0] + max_batch_size);
209  return std::make_pair(data_batch, obj_batch);
210  }
211  inline void reset() {
212  offsets[0] = 0;
213  }
214 protected:
215  inline void skip(std::size_t instances) {
216  offsets[0] = std::min((int) this->instances, (int) (offsets[0] + instances));
217  }
218 private:
219  typename Base::DataPtr obs;
220  typename Base::DataPtr obj;
221  Dimensions<std::size_t,Rank> obs_dims;
222  Dimensions<std::size_t,Rank> obj_dims;
223  std::size_t instances;
224  std::array<std::size_t,Base::DATA_RANK> offsets;
225  std::array<std::size_t,Base::DATA_RANK> data_extents;
226  std::array<std::size_t,Base::DATA_RANK> obj_extents;
227 };

240  inline bool has_more() {
241  for (; current_stream < data_streams.size(); ++current_stream) {
242  std::ifstream& data_stream = data_streams[current_stream];
243  if (data_stream && data_stream.peek() != EOF)
244  return true;
245  }
246  return false;
247  }
248  inline DataPair<Scalar,Rank,Sequential> get_data(std::size_t batch_size) {
249  if (!has_more())
250  throw std::out_of_range("no more data left to fetch");
251  DataPair<Scalar,Rank,Sequential> data_pair = _get_data(data_streams[current_stream],
252  batch_size);
253  assert(data_pair.first.dimension(0) == data_pair.second.dimension(0));
254  /* If the data contains fewer batches than expected, the end of the file has been reached and the
255  * rest of the data should be read from the next file. */
256  while (data_pair.first.dimension(0) < batch_size && has_more()) {
257  DataPair<Scalar,Rank,Sequential> add_data_pair = _get_data(data_streams[current_stream],
258  batch_size - data_pair.first.dimension(0));
259  assert(add_data_pair.first.dimension(0) == add_data_pair.second.dimension(0));
260  // It has to be evaluated into a temporary due to the dimension incompatibility.
261  typename Base::Data obs_concat =
262  data_pair.first.concatenate(std::move(add_data_pair.first),0);
263  data_pair.first = std::move(obs_concat);
264  typename Base::Data obj_concat =
265  data_pair.second.concatenate(std::move(add_data_pair.second), 0);
266  data_pair.second = std::move(obj_concat);
267  }
268  return data_pair;
269  }
270  inline void reset() {
271  std::size_t max_ind = std::min(current_stream, (std::size_t) (data_streams.size() - 1));
272  for (std::size_t i = 0; i <= max_ind; ++i) {
273  std::ifstream& data_stream = data_streams[i];
274  data_stream.clear();
275  _set_to_beg(data_stream);
276  }
277  current_stream = 0;
278  }
279 protected:

282  current_stream(0) {
283  assert(!dataset_paths.empty());
284  for (std::size_t i = 0; i < dataset_paths.size(); ++i) {
285  data_streams[i] = std::ifstream(dataset_paths[i], binary ? std::ios::binary : std::ios::in);
286  std::ifstream& data_stream = data_streams[i];
287  assert(data_stream.is_open());
288  _set_to_beg(data_stream);
289  }
290  }

467 public:
472  MNISTDataProvider(std::string obs_path, std::string labels_path) :
473  Base::SplitFileDataProvider(std::make_pair(obs_path, labels_path), true, true),

481  }

484  Base::_set_to_beg(obs_input_stream, obj_input_stream);
485  obs_input_stream.ignore(OBS_OFFSET);
486  obj_input_stream.ignore(LABEL_OFFSET);
487  }
488  inline DataPair<Scalar,3,false> _get_data(std::ifstream& obs_input_stream,
489  std::ifstream& obj_input_stream, std::size_t batch_size) {
490  Tensor<Scalar,4> obs(batch_size, 28u, 28u, 1u);
491  Tensor<Scalar,4> obj(batch_size, 10u, 1u, 1u);
492  obj.setZero();
493  std::size_t i;
494  for (i = 0; i < batch_size && obs_input_stream.read(obs_buffer, OBS_INSTANCE_LENGTH); ++i) {
495  // Read and set the label.
496  unsigned char label;
497  obj_input_stream >> label;
498  obj(i,(std::size_t) label,0u,0u) = (Scalar) 1;
499  // Set the image.
500  unsigned char* u_buffer = reinterpret_cast<unsigned char*>(obs_buffer);
501  std::size_t buffer_ind = 0;
502  for (std::size_t height = 0; height < 28; ++height) {
503  for (std::size_t width = 0; width < 28; ++width)
504  obs(i,height,width,0u) = (Scalar) u_buffer[buffer_ind++];
505  }
506  assert(buffer_ind == OBS_INSTANCE_LENGTH);
507  }
508  if (i == batch_size)
509  return std::make_pair(obs, obj);
510  std::array<std::size_t,4> offsets({ 0u, 0u, 0u, 0u });
511  std::array<std::size_t,4> obs_extents({ i, 28u, 28u, 1u });
512  std::array<std::size_t,4> obj_extents({ i, 10u, 1u, 1u });
513  return std::make_pair(obs.slice(offsets, obs_extents), obj.slice(offsets, obj_extents));
514  }
515  inline std::size_t _skip(std::ifstream& obs_input_stream, std::ifstream& obj_input_stream,
516  std::size_t instances) {
517  // Skip observations.
518  std::streampos curr_obs_pos = obs_input_stream.tellg();
519  obs_input_stream.seekg(0, std::ios::end);
520  std::size_t obs_skip_extent = obs_input_stream.tellg() - curr_obs_pos;
521  obs_input_stream.seekg(curr_obs_pos);
522  obs_input_stream.ignore(instances * OBS_INSTANCE_LENGTH);
523  std::size_t skipped_obs = std::min(instances, obs_skip_extent / OBS_INSTANCE_LENGTH);
524  // Skip labels.
525  std::streampos curr_label_pos = obj_input_stream.tellg();
526  obj_input_stream.seekg(0, std::ios::end);
527  std::size_t label_skip_extent = obj_input_stream.tellg() - curr_label_pos;
528  obj_input_stream.seekg(curr_label_pos);
529  obj_input_stream.ignore(instances * LABEL_INSTANCE_LENGTH);
530  std::size_t skipped_labels = std::min(instances, label_skip_extent / LABEL_INSTANCE_LENGTH);
531  assert(skipped_obs == skipped_labels);
532  return skipped_obs;
533  }
534 private:

536  const Dimensions<std::size_t,3> obj;

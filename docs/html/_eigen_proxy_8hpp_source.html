147 template<typename Scalar>
148 inline void serialize_binary(const Matrix<Scalar>& matrix, std::ostream& out_stream) {
149  unsigned short scalar_size = static_cast<unsigned short>(sizeof(Scalar));
150  out_stream.write(reinterpret_cast<const char*>(&scalar_size),
151  std::streamsize(sizeof(unsigned short)));
152  unsigned rows = static_cast<unsigned>(matrix.rows());
153  unsigned cols = static_cast<unsigned>(matrix.cols());
154  out_stream.write(reinterpret_cast<const char*>(&rows), std::streamsize(sizeof(unsigned)));
155  out_stream.write(reinterpret_cast<const char*>(&cols), std::streamsize(sizeof(unsigned)));
156  out_stream.write(reinterpret_cast<const char*>(matrix.data()),
157  std::streamsize(matrix.size() * sizeof(Scalar)));
158  out_stream << std::flush;
159 }

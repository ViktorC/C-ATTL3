38  inline FeedforwardNeuralNetwork(std::vector<LayerPtr<Scalar,Rank>>&& layers, bool foremost = true) :
39  layers(std::move(layers)),
40  foremost(foremost) {
41  assert(this->layers.size() > 0 && "layers must contain at least 1 element");
42  assert(this->layers[0] != nullptr);
43  Layer<Scalar,Rank>& first_layer = *this->layers[0];
44  input_dims = first_layer.get_input_dims();
45  output_dims = this->layers[this->layers.size() - 1]->get_output_dims();
46  typename Base::Dims prev_dims = first_layer.get_output_dims();
47  for (std::size_t i = 1; i < this->layers.size(); ++i) {
48  assert(this->layers[i] != nullptr && "layers contains null pointers");
49  assert(prev_dims == this->layers[i]->get_input_dims() && "incompatible layer dimensions");
50  prev_dims = this->layers[i]->get_output_dims();
51  }
52  first_layer.set_input_layer(foremost);
53  }

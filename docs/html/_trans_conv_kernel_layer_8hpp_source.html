115  inline Tensor<Scalar,4> _pass_forward(Tensor<Scalar,4> in, bool training) {
116  std::size_t rows = in.dimension(0);
117  std::size_t depth = ext_input_dims(2);
118  std::size_t receptor_vol = Base::weights->get_values().cols();
119  std::size_t total_patches = rows * patches_per_sample;
120  in_mat_cache = MatrixMap<Scalar>(in.data(), total_patches, depth);
121  Matrix<Scalar> out_conv_mat = in_mat_cache * Base::weights->get_values();
122  /* Given the values of the stretched out receptor patches, accumulate them in the output tensor. */
123  Tensor<Scalar,4> out(rows, padded_height, padded_width, ext_output_dims(2));
124  out.setZero();
125  std::size_t patch_ind = 0;
126  patch_extents[0] = rows;
127  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
128  patch_offsets[2] = i;
129  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
130  patch_offsets[1] = j;
131  // Accumulate the gradients where the receptor-patch-tensors overlap.
132  Matrix<Scalar> out_conv_mat_block = out_conv_mat.block(patch_ind, 0, rows, receptor_vol);
133  TensorMap<Scalar,4> out_patch(out_conv_mat_block.data(), rows, receptor_height,
134  receptor_width, ext_output_dims(2));
135  if (vertical_dilation > 0 || horizontal_dilation > 0)
136  out.slice(patch_offsets, patch_extents).stride(dil_strides) += out_patch;
137  else
138  out.slice(patch_offsets, patch_extents) += out_patch;
139  patch_ind += rows;
140  }
141  }
142  assert(patch_ind == total_patches);
143  if (vertical_padding > 0 || horizontal_padding > 0) {
144  // Cut off the padding.
145  no_padding_extents[0] = rows;
146  out = Tensor<Scalar,4>(out.slice(no_padding_offsets, no_padding_extents));
147  }
148  MatrixMap<Scalar> out_mat(out.data(), rows, ext_output_dims.get_volume());
149  out_mat.rowwise() += Base::bias->get_values().row(0);
150  return out;
151  }
152  inline Tensor<Scalar,4> _pass_back(Tensor<Scalar,4> out_grad) {
153  std::size_t rows = out_grad.dimension(0);
154  std::size_t depth = ext_input_dims(2);
155  std::size_t receptor_vol = Base::weights->get_values().cols();
156  std::size_t total_patches = rows * patches_per_sample;
157  std::size_t patch_ind = 0;
158  patch_extents[0] = rows;
159  // Compute the gradient of the bias.
160  Base::bias->accumulate_grad(MatrixMap<Scalar>(out_grad.data(), rows,
161  ext_output_dims.get_volume()).colwise().sum());
162  // Spatial padding.
163  if (vertical_padding > 0 || horizontal_padding > 0)
164  out_grad = Tensor<Scalar,4>(out_grad.pad(paddings));
165  Matrix<Scalar> out_grad_conv_mat(total_patches, receptor_vol);
166  for (std::size_t i = 0; i <= padded_width - dil_receptor_width; i += horizontal_stride) {
167  patch_offsets[2] = i;
168  for (std::size_t j = 0; j <= padded_height - dil_receptor_height; j += vertical_stride) {
169  patch_offsets[1] = j;
170  Tensor<Scalar,4> patch;
171  // If the patch is dilated, skip the spatial gaps when flattening it into a matrix.
172  if (vertical_dilation > 0 || horizontal_dilation > 0)
173  patch = out_grad.slice(patch_offsets, patch_extents).stride(dil_strides);
174  else
175  patch = out_grad.slice(patch_offsets, patch_extents);
176  out_grad_conv_mat.block(patch_ind, 0, rows, receptor_vol) = MatrixMap<Scalar>(patch.data(),
177  rows, receptor_vol);
178  patch_ind += rows;
179  }
180  }
181  assert(patch_ind == total_patches);
182  Base::weights->accumulate_grad(in_mat_cache.transpose() * out_grad_conv_mat);
183  if (Base::is_input_layer())
184  return Tensor<Scalar,4>();
185  Matrix<Scalar> prev_out_grad = out_grad_conv_mat * Base::weights->get_values().transpose();
186  prev_out_conversion_dims[0] = rows;
187  return TensorMap<Scalar,4>(prev_out_grad.data(), prev_out_conversion_dims);
188  }

200  inline static std::size_t calculate_spatial_output_dim(std::size_t input_dim, std::size_t receptor_size,
201  std::size_t padding, std::size_t dilation, std::size_t stride) {
202  return (input_dim - 1) * stride + receptor_size + (receptor_size - 1) * dilation - 2 * padding;
203  }
204  inline static Dimensions<std::size_t,3> calculate_output_dims(const Dimensions<std::size_t,3>& input_dims,
205  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width,
206  std::size_t vertical_padding, std::size_t horizontal_padding, std::size_t vertical_stride,
207  std::size_t horizontal_stride, std::size_t vertical_dilation, std::size_t horizontal_dilation) {
208  return { calculate_spatial_output_dim(input_dims(0), receptor_height, vertical_padding, vertical_dilation,
209  vertical_stride), calculate_spatial_output_dim(input_dims(1), receptor_width,
210  horizontal_padding, horizontal_dilation,horizontal_stride), filters };
211  }
212  inline static Dimensions<std::size_t,Rank> calculate_adjusted_output_dims(const typename Root::Dims& input_dims,
213  std::size_t filters, std::size_t receptor_height, std::size_t receptor_width,
214  std::size_t vertical_padding, std::size_t horizontal_padding, std::size_t vertical_stride,
215  std::size_t horizontal_stride, std::size_t vertical_dilation, std::size_t horizontal_dilation) {
216  auto output_dims = calculate_output_dims(input_dims.template extend<3 - Rank>(), filters, receptor_height,
217  receptor_width, vertical_padding, horizontal_padding, vertical_stride, horizontal_stride,
218  vertical_dilation, horizontal_dilation);
219  output_dims(2) /= filters;
220  output_dims(Rank - 1) *= filters;
221  return output_dims.template contract<3 - Rank>();
222  }
223  const Dimensions<std::size_t,3> ext_input_dims, ext_output_dims;
224  // Pre-computed values to improve propagation-time performance.
225  const std::size_t padded_height, padded_width, dil_receptor_height, dil_receptor_width, patches_per_sample;
226  Array4 prev_out_conversion_dims, patch_offsets, patch_extents, dil_strides,
227  no_padding_offsets, no_padding_extents;
228  PaddingsArray4 paddings;
229  // Staged computation caches
230  Matrix<Scalar> in_mat_cache;
231 };

126  inline typename Base::Data propagate(typename Base::Data input, bool training) {
127  assert(input_dims == (Dimensions<std::size_t,Base::DATA_RANK>(input.dimensions()).template demote<2>()));
128  pthread_attr_t attr;
129  pthread_t helper_thread;
130  int pthread_state;
131  pthread_state = pthread_attr_init(&attr);
132  assert(!pthread_state);
133  pthread_state = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
134  assert(!pthread_state);
135  PropArgs args;
136  args.obj = this;
137  args.training = training;
138  args.in = &input;
139  pthread_state = pthread_create(&helper_thread, &attr, propagate, &args);
140  assert(!pthread_state);
141  typename Base::Data forward_out = net->propagate(input, training);
142  pthread_state = pthread_join(helper_thread, nullptr);
143  assert(!pthread_state);
144  pthread_state = pthread_attr_destroy(&attr);
145  assert(!pthread_state);
146  assert(forward_out.dimension(1) == args.out.dimension(1));
147  if (MergeType == BIDIRECTIONAL_SUM)
148  return forward_out + args.out;
149  else if (MergeType == BIDIRECTIONAL_MUL) {
150  output = std::move(forward_out);
151  output_rev = std::move(args.out);
152  return output * output_rev;
153  } else
154  return forward_out.concatenate(args.out, +CONCAT_BATCH_RANK);
155  }
156  inline typename Base::Data backpropagate(typename Base::Data out_grad) {
157  Dimensions<std::size_t,Base::DATA_RANK> dims(out_grad.dimensions());
158  assert(output_dims == dims.template demote<2>());
159  pthread_attr_t attr;
160  pthread_t helper_thread;
161  int pthread_state;
162  pthread_state = pthread_attr_init(&attr);
163  assert(!pthread_state);
164  pthread_state = pthread_attr_setdetachstate(&attr, PTHREAD_CREATE_JOINABLE);
165  assert(!pthread_state);
166  BackpropArgs args;
167  args.obj = this;
168  typename Base::Data forward_prev_out_grad;
169  if (MergeType == BIDIRECTIONAL_SUM) {
170  args.out_grad = &out_grad;
171  pthread_state = pthread_create(&helper_thread, &attr, backpropagate, &args);
172  assert(!pthread_state);
173  forward_prev_out_grad = net->backpropagate(out_grad);
174  pthread_state = pthread_join(helper_thread, nullptr);
175  assert(!pthread_state);
176  out_grad = typename Base::Data();
177  } else if (MergeType == BIDIRECTIONAL_MUL) {
178  typename Base::Data out_grad_rev = output * out_grad;
179  args.out_grad = &out_grad_rev;
180  pthread_state = pthread_create(&helper_thread, &attr, backpropagate, &args);
181  assert(!pthread_state);
182  out_grad *= output_rev;
183  forward_prev_out_grad = net->backpropagate(std::move(out_grad));
184  pthread_state = pthread_join(helper_thread, nullptr);
185  assert(!pthread_state);
186  } else {
187  RankwiseArray offsets;
188  RankwiseArray extents = dims;
189  offsets.fill(0);
190  extents[+CONCAT_BATCH_RANK] /= 2;
191  offsets[+CONCAT_BATCH_RANK] += extents[+CONCAT_BATCH_RANK];
192  typename Base::Data backward_slice = out_grad.slice(offsets, extents);
193  args.out_grad = &backward_slice;
194  pthread_state = pthread_create(&helper_thread, &attr, backpropagate, &args);
195  assert(!pthread_state);
196  offsets[+CONCAT_BATCH_RANK] -= extents[+CONCAT_BATCH_RANK];
197  typename Base::Data forward_slice = out_grad.slice(offsets, extents);
198  out_grad = typename Base::Data();
199  forward_prev_out_grad = net->backpropagate(std::move(forward_slice));
200  // Make sure that backward_slice does not go out of scope before the thread terminates.
201  pthread_state = pthread_join(helper_thread, nullptr);
202  assert(!pthread_state);
203  }
204  pthread_state = pthread_attr_destroy(&attr);
205  assert(!pthread_state);
206  return forward_prev_out_grad + args.prev_out_grad;
207  }

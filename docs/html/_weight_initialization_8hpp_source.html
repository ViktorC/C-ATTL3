81 template<typename Scalar>
82 class GaussianWeightInitialization : public WeightInitialization<Scalar> {
83 public:
84  inline GaussianWeightInitialization(Scalar scaling_factor, Scalar bias) :
85  bias(bias),
86  scaling_factor(scaling_factor) { }
87  virtual ~GaussianWeightInitialization() = default;
88  inline virtual void apply(Matrix<Scalar>& weights) const {
89  int rows = weights.rows();
90  int cols = weights.cols();
91  std::default_random_engine gen;
92  std::normal_distribution<Scalar> dist(0, scaling_factor * sd(rows - 1, cols));
93  for (int i = 0; i < rows; ++i) {
94  if (i == rows - 1) // Bias row.
95  weights.row(i).setConstant(bias);
96  else {
97  for (int j = 0; j < cols; ++j)
98  weights(i,j) = dist(gen);
99  }
100  }
101  }
102 protected:
113  virtual Scalar sd(unsigned fan_ins, unsigned fan_outs) const = 0;
114 private:
115  const Scalar scaling_factor;
116  const Scalar bias;
117 };

196  weights.block(0, 0, rows, cols) = more_rows ?
197  svd.compute(weights, internal::SVDOptions::ComputeFullU).matrixU().block(0, 0, rows, cols) :
198  svd.compute(weights, internal::SVDOptions::ComputeFullV).matrixV().block(0, 0, rows, cols);
199  }
200 protected:

81 template<typename Scalar>
82 class IncrementalWeightInitialization : public WeightInitialization<Scalar> {
83 public:
90  inline IncrementalWeightInitialization(Scalar min = 0, Scalar max = .5, Scalar bias = 0) :
91  bias(bias),
92  min(min),
93  max(max) {
94  assert(max > min);
95  }
96  inline void apply(Matrix<Scalar>& weights) const {
97  std::size_t rows = weights.rows() - 1;
98  std::size_t cols = weights.cols();
99  std::size_t elements = (weights.rows() - 1) * weights.cols();
100  Scalar step_size = (max - min) / elements;
101  Scalar val = min;
102  for (std::size_t i = 0; i < cols; ++i) {
103  for (std::size_t j = 0; j < rows; ++j) {
104  weights(j,i) = val;
105  val += step_size;
106  }
107  }
108  weights.row(rows).setConstant(bias);
109  }
110 private:
111  const Scalar min;
112  const Scalar max;
113  const Scalar bias;
114 };

120 template<typename Scalar>
121 class GaussianWeightInitialization : public WeightInitialization<Scalar> {
122 public:
123  inline GaussianWeightInitialization(Scalar sd_scaling_factor = 1, Scalar bias = 0) :
124  bias(bias),
125  sd_scaling_factor(sd_scaling_factor) { }
126  virtual ~GaussianWeightInitialization() = default;
127  inline virtual void apply(Matrix<Scalar>& weights) const {
128  int rows = weights.rows();
129  int cols = weights.cols();
130  std::default_random_engine gen;
131  std::normal_distribution<Scalar> dist(0, sd_scaling_factor * _sd(rows - 1, cols));
132  for (int i = 0; i < rows; ++i) {
133  if (i == rows - 1) // Bias row.
134  weights.row(i).setConstant(bias);
135  else {
136  for (int j = 0; j < cols; ++j)
137  weights(i,j) = dist(gen);
138  }
139  }
140  }
141 protected:
152  virtual Scalar _sd(unsigned fan_ins, unsigned fan_outs) const {
153  return 1;
154  }
155 private:
156  const Scalar sd_scaling_factor;
157  const Scalar bias;
158 };

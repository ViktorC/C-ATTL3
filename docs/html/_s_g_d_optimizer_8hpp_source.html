18 template<typename Scalar, std::size_t Rank, bool Sequential>
19 class SGDOptimizer : public Optimizer<Scalar,Rank,Sequential> {
20  typedef Optimizer<Scalar,Rank,Sequential> Base;
21 public:
22  inline SGDOptimizer(LossSharedPtr<Scalar,Rank,Sequential> loss, std::size_t batch_size) :
23  Base::Optimizer(loss),
24  batch_size(batch_size) {
25  assert(batch_size > 0);
26  }
27  virtual ~SGDOptimizer() = default;
28  inline void fit(typename Base::Net& net) {
29  target_net_ptr = &net;
30  _fit(net.get_all_unique_params());
31  }
32 protected:
33  inline Scalar _train(typename Base::Net& net, typename Base::Provider& training_prov, std::size_t epoch,
34  bool verbose) {
35  assert(target_net_ptr == &net);
36  Scalar obj_loss = 0;
37  Scalar reg_loss = 0;
38  std::size_t instances = 0;
39  std::size_t updates = 0;
40  // Get all the optimizable parameters.
41  std::vector<Parameters<Scalar>*> params_vec = net.get_all_unique_params();
42  // Perform an entire training epoch.
43  while (training_prov.has_more()) {
44  DataPair<Scalar,Rank,Sequential> data_pair = training_prov.get_data(batch_size);
45  instances += data_pair.first.dimension(0);
46  typename Base::Data out = net.propagate(std::move(data_pair.first), true);
47  obj_loss += Base::loss->function(out, data_pair.second).sum();
48  /* Divide the gradient by the batch size to decouple the learning rate and the batch
49  * size hyper-parameters. Use the nominal batch size as the denominator even if the
50  * actual batch size is different (in case the number of samples in the data set is
51  * not divisible by the batch size and the last batch of the epoch contains fewer
52  * instances than the others) to make sure that the magnitude of the gradient is
53  * proportional to the batch size (just like its 'accuracy' is). */
54  net.backpropagate(Base::loss->d_function(std::move(out),
55  std::move(data_pair.second)) / (Scalar) batch_size);
56  // Update the values of the parameters.
57  std::size_t i = 0;
58  for (auto params_ptr : params_vec) {
59  if (!params_ptr->are_optimizable() || params_ptr->are_frozen())
60  continue;
61  reg_loss += params_ptr->get_regularization_penalty();
62  params_ptr->regularize();
63  }
64  _update_params(params_vec, epoch - 1);
65  ++updates;
66  // Reset the gradients of the optimizable parameters.
67  for (auto params_ptr : params_vec)
68  params_ptr->reset_grad();
69  }
70  Scalar mean_obj_loss = obj_loss / instances;
71  Scalar mean_reg_loss = reg_loss / updates;
72  if (verbose) {
73  std::cout << std::left << std::setw(20) << "\ttraining obj loss: " << std::right <<
74  std::to_string(mean_obj_loss) << std::endl;
75  std::cout << std::left << std::setw(20) << "\ttraining reg loss: " << std::right <<
76  std::to_string(mean_reg_loss) << std::endl;
77  }
78  return mean_obj_loss + mean_reg_loss;
79  }
80  inline Scalar _test(typename Base::Net& net, typename Base::Provider& test_prov, std::size_t epoch,
81  bool verbose) {
82  assert(target_net_ptr == &net);
83  Scalar obj_loss = 0;
84  Scalar instances = 0;
85  std::vector<Parameters<Scalar>*> params_vec = net.get_all_unique_params();
86  // Perform an entire test epoch.
87  while (test_prov.has_more()) {
88  DataPair<Scalar,Rank,Sequential> data_pair = test_prov.get_data(batch_size);
89  instances += data_pair.first.dimension(0);
90  obj_loss += Base::loss->function(net.infer(std::move(data_pair.first)),
91  std::move(data_pair.second)).sum();
92  }
93  Scalar mean_obj_loss = obj_loss / instances;
94  Scalar reg_loss = 0;
95  for (auto params_ptr : params_vec) {
96  if (params_ptr->are_optimizable() && !params_ptr->are_frozen())
97  reg_loss += params_ptr->get_regularization_penalty();
98  }
99  if (verbose) {
100  std::cout << std::left << std::setw(20) << "\ttest obj loss: " << std::right <<
101  std::to_string(mean_obj_loss) << std::endl;
102  std::cout << std::left << std::setw(20) << "\ttest reg loss: " << std::right <<
103  std::to_string(reg_loss) << std::endl;
104  }
105  return mean_obj_loss + reg_loss;
106  }
112  virtual void _fit(const std::vector<Parameters<Scalar>*>& params_vec) = 0;
119  virtual void _update_params(const std::vector<Parameters<Scalar>*>& params_vec, std::size_t epoch) = 0;
120  const std::size_t batch_size;
121 private:
122  const typename Base::Net* target_net_ptr;
123 };

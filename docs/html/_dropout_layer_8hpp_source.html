76  assert(in.dimension(0) > 0);

80  dropout_mask = in.random().unaryExpr([this,scaling_factor](Scalar e) {
81  return (Scalar) (e <= dropout_prob ? 0 : scaling_factor);

83  return in * dropout_mask;

89  assert(out_grad.dimension(0) > 0 && dropout_mask.dimension(0) == out_grad.dimension(0));
90  if (input_layer)
91  return typename Base::Data();
92  // The derivative of the dropout function.
93  return out_grad * dropout_mask;
94  }

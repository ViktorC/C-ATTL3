66  inline void _update_params(const std::vector<Parameters<Scalar>*>& params_vec, std::size_t epoch) {
67  Scalar l1_corr = (Scalar) 1 / (1 - pow(1 - l1_decay, epoch + 1) + epsilon);
68  Scalar l2_corr = (Scalar) 1 / (1 - pow(1 - l2_decay, epoch + 1) + epsilon);
69  std::size_t i = 0;
70  for (auto params_ptr : params_vec) {
71  if (params_ptr->are_frozen())
72  continue;
73  ParamsGradNorms& grad_norms = pgn_vec[i++];
74  const Matrix<Scalar>& params_grad = params_ptr->get_grad();
75  grad_norms.params_grad_l1 = grad_norms.params_grad_l1 * (1 - l1_decay) + params_grad * l1_decay;
76  grad_norms.params_grad_l2 = grad_norms.params_grad_l2 * (1 - l2_decay) +
77  params_grad.cwiseProduct(params_grad) * l2_decay;
78  params_ptr->set_values(params_ptr->get_values() -
79  ((grad_norms.params_grad_l1 * (learning_rate * l1_corr)).array() /
80  ((grad_norms.params_grad_l2 * l2_corr).array() + epsilon).sqrt()).matrix());
81  }
82  }
